156 State update() {
157 int32_t percent = 100;
158 int32_t rssiAvg = 0;
159
160 if (connection->state == Connection::State::CONNECTED) {
161 if (qualityTimeout.check()) {
162 qualityTimeout.set(RESET_CYCLE_PERIOD);
163 previousRetries = connection->info.retriesCounter;
164 if (assessmentMode == Mode::RUNTIME) {
165 if (retriesPenalty > 0) --retriesPenalty;
166 if (transmitTimePenalty > 0) --transmitTimePenalty;
167 if (rssiPenalty > 0) --rssiPenalty;
168 }
169 }
170 if (connection->info.retriesCounter != previousRetries) {
171 if (connection->info.retriesCounter > previousRetries) {
172 retriesPenalty = std::min(retriesPenalty + 50, 100);
173 }
174 previousRetries = connection->info.retriesCounter;
175 }
176 if (previousTransmitPeriod != connection->info.transmitPeriod.getRawValue()) {
177 previousTransmitPeriod = connection->info.transmitPeriod.getRawValue();
178 if (connection->info.transmitPeriod > TRANSMIT_TIME_THRESHOLD) {
179 transmitTimePenalty = std::min(transmitTimePenalty + 5, 100);
180 }
181 }
182 rssiAvg = connection->info.signalStrength;
183 rssiAvg = ((previousRssi * 2) + rssiAvg) / 3;
184 previousRssi = static_cast<uint8_t>(rssiAvg);
185 if (rssiAvg < RSSI_MIN_THRESHOLD) rssiPenalty = (RSSI_MIN_THRESHOLD - rssiAvg);
186 percent -= retriesPenalty;
187 percent -= transmitTimePenalty;
188 percent -= rssiPenalty;
189 if (percent < 0) percent = 0;
190 qualityPercent = static_cast<uint8_t>(percent);
191 if (qualityPercent > 90) qualityState = State::AMAZING;
192 else if (qualityPercent > 75) qualityState = State::VERY_GOOD;
193 else if (qualityPercent > 50) qualityState = State::OKAY;
194 else if (qualityPercent > 30) qualityState = State::NOT_GOOD;
195 else qualityState = State::UNUSABLE;
196 if (onUpdate) onUpdate(this);
197 } else {
198 qualityState = State::DISCONNECTED;
199 qualityPercent = 0;
200 }
201 return qualityState;
202 }
