Lib::Canbus::ClientBase

Example usage of the CAN bus client base class.

This example demonstrates how to derive from the CAN bus client base class and implement the required task function for periodic processing.

Example usage:
#include "ClientBase.hpp"
class MyCanClient : public Lib::Canbus::ClientBase {
public:
explicit MyCanClient(CAN_HandleTypeDef *handle) : ClientBase(handle) {}
void taskFn() override {
// Implement periodic or event-driven logic here.
// For example, send a message every cycle.
send();
}
};
int main() {
CAN_HandleTypeDef canHandle{};
MyCanClient client(&canHandle);
client.start(1);
// Main loop
while (true) {
client.taskFn();
}
return 0;
}
