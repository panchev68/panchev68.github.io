248 void handleTickEvent() {
249 if (++tickCounter % 1000 != 0) return;
250 if (!lock(0)) return;
251
252 if (readDateTime(dateTime)) {
253 if (dateTime.time.seconds != lastSecond) {
254 lastSecond = dateTime.time.seconds;
255 if (onTimeChanged) onTimeChanged(dateTime);
256 }
257 }
258 unlock();
259 }
