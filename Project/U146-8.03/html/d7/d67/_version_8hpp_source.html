36 static Lib::Helper::DateTime stringToDateTime(const char * timeStr, const char * dateStr) {
37 assert(timeStr && dateStr);
38 Lib::Helper::DateTime dt{};
39 // Parse time: "HH:MM:SS"
40 dt.time.hours = static_cast<uint8_t>((timeStr[0] - '0') * 10 + (timeStr[1] - '0'));
41 dt.time.minutes = static_cast<uint8_t>((timeStr[3] - '0') * 10 + (timeStr[4] - '0'));
42 dt.time.seconds = static_cast<uint8_t>((timeStr[6] - '0') * 10 + (timeStr[7] - '0'));
43 // Parse date: "Mon DD YYYY"
44 constexpr std::array<const char *, 12> months = {"Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
45 uint8_t month = 1;
46 for (uint8_t i = 0; i < months.size(); ++i) {
47 if (dateStr[0] == months[i][0] && dateStr[1] == months[i][1] && dateStr[2] == months[i][2]) {
48 month = i + 1;
49 break;
50 }
51 }
52 dt.date.month = month;
53 dt.date.day = static_cast<uint8_t>((dateStr[4] == ' ' ? 0 : (dateStr[4] - '0')) * 10 + (dateStr[5] - '0'));
54 dt.date.year = static_cast<uint16_t>((dateStr[7] - '0') * 1000 + (dateStr[8] - '0') * 100 + (dateStr[9] - '0') * 10 + (dateStr[10] - '0'));
55 return dt;
56 }

63 static uint32_t stringToU32(const std::string & string) {
64 uint32_t major = 0, minor = 0;
65 size_t dot = string.find('.');
66 if (dot != std::string::npos) {
67 major = static_cast<uint32_t>(std::stoul(string.substr(0, dot)));
68 minor = static_cast<uint32_t>(std::stoul(string.substr(dot + 1)));
69 } else {
70 major = static_cast<uint32_t>(std::stoul(string));
71 }
72 return (major << 8) | minor;
73 }
