143 void handleScanRequest() {
144 if (!impl->timeout.check()) return;
145 if (!impl->localAtCommand->send({"ED"})) {
146 impl->timeout.set(SCAN_ERROR_RETRY_MS);
147 return;
148 }
149 impl->timeout.set(SCAN_RESPONSE_TIMEOUT_MS);
150 impl->hasWaitForResponse = true;
151 }
152
156 void handleScanResponse() {
157 static std::vector<uint8_t> response;
158 if (impl->timeout.check()) {
159 impl->hasWaitForResponse = false;
160 impl->timeout.set(SCAN_ERROR_RETRY_MS);
161 return;
162 }
163 if (!impl->localAtCommand->receive(response, 0)) return;
164 processSpectrumData(response);
165 impl->isNewDataFlag = true;
166 impl->hasWaitForResponse = false;
167 impl->timeout.set(SCAN_INTERVAL_MS);
168 }
169
174 void processSpectrumData(std::vector<uint8_t> & response) {
175 if (response.size() != impl->currentArray.size()) return;
176 std::transform(response.begin(), response.end(), response.begin(),
177 [](uint8_t value) { return Lib::Helper::Convert::dBM::toDigiPercent(value); });
178 for (size_t i = 0; i < response.size(); ++i) { updateChannelData(i, response[i]); }
179 }
180
186 void updateChannelData(size_t index, uint8_t newValue) {
187 const uint8_t currentValue = static_cast<uint8_t>((newValue + impl->currentArray[index]) / 2);
188 impl->currentArray[index] = currentValue;
189 updateAverageValue(index, currentValue);
190 updateMaxValue(index, currentValue);
191 }
192
198 void updateAverageValue(size_t index, uint8_t currentValue) {
199 auto & avg = impl->averageArray[index];
200 if (avg > currentValue) avg -= 1;
201 else if (avg < currentValue) avg += 2;
202 }
203
209 void updateMaxValue(size_t index, uint8_t currentValue) {
210 auto & maxVal = impl->maxArray[index];
211 if (currentValue > maxVal) maxVal = currentValue;
212 }
