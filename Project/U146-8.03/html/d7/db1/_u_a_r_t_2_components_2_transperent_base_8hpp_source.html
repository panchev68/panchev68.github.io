229 void emitCallback(HAL_UART_CallbackIDTypeDef id) {
230 if (id == HAL_UART_TX_COMPLETE_CB_ID) {
231 if (dir & DriverDataDirection::TX) {
232 if (!semaphoreTxCompleted.giveFromISR()) assert(false);
233 }
234 } else if (id == HAL_UART_RX_COMPLETE_CB_ID) {
235 if (dir & DriverDataDirection::RX) {
236 __HAL_UART_FLUSH_DRREGISTER(getHandle());
237 if (msgBufferRxData.sendFromISR(&rxChar, sizeof(uint8_t)) != sizeof(uint8_t)) assert(false);
238 beginReceive();
239 }
240 } else if (id == HAL_UART_ERROR_CB_ID) {
241 __HAL_UART_FLUSH_DRREGISTER(getHandle());
242 auto errorCode = HAL_UART_GetError(getHandle());
243 __HAL_UART_CLEAR_FLAG(getHandle(), UART_IT_ERR);
244 if (errorCode & (HAL_UART_ERROR_ORE | HAL_UART_ERROR_FE | HAL_UART_ERROR_NE | HAL_UART_ERROR_PE))
245 errorCallback(errorCode);
246 if (dir & DriverDataDirection::RX) beginReceive();
247 }
