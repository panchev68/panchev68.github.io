159 void handleReceivedChar(RxCharParameters & params) {
160 // Timeout handling for in-progress reception
161 if (params.inProgress && params.timeout.check()) {
162 diagnostic.ignoredMessagesByTimeout++;
163 params.rxFrame.clear();
164 params.remain = -1;
165 params.inProgress = false;
166 }
167
168 const size_t pos = params.rxFrame.getSize();
169
170 // Start delimiter detection
171 if (pos == offsetof(Frame::Header, Frame::Header::startDelimiter)) {
172 if (params.rxChar == Protocol::START_DELIMITER) {
173 if (params.inProgress) { assert(false); }
174 signalBeginMessageReceive();
175 params.inProgress = true;
176 params.rxFrame.clear();
177 params.remain = -1;
178 params.rxFrame.append(params.rxChar);
179 params.timeout.set(5);
180 } else {
181 params.timeout.set(5);
182 }
183 return;
184 }
185
186 params.rxFrame.append(params.rxChar);
187
188 // Length MSB
189 if (pos == offsetof(Frame::Header, Frame::Header::lengthMSB)) {
190 params.payloadLen = static_cast<size_t>(params.rxChar) << 8;
191 return;
192 }
193
194 // Length LSB
195 if (pos == offsetof(Frame::Header, Frame::Header::lengthLSB)) {
196 params.payloadLen |= params.rxChar;
197 constexpr size_t MAX_PAYLOAD = (8 * 8) + 1;
198 if (params.payloadLen >= MAX_PAYLOAD) {
199 diagnostic.maxMsgLengthErrors++;
200 params.rxFrame.clear();
201 signalEndMessageReceived();
202 params.inProgress = false;
203 params.remain = -1;
204 return;
205 }
206 params.remain = static_cast<int32_t>(params.payloadLen) + static_cast<int32_t>(sizeof(uint8_t));
207 return;
208 }
209
210 // Payload and checksum
211 if (params.remain > 0) {
212 params.remain--;
213 }
214 if (params.remain == 0) {
215 if (Protocol::instance().validate(params.rxFrame)) {
216 Protocol::instance().handleIncomingMessage(params.rxFrame);
217 } else {
218 diagnostic.invalidMessages++;
219 assert(false);
220 }
221 params.rxFrame.clear();
222 signalEndMessageReceived();
223 params.inProgress = false;
224 params.remain = -1;
225 }
226 }
