50 bool parse(const DigiApiFrameBuffer & apiFrame, Frame::Data & frameData) {
51 if (!validate(apiFrame)) return false;
52 const uint8_t * data = apiFrame.getData();
53 const auto * header = reinterpret_cast<const Frame::Header *>(data);
54 size_t payloadLen = (static_cast<size_t>(header->lengthMSB) << 8) | header->lengthLSB;
55 data += sizeof(Frame::Header);
56 if (payloadLen == 0) return false;
57 frameData.type = *(data++);
58 payloadLen--;
59 frameData.clear();
60 frameData.append(data, payloadLen);
61 return true;
62 }

69 bool handleIncomingMessage(const DigiApiFrameBuffer & apiFrame) {
70 auto frameData = std::make_unique<Frame::Data>();
71 ResponseEvents & eventHandler = ResponseEvents::instance();
72 if (!parse(apiFrame, *frameData)) {
73 assert(false);
74 return false;
75 }
76 switch (static_cast<Frame::Type>(frameData->type)) {
77 case Frame::Type::RECEIVE_PACKET_64BIT:
78 if (eventHandler.onReceivePacket64bit) eventHandler.onReceivePacket64bit(*frameData);

80 case Frame::Type::RECEIVE_PACKET_16BIT:
81 if (eventHandler.onReceivePacket16bit) eventHandler.onReceivePacket16bit(*frameData);
82 break;
83 case Frame::Type::LOCAL_AT_COMMAND_RESPONSE:

85 break;

98 case Frame::Type::REMOTE_AT_COMMAND_RESPONSE:

111 bool validate(const DigiApiFrameBuffer & apiFrame) {
112 size_t msgSize = apiFrame.getSize();
113 if (msgSize < sizeof(Frame::Header) + 1) return false;
114 const auto & header = *reinterpret_cast<const Frame::Header *>(apiFrame.getData());
115 if (header.startDelimiter != START_DELIMITER) return false;
116 size_t headerLen = (static_cast<size_t>(header.lengthMSB) << 8) | header.lengthLSB;
117 size_t crcLen = sizeof(Frame::Header) + headerLen;
118 if (crcLen > msgSize - 1) return false;
119 return Checksum(apiFrame) == apiFrame.getData()[crcLen];
120 }

127 bool validate(const uint8_t * data) {
128 assert(data);
129 const auto & header = *reinterpret_cast<const Frame::Header *>(data);
130 if (header.startDelimiter != START_DELIMITER) return false;
131 size_t headerLen = (static_cast<size_t>(header.lengthMSB) << 8) | header.lengthLSB;
132 size_t crcLen = sizeof(Frame::Header) + headerLen;
133 return Checksum(data) == data[crcLen];

141 const DigiApiFrameBuffer & handleOutgoingMessage(const Frame::Data & frameData) {
142 static std::unique_ptr<DigiApiFrameBuffer> apiFrame = std::make_unique<DigiApiFrameBuffer>();
143
144 assert(API_MODE == ApiMode::API1);
145 size_t frameSize = frameData.size() + 1;
146 apiFrame->clear();
147 apiFrame->append(START_DELIMITER);
148 apiFrame->append(static_cast<uint8_t>(frameSize >> 8));
149 apiFrame->append(static_cast<uint8_t>(frameSize & 0xFF));
150 apiFrame->append(frameData.type);
151 apiFrame->append(frameData.data(), frameData.size());
152 apiFrame->append(Checksum(*apiFrame));
153 return *apiFrame;
