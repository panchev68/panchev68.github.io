189 void invalidate() {
190 const int16_t yMax = scrollableContainer->getHeight();
191 int16_t ySelectBegin = 0, ySelectEnd = 0;
192 for (size_t i = 0; i < buttons.size(); ++i) {
193 auto * button = buttons[i];
194 const bool selected = static_cast<int16_t>(i) == selectPosition;
195 button->forceState(selected);
196 if (selected) {
197 ySelectBegin = button->getY();
198 ySelectEnd = ySelectBegin + button->getHeight();
199 }
200 }
201 if (ySelectEnd > yMax) {
202 scrollableContainer->moveChildrenRelative(0, yMax - ySelectEnd);
203 } else if (ySelectBegin < 0) {
204 scrollableContainer->moveChildrenRelative(0, -ySelectBegin);
205 }
206 scrollableContainer->invalidate();
207 }

213 void setup() {

215 if (buttonParm.width == SIZE_MAX || buttonParm.height == SIZE_MAX) {
216 const auto & rect = buttons[0]->getRect();
217 buttonParm.width = rect.width;
218 buttonParm.height = rect.height;
219 }
220 int16_t y = 0;
221 for (auto * button : buttons) {
222 button->setPosition(0, y, buttonParm.width, buttonParm.height);
223 button->setBoxWithBorderPosition(0, 0, buttonParm.width, buttonParm.height);
224 button->setTextY(buttonParm.textYOffset);
225 y += static_cast<int16_t>(buttonParm.height + buttonParm.buttonsOffset);
226 if (buttonParm.textColor != UINT32_MAX && buttonParm.textCursorColor != UINT32_MAX) {
227 button->setTextColors(buttonParm.textColor, buttonParm.textCursorColor);
228 }
229 if (buttonParm.boxColor != UINT32_MAX && buttonParm.boxCursorColor != UINT32_MAX) {
230 button->setBoxWithBorderColors(
231 buttonParm.boxColor, buttonParm.boxCursorColor, buttonParm.boxColor, buttonParm.boxCursorColor);
232 }
233 scrollableContainer->add(*button);
234 }
