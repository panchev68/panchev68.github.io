138 bool write8(uint32_t eepAddr, const void * src, size_t len) {
139 if (!impl->isReady) {
140 assert(false);
141 return false;
142 }
143 if ((eepAddr + len) > impl->chipSize) {
144 assert((eepAddr + len) <= impl->chipSize);
145 return false;
146 }
147 const uint8_t * ptr = static_cast<const uint8_t *>(src);
148 while (len > 0) {
149 const size_t pageOffset = eepAddr % impl->pageSize;
150 const size_t wlen = std::min(len, impl->pageSize - pageOffset);
151 const uint8_t addrH = static_cast<uint8_t>((impl->devAddr & 0xF0) | ((eepAddr >> 7) & 0x0E));
152 const uint8_t addrL = static_cast<uint8_t>(eepAddr & 0xFF);
153 if (!impl->i2c.send(addrH, HAL::I2C::Master::MemAddrSize::ADDR_8BITS, addrL,
154 const_cast<void *>(static_cast<const void *>(ptr)), wlen)) {
155 assert(false);
156 return false;
157 }
158 vTaskDelay(5);
159 eepAddr += wlen;
160 ptr += wlen;
161 len -= wlen;
162 }
163 return true;
164 }

174 bool write16(uint32_t eepAddr, const void * src, size_t len) {
175 if (!impl->isReady) {
176 assert(false);

179 if ((eepAddr + len) > impl->chipSize) {

182 }
183 const uint8_t * ptr = static_cast<const uint8_t *>(src);
184 std::vector<uint8_t> buf;
185 while (len > 0) {
186 const size_t pageOffset = eepAddr % impl->pageSize;
187 const size_t wlen = std::min(len, impl->pageSize - pageOffset);
188 buf.resize(2 + wlen);
189 buf[0] = static_cast<uint8_t>((eepAddr >> 8) & 0xFF);
190 buf[1] = static_cast<uint8_t>(eepAddr & 0xFF);
191 std::copy_n(ptr, wlen, buf.begin() + 2);
192 if (!impl->i2c.send(impl->devAddr, buf.data(), buf.size())) {
193 assert(false);
194 return false;
195 }
196 vTaskDelay(5);
197 eepAddr += wlen;
198 ptr += wlen;
199 len -= wlen;
200 }
