134 bool write8(uint32_t memoryAddress, const void * source, size_t size) {
135 if (!impl->isReady) {
136 assert(false);
137 return false;
138 }
139 if ((memoryAddress + size) > impl->chipSize) {
140 assert((memoryAddress + size) <= impl->chipSize);
141 return false;
142 }
143 const uint8_t * srcPtr = static_cast<const uint8_t *>(source);
144 size_t remaining = size;
145 while (remaining > 0) {
146 size_t pageOffset = memoryAddress % impl->pageSize;
147 size_t writeSize = std::min(remaining, impl->pageSize - pageOffset);
148 uint8_t addressHigh = static_cast<uint8_t>((impl->deviceAddress & 0xF0) | ((memoryAddress >> 7) & 0x0E));
149 uint8_t addressLow = static_cast<uint8_t>(memoryAddress & 0xFF);
150 if (!impl->i2c.send(
151 addressHigh, HAL::I2C::Master::MemAddrSize::ADDR_8BITS, addressLow, const_cast<void *>(static_cast<const void *>(srcPtr)), writeSize)) {
152 assert(false);
153 return false;
154 }
155 vTaskDelay(5);
156 memoryAddress += writeSize;
157 srcPtr += writeSize;
158 remaining -= writeSize;
159 }
160 return true;
161 }

170 bool write16(uint32_t memoryAddress, const void * source, size_t size) {
171 if (!impl->isReady) {
172 assert(false);
173 return false;
174 }
175 if ((memoryAddress + size) > impl->chipSize) {
176 assert(false);
177 return false;
178 }
179 const uint8_t * srcPtr = static_cast<const uint8_t *>(source);
180 std::vector<uint8_t> buffer;
181 size_t remaining = size;
182 while (remaining > 0) {
183 size_t pageOffset = memoryAddress % impl->pageSize;
184 size_t writeSize = std::min(remaining, impl->pageSize - pageOffset);
185 buffer.resize(2 + writeSize);
186 buffer[0] = static_cast<uint8_t>((memoryAddress >> 8) & 0xFF);
187 buffer[1] = static_cast<uint8_t>(memoryAddress & 0xFF);
188 std::copy_n(srcPtr, writeSize, buffer.begin() + 2);
189 if (!impl->i2c.send(impl->deviceAddress, buffer.data(), buffer.size())) {
190 assert(false);
191 return false;
192 }
193 vTaskDelay(5);
194 memoryAddress += writeSize;
195 srcPtr += writeSize;
196 remaining -= writeSize;
197 }
