161 void handleReceivedChar(RxCharParameters & params) {
162 if (params.inProgress && params.timeout.check()) {
163 diagnostic.ignoredMessagesByTimeout++;
164 params.rxFrame.clear();
165 params.remain = -1;
166 params.inProgress = false;
167 }
168 size_t pos = params.rxFrame.getSize();
169 if (pos == offsetof(Frame::Header, Frame::Header::startDelimiter)) {
170 if (params.rxChar == Protocol::START_DELIMITER) {
171 if (params.inProgress) assert(false);
172 emitBeginMessageReceive();
173 params.inProgress = true;
174 params.rxFrame.clear();
175 params.remain = -1;
176 params.rxFrame.append(params.rxChar);
177 params.timeout.set(5);
178 return;
179 }
180 params.timeout.set(5);
181 return;
182 }
183 params.rxFrame.append(params.rxChar);
184 if (pos == offsetof(Frame::Header, Frame::Header::lengthMSB)) {
185 params.payloadLen = static_cast<size_t>(params.rxChar) << 8;
186 } else if (pos == offsetof(Frame::Header, Frame::Header::lengthLSB)) {
187 params.payloadLen |= params.rxChar;
188 if (params.payloadLen >= ((8U * 8U) + 1U)) {
189 diagnostic.maxMsgLengthErrors++;
190 goto exit;
191 }
192 params.remain = static_cast<int32_t>(params.payloadLen) + static_cast<int32_t>(sizeof(uint8_t));
193 } else {
194 if (params.remain > 0) params.remain--;
195 if (params.remain == 0) {
196 if (Protocol::validate(params.rxFrame)) {
197 Protocol::handleIncomingMessage(params.rxFrame);
198 } else {
199 diagnostic.invalidMessages++;
200 assert(false);
201 }
202 exit:
203 params.rxFrame.clear();
204 emitEndMessageReceived();
205 params.inProgress = false;
206 params.remain = -1;
207 }
208 }
