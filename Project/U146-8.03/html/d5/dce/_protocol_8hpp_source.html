43 static bool parse(const DigiApiFrameBuffer & apiFrame, Frame::Data & frameData) {
44 if (!validate(apiFrame)) return false;
45 const uint8_t * data = apiFrame.getData();
46 const auto * header = reinterpret_cast<const Frame::Header *>(data);
47 size_t payloadLen = (static_cast<size_t>(header->lengthMSB) << 8) | header->lengthLSB;
48 data += sizeof(Frame::Header);
49 frameData.type = *(data++);
50 payloadLen--;
51 frameData.vector.clear();
52 frameData.vector.insert(frameData.vector.end(), data, data + payloadLen);
53 return true;
54 }

61 static bool handleIncomingMessage(const DigiApiFrameBuffer & apiFrame) {
62 static auto frameData = std::make_unique<Frame::Data>();
63 ResponseEvents & responseEvents = ResponseEvents::getInstance();
64 if (!parse(apiFrame, *frameData)) {
65 assert(false);
66 return false;
67 }
68 switch (static_cast<Frame::Type>(frameData->type)) {
69 case Frame::Type::RECEIVE_PACKET_64BIT:
70 if (responseEvents.onReceivePacket64bit) responseEvents.onReceivePacket64bit({*frameData});

72 case Frame::Type::RECEIVE_PACKET_16BIT:
73 if (responseEvents.onReceivePacket16bit) responseEvents.onReceivePacket16bit({*frameData});

105 static bool validate(const DigiApiFrameBuffer & apiFrame) {
106 size_t msgSize = apiFrame.getSize();
107 if (msgSize < sizeof(Frame::Header) + 1) return false;
108 const auto & header = *reinterpret_cast<const Frame::Header *>(apiFrame.getData());
109 if (header.startDelimiter != START_DELIMITER) return false;
110 size_t headerLen = (static_cast<size_t>(header.lengthMSB) << 8) | header.lengthLSB;
111 size_t crcLen = sizeof(Frame::Header) + headerLen;
112 if (crcLen > msgSize - 1) return false;
113 return Checksum(apiFrame) == apiFrame.getData()[crcLen];
114 }

121 static bool validate(const uint8_t * data) {
122 assert(data);
123 const auto & header = *reinterpret_cast<const Frame::Header *>(data);
124 if (header.startDelimiter != START_DELIMITER) return false;
125 size_t headerLen = (static_cast<size_t>(header.lengthMSB) << 8) | header.lengthLSB;
126 size_t crcLen = sizeof(Frame::Header) + headerLen;
127 return Checksum(data) == data[crcLen];

135 const static DigiApiFrameBuffer & handleOutgoingMessage(const Frame::Data & frameData) {
136 static auto msg = std::make_unique<DigiApiFrameBuffer>();
137 assert(API_MODE == ApiMode::API1);
138 size_t frameSize = frameData.vector.size() + 1;
139 msg->clear();
140 msg->append(START_DELIMITER);
141 msg->append(static_cast<uint8_t>(frameSize >> 8));
142 msg->append(static_cast<uint8_t>(frameSize & 0xFF));
143 msg->append(frameData.type);
144 msg->append(frameData.vector);
145 msg->append(Checksum(*msg));
146 return *msg;
147 }
