319 void process() {
320 constexpr size_t RX_TIMEOUT = 20; // Timeout for receiving frames in milliseconds
321 while (1) {
322 apiFrame.clear();
323 auto * data = apiFrame.getData();
324 signalEndMessageReceived();
325 // Wait for start delimiter
326 do {
327 rxPort.receive(data, sizeof(uint8_t), portMAX_DELAY);
328 } while (*data != Protocol::START_DELIMITER);
329 signalBeginMessageReceive();
330
331 data += sizeof(uint8_t);
332 // Wait for length field
333 if (!rxPort.receive(data, sizeof(uint16_t), RX_TIMEOUT)) {
334 diagnostic.ignoredMessagesByTimeout++;
335 assert(false);
336 continue;
337 }
338
339 size_t remaind = (static_cast<size_t>(data[0]) << 8) | static_cast<size_t>(data[1]) + sizeof(uint8_t);
340 if (remaind > DigiApiFrameBuffer::BUFFER_SIZE) {
341 diagnostic.maxMsgLengthErrors++;
342 assert(false);
343 continue;
344 }
345 data += sizeof(uint16_t);
346
347 // Wait for message body
348 if (!rxPort.receive(data, remaind, RX_TIMEOUT)) {
349 diagnostic.ignoredMessagesByTimeout++;
350 assert(false);
351 continue;
352 }
353 data += remaind;
354 apiFrame.setSize(data - apiFrame.getData());
355 // Validate the received API frame
356 if (!Protocol::instance().validate(apiFrame)) {
357 diagnostic.invalidMessages++;
358 assert(false);
359 continue;
360 }
361 // Handle the valid API frame
362 Protocol::instance().handleIncomingMessage(apiFrame);
363 }
364 }
