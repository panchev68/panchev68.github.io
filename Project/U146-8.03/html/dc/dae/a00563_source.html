238 void receivedPacked16BitsCallback(const Lib::Xbee::Core::Frame0x81 & frame0x81) {
239 if (state != State::RUNNING) return;
240 connection.info.signalStrength = [](uint32_t value) -> uint8_t {
241 constexpr uint32_t MAX_RSSI_VAL = 90;
242 if (value > MAX_RSSI_VAL) value = MAX_RSSI_VAL;
243 return static_cast<uint8_t>(((MAX_RSSI_VAL - value) * 100U) / MAX_RSSI_VAL);
244 }(frame0x81.rssi);
245 diagnostic.receivedDataMessages++;
246 Core::Message msg(frame0x81.rfData);
247 if (!queueRxMessage.overwrite(&msg)) assert(false);
248 }

254 void receivedTransmitStausCallback(const Lib::Xbee::Core::Frame0x89 frame0x89) {

260 }
