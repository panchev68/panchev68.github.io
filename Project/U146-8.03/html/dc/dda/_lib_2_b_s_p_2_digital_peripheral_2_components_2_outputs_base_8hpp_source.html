186 void preProcessing() {

188 for (size_t i = 0U; i < T_PIN_NUM; ++i) {
189 const auto & map = mapFields[i];
190 auto & outReg = txBuffer[map.port + MC33879::TxPort::OUTPUTS + 1U];
191 auto & ctrlReg = txBuffer[map.port + MC33879::TxPort::CONTROL + 1U];
192 const auto mask = map.mask;
193 if ((groupValue & BitMask::at(i)) != 0U) {
194 outReg |= mask;
195 ctrlReg |= mask;

208 bool postProcessing() {
209 if (!isSpiError) {
210 if ((rxBuffer.at(BUFFER_SIZE - 1U) != txBuffer.at(0)) && (txBuffer.at(0) != 0U)) {
211 isSpiError = true;
212 spiErrorsCounter++;
213 emitErrorEvent(true);
214 assert(false);
215 return false;
216 }
217 } else {
218 isSpiError = false;
219 emitErrorEvent(false);
220 }
221 uint64_t value = 0U;
222 for (size_t i = 0U; i < T_PIN_NUM; ++i) {
223 const auto & map = mapFields[i];
224 if ((rxBuffer[map.port + MC33879::RxPort::STATUS + 1U] & map.mask) != 0U) value |= BitMask::at(i);
225 }
226 statusRegister = value;

318 void taskFn() {
319 uint64_t value;
320 for (;;) {
321 queueValueChanged.receive(&value, portMAX_DELAY);
322 outputsChangedCounter++;
323 if (parameters.enable) {
324 value &= OutputsBase<T_PIN_NUM>::VALUE_MASK_AND;
325 uint64_t diffBits = value ^ result;
326 uint64_t diffActiveBits = diffBits & value;
327 size_t diffBitCounter = std::bitset<64>(diffActiveBits).count();
328 if (diffBitCounter >= parameters.outputsNumber) {
329 uint64_t pendingBits = diffActiveBits;
330 while (pendingBits) {
331 uint64_t groupBitMask = 0U;
332 uint32_t count = 0U;
333 for (size_t i = 0U; i < T_PIN_NUM && count < parameters.outputsNumber; ++i) {
334 uint64_t bit = BitMask::at(i);
335 if ((pendingBits & bit) != 0U) {
336 groupBitMask |= bit;
337 pendingBits &= ~bit;
338 count++;
339 }
340 }
341 if (groupBitMask != 0U) {
342 result = (result & ~groupBitMask) | (value & groupBitMask);
343 vTaskDelay(parameters.switchPeriod);
344 }
345 }
346 } else {
347 result = value;
348 }
349 } else {
350 result = value;
351 }
352 }
