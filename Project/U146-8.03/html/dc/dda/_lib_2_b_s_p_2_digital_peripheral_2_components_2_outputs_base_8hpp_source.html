174 void preProcessing() {
175 const uint64_t groupValue = grouping->getValue();
176 for (size_t i = 0U; i < T_PIN_NUM; ++i) {
177 const auto & map = mapFields[i];
178 auto & outReg = txBuffer[map.port + MC33879::TxPort::OUTPUTS + 1U];
179 auto & ctrlReg = txBuffer[map.port + MC33879::TxPort::CONTROL + 1U];
180 const auto mask = map.mask;
181 if ((groupValue & BitMask::at(i)) != 0U) {
182 outReg |= mask;
183 ctrlReg |= mask;
184 } else {
185 outReg &= ~mask;
186 }
187 }
188 txBuffer[0] = diagCounter;
189 if ((++diagCounter) == 0U) diagCounter = 1U;
190 }
191
196 bool postProcessing() {
197 if (!isSpiError) {
198 if ((rxBuffer.at(BUFFER_SIZE - 1U) != txBuffer.at(0)) && (txBuffer.at(0) != 0U)) {
199 isSpiError = true;
200 spiErrorsCounter++;
201 emitErrorEvent(true);
202 assert(false);
203 return false;
204 }
205 } else {
206 isSpiError = false;
207 emitErrorEvent(false);
208 }
209 uint64_t value = 0U;
210 for (size_t i = 0U; i < T_PIN_NUM; ++i) {
211 const auto & map = mapFields[i];
212 if ((rxBuffer[map.port + MC33879::RxPort::STATUS + 1U] & map.mask) != 0U) value |= BitMask::at(i);
213 }
214 statusRegister = value;
215 return true;
216 }

297 void taskFn() {
298 uint64_t value;
299 for (;;) {
300 queueValueChanged.receive(&value, portMAX_DELAY);
301 outputsChangedCounter++;
302 if (parameters.enable) {
303 value &= OutputsBase<T_PIN_NUM>::VALUE_MASK_AND;
304 uint64_t diffBits = value ^ result;
305 uint64_t diffActiveBits = diffBits & value;
306 size_t diffBitCounter = std::bitset<64>(diffActiveBits).count();
307 if (diffBitCounter >= parameters.outputsNumber) {
308 uint64_t pendingBits = diffActiveBits;
309 while (pendingBits) {
310 uint64_t groupBitMask = 0U;
311 uint32_t count = 0U;
312 for (size_t i = 0U; i < T_PIN_NUM && count < parameters.outputsNumber; ++i) {
313 uint64_t bit = BitMask::at(i);
314 if ((pendingBits & bit) != 0U) {
315 groupBitMask |= bit;
316 pendingBits &= ~bit;
317 count++;
318 }
319 }
320 if (groupBitMask != 0U) {
321 result = (result & ~groupBitMask) | (value & groupBitMask);
322 vTaskDelay(parameters.switchPeriod);
323 }
324 }
325 } else {
326 result = value;
327 }
328 } else {
329 result = value;
330 }
331 }
332 }
