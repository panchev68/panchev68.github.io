162 template <int T_BIT> bool waitForComplete(uint32_t timeout) {
163 assert(isOpen);
164 if (timeout == 0) return true;
165 auto result = notify.waitBits(T_BIT, true, false, timeout);
166 assert(result & T_BIT);
167 return (result & T_BIT);
168 }

264 void emitCallback(HAL_SPI_CallbackIDTypeDef id) {
265 assert(isOpen);
266 switch (id) {
267 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_TX_COMPLETE_CB_ID:
268 emitTxComplete();
269 break;
270 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_RX_COMPLETE_CB_ID:
271 emitRxComplete();
272 break;
273 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_TX_RX_COMPLETE_CB_ID:
274 emitTxRxComplete();
275 break;
276 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_ABORT_CB_ID:
277 emitAbortComplete();
278 break;
279 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_RX_HALF_COMPLETE_CB_ID:
280 emitRxHalfComplete();
281 break;
282 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_TX_HALF_COMPLETE_CB_ID:
283 emitTxHalfComplete();
284 break;
285 case HAL_SPI_CallbackIDTypeDef::HAL_SPI_TX_RX_HALF_COMPLETE_CB_ID:
286 emitTxRxHalfComplete();
287 break;
288 default:
289 break;
290 }
