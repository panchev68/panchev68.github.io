86 void handleModemStatus(const Frame0x8a & frame) {
87 using Status = Frame0x8a::ModemStatus;
88 static constexpr size_t STATUS_COUNT = static_cast<size_t>(Status::STACK_ERROR) + 1;
89 using Callback = std::function<void()> ModemStatus::*;
90 static constexpr Callback callbacks[STATUS_COUNT] = {&ModemStatus::onHardwareReset, &ModemStatus::onWatchdogTimerReset,
91 &ModemStatus::onJoinedNetwork, &ModemStatus::onLeftNetwork, &ModemStatus::onCoordinatorStarted,
92 &ModemStatus::onNetworkSecurityKeyWasUpdated, &ModemStatus::onNetworkWokeUp, &ModemStatus::onNetworkWentToSleep,
93 &ModemStatus::onVoltageSupplyLimitExceeded, &ModemStatus::onRemoteManagerConnected, &ModemStatus::onRemoteManagerDisconnected,
94 &ModemStatus::onModemConfigurationChangedWhileJoinInProgress, &ModemStatus::onAccessFault, &ModemStatus::onFatalError,
95 &ModemStatus::onSecureSessionSuccessfullyEstablished, &ModemStatus::onSecureSessionEnded,
96 &ModemStatus::onSecureEessionAuthenticationFailed, &ModemStatus::onCoordinatorDetectedAPanIdConflictButTookNoAction,
97 &ModemStatus::onCoordinatorChangedPanIdDueToAConflict, &ModemStatus::onBLEConnect, &ModemStatus::onBLEDisconnect,
98 &ModemStatus::onBandmaskConfigurationFailed, &ModemStatus::onCellularComponentUpdateStarted,
99 &ModemStatus::onCellularComponentUpdateFailed, &ModemStatus::onCellularComponentUpdateCompleted,
100 &ModemStatus::onXbeeFirmwareUpdateStarted, &ModemStatus::onXbeeFirmwareUpdateFailed, &ModemStatus::onXbeeFirmwareUpdateApplying,
101 &ModemStatus::onRouterPANIDWasShangedByCoordinatorDueToConflict, &ModemStatus::onNetworkWatchdogTimeoutExpired,
102 &ModemStatus::onStackError};
103 auto statusValue = static_cast<size_t>(frame.modemStatus);
104 if (statusValue < STATUS_COUNT) {
105 auto & cb = this->*callbacks[statusValue];
106 if (cb) cb();
107 } else {
108 if (onStackError) onStackError();
109 assert(false);
110 }
111 }
