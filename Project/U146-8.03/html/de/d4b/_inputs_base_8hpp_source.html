108 const auto & map = mapFields[i];
109 auto & input = array[i];
110 const bool isActive = (rxBuffer[map.port] & map.mask) != 0U;
111 if (input.update(isActive)) { value |= BitMask::at(i); }
112 }
113 if (value != lastValue) {
114 lastValue = value;
115 ++changesCounter;
116 handleValueChangedEvent(value);
