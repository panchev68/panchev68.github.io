169 void invalidate() {
170 const int16_t yMax = scrollableContainer->getHeight();
171 int16_t ySelectBegin = 0, ySelectEnd = 0;
172 for (size_t i = 0; i < buttons.size(); ++i) {
173 auto * button = buttons[i];
174 const bool selected = static_cast<int16_t>(i) == selectPosition;
175 button->forceState(selected);
176 if (selected) {
177 ySelectBegin = button->getY();
178 ySelectEnd = ySelectBegin + button->getHeight();
179 }
180 }
181 if (ySelectEnd > yMax) {
182 scrollableContainer->moveChildrenRelative(0, yMax - ySelectEnd);
183 } else if (ySelectBegin < 0) {
184 scrollableContainer->moveChildrenRelative(0, -ySelectBegin);
185 }
186 scrollableContainer->invalidate();
187 }

199 int16_t y = 0;
200 for (auto * button : buttons) {
201 button->setPosition(0, y, buttonParm.width, buttonParm.height);
202 button->setBoxWithBorderPosition(0, 0, buttonParm.width, buttonParm.height);
203 button->setTextY(buttonParm.textYOffset);
204 y += static_cast<int16_t>(buttonParm.height + buttonParm.buttonsOffset);
205 if (buttonParm.textColor != UINT32_MAX && buttonParm.textCursorColor != UINT32_MAX) {

207 }
208 if (buttonParm.boxColor != UINT32_MAX && buttonParm.boxCursorColor != UINT32_MAX) {
