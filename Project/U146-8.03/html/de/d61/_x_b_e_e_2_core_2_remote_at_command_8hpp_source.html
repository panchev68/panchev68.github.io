148 void handleReceivedChar(const Frame0x97 & response) {
149 if ((response.frameId != 0x17) || (impl->sendFrame.atCmd != response.atCmd)) return;
150 if ((response.commandStatus != 0) && (response.commandStatus != 4)) {
151 assert(false);
152 return;
153 }
154 QueueMessage msg{};
155 msg.size = response.parameters.size();
156 assert(msg.size < msg.DATA_SIZE);
157 std::memcpy(msg.data.data(), response.parameters.data(), msg.size);
158 if (!impl->queueMessage.overwrite(&msg)) assert(false);
