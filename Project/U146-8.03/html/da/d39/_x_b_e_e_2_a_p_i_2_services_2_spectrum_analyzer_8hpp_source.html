134 void handleScanRequest() {
135 if (!impl->timeout.check()) return;
136 if (!impl->localAtCommand->send({"ED"})) {
137 impl->timeout.set(SCAN_ERROR_RETRY_MS);
138 return;
139 }
140 impl->timeout.set(SCAN_RESPONSE_TIMEOUT_MS);
141 impl->hasWaitForResponse = true;
142 }
143
147 void handleScanResponse() {
148 static std::vector<uint8_t> response;
149 if (impl->timeout.check()) {
150 impl->hasWaitForResponse = false;
151 impl->timeout.set(SCAN_ERROR_RETRY_MS);
152 return;
153 }
154 if (!impl->localAtCommand->receive(response, 0)) return;
155 processSpectrumData(response);
156 impl->isNewDataFlag = true;
157 impl->hasWaitForResponse = false;
158 impl->timeout.set(SCAN_INTERVAL_MS);
159 }
160
165 void processSpectrumData(std::vector<uint8_t> & response) {
166 if (response.size() != impl->currentArray.size()) return;
167 std::transform(
168 response.begin(), response.end(), response.begin(),
169 [](uint8_t value) { return Lib::Helper::Convert::dBM::toDigiPercent(value); });
170 for (size_t index = 0; index < response.size(); ++index) {
171 updateChannelData(index, response[index]);
172 }
173 }
174
180 void updateChannelData(size_t index, uint8_t newValue) {
181 uint8_t & current = impl->currentArray[index];
182 current = static_cast<uint8_t>((newValue + current) >> 1);
183 updateAverageValue(index, current);
184 updateMaxValue(index, current);
185 }
186
192 void updateAverageValue(size_t index, uint8_t currentValue) {
193 uint8_t & avg = impl->averageArray[index];
194 if (avg > currentValue) {
195 --avg;
196 } else if (avg < currentValue) {
197 avg = static_cast<uint8_t>(avg + 2);
198 }
199 }
200
206 void updateMaxValue(size_t index, uint8_t currentValue) {
207 uint8_t & maxVal = impl->maxArray[index];
208 if (currentValue > maxVal) {
209 maxVal = currentValue;
210 }
211 }
