usage of Lib::Helper::ValueInfo.

usage of Lib::Helper::ValueInfo.Demonstrates how to use ValueInfo to store a variable pointer and its type information, and how to retrieve and check the type at runtime.

#include <Helper/ValueInfo.hpp>
uint32_t test = 0x54;
__root Lib::Helper::ValueInfo info(&test, typeid(decltype(test)));
uint32_t main() {
if (info.getTypeInfo() == typeid(uint32_t)) {
return *reinterpret_cast<uint32_t*>(info.getData());
}
assert(false);
return 0;
}
