100 bool add(TransparentBase * device) override {
101 DeviceManagerBase::add(device);
102 auto handle = device->getHandle();
103 if (handle != nullptr) {
104 bool ok = true;
105 ok &= (HAL_ADC_RegisterCallback(handle, HAL_ADC_CONVERSION_COMPLETE_CB_ID, callbackConversionComplete) == HAL_OK);
106 ok &= (HAL_ADC_RegisterCallback(handle, HAL_ADC_CONVERSION_HALF_CB_ID, callbackConversionHalf) == HAL_OK);
