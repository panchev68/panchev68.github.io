138 void handleTaskLoop() override {
139 using State = TaskFuncData::State;
140 using DeliveryStatus = TaskFuncData::DeliveryStatus;
141 auto & data = *taskData;
142 auto isConnected = [this]() { return connection.state == Connection::State::CONNECTED; };
143 auto disconnect = [this, &data] { data.timeoutConnection.set(0); };
144 auto connect = [this] { connection.state = Connection::State::CONNECTED; };
145 auto refreshConnectionTimeout = [this, &data]() { data.timeoutConnection.set(ApplicationBase::CONNECTION_TIMEOUT); };
146
147 for (;;) {
148 switch (data.state) {
149 case State::BEGIN_TRANSACTION:
150 if (!data.timeoutMessage.check()) break;
151 data.timeoutMessage.set(25);
152 data.state = State::SEND_MESSAGE;
153 continue;
154 case State::SEND_MESSAGE:
155 if (isConnected()) connection.info.transmitPeriod.beginMeasurement();
156 handleOutgoingMessage(data.txMessageBuffer);
157 if (queueTransmitDeliveryStatus.receive(&data.transmitDeliveryStatus, 0)) connection.info.deletedTransmitStatus++;
158 if (!sendData(data.txMessageBuffer)) {
159 assert(false);
160 data.state = State::ERROR;
161 break;
162 }
163 data.state = State::WAIT_FOR_SENDING_COMPLETED;
164 continue;
165 case State::WAIT_FOR_SENDING_COMPLETED:
166 if (queueTransmitDeliveryStatus.receive(&data.transmitDeliveryStatus, MAX_RESPONSE_TIME / 2)) {
167 data.isError = (data.transmitDeliveryStatus != DeliveryStatus::SUCCESS);
168 if (data.isError) {
169 if (data.transmitDeliveryStatus == DeliveryStatus::NO_ACK_RECEIVED) {
170 disconnect();
171 vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
172 } else connection.info.unsuccessTransmitStatus++;
173 }
174 } else {
175 connection.info.noTransmitStatus++;
176 data.isError = true;
177 }
178 if (data.isError) {
179 data.state = State::ERROR;
180 break;
181 }
182 if (queueRxMessage.receive(&data.rxMessage, 0)) connection.info.deletedMessages++;
183 data.state = State::WAITING_FOR_RECEIVE_INCOMING_MESSAGE;
184 continue;
185 case State::WAITING_FOR_RECEIVE_INCOMING_MESSAGE:
186 if (queueRxMessage.receive(&data.rxMessage, ApplicationBase::MAX_RESPONSE_TIME)) {
187 data.isError = false;
188 if (handleIncomingMessage(data.rxMessage)) {
189 if (!isConnected()) connect();
190 connection.info.transmitPeriod.endMeasurement();
191 outputsUpdatedSucceessfulCallback();
192 refreshConnectionTimeout();
193 }
194 } else {
195 data.isError = true;
196 if (isConnected()) connection.info.retriesCounter++;
197 else vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
198 }
199 if (data.isError) {
200 data.state = State::ERROR;
201 break;
202 }
203 data.state = State::END_TRANSACTION;
204 continue;
205 case State::END_TRANSACTION:
206 if (data.timeoutConnection.check() && isConnected()) {
207 connection.info.signalStrength = 0;
208 connection.info.transmitPeriod.reset();
209 connection.info.reconnectionCounter++;
210 connection.state = Connection::State::DISCONNECTED;
211 }
212 data.state = State::BEGIN_TRANSACTION;
213 break;
214 case State::ERROR: data.state = State::END_TRANSACTION; break;
215 }
216 break;
217 }
218 }
