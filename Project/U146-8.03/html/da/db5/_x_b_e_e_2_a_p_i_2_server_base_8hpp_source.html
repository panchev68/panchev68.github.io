100 enum State {

102 SEND_MESSAGE,
103 WAIT_FOR_SENDING_COMPLETED,

110 DeliveryStatus lastDeliveryStatus = DeliveryStatus::SUCCESS;
111 FreeRTOS::Timeout connectionTimeout;

133 void update() override {
134 auto & stateData = *taskState;
135 auto isConnected = [this]() { return connection.state == Connection::State::CONNECTED; };
136 auto disconnect = [this, &stateData] { stateData.connectionTimeout.set(0); };
137 auto connect = [this] { connection.state = Connection::State::CONNECTED; };
138 auto refreshConnectionTimeout = [this, &stateData]() { stateData.connectionTimeout.set(ApplicationBase::CONNECTION_TIMEOUT); };
139
140 while (true) {
141 switch (stateData.currentState) {
142 case TaskState::BEGIN_TRANSACTION:
143 if (!stateData.messageTimeout.check()) break;
144 stateData.messageTimeout.set(25);
145 stateData.currentState = TaskState::SEND_MESSAGE;
146 continue;
147 case TaskState::SEND_MESSAGE:
148 if (isConnected()) connection.info.transmitPeriod.beginMeasurement();
149 handleOutgoingMessage(stateData.txBuffer);
150 if (queueTransmitDeliveryStatus.receive(&stateData.lastDeliveryStatus, 0)) connection.info.deletedTransmitStatus++;
151 if (!sendData(stateData.txBuffer)) {
152 assert(false);
153 stateData.currentState = TaskState::ERROR;
154 break;
155 }
156 stateData.currentState = TaskState::WAIT_FOR_SENDING_COMPLETED;
157 continue;
158 case TaskState::WAIT_FOR_SENDING_COMPLETED:
159 if (queueTransmitDeliveryStatus.receive(&stateData.lastDeliveryStatus, MAX_RESPONSE_TIME / 2)) {
160 stateData.hasError = (stateData.lastDeliveryStatus != TaskState::DeliveryStatus::SUCCESS);
161 if (stateData.hasError) {
162 if (stateData.lastDeliveryStatus == TaskState::DeliveryStatus::NO_ACK_RECEIVED) {
163 disconnect();
164 vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
165 } else {
166 connection.info.unsuccessTransmitStatus++;
167 }
168 }
169 } else {
170 connection.info.noTransmitStatus++;
171 stateData.hasError = true;
172 }
173 if (stateData.hasError) {
174 stateData.currentState = TaskState::ERROR;
175 break;
176 }
177 if (queueRxMessage.receive(&stateData.rxBuffer, 0)) connection.info.deletedMessages++;
178 stateData.currentState = TaskState::WAITING_FOR_RECEIVE_INCOMING_MESSAGE;
179 continue;
180 case TaskState::WAITING_FOR_RECEIVE_INCOMING_MESSAGE:
181 if (queueRxMessage.receive(&stateData.rxBuffer, ApplicationBase::MAX_RESPONSE_TIME)) {
182 stateData.hasError = false;
183 if (handleIncomingMessage(stateData.rxBuffer)) {
184 if (!isConnected()) connect();
185 connection.info.transmitPeriod.endMeasurement();
186 outputsUpdatedSucceessfulCallback();
187 refreshConnectionTimeout();
188 }
189 } else {
190 stateData.hasError = true;
191 if (isConnected()) connection.info.retriesCounter++;
192 else vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
193 }
194 if (stateData.hasError) {
195 stateData.currentState = TaskState::ERROR;
196 break;
197 }
198 stateData.currentState = TaskState::END_TRANSACTION;
199 continue;
200 case TaskState::END_TRANSACTION:
201 if (stateData.connectionTimeout.check() && isConnected()) {
202 connection.info.signalStrength = 0;
203 connection.info.transmitPeriod.reset();
204 connection.info.reconnectionCounter++;
205 connection.state = Connection::State::DISCONNECTED;
206 }
207 stateData.currentState = TaskState::BEGIN_TRANSACTION;
208 break;
209 case TaskState::ERROR:
210 stateData.currentState = TaskState::END_TRANSACTION;
211 break;
212 }
213 break;
214 }
215 }
