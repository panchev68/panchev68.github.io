218 static double calcMeanObliquityOfEcliptic(double t) {
219 double seconds = 21.448 - t * (46.8150 + t * (0.00059 - t * 0.001813));

228 static double calcGeomMeanLongSun(double t) {

268 static double calcEquationOfTime(double t) {
269 double epsilon = calcObliquityCorrection(t);
270 double l0 = calcGeomMeanLongSun(t);
271 double e = calcEccentricityEarthOrbit(t);
272 double m = calcGeomMeanAnomalySun(t);
273 double y = std::tan(degToRad(epsilon) / 2.0);
274 y *= y;
275 double sin2l0 = std::sin(2.0 * degToRad(l0));
276 double sinm = std::sin(degToRad(m));
277 double cos2l0 = std::cos(2.0 * degToRad(l0));
278 double sin4l0 = std::sin(4.0 * degToRad(l0));
279 double sin2m = std::sin(2.0 * degToRad(m));
280 double eqTime =
281 y * sin2l0 - 2.0 * e * sinm + 4.0 * e * y * sinm * cos2l0 - 0.5 * y * y * sin4l0 - 1.25 * e * e * sin2m;
282 return radToDeg(eqTime) * 4.0;
283 }

328 static double calcHourAngleSunrise(double latitude, double solarDeclination, double zenith) {
329 double latRad = degToRad(latitude);

358 static double calcJulianDate(int year, int month, int day) {
359 int y = year;
360 int m = month;
361 if (m <= 2) {
362 y -= 1;
363 m += 12;
364 }
365 double A = std::floor(y / 100.0);
366 double B = 2.0 - A + std::floor(A / 4.0);

382 static double calcSunEqOfCenter(double t) {
383 double m = calcGeomMeanAnomalySun(t);
384 double mrad = degToRad(m);
385 return std::sin(mrad) * (1.914602 - t * (0.004817 + 0.000014 * t)) +
386 std::sin(2.0 * mrad) * (0.019993 - 0.000101 * t) + std::sin(3.0 * mrad) * 0.000289;

394 double calcAbsSunrise(double zenith) const {
395 double t = calcTimeJulianCent(julianDate);
396 double eqTime = calcEquationOfTime(t);
397 double solarDeclination = calcSunDeclination(t);
398 double hourAngle = calcHourAngleSunrise(latitude, solarDeclination, zenith);
399 double delta = longitude + radToDeg(hourAngle);
400 double timeDiff = 4.0 * delta;
401 double timeUTC = 720.0 - timeDiff - eqTime;
402 double newt = calcTimeJulianCent(calcJulianDateFromJulianCent(t) + timeUTC / 1440.0);
403 eqTime = calcEquationOfTime(newt);
404 solarDeclination = calcSunDeclination(newt);
405 hourAngle = calcHourAngleSunrise(latitude, solarDeclination, zenith);
406 delta = longitude + radToDeg(hourAngle);
407 timeDiff = 4.0 * delta;
408 timeUTC = 720.0 - timeDiff - eqTime;
