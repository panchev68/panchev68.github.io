255 static double calcEquationOfTime(double t) {
256 double epsilon = calcObliquityCorrection(t);
257 double l0 = calcGeomMeanLongSun(t);
258 double e = calcEccentricityEarthOrbit(t);
259 double m = calcGeomMeanAnomalySun(t);
260 double y = std::tan(degToRad(epsilon) / 2.0);
261 y *= y;
262 double sin2l0 = std::sin(2.0 * degToRad(l0));
263 double sinm = std::sin(degToRad(m));
264 double cos2l0 = std::cos(2.0 * degToRad(l0));
265 double sin4l0 = std::sin(4.0 * degToRad(l0));
266 double sin2m = std::sin(2.0 * degToRad(m));
267 double eqTime = y * sin2l0 - 2.0 * e * sinm + 4.0 * e * y * sinm * cos2l0 - 0.5 * y * y * sin4l0 - 1.25 * e * e * sin2m;
268 return radToDeg(eqTime) * 4.0;
269 }

342 static double calcJulianDate(int year, int month, int day) {
343 int y = year;
344 int m = month;
345 if (m <= 2) {
346 y -= 1;
347 m += 12;
348 }
349 double A = std::floor(y / 100.0);
350 double B = 2.0 - A + std::floor(A / 4.0);
351 return std::floor(365.25 * (y + 4'716)) + std::floor(30.6001 * (m + 1)) + day + B - 1524.5;
352 }

366 static double calcSunEqOfCenter(double t) {
367 double m = calcGeomMeanAnomalySun(t);
368 double mrad = degToRad(m);
369 return std::sin(mrad) * (1.914602 - t * (0.004817 + 0.000014 * t)) + std::sin(2.0 * mrad) * (0.019993 - 0.000101 * t) +
370 std::sin(3.0 * mrad) * 0.000289;

378 double calcAbsSunrise(double zenith) const {
379 double t = calcTimeJulianCent(julianDate);
380 double eqTime = calcEquationOfTime(t);
381 double solarDeclination = calcSunDeclination(t);
382 double hourAngle = calcHourAngleSunrise(latitude, solarDeclination, zenith);
383 double delta = longitude + radToDeg(hourAngle);
384 double timeDiff = 4.0 * delta;
385 double timeUTC = 720.0 - timeDiff - eqTime;
386 double newt = calcTimeJulianCent(calcJulianDateFromJulianCent(t) + timeUTC / 1440.0);
387 eqTime = calcEquationOfTime(newt);
388 solarDeclination = calcSunDeclination(newt);
389 hourAngle = calcHourAngleSunrise(latitude, solarDeclination, zenith);
390 delta = longitude + radToDeg(hourAngle);
391 timeDiff = 4.0 * delta;
392 timeUTC = 720.0 - timeDiff - eqTime;
