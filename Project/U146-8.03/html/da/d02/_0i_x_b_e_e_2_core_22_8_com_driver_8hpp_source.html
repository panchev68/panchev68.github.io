326 void process() {
327 constexpr size_t RX_TIMEOUT = 20; // Timeout for receiving frames in milliseconds
328 while (1) {
329 apiFrame.clear(); // Clear the frame buffer for new data
330 auto * data = apiFrame.getData(); // Get a pointer to the frame data
331 signalEndMessageReceived(); // Signal the end of message reception
332
333 // Wait for start delimiter
334 do {
335 rxPort.receive(data, sizeof(uint8_t), portMAX_DELAY);
336 } while (*data != Protocol::START_DELIMITER);
337 signalBeginMessageReceive();
338
339 data += sizeof(uint8_t);
340
341 // Wait for length field
342 if (!rxPort.receive(data, sizeof(uint16_t), RX_TIMEOUT)) {
343 diagnostic.ignoredMessagesByTimeout++;
344 assert(false);
345 continue;
346 }
347 // Calculate remaining bytes to read
348 size_t remaind = (static_cast<size_t>(data[0]) << 8) | static_cast<size_t>(data[1]) + sizeof(uint8_t);
349 if (remaind > DigiApiFrameBuffer::BUFFER_SIZE) {
350 diagnostic.maxMsgLengthErrors++;
351 assert(false);
352 continue;
353 }
354
355 data += sizeof(uint16_t);
356
357 // Wait for message body
358 if (!rxPort.receive(data, remaind, RX_TIMEOUT)) {
359 diagnostic.ignoredMessagesByTimeout++;
360 assert(false);
361 continue;
362 }
363 data += remaind;
364 apiFrame.setSize(data - apiFrame.getData());
365 // Validate the received API frame
366 if (!Protocol::instance().validate(apiFrame)) {
367 diagnostic.invalidMessages++;
368 assert(false);
369 continue;
370 }
371 // Handle the valid API frame
372 Protocol::instance().handleIncomingMessage(apiFrame);
373 }
374 }
