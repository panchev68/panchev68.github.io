JaggedArray

Example usage of Lib::Helper::JaggedArray.

Example usage of Lib::Helper::JaggedArray.Demonstrates how to use the JaggedArray container to store non-owning pointers, add and remove elements, and access elements by index.

#include "JaggedArray.hpp"
#include <iostream>
struct MyStruct {
int value;
};
int main() {
using namespace Lib::Helper;
MyStruct a{1}, b{2}, c{3};
JaggedArray<MyStruct> arr;
arr.add(a);
arr.add(&b);
arr.add({&c});
std::cout << "JaggedArray contents:" << std::endl;
for (size_t i = 0; i < arr.size(); ++i) {
std::cout << arr[i].value << std::endl;
}
arr.remove(b);
std::cout << "After removing b:" << std::endl;
for (size_t i = 0; i < arr.size(); ++i) {
std::cout << arr[i].value << std::endl;
}
arr.clear();
std::cout << "Size after clear: " << arr.size() << std::endl;
return 0;
}
