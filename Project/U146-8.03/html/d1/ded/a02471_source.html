256 bool readStaticParameters(Data & data) {
257 std::vector<uint8_t> response;
258 assert(localAtCommand->isOpen());
259
260 struct AtCmd {
261 const char * cmd;
262 size_t expectedSize;
263 void (*assign)(Data &, const std::vector<uint8_t> &);
264 };
265
266 static constexpr AtCmd cmds[] = {
267 {"NI", 0,
268 [](Data & d, const std::vector<uint8_t> & r) {
269 std::string s(r.begin(), r.end());
270 d.nodeIdentifier = std::move(s);
271 } },
272 {"SL", sizeof(uint32_t),
273 [](Data & d, const std::vector<uint8_t> & r) { d.serialNumber.u32.low = getU32(r); } },
274 {"SH", sizeof(uint32_t),
275 [](Data & d, const std::vector<uint8_t> & r) { d.serialNumber.u32.high = getU32(r); }},
276 {"ID", sizeof(uint16_t), [](Data & d, const std::vector<uint8_t> & r) { d.panId = getU16(r); } },
277 {"CH", 1, [](Data & d, const std::vector<uint8_t> & r) { d.channel = getU8(r); } },
278 {"VR", sizeof(uint16_t),
279 [](Data & d, const std::vector<uint8_t> & r) { d.firmwareVersion = getU16(r); } },
280 {"HV", sizeof(uint16_t),
281 [](Data & d, const std::vector<uint8_t> & r) { d.hardwareVersion = getU16(r); } },
282 };
283 for (const auto & cmd : cmds) {
284 if (!localAtCommand->send({cmd.cmd})) {
285 assert(false);
286 return false;
287 }
288 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) {
289 assert(false);
290 return false;
291 }
292 if (cmd.expectedSize && response.size() != cmd.expectedSize) {
293 assert(false);
294 return false;
295 }
296 if (cmd.cmd[0] == 'N' && cmd.cmd[1] == 'I') response.push_back(0);
297 cmd.assign(data, response);
298 }
299 return true;
300 }

309 bool readDynamicParameters(Data & data, uint32_t mask = static_cast<uint32_t>(-1)) {
310 std::vector<uint8_t> response;
311 assert(localAtCommand->isOpen());
312
313 struct DynCmd {
314 uint32_t bit;
315 const char * cmd;
316 size_t expectedSize;
317 void (*assign)(Data &, const std::vector<uint8_t> &);
318 };
319
320 static constexpr DynCmd dynCmds[] = {
321 {0x0001, "EA", 0, nullptr},
322 {0x0002, nullptr, sizeof(uint16_t),
323 [](Data & d, const std::vector<uint8_t> & r) { d.ackFailures = getU16(r); }},
324 {0x0004, "EC", 0, nullptr},
325 {0x0008, nullptr, sizeof(uint16_t),
326 [](Data & d, const std::vector<uint8_t> & r) { d.ccaFailures = getU16(r); }},
327 {0x0010, "%V", 0, nullptr},
328 {0x0020, nullptr, sizeof(uint16_t),
329 [](Data & d, const std::vector<uint8_t> & r) {
330 d.voltage = static_cast<float>(getU16(r)) / 1'000;
331 }},
332 {0x0040, "TP", 0, nullptr},
333 {0x0080, nullptr, sizeof(uint16_t),
334 [](Data & d, const std::vector<uint8_t> & r) {
335 d.temperature = Lib::Helper::Convert::BcdHelper::toHex(getU16(r));
336 }},
337 {0x0100, "CH", 0, nullptr},
338 {0x0200, nullptr, 1, [](Data & d, const std::vector<uint8_t> & r) { d.channel = getU8(r); }},
339 };
340 for (const auto & cmd : dynCmds) {
341 if (!(mask & cmd.bit)) continue;
342 if (cmd.cmd) {
343 taskYIELD();
344 if (!localAtCommand->send({cmd.cmd})) {
345 assert(false);
346 return false;
347 }
348 } else if (cmd.assign) {
349 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) {
350 assert(false);
351 return false;
352 }
353 if (cmd.expectedSize && response.size() != cmd.expectedSize) {
354 assert(false);
355 return false;
356 }
357 cmd.assign(data, response);
358 }
359 }
360 return true;
361 }

367 void updateCallback(Data & data) {
368 static std::vector<uint8_t> response;
369 using Convert = Core::Frame::Helper::Convert;
370 if (!pause) {
371 if (state == State::IDLE && timeoutUpdate.check()) {
372 timeoutUpdate.set(50);
373 auto mask = 1U << (tickCounter++ % 10U);
374 readDynamicParameters(data, mask);
375 }
376 }
377 switch (state) {
378 case State::IDLE:
379 break;
380 case State::SET_PARAMETERS_REMOTELY:
381 state = (remotePanId == static_cast<uint64_t>(-1)) ? State::GET_REMOTE_PAN_ID
382 : State::SET_CHANNEL_REMOTELY;
383 break;
384 case State::GET_REMOTE_PAN_ID:
385 if (newData.isChannelChanged && newData.channel != data.channel) {
386 if (!localAtCommand->send({"ND"})) assert(false);
387 state = State::GET_REMOTE_PAN_ID_RESPONSE;
388 } else state = State::END;
389 break;
390 case State::GET_REMOTE_PAN_ID_RESPONSE:
391 if (!localAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
392 else remotePanId = Convert::to<uint64_t>(response.data() + 2);
393 state = State::SET_CHANNEL_REMOTELY;
394 break;
395 case State::SET_CHANNEL_REMOTELY:
396 if (newData.isChannelChanged && newData.channel != data.channel) {
397 if (!remoteAtCommand->send({"CH"}, remotePanId, {newData.channel})) assert(false);
398 else state = State::SET_CHANNEL_REMOTELY_RESPONSE;
399 } else state = State::END;
400 break;
401 case State::SET_CHANNEL_REMOTELY_RESPONSE:
402 if (!remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
403 state = State::WRITE_CHANGES_REMOTELY;
404 break;
405 case State::WRITE_CHANGES_REMOTELY:
406 if (!remoteAtCommand->send({"WR"}, true)) assert(false);
407 state = State::WRITE_CHANGES_REMOTELY_RESPONSE;
408 break;
409 case State::WRITE_CHANGES_REMOTELY_RESPONSE:
410 if (!remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
411 state = State::SET_PARAMETERS_LOCAL;
412 break;
413 case State::SET_PARAMETERS_LOCAL:
414 hasWrite = false;
415 state = State::SET_CHANNEL_LOCALLY;
416 break;
417 case State::SET_CHANNEL_LOCALLY:
418 if (newData.isChannelChanged && newData.channel != data.channel) {
419 if (!localAtCommand->send({"CH"}, {static_cast<uint8_t>(newData.channel)})) assert(false);
420 else {
421 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
422 data.channel = newData.channel;
423 hasWrite = true;
424 }
425 }
426 state = State::SET_PAN_ID_LOCALLY;
427 break;
428 case State::SET_PAN_ID_LOCALLY:
429 if (newData.isPanIdChanged && newData.panId != data.panId) {
430 if (!localAtCommand->send({"ID"}, {static_cast<uint8_t>(newData.panId >> 8),
431 static_cast<uint8_t>(newData.panId & 0xff)}))
432 assert(false);
433 else {
434 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
435 data.panId = newData.panId;
436 hasWrite = true;
437 }
438 }
439 state = State::WRITE_CHANGES_LOCALLY;
440 break;
441 case State::WRITE_CHANGES_LOCALLY:
442 if (hasWrite) {
443 if (localAtCommand->send({"WR"})) {
444 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
445 }
446 state = State::APPLY_CHANGES_LOCALLY;
447 } else state = State::END;
448 break;
449 case State::APPLY_CHANGES_LOCALLY:
450 if (localAtCommand->send({"AC"})) {
451 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
452 }
453 connection->quality.clear();
454 state = State::END;
455 break;
456 case State::END:
457 newData.isChannelChanged = false;
458 newData.isPanIdChanged = false;
459 state = State::IDLE;
460 break;
461 }
462 }

481 static uint16_t getU16(const std::vector<uint8_t> & src) {

492 static uint32_t getU32(const std::vector<uint8_t> & src) {
493 assert(src.size() == sizeof(uint32_t));

495 (static_cast<uint32_t>(src.at(2)) << 8) | src.at(3);
