146 bool read(uint32_t address, uint8_t * buffer, size_t len) {
147 assert(buffer);
148 assert(len);
149 assert((address + len) < impl->getChipInfo().memoryCapacity);
150 if (!impl->isPresent) return false;
151 while (len > 0) {
152 auto rlen = std::min(len, BLOCK_READ_SIZE);
153 impl->select();
154 if (!impl->sendCommandA24(Register::READ_DATA, address)) {
155 assert(false);
156 impl->deselect();
157 return false;
158 }
159 if (!impl->receive(buffer, rlen, 200)) {
160 assert(false);
161 impl->deselect();
162 return false;
163 }
164 address += rlen;
165 buffer += rlen;
166 len -= rlen;
167 impl->deselect();
168 }
169 return true;
170 }

180 bool readSector(uint8_t * buffer, uint32_t sector, uint32_t count) {
181 assert(buffer);
182 assert(count);
183 if (!impl->isPresent) return false;
184 auto address = sector * impl->getChipInfo().sectorSize;
185 for (auto i = 0U; i < count; i++) {

192 }

204 bool write(uint32_t address, const uint8_t * buffer, size_t len) {
205 bool result = false;
206 assert(buffer);
207 assert((address + len) < impl->getChipInfo().memoryCapacity);
208 auto dataPtr = const_cast<uint8_t *>(buffer);
209 if (!impl->isPresent) return false;
210 while (len > 0) {
211 auto pageOffs = address & 0xFFU;
212 auto blockSize = std::min(len, static_cast<size_t>(256 - pageOffs));
213 if (!impl->writeEnable()) {
214 assert(false);
215 goto exit;
216 }
217 impl->select();
218 if (!impl->sendCommandA24(Register::PAGE_PROGRAM, address)) {
219 assert(false);
220 goto exit;
221 }
222 if (!impl->send(dataPtr, blockSize)) {
223 assert(false);
224 goto exit;
225 }
226 impl->deselect();
227 if (!impl->waitForWrtiteCompleted()) {
228 assert(false);
229 goto exit;
230 }
231 address += blockSize;
232 dataPtr += blockSize;
233 len -= blockSize;
234 }
235 result = true;
236 exit:
237 if (!impl->writeDisable()) {
238 assert(false);
239 result = false;
240 }
241 return result;
242 }

253 bool writeSector(const uint8_t * buffer, uint32_t sector, uint32_t count, bool hasSectorErase = true) {
254 auto & chipInfo = impl->getChipInfo();
255 auto dataPtr = const_cast<uint8_t *>(buffer);
256 assert(buffer);
257 assert(sector < chipInfo.sectorCount);
258 if (!impl->isPresent) return false;
259 for (auto i = 0U; i < count; i++) {
260 auto address = (sector + i) * chipInfo.sectorSize;
261 if (hasSectorErase) {
262 if (!eraseSector(address)) {
263 assert(false);
264 return false;
265 }
266 }
267 for (auto j = 0U; j < chipInfo.pagePerSector; j++) {
268 if (!writePage(dataPtr, address)) {
269 assert(false);
270 return false;
271 }
272 address += chipInfo.pageSize;
273 dataPtr += chipInfo.pageSize;
274 }
275 }
276 return true;

515 uint8_t chipManufacturer = 0;
516 uint8_t memoryInterfaceType = 0;
517 uint8_t memoryDensity = 0;

520 static W25Q128JVSIQ::Info info;
521 RawChipInfo rawInfo;
522 if (info.memoryCapacity != 0) return info;
523 select();
524 if (!sendCommand(Register::READ_JEDEC_ID)) goto exit;
525 if (!receive(reinterpret_cast<uint8_t *>(&rawInfo), sizeof(RawChipInfo))) goto exit;
526 info.chipManufacturer = rawInfo.chipManufacturer;
527 if (rawInfo.memoryDensity == 0x18) {
528 info.memoryCapacity = 128U * (1'024U * 1'024U / 8U);
529 info.numberOfPages = 65'536U;
530 info.pageSize = 256U;
531 info.sectorSize = 4'096U;
532 info.sectorCount = info.memoryCapacity / info.sectorSize;
533 info.pagePerSector = info.sectorSize / 256U;
534 } else {
535 assert(false);
536 info.memoryCapacity = 0;
537 }
538 exit:
539 deselect();
540 return info;

548 bool waitForWrtiteCompleted(uint32_t timeout = MAX_WRITE_TIME) {
549 uint8_t statusReg;
550 auto endTicks = HAL_GetTick() + timeout;
551 if (!readStatusRegister(statusReg)) {
552 assert(false);
553 return false;
554 }
555 while (HAL_GetTick() < endTicks) {
556 if (!readStatusRegister(statusReg)) {
557 assert(false);
558 return false;
559 }
560 if ((statusReg & 0x01U) == 0U) return true;
561 }
562 return false;
563 }
