375 static DayOfWeek dayOfWeek(const Date & date) {
376 int y = date.year;
377 int m = date.month;
378 int d = date.day;
379 if (m < 3) {
380 m += 12;
381 y -= 1;
382 }
383 int k = y % 100;
384 int j = y / 100;
385 int h = (d + 13 * (m + 1) / 5 + k + k / 4 + j / 4 + 5 * j) % 7;
386 int dow = ((h + 5) % 7) + 1;
387 return static_cast<DayOfWeek>(dow);
388 }
