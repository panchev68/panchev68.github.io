346 static DayOfWeek dayOfWeek(const Date & date) {
347 int y = date.year;
348 int m = date.month;
349 int d = date.day;
350 if (m < 3) {
351 m += 12;
352 y -= 1;
353 }
354 int k = y % 100;
355 int j = y / 100;
356 int h = (d + 13 * (m + 1) / 5 + k + k / 4 + j / 4 + 5 * j) % 7;
357 int dow = ((h + 5) % 7) + 1;
358 return static_cast<DayOfWeek>(dow);
359 }
