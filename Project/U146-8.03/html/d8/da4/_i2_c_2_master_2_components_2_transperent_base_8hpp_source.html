366 void emitCallback(HAL_I2C_CallbackIDTypeDef id) {
367 if (id == HAL_I2C_CallbackIDTypeDef::HAL_I2C_MASTER_TX_COMPLETE_CB_ID ||
368 id == HAL_I2C_MEM_TX_COMPLETE_CB_ID) {
369 if (isWaitForComplete) {
370 if (!semComplete.giveFromISR()) assert(false);
371 } else {
372 emitTxCompleteEvent();
373 }
374 } else if (id == HAL_I2C_CallbackIDTypeDef::HAL_I2C_MASTER_RX_COMPLETE_CB_ID ||
375 id == HAL_I2C_MEM_RX_COMPLETE_CB_ID) {
376 if (isWaitForComplete) {
377 if (!semComplete.giveFromISR()) assert(false);
378 } else {
379 emitRxCompleteEvent();
380 }
381 } else if (id == HAL_I2C_CallbackIDTypeDef::HAL_I2C_ERROR_CB_ID) {
382 emitErrorEvent();
383 } else if (id == HAL_I2C_CallbackIDTypeDef::HAL_I2C_ABORT_CB_ID) {
384 emitAbortEvent();
385 }
