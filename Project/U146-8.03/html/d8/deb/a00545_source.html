75 static uint32_t stringToU32(const std::string & string) {
76 std::array<uint8_t, 2> arr = {0, 0};
77 size_t begin = 0;
78 for (size_t i = 0; i < arr.size(); ++i) {
79 size_t found = string.find('.', begin);
80 if (found == std::string::npos) found = string.size();
81 arr[i] = static_cast<uint8_t>(std::stol(string.substr(begin, found - begin)));
82 begin = found + 1;
83 if (found == string.size()) break;
84 }
85 return (static_cast<uint32_t>(arr[0]) << 8) | static_cast<uint32_t>(arr[1]);
86 }
