168 void taskFn() {
169 auto checkPressed = [this]() -> bool {
170 bool pin = (data->nirq.read() == 0);
171 if (data->debounceTime == 0) return pin;
172 if (pin) {
173 if (!data->lastPressState) {
174 data->lastPressState = true;
175 data->timeoutDebounce.set(data->debounceTime);
176 } else {
177 return data->timeoutDebounce.check();
178 }
179 } else {
180 data->lastPressState = false;
181 }
182 return false;
183 };
184
185 auto writeByte = [this](uint8_t val) {
186 for (int i = 0; i < 8; ++i) {
187 data->si.write((val & 0x80) != 0);
188 __NOP();
189 val <<= 1;
190 data->sck.write(0);
191 __NOP();
192 data->sck.write(1);
193 __NOP();
194 }
195 };
196
197 auto readAdc = [this, &writeByte](uint8_t cmd) -> uint16_t {
198 uint32_t num = 0;
199 data->si.write(0);
200 data->nss.write(0);
201 writeByte(cmd);
202 data->sck.write(0);
203 __NOP();
204 data->sck.write(1);
205 __NOP();
206 for (int i = 0; i < 16; ++i) {
207 num <<= 1;
208 data->sck.write(0);
209 __NOP();
210 data->sck.write(1);
211 __NOP();
212 if (data->so.read() == 1) ++num;
213 }
214 num >>= 4;
215 data->nss.write(1);
216 return static_cast<uint16_t>(num);
217 };
218
219 data->sck.write(1);
220 data->nss.write(1);
221 bool isPressed = false;
222 for (;;) {
223 if (checkPressed()) {
224 if (!isPressed) {
225 isPressed = true;
226 data->integratorX.reset();
227 data->integratorY.reset();
228 data->position = {-1, -1};
229 }
230 int32_t rawX = data->integratorX.process(readAdc(CMD_READ_X));
231 int32_t rawY = data->integratorY.process(readAdc(CMD_READ_Y));
232 if (!data->integratorX.isFull()) {
233 vTaskDelay(5);
234 continue;
235 }
236 auto clamp = [](int32_t v, int32_t lo, int32_t hi) -> int32_t {
237 return v < lo ? lo : (v > hi ? hi : v);
238 };
239 int32_t clampedX = clamp(rawX, parameters.xMin, parameters.xMax);
240 int32_t clampedY = clamp(rawY, parameters.yMin, parameters.yMax);
241 int32_t adjX = clampedX - parameters.xMin;
242 int32_t adjY = clampedY - parameters.yMin;
243 int32_t posX = static_cast<int32_t>(adjX / data->devX);
244 int32_t posY = static_cast<int32_t>(adjY / data->devY);
245 data->tempRawPosition = {clampedX, clampedY};
246 data->tempPosition = {posX, posY};
247 data->rawPosition = {rawX, rawY};
248 data->isReady = true;
249 switch (data->rotation) {
250 case 0:
251 data->position = data->tempPosition;
252 break;
253 case 1:
254 data->position = {parameters.dispWidth - posX, posY};
255 break;
256 case 2:
257 data->position = {parameters.dispWidth - posX, parameters.dispHeight - posY};
258 break;
259 case 3:
260 data->position = {posX, parameters.dispHeight - posY};
261 break;
262 case 4:
263 data->position = {parameters.dispWidth - posX, parameters.dispHeight - posY};
264 break;
265 default:
266 assert(false);
267 break;
268 }
269 } else {
270 isPressed = false;
271 data->isReady = false;
272 }
273 vTaskDelay(30);
274 }
275 }
