164 void taskFn() {
165 auto checkPressed = [this]() -> bool {
166 bool pin = (impl->nirq.read() == 0);
167 if (impl->debounceTime == 0) return pin;
168 if (pin) {
169 if (!impl->lastPressState) {
170 impl->lastPressState = true;
171 impl->timeoutDebounce.set(impl->debounceTime);
172 } else {
173 return impl->timeoutDebounce.check();
174 }
175 } else {
176 impl->lastPressState = false;
177 }
178 return false;
179 };
180
181 auto writeByte = [this](uint8_t val) {
182 for (int index = 0; index < 8; ++index) {
183 impl->si.write((val & 0x80) != 0);
184 __NOP();
185 val <<= 1;
186 impl->sck.write(0);
187 __NOP();
188 impl->sck.write(1);
189 __NOP();
190 }
191 };
192
193 auto readAdc = [this, &writeByte](uint8_t cmd) -> uint16_t {
194 uint32_t num = 0;
195 impl->si.write(0);
196 impl->nss.write(0);
197 writeByte(cmd);
198 impl->sck.write(0);
199 __NOP();
200 impl->sck.write(1);
201 __NOP();
202 for (int index = 0; index < 16; ++index) {
203 num <<= 1;
204 impl->sck.write(0);
205 __NOP();
206 impl->sck.write(1);
207 __NOP();
208 if (impl->so.read() == 1) ++num;
209 }
210 num >>= 4;
211 impl->nss.write(1);
212 return static_cast<uint16_t>(num);
213 };
214
215 impl->sck.write(1);
216 impl->nss.write(1);
217 bool isPressed = false;
218 for (;;) {
219 if (checkPressed()) {
220 if (!isPressed) {
221 isPressed = true;
222 impl->integratorX.reset();
223 impl->integratorY.reset();
224 impl->position = {-1, -1};
225 }
226 int32_t rawX = impl->integratorX.process(readAdc(CMD_READ_X));
227 int32_t rawY = impl->integratorY.process(readAdc(CMD_READ_Y));
228 if (!impl->integratorX.isFull()) {
229 vTaskDelay(5);
230 continue;
231 }
232 int32_t clampedX = std::max(parameters.xMin, std::min(rawX, parameters.xMax));
233 int32_t clampedY = std::max(parameters.yMin, std::min(rawY, parameters.yMax));
234 int32_t adjX = clampedX - parameters.xMin;
235 int32_t adjY = clampedY - parameters.yMin;
236 int32_t posX = static_cast<int32_t>(adjX / impl->devX);
237 int32_t posY = static_cast<int32_t>(adjY / impl->devY);
238 impl->tempRawPosition = {clampedX, clampedY};
239 impl->tempPosition = {posX, posY};
240 impl->rawPosition = {rawX, rawY};
241 impl->isReady = true;
242 switch (impl->rotation) {
243 case 0:
244 impl->position = impl->tempPosition;
245 break;
246 case 1:
247 impl->position = {parameters.dispWidth - posX, posY};
248 break;
249 case 2:
250 case 4:
251 impl->position = {parameters.dispWidth - posX, parameters.dispHeight - posY};
252 break;
253 case 3:
254 impl->position = {posX, parameters.dispHeight - posY};
255 break;
256 default:
257 assert(false);
258 break;
259 }
260 } else {
261 isPressed = false;
262 impl->isReady = false;
263 }
264 vTaskDelay(30);
265 }
266 }
