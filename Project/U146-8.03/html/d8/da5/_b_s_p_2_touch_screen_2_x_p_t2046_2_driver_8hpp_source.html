180 bool checkPressed() {
181 const bool pin = (nirq.read() == 0);
182 if (debounceTime == 0) return pin;
183 if (pin) {
184 if (!lastPressState) {
185 lastPressState = true;
186 timeoutDebounce.set(debounceTime);
187 } else {
188 return timeoutDebounce.check();
189 }
190 } else {
191 lastPressState = false;
192 }
193 return false;
194 }

201 void writeByte(uint8_t val) {
202 for (uint8_t i = 0; i < 8U; ++i) {
203 si.write((val & 0x80U) != 0U);
204 __NOP();
205 val <<= 1U;
206 sck.write(0);
207 __NOP();
208 sck.write(1);
209 __NOP();
210 }
211 }

219 uint16_t readAdc(uint8_t cmd) {
220 uint32_t num = 0U;
221 si.write(0);
222 nss.write(0);
223 writeByte(cmd);
224 sck.write(0);
225 __NOP();
226 sck.write(1);
227 __NOP();
228 for (uint8_t i = 0; i < 16U; ++i) {
229 num <<= 1U;
230 sck.write(0);
231 __NOP();
232 sck.write(1);
233 __NOP();
234 num += so.read();
235 }
236 num >>= 4U;
237 nss.write(1);
238 return static_cast<uint16_t>(num);
239 }

245 void taskFn() {
246 sck.write(1);
247 nss.write(1);
248 bool pressed = false;
249 for (;;) {
250 if (checkPressed()) {
251 if (!pressed) {
252 pressed = true;
253 integratorX.reset();
254 integratorY.reset();
255 position = {-1, -1};
256 }
257 const int32_t rawX = integratorX.process(readAdc(CMD_READ_X));
258 const int32_t rawY = integratorY.process(readAdc(CMD_READ_Y));
259 if (!integratorX.isFull()) {
260 vTaskDelay(1);
261 continue;
262 }
263 const int32_t clampedX = clamp(rawX, parameters.xMin, parameters.xMax);
264 const int32_t clampedY = clamp(rawY, parameters.yMin, parameters.yMax);
265 const int32_t adjX = clampedX - parameters.xMin;
266 const int32_t adjY = clampedY - parameters.yMin;
267 const int32_t posX = static_cast<int32_t>(adjX / devX);
268 const int32_t posY = static_cast<int32_t>(adjY / devY);
269 tempRawPosition = {clampedX, clampedY};
270 tempPosition = {posX, posY};
271 rawPosition = {rawX, rawY};
272 isReady = true;
273 switch (rotation) {
274 case 0:
275 position = tempPosition;
276 break;
277 case 1:
278 position = {parameters.dispWidth - posX, posY};
279 break;
280 case 2:
281 case 4:
282 position = {parameters.dispWidth - posX, parameters.dispHeight - posY};
283 break;
284 case 3:
285 position = {posX, parameters.dispHeight - posY};
286 break;
287 default:
288 assert(false);
289 break;
290 }
291 } else {
292 pressed = false;
293 isReady = false;
294 }
295 vTaskDelay(20);
296 }
297 }
298 };
