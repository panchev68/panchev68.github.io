177 bool checkPressed() {
178 bool pin = (nirq.read() == 0);
179 if (debounceTime == 0) return pin;
180 if (pin) {
181 if (!lastPressState) {
182 lastPressState = true;
183 timeoutDebounce.set(debounceTime);
184 } else {
185 return timeoutDebounce.check();
186 }
187 } else {
188 lastPressState = false;
189 }
190 return false;
191 }

197 void writeByte(uint8_t val) {
198 // Write 8 bits, MSB first
199 for (uint8_t index = 0; index < 8; ++index) {
200 si.write((val & 0x80) != 0);
201 __NOP();
202 val <<= 1;
203 sck.write(0);
204 __NOP();
205 sck.write(1);
206 __NOP();
207 }
208 }

215 uint16_t readAdc(uint8_t cmd) {
216 uint32_t value = 0;
217 si.write(0);
218 nss.write(0);
219 writeByte(cmd);
220 sck.write(0);
221 __NOP();
222 sck.write(1);
223 __NOP();
224 for (uint8_t index = 0; index < 16; ++index) {
225 value <<= 1;
226 sck.write(0);
227 __NOP();
228 sck.write(1);
229 __NOP();
230 value |= so.read();
231 }
232 value >>= 4;
233 nss.write(1);
234 return static_cast<uint16_t>(value);
235 }

241 void process() {
242 sck.write(1);
243 nss.write(1);
244 bool pressed = false;
245 for (;;) {
246 if (!checkPressed()) {
247 pressed = false;
248 isReady = false;
249 vTaskDelay(20);
250 continue;
251 }
252 if (!pressed) {
253 pressed = true;
254 integratorX.reset();
255 integratorY.reset();
256 position = {-1, -1};
257 }
258 int32_t rawX = integratorX.process(readAdc(CMD_READ_X));
259 int32_t rawY = integratorY.process(readAdc(CMD_READ_Y));
260 if (!integratorX.isFull()) {
261 vTaskDelay(1);
262 continue;
263 }
264 int32_t clampedX = clamp(rawX, parameters.xMin, parameters.xMax);
265 int32_t clampedY = clamp(rawY, parameters.yMin, parameters.yMax);
266 int32_t posX = static_cast<int32_t>((clampedX - parameters.xMin) / devX);
267 int32_t posY = static_cast<int32_t>((clampedY - parameters.yMin) / devY);
268 tempRawPosition = {clampedX, clampedY};
269 tempPosition = {posX, posY};
270 rawPosition = {rawX, rawY};
271 isReady = true;
272 switch (rotation) {
273 case 0:
274 position = tempPosition;
275 break;
276 case 1:
277 position = {parameters.dispWidth - posX, posY};
278 break;
279 case 2:
280 case 4:
281 position = {parameters.dispWidth - posX, parameters.dispHeight - posY};
282 break;
283 case 3:
284 position = {posX, parameters.dispHeight - posY};
285 break;
286 default:
287 assert(false);
288 break;
289 }
290 vTaskDelay(20);
291 }
292 }
