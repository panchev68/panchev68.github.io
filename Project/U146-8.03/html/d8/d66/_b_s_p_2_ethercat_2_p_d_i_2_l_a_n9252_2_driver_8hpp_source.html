79 bool resetChip() {
80 FreeRTOS::Timeout timeout;
81 pinReset.write(0);
82 vTaskDelay(50);
83 pinReset.write(1);
84 timeout.set(1000);
85 for (;;) {
86 if (verifyChipId()) {
87 state = State::READY;
88 vTaskDelay(100);
89 return true;
90 }
91 if (timeout.check()) {
92 state = State::ERROR;
93 return false;
94 }
95 }
96 }

195 void readCsr(uint16_t address, void * destination, size_t length) {
196 Helper::U32 param = {0};
197 Helper::U16 addressUnion;
198 addressUnion.value = address;
199 param.bytes[0] = addressUnion.byte.lowByte;
200 param.bytes[1] = addressUnion.byte.highByte;
201 param.bytes[2] = static_cast<uint8_t>(length);
202 param.bytes[3] = Registers::ESC_READ_BYTE;
203 writeU32(Registers::ESC_CSR_CMD_REG, param.value);
204 do { param.value = readU32(Registers::ESC_CSR_CMD_REG); } while (param.bytes[3] & Registers::ESC_CSR_BUSY);
205 param.value = readU32(Registers::ESC_CSR_DATA_REG);
206 std::memcpy(destination, param.bytes, length);
207 }

213 uint32_t readBurst() {
214 Helper::U32 valueUnion;
215 uint8_t readBytes[4] = {0};
216 readWrite(nullptr, 0, readBytes, 4);
217 valueUnion.byte.lowByte = readBytes[0];
218 valueUnion.byte.highByte = readBytes[1];
219 valueUnion.byte.upperByte = readBytes[2];
220 valueUnion.byte.mostByte = readBytes[3];

230 void writeCsr(uint16_t address, const uint8_t * source, uint8_t length) {
231 Helper::U32 param = {0};
232 Helper::U16 waddr;
233 std::memcpy(param.bytes, source, length);
234 writeU32(Registers::ESC_CSR_DATA_REG, param.value);
235 waddr.value = address;
236 param.bytes[0] = waddr.byte.lowByte;
237 param.bytes[1] = waddr.byte.highByte;
238 param.bytes[2] = length;
239 param.bytes[3] = Registers::ESC_WRITE_BYTE;
240 writeU32(Registers::ESC_CSR_CMD_REG, param.value);
241 do {
242 param.value = readU32(Registers::ESC_CSR_CMD_REG);
243 } while (param.bytes[3] & Registers::ESC_CSR_BUSY);
244 }
