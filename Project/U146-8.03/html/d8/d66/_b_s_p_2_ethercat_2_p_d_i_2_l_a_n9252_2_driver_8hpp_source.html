76 bool resetChip() {
77 FreeRTOS::Timeout timeout;
78 pinReset.write(0);
79 vTaskDelay(50);
80 pinReset.write(1);
81 timeout.set(1'000);
82 for (;;) {
83 if (verifyChipID()) {
84 state = State::READY;
85 vTaskDelay(100);
86 return true;
87 }
88 if (timeout.check()) {
89 state = State::ERROR;
90 return false;
91 }
92 }

192 void readCSR(uint16_t address, void * destination, size_t length) {
193 Helper::UU32 param = {0};
194 Helper::UU16 waddr;
195 waddr.val = address;
196 param.v[0] = waddr.bytes.lb;
197 param.v[1] = waddr.bytes.hb;
198 param.v[2] = static_cast<uint8_t>(length);
199 param.v[3] = Registers::ESC_READ_BYTE;
200 writeU32(Registers::ESC_CSR_CMD_REG, param.val);
201 do { param.val = readU32(Registers::ESC_CSR_CMD_REG); } while (param.v[3] & Registers::ESC_CSR_BUSY);
202 param.val = readU32(Registers::ESC_CSR_DATA_REG);
203 std::memcpy(destination, param.v, length);
204 }

228 void writeCSR(uint16_t address, const uint8_t * source, uint8_t length) {
229 Helper::UU32 param = {0};
230 Helper::UU16 waddr;
231 std::memcpy(param.v, source, length);
232 writeU32(Registers::ESC_CSR_DATA_REG, param.val);
233 waddr.val = address;
234 param.v[0] = waddr.bytes.lb;
235 param.v[1] = waddr.bytes.hb;
236 param.v[2] = length;
237 param.v[3] = Registers::ESC_WRITE_BYTE;
238 writeU32(Registers::ESC_CSR_CMD_REG, param.val);
239 do { param.val = readU32(Registers::ESC_CSR_CMD_REG); } while (param.v[3] & Registers::ESC_CSR_BUSY);
240 }
