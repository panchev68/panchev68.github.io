242 void handleReceivedPacked16Bits(const Lib::Xbee::Core::Frame0x81 & frame) {
243 if (state != State::RUNNING) {
244 return;

246 constexpr uint32_t MAX_RSSI_VAL = 90;
247 uint32_t rssi = frame.rssi > MAX_RSSI_VAL ? MAX_RSSI_VAL : frame.rssi;
248 connection.info.signalStrength = static_cast<uint8_t>(((MAX_RSSI_VAL - rssi) * 100) / MAX_RSSI_VAL);
249 diagnostic.receivedDataMessages++;
