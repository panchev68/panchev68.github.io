155 void handleReceivedChar(const Frame0x97 & frame) {
156 if ((frame.frameId != 0x17) || (impl->sendFrame.atCmd != frame.atCmd)) return;
157 if ((frame.commandStatus != 0) && (frame.commandStatus != 4)) {
158 assert(false);
