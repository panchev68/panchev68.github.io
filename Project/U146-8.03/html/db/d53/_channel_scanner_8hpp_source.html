163 void taskFn() {
164 enum class State { IDLE, BEGIN, OPEN_THE_CHANNEL, SCAN, NEXT_CHANNEL, END };
165 State state = State::BEGIN;
166 size_t index = 0U;
167 uint32_t scanStartTime = 0U;
168 Channel * current = nullptr;
169 auto emitError = [this, &state](Error error) {
170 if (onError) onError(error);
171 state = State::IDLE;
172 };
173 for (;;) {
174 const uint32_t currentTime = xTaskGetTickCount();
175 switch (state) {
176 case State::IDLE:
177 break;
178 case State::BEGIN:
179 moduleParameters->open();
180 if (!connection->state.isConnected()) {
181 assert(false);
182 emitError(Error::MODULE_IS_NOT_CONNECTED);
183 break;
184 }
185 for (size_t i = 0U; i < channels.size(); ++i) {
186 channels[i] = static_cast<uint8_t>(i);
187 channels[i].setConnectionQuality(0U);
188 }
189 index = 0U;
190 state = State::OPEN_THE_CHANNEL;
191 break;
192 case State::OPEN_THE_CHANNEL:
193 current = &channels[index];
194 moduleParameters->setChannel(*current);
195 moduleParameters->writeChanges();
196 connection->quality.setScannerMode();
197 scanStartTime = currentTime;
198 state = State::SCAN;
199 break;
200 case State::SCAN: {
201 Connection::Quality * sender = nullptr;
202 if (queUpdateQuality.receive(&sender, 0U)) {
203 current->setConnectionQuality(sender->getPercent());
204 state = State::NEXT_CHANNEL;
205 } else if ((currentTime - scanStartTime) >= pdMS_TO_TICKS(SCAN_CHANNEL_PERIOD)) {
206 state = State::NEXT_CHANNEL;
207 }
208 break;
209 }
210 case State::NEXT_CHANNEL:
211 ++index;
212 state = (index >= channels.size()) ? State::END : State::OPEN_THE_CHANNEL;
213 break;
214 case State::END:
215 moduleParameters->close();
216 state = State::IDLE;
217 break;
218 }
219 vTaskDelay(1U);
220 }
221 }

229 }
