BSP::Outputs<OutputVersion::V1>

Example usage of the Outputs<OutputVersion::V1> class.

Example usage of the Outputs<OutputVersion::V1> class.Demonstrates configuration, initialization, error handling, and resource management for OutputVersion::V1 digital output devices in STM32 embedded systems. The example shows how to set up SPI and GPIO handles, initialize the Outputs controller, bind an error handler, and check error state.

Example usage:
#include "Outputs.V1.hpp"
#include <Lib/HAL/SPI.hpp>
#include <Lib/HAL/GPIO.hpp>
void errorHandler(void * context, uint32_t errorCode) {
// Handle error (logging, recovery, etc.)
}
int main() {
Lib::HAL::SPI::Handle spiHandle;
Lib::HAL::GPIO::Handle ssPinHandle;
Lib::HAL::GPIO::Handle csPinHandle;
BSP::Outputs<OutputVersion::V1>::Config config{ &spiHandle, ssPinHandle, csPinHandle };
BSP::Outputs<OutputVersion::V1> outputs(config);
if (outputs.init()) {
outputs.bindErrorHandler(errorHandler);
// Use outputs controller for device operations.
bool hasError = outputs.getErrorState();
// Check error state and handle as needed.
} else {
// Initialization failed; handle error.
}
outputs.deinit();
return 0;
}
