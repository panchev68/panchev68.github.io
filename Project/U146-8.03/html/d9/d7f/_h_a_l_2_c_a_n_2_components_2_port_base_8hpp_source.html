99 virtual void emitCallback(HAL_CAN_CallbackIDTypeDef id) {
100 RxMessage rxMessage;
101 switch (id) {
102 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_ERROR_CB_ID:
103 semaphoreError.giveFromISR();
104 break;
105 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX0_COMPLETE_CB_ID:
106 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX1_COMPLETE_CB_ID:
107 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX2_COMPLETE_CB_ID:
108 semaphoreTxComplete.giveFromISR();
109 break;
110 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_RX_FIFO0_MSG_PENDING_CB_ID:
111 if (HAL_CAN_GetRxMessage(getHandle(), CAN_RX_FIFO0, &rxMessage, rxMessage.getData()) == HAL_OK) {
112 queueRxMessage.sendToBackFromISR(&rxMessage);
113 }
114 break;
115 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_RX_FIFO1_MSG_PENDING_CB_ID:
116 if (HAL_CAN_GetRxMessage(getHandle(), CAN_RX_FIFO1, &rxMessage, rxMessage.getData()) == HAL_OK) {
117 queueRxMessage.sendToBackFromISR(&rxMessage);

217 return false;

219 return true;
220 };
221 for (;;) {
222 auto selected = queueSet.select(portMAX_DELAY);
223 if (selected == queueRxMessage) {
224 if (queueRxMessage.receive(&rxMessage, 0)) { messageReceivedCallback(rxMessage); }
225 } else if (selected == queueTxMessage) {
226 if (!queueTxMessage.receive(&txMessage, 0)) {
227 assert(false);
228 } else {
229 preSendCallback();
230 if (sendFunc(txMessage, txMailbox)) {
231 if (!semaphoreTxComplete.take(100)) {
232 // Transmission timeout
233 } else {
234 semaphoreTxCompleted.give();
235 }
236 }
237 postSendCallback();
238 }
239 } else if (selected == semaphoreError) {
