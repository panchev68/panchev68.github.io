96 virtual void emitCallback(HAL_CAN_CallbackIDTypeDef id) {
97 RxMessage rxMessage;
98 switch (id) {
99 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_ERROR_CB_ID: semaphoreError.giveFromISR(); break;
100 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX0_COMPLETE_CB_ID:
101 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX1_COMPLETE_CB_ID:
102 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_TX_MAILBOX2_COMPLETE_CB_ID: semaphoreTxComplete.giveFromISR(); break;
103 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_RX_FIFO0_MSG_PENDING_CB_ID:
104 if (HAL_CAN_GetRxMessage(getHandle(), CAN_RX_FIFO0, &rxMessage, rxMessage.getData()) == HAL_OK) {
105 queueRxMessage.sendToBackFromISR(&rxMessage);
106 }
107 break;
108 case HAL_CAN_CallbackIDTypeDef::HAL_CAN_RX_FIFO1_MSG_PENDING_CB_ID:
109 if (HAL_CAN_GetRxMessage(getHandle(), CAN_RX_FIFO1, &rxMessage, rxMessage.getData()) == HAL_OK) {
110 queueRxMessage.sendToBackFromISR(&rxMessage);
111 }

201 void taskFn() {
202 auto sendFunc = [this](const TxMessage & msg, uint32_t & txMailbox) {
203 if (HAL_CAN_AddTxMessage(getHandle(), &msg, const_cast<uint8_t *>(msg.getData()), &txMailbox) != HAL_OK) { return false; }
204 return true;
205 };
206 for (;;) {
207 auto selected = queueSet.select(portMAX_DELAY);
208 if (selected == queueRxMessage) {
209 if (queueRxMessage.receive(&rxMessage, 0)) { messageReceivedCallback(rxMessage); }
210 } else if (selected == queueTxMessage) {
211 if (!queueTxMessage.receive(&txMessage, 0)) {
212 assert(false);
213 } else {
214 preSendCallback();
215 if (sendFunc(txMessage, txMailbox)) {
216 if (!semaphoreTxComplete.take(100)) {
217 // TX complete timeout
218 } else {
219 semaphoreTxCompleted.give();
220 }
221 }
222 postSendCallback();
223 }
224 } else if (selected == semaphoreError) {
225 errorCallback(HAL_CAN_GetError(getHandle()));
226 }
227 }
228 }
