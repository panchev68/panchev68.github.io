196 void taskFn() {
197 uint8_t lastSecond = 0xff;
198 uint8_t lastDate = 0xff;
199 for (;;) {
200 auto dt = get();
201 if (dt.time.seconds != lastSecond) {
202 lastSecond = dt.time.seconds;
203 if (onTimeChanged) onTimeChanged(dt);
204 }
205 if (dt.date.day != lastDate) {
206 lastDate = dt.date.day;
