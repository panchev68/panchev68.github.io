145 void processScanning() {
146 enum class State { IDLE,
147 BEGIN,
148 OPEN_CHANNEL,
149 SCAN,
150 NEXT_CHANNEL,
151 END };
152 State state = State::BEGIN;
153 size_t index = 0;
154 uint32_t scanStartTime = 0;
155 Channel * current = nullptr;
156 auto emitError = [this, &state](Error error) {
157 if (onError) onError(error);
158 state = State::IDLE;
159 };
160 for (;;) {
161 const uint32_t currentTime = xTaskGetTickCount();
162 switch (state) {
163 case State::IDLE:
164 vTaskDelay(10);
165 continue;
166 case State::BEGIN:
167 moduleParameters->open();
168 if (!connection->state.isConnected()) {
169 emitError(Error::MODULE_IS_NOT_CONNECTED);
170 break;
171 }
172 for (size_t i = 0; i < channels.size(); ++i) {
173 channels[i] = static_cast<uint8_t>(i);
174 channels[i].setConnectionQuality(0);
175 }
176 index = 0;
177 state = State::OPEN_CHANNEL;
178 break;
179 case State::OPEN_CHANNEL:
180 current = &channels[index];
181 moduleParameters->setChannel(*current);
182 moduleParameters->writeChanges();
183 connection->quality.setScannerMode();
184 scanStartTime = currentTime;
185 state = State::SCAN;
186 break;
187 case State::SCAN: {
188 Connection::Quality * sender = nullptr;
189 if (queUpdateQuality.receive(&sender, 0)) {
190 current->setConnectionQuality(sender->getPercent());
191 state = State::NEXT_CHANNEL;
192 } else if ((currentTime - scanStartTime) >= pdMS_TO_TICKS(SCAN_CHANNEL_PERIOD)) {
193 state = State::NEXT_CHANNEL;
194 }
195 break;
196 }
197 case State::NEXT_CHANNEL:
198 ++index;
199 state = (index >= channels.size()) ? State::END : State::OPEN_CHANNEL;
200 break;
201 case State::END:
202 moduleParameters->close();
203 state = State::IDLE;
204 break;
205 }
206 vTaskDelay(1);
207 }
208 }

216 }
