281 bool readStaticParameters(Data & data) {
282 assert(localAtCommand->isOpen());
283 std::vector<uint8_t> response;
284
285 struct AtCommand {
286 const char * cmd;
287 size_t sz;
288 StaticAssignFn fn;
289 };
290
291 static constexpr AtCommand cmds[] = {
292 {"NI", 0, assignNodeId},
293 {"SL", 4, assignSL},
294 {"SH", 4, assignSH},
295 {"ID", 2, assignPanId},
296 {"CH", 1, assignChannel},
297 {"VR", 2, assignFw},
298 {"HV", 2, assignHw}};
299 for (const auto & c : cmds) {
300 response.clear();
301 if (!localAtCommand->send({c.cmd}) || !localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) return false;
302 if (c.sz && response.size() != c.sz) return false;
303 if (c.cmd[0] == 'N' && c.cmd[1] == 'I') response.push_back(0);
304 c.fn(data, response);
305 }
306 return true;
307 }

342 bool readDynamicParameters(Data & data, uint32_t mask = static_cast<uint32_t>(-1)) {
343 assert(localAtCommand->isOpen());
344 std::vector<uint8_t> response;
345
346 struct DynamicCommand {
347 uint32_t bit;
348 const char * cmd;
349 size_t sz;
350 DynamicAssignFn fn;
351 };
352
353 static constexpr DynamicCommand cmds[] = {
354 {0x0001, "EA", 0, nullptr},
355 {0x0002, nullptr, 2, assignAck},
356 {0x0004, "EC", 0, nullptr},
357 {0x0008, nullptr, 2, assignCca},
358 {0x0010, "%V", 0, nullptr},
359 {0x0020, nullptr, 2, assignVolt},
360 {0x0040, "TP", 0, nullptr},
361 {0x0080, nullptr, 2, assignTemp},
362 {0x0100, "CH", 0, nullptr},
363 {0x0200, nullptr, 1, assignChan}};
364 for (const auto & c : cmds) {
365 if (!(mask & c.bit)) continue;
366 if (c.cmd) {
367 taskYIELD();
368 if (!localAtCommand->send({c.cmd})) return false;
369 } else if (c.fn) {
370 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) return false;
371 if (c.sz && response.size() != c.sz) return false;
372 c.fn(data, response);
373 response.clear();
374 }
375 }
376 return true;
377 }

383 void process(Data & data) {
384 static std::vector<uint8_t> response;
385 using Convert = Core::Frame::Helper::Convert;
386
387 if (!pause && state == State::IDLE && timeoutUpdate.check()) {
388 timeoutUpdate.set(50);
389 const auto mask = 1u << (tickCounter++ % 10);
390 readDynamicParameters(data, mask);
391 }
392
393 switch (state) {
394 case State::IDLE:
395 break;
396 case State::SET_PARAMETERS_REMOTELY:
397 state = (remotePanId == static_cast<uint64_t>(-1)) ? State::GET_REMOTE_PAN_ID : State::SET_CHANNEL_REMOTELY;
398 break;
399 case State::GET_REMOTE_PAN_ID:
400 if (newData.isChannelChanged && newData.channel != data.channel) {
401 state = localAtCommand->send({"ND"}) ? State::GET_REMOTE_PAN_ID_RESPONSE : State::END;
402 } else {
403 state = State::END;
404 }
405 break;
406 case State::GET_REMOTE_PAN_ID_RESPONSE:
407 if (localAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) {
408 remotePanId = Convert::to<uint64_t>(response.data() + 2);
409 }
410 state = State::SET_CHANNEL_REMOTELY;
411 break;
412 case State::SET_CHANNEL_REMOTELY:
413 if (newData.isChannelChanged && newData.channel != data.channel) {
414 state = remoteAtCommand->send({"CH"}, remotePanId, {newData.channel}) ? State::SET_CHANNEL_REMOTELY_RESPONSE : State::END;
415 } else {
416 state = State::END;
417 }
418 break;
419 case State::SET_CHANNEL_REMOTELY_RESPONSE:
420 state = remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND) ? State::WRITE_CHANGES_REMOTELY : State::END;
421 break;
422 case State::WRITE_CHANGES_REMOTELY:
423 state = remoteAtCommand->send({"WR"}, true) ? State::WRITE_CHANGES_REMOTELY_RESPONSE : State::END;
424 break;
425 case State::WRITE_CHANGES_REMOTELY_RESPONSE:
426 remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND);
427 state = State::SET_PARAMETERS_LOCAL;
428 break;
429 case State::SET_PARAMETERS_LOCAL:
430 hasWrite = false;
431 state = State::SET_CHANNEL_LOCALLY;
432 break;
433 case State::SET_CHANNEL_LOCALLY:
434 if (newData.isChannelChanged && newData.channel != data.channel) {
435 if (localAtCommand->send({"CH"}, {newData.channel}) && localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) {
436 data.channel = newData.channel;
437 hasWrite = true;
438 }
439 }
440 state = State::SET_PAN_ID_LOCALLY;
441 break;
442 case State::SET_PAN_ID_LOCALLY:
443 if (newData.isPanIdChanged && newData.panId != data.panId) {
444 if (localAtCommand->send({"ID"}, {static_cast<uint8_t>(newData.panId >> 8), static_cast<uint8_t>(newData.panId & 0xFF)}) &&
445 localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) {
446 data.panId = newData.panId;
447 hasWrite = true;
448 }
449 }
450 state = State::WRITE_CHANGES_LOCALLY;
451 break;
452 case State::WRITE_CHANGES_LOCALLY:
453 if (hasWrite) {
454 if (localAtCommand->send({"WR"})) localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND);
455 state = State::APPLY_CHANGES_LOCALLY;
456 } else {
457 state = State::END;
458 }
459 break;
460 case State::APPLY_CHANGES_LOCALLY:
461 if (localAtCommand->send({"AC"})) localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND);
462 connection->quality.clear();
463 state = State::END;
464 break;
465 case State::END:
466 newData.isChannelChanged = false;
467 newData.isPanIdChanged = false;
468 state = State::IDLE;
469 break;
470 }
471 }
