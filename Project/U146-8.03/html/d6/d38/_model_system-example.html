ModelSystem

Example usage of ModelSystem for system metadata access and UI integration.

Demonstrates how to create a ModelSystem instance, retrieve build/hardware version, serial number, last reset state, and interact with production dates for UI display.

Example usage:

#include "ModelSystem.hpp"
#include <iostream>
int main() {
ModelSystem modelSystem;
std::cout << "Build Version: " << modelSystem.getBuildVersion() << std::endl;
std::cout << "Hardware Version: " << modelSystem.getHardwareVersion() << std::endl;
std::cout << "Serial Number: " << modelSystem.getSerialNumber() << std::endl;
std::cout << "Last Reset State: " << modelSystem.getLastResetStateString() << std::endl;
std::cout << "Last Reset State Color: " << modelSystem.getLastResetStateColor() << std::endl;
auto startupMode = modelSystem.getStartupMode();
modelSystem.setStartupMode(startupMode);
if (modelSystem.produceDates) {
std::cout << "Produce Date: " << std::string(modelSystem.produceDates->produceDate) << std::endl;
std::cout << "Sale Date: " << std::string(modelSystem.produceDates->saleDate) << std::endl;
}
bool ready = modelSystem.waitForReadyAfterStartup(1000);
std::cout << "System Ready: " << (ready ? "Yes" : "No") << std::endl;
return 0;
}
