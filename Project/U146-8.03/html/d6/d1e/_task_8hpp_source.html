122 bool create(std::string name, std::function<void()> function, size_t stack = configMINIMAL_STACK_SIZE, uint32_t priority = 2) {
123 assert(handle == nullptr);
124 assert(function);
125
126 this->function = std::move(function);
127 auto entry = [](void * arg) {
128 assert(arg);
129 auto task = static_cast<Task *>(arg);
130 assert(task->function);
131 task->function();
132 task->release();
133 };
134
135 BaseType_t result = xTaskCreate(entry, name.c_str(), stack, this, tskIDLE_PRIORITY + priority, &handle);
136 if (result != pdPASS) {
137 handle = nullptr;
138 return false;
139 }
140
141 assert(handle);
142 return true;
143 }
