92 void handleModemStatus(const Frame0x8a & frame) {
93 using Status = Frame0x8a::ModemStatus;
94 static constexpr size_t STATUS_COUNT = static_cast<size_t>(Status::STACK_ERROR) + 1;
95 using Callback = std::function<void()> ModemStatus::*;
96 static constexpr Callback callbacks[STATUS_COUNT] = {&ModemStatus::onHardwareReset,
97 &ModemStatus::onWatchdogTimerReset, &ModemStatus::onJoinedNetwork, &ModemStatus::onLeftNetwork,
98 &ModemStatus::onCoordinatorStarted, &ModemStatus::onNetworkSecurityKeyWasUpdated, &ModemStatus::onNetworkWokeUp,
99 &ModemStatus::onNetworkWentToSleep, &ModemStatus::onVoltageSupplyLimitExceeded,
100 &ModemStatus::onRemoteManagerConnected, &ModemStatus::onRemoteManagerDisconnected,
101 &ModemStatus::onModemConfigurationChangedWhileJoinInProgress, &ModemStatus::onAccessFault,
102 &ModemStatus::onFatalError, &ModemStatus::onSecureSessionSuccessfullyEstablished,
103 &ModemStatus::onSecureSessionEnded, &ModemStatus::onSecureEessionAuthenticationFailed,
104 &ModemStatus::onCoordinatorDetectedAPanIdConflictButTookNoAction,
105 &ModemStatus::onCoordinatorChangedPanIdDueToAConflict, &ModemStatus::onBLEConnect, &ModemStatus::onBLEDisconnect,
106 &ModemStatus::onBandmaskConfigurationFailed, &ModemStatus::onCellularComponentUpdateStarted,
107 &ModemStatus::onCellularComponentUpdateFailed, &ModemStatus::onCellularComponentUpdateCompleted,
108 &ModemStatus::onXbeeFirmwareUpdateStarted, &ModemStatus::onXbeeFirmwareUpdateFailed,
109 &ModemStatus::onXbeeFirmwareUpdateApplying, &ModemStatus::onRouterPANIDWasShangedByCoordinatorDueToConflict,
110 &ModemStatus::onNetworkWatchdogTimeoutExpired, &ModemStatus::onStackError};
111 auto statusValue = static_cast<size_t>(frame.modemStatus);
112 if (statusValue < STATUS_COUNT) {
113 auto & cb = this->*callbacks[statusValue];
114 if (cb) cb();
115 } else {
116 if (onStackError) onStackError();
117 assert(false);
118 }
119 }
