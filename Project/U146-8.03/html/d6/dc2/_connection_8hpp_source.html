163 State update() {
164 int32_t result = 100;
165 int32_t rssiValue = 0;
166 if (connection->state == Connection::State::CONNECTED) {
167 if (timeout.check()) {
168 timeout.set(RESET_CYCLE_PERIOD);
169 lastRetriesCounter = connection->info.retriesCounter;
170 if (mode == Mode::RUNTIME) {
171 if (retriesPenalty > 0) retriesPenalty--;
172 if (transmitTimePenalty > 0) transmitTimePenalty--;
173 if (rssiPenalty > 0) rssiPenalty--;
174 }
175 }
176 if (connection->info.retriesCounter != lastRetriesCounter) {
177 if (connection->info.retriesCounter > lastRetriesCounter) {
178 retriesPenalty = std::min(retriesPenalty + 50, 100);
179 }
180 lastRetriesCounter = connection->info.retriesCounter;
181 }
182 if (lastTransmitPeriod != connection->info.transmitPeriod.getRawValue()) {
183 lastTransmitPeriod = connection->info.transmitPeriod.getRawValue();
184 if (connection->info.transmitPeriod > TRANSMIT_TIME_THRESHOLD) {
185 transmitTimePenalty = std::min(transmitTimePenalty + 5, 100);
186 }
187 }
188 rssiValue = connection->info.signalStrength;
189 rssiValue = ((lastRssi * 2) + rssiValue) / 3;
190 lastRssi = static_cast<uint8_t>(rssiValue);
191 if (rssiValue < RSSI_MIN_THRESHOLD) rssiPenalty = (RSSI_MIN_THRESHOLD - rssiValue);
192 result -= retriesPenalty;
193 result -= transmitTimePenalty;
194 result -= rssiPenalty;
195 percent = static_cast<uint8_t>(std::max(result, 0));
196 if (percent > 90) state = State::AMAZING;
197 else if (percent > 75) state = State::VERY_GOOD;
198 else if (percent > 50) state = State::OKAY;
199 else if (percent > 30) state = State::NOT_GOOD;
200 else state = State::UNUSABLE;
201 if (onUpdate) onUpdate(this);
202 } else {
203 state = State::DISCONNECTED;
204 percent = 0;
205 }
206 return state;
207 }
