Interpreter Design Pattern

Въведение

Патърнът Interpreter е поведенчески дизайн патърн, който предоставя начин за оценка на изрази, представени като абстрактно синтактично дърво. Този патърн е полезен, когато имате език за интерпретиране и искате да оцените изрази в този език.

Пример

Ето един пример за имплементация на патърна Interpreter в C++:

/**
* @file Interpreter.hpp
* This file implements the Interpreter design pattern for evaluating arithmetic expressions using an abstract syntax
* tree. It provides an interface for expressions, concrete classes for numbers, variables, and addition, and a sample
* usage in the main function. The implementation demonstrates how to interpret expressions with a context of variable
* values using C++17 features and modern memory management practices.
*/
#include <iostream>
#include <memory>
#include <string>
#include <unordered_map>
/**
* Interface for expression evaluation.
* Defines the contract for interpreting expressions with a given context mapping variable names to integer values.
*/
class Expression {
public:
/**
* Evaluates the expression using the provided context mapping variable names to integer values.
* @param context The context mapping variable names to integer values.
* @return The result of evaluating the expression as an integer.
* @pre The context must contain all variable names used in the expression.
*/
virtual int interpret(const std::unordered_map<std::string, int> & context) const = 0;
/**
* Destroys the Expression object.
*/
virtual ~Expression() = default;
};
/**
* Concrete expression representing a numeric constant.
* Implements the Expression interface to return a fixed integer value regardless of the context provided.
*/
class Number : public Expression {
public:
/**
* Constructs a Number expression with the specified integer value.
* @param value The integer value to be represented by this expression.
*/
explicit Number(int value) : value(value) {}
/**
* Returns the stored integer value, ignoring the context.
* @param context The context mapping variable names to integer values.
* @return The stored integer value.
*/
int interpret(const std::unordered_map<std::string, int> & context) const override { return value; }
private:
int value; ///< The integer value represented by this expression.
};
/**
* Concrete expression representing a variable.
* Implements the Expression interface to look up the variable's value in the provided context.
*/
class Variable : public Expression {
public:
/**
* Constructs a Variable expression with the specified variable name.
* @param name The name of the variable.
*/
explicit Variable(const std::string & name) : name(name) {}
/**
* Returns the value of the variable from the context.
* @param context The context mapping variable names to integer values.
* @return The value of the variable from the context.
* @pre The context must contain the variable name.
*/
int interpret(const std::unordered_map<std::string, int> & context) const override { return context.at(name); }
private:
std::string name; ///< The name of the variable.
};
/**
* Concrete expression representing addition of two expressions.
* Implements the Expression interface to evaluate the sum of two sub-expressions.
*/
class Add : public Expression {
public:
/**
* Constructs an Add expression with left and right operand expressions.
* @param left The left operand expression.
* @param right The right operand expression.
*/
Add(std::unique_ptr<Expression> left, std::unique_ptr<Expression> right) :
left(std::move(left)), right(std::move(right)) {}
/**
* Evaluates the sum of the left and right operand expressions using the provided context.
* @param context The context mapping variable names to integer values.
* @return The sum of the evaluated left and right expressions.
*/
int interpret(const std::unordered_map<std::string, int> & context) const override {
return left->interpret(context) + right->interpret(context);
}
private:
std::unique_ptr<Expression> left; ///< The left operand expression.
std::unique_ptr<Expression> right; ///< The right operand expression.
};
/**
* Entry point for demonstrating the Interpreter pattern with arithmetic expressions.
* @return Exit code indicating successful execution.
*/
int main() {
std::unordered_map<std::string, int> context = {
{"x", 5 },
{"y", 10}
};
auto expression = std::make_unique<Add>(std::make_unique<Variable>("x"), std::make_unique<Number>(20));
std::cout << "Result: " << expression->interpret(context) << std::endl;
return 0;
}

В този пример, Expression е интерфейсът, който дефинира метода interpret. Number и Variable са конкретни изрази, които имплементират този интерфейс. Add е конкретен израз, който събира два израза. В main функцията се създава контекст с променливи и се интерпретира израз, който събира променлива и число.
