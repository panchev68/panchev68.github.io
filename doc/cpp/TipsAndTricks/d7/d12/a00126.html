Mediator Design Pattern

Въведение

В този пример класът Mediator централизира комуникацията между различни обекти, наречени Colleague. Вместо да обменят съобщения директно, обектите Colleague използват посредничеството на Mediator, което намалява зависимостите между тях и улеснява разширяемостта и поддръжката на системата.

Примерен код:

/**
* @file Mediator.hpp
* Demonstrates the Mediator design pattern in C++ for embedded and UI codebases. This file defines abstract colleague
* and mediator interfaces, along with concrete implementations for managing communication between colleague objects.
* The pattern centralizes complex communications and control logic between related objects, promoting loose coupling
* and easier maintenance in embedded and UI application architectures.
*/
#pragma once
#include <iostream>
#include <string>
#include <vector>
/**
* Abstract interface for colleague objects participating in mediated communication.
* This class defines the interface for objects that communicate through a mediator, decoupling direct interactions
* and enabling centralized message distribution in collaborative systems.
*/
class Colleague {
public:
/**
* Virtual destructor for Colleague.
*/
virtual ~Colleague() = default;
/**
* Sends a message to other colleagues via the mediator.
* @param message The message to be sent.
*/
virtual void send(const std::string & message) = 0;
/**
* Receives a message from another colleague via the mediator.
* @param message The message received.
*/
virtual void receive(const std::string & message) = 0;
};
/**
* Mediator class for managing communication between colleague objects.
* This class provides mechanisms to register colleagues and distribute messages among them, centralizing
* communication logic and reducing dependencies between individual colleagues.
*/
class Mediator {
public:
/**
* Adds a colleague to the mediator's list of participants.
* @param colleague Pointer to the colleague to be added.
*/
void addColleague(Colleague * colleague) { colleagues.push_back(colleague); }
/**
* Distributes a message from the sender to all other colleagues.
* @param sender Pointer to the colleague sending the message.
* @param message The message to be distributed.
*/
void distributeMessage(Colleague * sender, const std::string & message) {
for (Colleague * colleague : colleagues) {
if (colleague != sender) { colleague->receive(message); }
}
}
private:
std::vector<Colleague *> colleagues; ///< List of colleagues managed by the mediator.
};
/**
* Concrete implementation of a colleague participating in mediated communication.
* This class registers itself with a mediator and provides mechanisms to send and receive messages, enabling
* decoupled communication with other colleagues through the mediator.
*/
class ConcreteColleague : public Colleague {
public:
/**
* Constructs a ConcreteColleague and registers it with the given mediator.
* @param mediator Reference to the mediator managing communication.
* @param name The name of the colleague.
*/
ConcreteColleague(Mediator & mediator, const std::string & name) : mediator(mediator), name(name) {
mediator.addColleague(this);
}
/**
* Sends a message to other colleagues via the mediator.
* @param message The message to be sent.
*/
void send(const std::string & message) override {
std::cout << name << " sends message: " << message << std::endl;
mediator.distributeMessage(this, message);
}
/**
* Receives a message from another colleague via the mediator.
* @param message The message received.
*/
void receive(const std::string & message) override {
std::cout << name << " receives message: " << message << std::endl;
}
private:
Mediator & mediator; ///< Reference to the mediator managing communication.
std::string name; ///< Name of the colleague.
};
#if defined(MEDIATOR_MAIN) || defined(INTELLISENSE) || defined(DOXYGEN)
/**
* Entry point for demonstrating the Mediator pattern.
* @return Exit status code.
*/
int main() {
Mediator mediator;
ConcreteColleague colleague1(mediator, "Colleague 1");
ConcreteColleague colleague2(mediator, "Colleague 2");
ConcreteColleague colleague3(mediator, "Colleague 3");
colleague1.send("Hello everyone!");
colleague2.send("Hello, Colleague 1!");
return 0;
}
#endif

В този пример обектите от тип ConcreteColleague комуникират чрез посредничеството на Mediator. Когато един Colleague изпрати съобщение, Mediator го препраща към всички останали регистрирани Colleague обекти.
