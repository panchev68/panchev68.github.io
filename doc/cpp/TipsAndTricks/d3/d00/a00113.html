Abstract Factory Design Pattern

ABSTRACT_FACTORY_INTRO Въведение

Абстрактната фабрика е породителен шаблон за проектиране, който позволява създаването на семейства от свързани обекти, без да се посочват техните конкретни класове. Този шаблон предоставя интерфейс за създаване на обекти в суперклас, позволявайки на подкласовете да променят типа на създаваните обекти. Използва се често за гарантиране на съвместимост между продуктите, създавани от фабриката, и за инкапсулиране на логиката по създаване на обекти. Абстрактната фабрика дефинира набор от методи за създаване на различни продукти, които обикновено са част от едно и също продуктово семейство. Всеки конкретен клас-фабрика реализира тези методи, като създава конкретни варианти на продуктите, които са съвместими помежду си. Това позволява на клиентския код да работи с фабриките и продуктите чрез техните абстрактни интерфейси, без да зависи от конкретните реализации. Основното предимство на шаблона е, че улеснява разширяемостта и поддръжката на кода, като позволява лесно добавяне на нови продуктови семейства или варианти, без да се променя клиентският код. Абстрактната фабрика е особено полезна, когато системата трябва да работи с множество взаимозаменяеми продуктови семейства и когато е важно да се гарантира съвместимостта между продуктите, създавани от различните фабрики.

/**
* @file AbstractFactory.hpp
* Demonstrates the Abstract Factory design pattern for embedded and UI codebases, defining abstract product interfaces,
* concrete product implementations, an abstract factory interface, and concrete factories. This pattern enables the
* creation of related product families without specifying their concrete classes, supporting flexibility and
* scalability in embedded and UI application architectures.
*/
#pragma once
#include <iostream>
#include <memory>
/**
* Abstract interface for ProductA family.
* Defines the interface for ProductA objects, enabling polymorphic usage of different concrete implementations in the
* Abstract Factory pattern for embedded and UI systems.
*/
class AbstractProductA {
public:
/**
* Destructor for AbstractProductA.
*/
virtual ~AbstractProductA() = default;
/**
* Performs the operation specific to ProductA.
*/
virtual void operationA() const = 0;
};
/**
* Concrete implementation of ProductA, variant 1.
* Provides a specific implementation of the ProductA interface, representing one variant in the product family created
* by a concrete factory for embedded or UI applications.
*/
class ConcreteProductA1 : public AbstractProductA {
public:
/**
* Performs the operation specific to ConcreteProductA1.
*/
void operationA() const override { std::cout << "ConcreteProductA1 operationA\n"; }
};
/**
* Concrete implementation of ProductA, variant 2.
* Provides an alternative implementation of the ProductA interface, representing another variant in the product family
* created by a different concrete factory for embedded or UI applications.
*/
class ConcreteProductA2 : public AbstractProductA {
public:
/**
* Performs the operation specific to ConcreteProductA2.
*/
void operationA() const override { std::cout << "ConcreteProductA2 operationA\n"; }
};
/**
* Abstract interface for ProductB family.
* Defines the interface for ProductB objects, enabling polymorphic usage of different concrete implementations in the
* Abstract Factory pattern for embedded and UI systems.
*/
class AbstractProductB {
public:
/**
* Destructor for AbstractProductB.
*/
virtual ~AbstractProductB() = default;
/**
* Performs the operation specific to ProductB.
*/
virtual void operationB() const = 0;
};
/**
* Concrete implementation of ProductB, variant 1.
* Provides a specific implementation of the ProductB interface, representing one variant in the product family created
* by a concrete factory for embedded or UI applications.
*/
class ConcreteProductB1 : public AbstractProductB {
public:
/**
* Performs the operation specific to ConcreteProductB1.
*/
void operationB() const override { std::cout << "ConcreteProductB1 operationB\n"; }
};
/**
* Concrete implementation of ProductB, variant 2.
* Provides an alternative implementation of the ProductB interface, representing another variant in the product family
* created by a different concrete factory for embedded or UI applications.
*/
class ConcreteProductB2 : public AbstractProductB {
public:
/**
* Performs the operation specific to ConcreteProductB2.
*/
void operationB() const override { std::cout << "ConcreteProductB2 operationB\n"; }
};
/**
* Abstract factory interface for creating product families.
* Defines the interface for factories that create families of related products (ProductA and ProductB), enabling
* decoupling of product creation from client code in embedded and UI systems.
*/
class AbstractFactory {
public:
/**
* Destructor for AbstractFactory.
*/
virtual ~AbstractFactory() = default;
/**
* Creates a new instance of ProductA.
* @return Unique pointer to the created ProductA instance.
*/
virtual std::unique_ptr<AbstractProductA> createProductA() const = 0;
/**
* Creates a new instance of ProductB.
* @return Unique pointer to the created ProductB instance.
*/
virtual std::unique_ptr<AbstractProductB> createProductB() const = 0;
};
/**
* Concrete factory for creating ProductA1 and ProductB1.
* Implements the AbstractFactory interface, producing a specific family of products (ConcreteProductA1 and
* ConcreteProductB1) for use in embedded or UI applications.
*/
class ConcreteFactory1 : public AbstractFactory {
public:
/**
* Creates a new instance of ConcreteProductA1.
* @return Unique pointer to the created ConcreteProductA1 instance.
*/
std::unique_ptr<AbstractProductA> createProductA() const override { return std::make_unique<ConcreteProductA1>(); }
/**
* Creates a new instance of ConcreteProductB1.
* @return Unique pointer to the created ConcreteProductB1 instance.
*/
std::unique_ptr<AbstractProductB> createProductB() const override { return std::make_unique<ConcreteProductB1>(); }
};
/**
* Concrete factory for creating ProductA2 and ProductB2.
* Implements the AbstractFactory interface, producing an alternative family of products (ConcreteProductA2 and
* ConcreteProductB2) for use in embedded or UI applications.
*/
class ConcreteFactory2 : public AbstractFactory {
public:
/**
* Creates a new instance of ConcreteProductA2.
* @return Unique pointer to the created ConcreteProductA2 instance.
*/
std::unique_ptr<AbstractProductA> createProductA() const override { return std::make_unique<ConcreteProductA2>(); }
/**
* Creates a new instance of ConcreteProductB2.
* @return Unique pointer to the created ConcreteProductB2 instance.
*/
std::unique_ptr<AbstractProductB> createProductB() const override { return std::make_unique<ConcreteProductB2>(); }
};
/**
* Demonstrates usage of the AbstractFactory by creating and operating on products.
* @param factory Reference to the AbstractFactory used for product creation.
* @see AbstractFactory
*/
inline void clientCode(const AbstractFactory & factory) {
auto productA = factory.createProductA();
auto productB = factory.createProductB();
productA->operationA();
productB->operationB();
}
/**
* Entry point for demonstrating the Abstract Factory pattern.
* @return Exit status code.
*/
int main() {
std::cout << "Client: Testing client code with the first factory type:\n";
ConcreteFactory1 factory1;
clientCode(factory1);
std::cout << "Client: Testing the same client code with the second factory type:\n";
ConcreteFactory2 factory2;
clientCode(factory2);
return 0;
}
