Abstract Factory Design Pattern

Въведение

Абстрактната фабрика е създаващ дизайн патърн, който позволява създаването на семейства свързани обекти, без да се специфицират техните конкретни класове. Този патърн предоставя интерфейс за създаване на обекти в суперклас, като позволява на подкласовете да променят типа на създаваните обекти.

Пример

Ето един пример за имплементация на Абстрактната фабрика в C++:

#include <iostream>
#include <memory>
// Продукт A
class AbstractProductA {
public:
virtual ~AbstractProductA() = default;
virtual void operationA() const = 0;
};
class ConcreteProductA1 : public AbstractProductA {
public:
void operationA() const override {
std::cout << "ConcreteProductA1 operationA\n";
}
};
class ConcreteProductA2 : public AbstractProductA {
public:
void operationA() const override {
std::cout << "ConcreteProductA2 operationA\n";
}
};
// Продукт B
class AbstractProductB {
public:
virtual ~AbstractProductB() = default;
virtual void operationB() const = 0;
};
class ConcreteProductB1 : public AbstractProductB {
public:
void operationB() const override {
std::cout << "ConcreteProductB1 operationB\n";
}
};
class ConcreteProductB2 : public AbstractProductB {
public:
void operationB() const override {
std::cout << "ConcreteProductB2 operationB\n";
}
};
// Абстрактна фабрика
class AbstractFactory {
public:
virtual ~AbstractFactory() = default;
virtual std::unique_ptr<AbstractProductA> createProductA() const = 0;
virtual std::unique_ptr<AbstractProductB> createProductB() const = 0;
};
class ConcreteFactory1 : public AbstractFactory {
public:
std::unique_ptr<AbstractProductA> createProductA() const override {
return std::make_unique<ConcreteProductA1>();
}
std::unique_ptr<AbstractProductB> createProductB() const override {
return std::make_unique<ConcreteProductB1>();
}
};
class ConcreteFactory2 : public AbstractFactory {
public:
std::unique_ptr<AbstractProductA> createProductA() const override {
return std::make_unique<ConcreteProductA2>();
}
std::unique_ptr<AbstractProductB> createProductB() const override {
return std::make_unique<ConcreteProductB2>();
}
};
void clientCode(const AbstractFactory& factory) {
auto productA = factory.createProductA();
auto productB = factory.createProductB();
productA->operationA();
productB->operationB();
}
int main() {
std::cout << "Client: Testing client code with the first factory type:\n";
ConcreteFactory1 factory1;
clientCode(factory1);
std::cout << "Client: Testing the same client code with the second factory type:\n";
ConcreteFactory2 factory2;
clientCode(factory2);
return 0;
}

Изходът от програмата е:

Client: Testing client code with the first factory type:
ConcreteProductA1 operationA
ConcreteProductB1 operationB
Client: Testing the same client code with the second factory type:
ConcreteProductA2 operationA
ConcreteProductB2 operationB
