Observer Design Pattern

Въведение

Патернът Observer дефинира зависимост едно към много между обекти, така че когато един обект промени състоянието си, всички зависими обекти биват уведомени и автоматично обновени.

#include <iostream>
#include <vector>
#include <string>
class Observer {
public:
virtual void update(const std::string &message) = 0;
};
class Subject {
std::vector<Observer *> observers;
std::string message;
public:
void attach(Observer *observer) {
observers.push_back(observer);
}
void detach(Observer *observer) {
observers.erase(std::remove(observers.begin(), observers.end(), observer), observers.end());
}
void notify() {
for (Observer *observer : observers) {
observer->update(message);
}
}
void createMessage(const std::string &newMessage) {
message = newMessage;
notify();
}
};
class ConcreteObserver : public Observer {
std::string name;
public:
ConcreteObserver(const std::string &name) : name(name) {}
void update(const std::string &message) override {
std::cout << name << " получи съобщение: " << message << std::endl;
}
};
int main() {
Subject subject;
ConcreteObserver observer1("Наблюдател 1");
ConcreteObserver observer2("Наблюдател 2");
subject.attach(&observer1);
subject.attach(&observer2);
subject.createMessage("Здравейте на всички!");
subject.detach(&observer1);
subject.createMessage("Здравей, Наблюдател 2!");
return 0;
}

В този пример, класът Subject представлява наблюдавания обект, който съдържа списък от наблюдатели. Когато състоянието на Subject се промени, всички наблюдатели биват уведомени и обновени. Класът ConcreteObserver представлява конкретен наблюдател, който реагира на промените в състоянието на Subject.
