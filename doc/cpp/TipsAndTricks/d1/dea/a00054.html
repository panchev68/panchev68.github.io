class Target {
public:
virtual void request() const {
std::cout << "Target: The default target's behavior." << std::endl;
}
};
class Adaptee {
public:
void specificRequest() const {
std::cout << "Adaptee: Specific request." << std::endl;
}
};
class Adapter : public Target {
private:
Adaptee *adaptee_;
public:
Adapter(Adaptee *adaptee) : adaptee_(adaptee) {}
void request() const override {
adaptee_->specificRequest();
}
};
int main() {
Adaptee *adaptee = new Adaptee;
Target *target = new Adapter(adaptee);
target->request();
delete adaptee;
delete target;
return 0;
}

В този пример, класът Adapter преобразува интерфейса на Adaptee в интерфейс, който Target очаква. Това позволява на Adaptee и Target да работят заедно, въпреки че имат различни интерфейси.
