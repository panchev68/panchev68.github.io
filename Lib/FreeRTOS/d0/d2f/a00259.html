Queue

This example demonstrates how to use the Queue class to create, send, and receive messages using FreeRTOS.

/**
* @file Queue.cpp
*
* @brief Queue class definition
*
* @version 1.0
* @date 11.01.2025
*/
#include <iostream>
#include <RTOS.hpp>
/**
* @brief Main function demonstrating the usage of a RTOS queue.
*
* This function creates a queue for integers with a length of 5, sends a message to the queue,
* receives a message from the queue, and prints the received message to the standard output.
* If any operation fails, an error message is printed to the standard error output and the
* program returns -1.
*
* @return int Returns 0 on success, -1 on failure.
*/
int main () {
// Create a queue for integers with a length of 5
RTOS::Queue<int, 5> queue;
if (!queue.create ()) {
std::cerr << "Failed to create queue" << std::endl;
return -1;
}
// Send a message to the queue
int message = 42;
if (!queue.send (message, 100)) {
std::cerr << "Failed to send message to queue" << std::endl;
return -1;
}
// Receive a message from the queue
int receivedMessage;
if (!queue.receive (receivedMessage, 100)) {
std::cerr << "Failed to receive message from queue" << std::endl;
return -1;
}
std::cout << "Received message: " << receivedMessage << std::endl;
// Destroy the queue
queue.destroy ();
return 0;
}
