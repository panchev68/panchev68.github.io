* @brief Main function demonstrating the usage of a binary semaphore in FreeRTOS.

*
*/

// Create a semaphore object
Lib::FreeRTOS::Semaphore semaphore;
// Create the semaphore
if (semaphore.create ()) {
// Successfully created the semaphore
// Take the semaphore
if (semaphore.take (100)) {
// Successfully took the semaphore
// Perform some critical section code here
// Give the semaphore back
semaphore.give ();
} else {
// Failed to take the semaphore within the specified time
}
} else {
// Failed to create the semaphore
}
}
