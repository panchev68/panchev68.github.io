TickCounter

This example demonstrates how to use the TickCounter class to get the current tick count from the FreeRTOS scheduler.

/**
* @file TickCounter.cpp
* @brief Example to demonstrate the usage of the TickCounter class.
*
* This file contains an example task function that demonstrates the usage of the TickCounter class.
* The task gets the current tick count from the FreeRTOS scheduler and prints it.
* @date 11.01.2025
*
*/
#include <cstdio>
#include <RTOS.hpp>
using RTOS::Task;
using RTOS::TickCounter;
/**
* @brief Example task function to demonstrate the usage of the TickCounter class.
*
* This function gets the current tick count from the FreeRTOS scheduler and prints it.
*/
void example () {
// Get the current tick count
uint32_t tickCount = RTOS::TickCounter::get ();
printf ("Current tick count: %u\n", tickCount);
// Get the current tick count from ISR context
uint32_t tickCountFromISR = RTOS::TickCounter::getFromISR ();
printf ("Current tick count from ISR: %u\n", tickCountFromISR);
// Use the conversion operator to get the tick count
uint32_t tickCountConverted = TickCounter::get ();
printf ("Tick count using conversion operator: %u\n", tickCountConverted);
// Delay the task for 1000 milliseconds
vTaskDelay (pdMS_TO_TICKS (1000));
}
