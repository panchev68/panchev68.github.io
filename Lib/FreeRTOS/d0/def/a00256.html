Tick

This example demonstrates how to use the Tick class to manage tick events in the FreeRTOS environment. It shows how to create a client that handles tick events and register it with the Tick class. The example also demonstrates how to start the tick event handling and receive tick events.

#include <FreeRTOS.hpp>
/**
* @brief Main function to demonstrate the Tick class.
*
* This function demonstrates how to use the Tick class to manage tick events in the FreeRTOS environment.
* It creates an instance of the Tick class, registers a client to handle tick events, and starts the tick event handling.
*
* @return int The exit status of the application.
*/
int main () {
auto & tick = Lib::FreeRTOS::Tick::getInstance ();
tick.start ();
MyTickClient client;
// Register the client to receive tick events
if (tick.registerClient (&client)) {
std::cout << "Client registered successfully." << std::endl;
} else {
std::cout << "Failed to register client." << std::endl;
}
// Unregister the client to stop receiving tick events
if (tick.unregisterClient (&client)) {
std::cout << "Client unregistered successfully." << std::endl;
} else {
std::cout << "Failed to unregister client." << std::endl;
}
return 0;
}
