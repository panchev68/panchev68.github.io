class TaskExample1
{
public:
void start ()
{
auto pool = [this] {
for (;;)
{
// task function code
vTaskDelay (1);
}
};
FreeRTOS::Task::run ("Task Name", pool, 256);
// or FreeRTOS::Task::run ({"Task Name", pool, 256});
}
}
