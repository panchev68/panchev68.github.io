TaskHelper

This example demonstrates how to use the TaskHelper class to interact with FreeRTOS tasks.

/**
* @file TaskHelper.cpp
* @brief Example demonstrating the usage of the TaskHelper class in FreeRTOS.
*
* This file contains an example of how to use the TaskHelper class to manage
* FreeRTOS tasks. The example includes creating a task, printing the task name,
* setting task priority, getting the current tick count, delaying the task, and
* suspending the task.
*
* @author Bobby
* @date 11.01.2025
*/
#include "TaskHelper.hpp"
#include "Task.hpp"
#include <cstdio>
#include <string>
using Lib::FreeRTOS::Task;
using Lib::FreeRTOS::TaskHelper;
void taskFn () {
// Get the handle of the current task
TaskHelper::Handle currentTaskHandle = TaskHelper::getCurrentTaskHandler ();
// Print the name of the current task
std::string taskName = TaskHelper::getName (currentTaskHandle);
printf ("Task name: %s\n", taskName.c_str ());
// Set the priority of the current task
TaskHelper::setPriority (currentTaskHandle, 2);
// Get the current tick count
uint32_t tickCount = TaskHelper::getTickCount ();
printf ("Current tick count: %u\n", tickCount);
// Delay the task for 1000 milliseconds
TaskHelper::delay (1000);
// Suspend the current task
TaskHelper::suspend (currentTaskHandle);
}
/**
* @brief Main function to demonstrate the usage of the TaskHelper class.
*
* This function creates a new task using the Task class and runs it. The task
* prints the name of the current task, sets its priority, gets the current tick
* count, delays for 1000 milliseconds, and suspends itself.
*/
int main () {
// Create a task using FreeRTOS API
Task::run ("ExampleTask", taskFn);
// Start the scheduler
vTaskStartScheduler ();
// The program should never reach here
for (;;);
}
