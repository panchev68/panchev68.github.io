This is an example of how to use the SemaphoreCounting class.

/**
* @file SemaphoreCounting.cpp
* @brief Example demonstrating the usage of a counting semaphore.
*
* This file contains a main function that demonstrates the following operations:
* - Creates a counting semaphore with a maximum count of 10 and an initial count of 5.
* - Retrieves and prints the current count of the semaphore.
* - Attempts to take the semaphore with a timeout of 100 ticks and prints the result.
* - Gives the semaphore and prints the result.
* - Releases the semaphore.
*
* If any of the operations fail, an appropriate message is printed.
*
* @date 11.01.2025
*/
#include "SemaphoreCounting.hpp"
/**
* @brief Main function demonstrating the usage of a counting semaphore.
*
* This function creates a counting semaphore with a maximum count of 10 and an initial count of 5.
* It then performs the following operations:
* - Creates the semaphore.
* - Retrieves and prints the current count of the semaphore.
* - Attempts to take the semaphore with a timeout of 100 ticks and prints the result.
* - Gives the semaphore and prints the result.
* - Releases the semaphore.
*
* If any of the operations fail, an appropriate message is printed.
*
* @return int Exit status of the program.
*/
int main () {
// Create a counting semaphore with a maximum count of 10 and an initial count of 5
Lib::FreeRTOS::SemaphoreCounting semaphore (10, 5);
// Create the semaphore
if (semaphore.create ()) {
// Successfully created the semaphore
// Get the current count of the semaphore
uint32_t count = semaphore.getCount ();
printf ("Semaphore count: %u\n", count);
// Take the semaphore
if (semaphore.take (100)) {
// Successfully took the semaphore
printf ("Semaphore taken\n");
} else {
// Failed to take the semaphore
printf ("Failed to take semaphore\n");
}
// Give the semaphore
if (semaphore.give ()) {
// Successfully gave the semaphore
printf ("Semaphore given\n");
} else {
// Failed to give the semaphore
printf ("Failed to give semaphore\n");
}
// Release the semaphore
semaphore.release ();
} else {
// Failed to create the semaphore
printf ("Failed to create semaphore\n");
}
}
