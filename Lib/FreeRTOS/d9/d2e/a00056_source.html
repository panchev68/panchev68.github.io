29 * This class provides static functions for allocating and freeing memory
30 * using the FreeRTOS memory management functions. It also includes functions
31 * for retrieving the amount of free heap memory available.

33 class MemoryManager
34 {
35 public:

37#if (MEMODY_MANAGER_DIAG > 0)
38 /**
39 * @struct Diagnostic
40 * @brief A structure for tracking memory allocation and deallocation.
41 *
42 * This structure is used to track the number of memory allocations and
43 * deallocations that have occurred. It also tracks the number of active
44 * memory blocks that have been allocated.

46 struct Diagnostic

48 uint32_t allocated = 0; ///< The total number of memory blocks allocated.
49 uint32_t deallocated = 0; ///< The total number of memory blocks deallocated.
50 uint32_t active = 0; ///< The number of active memory blocks.
51 };

61 /**
62 * @brief Allocates memory using the FreeRTOS memory management functions.
63 *
64 * This function allocates memory using the FreeRTOS function pvPortMalloc.
65 * If the allocation fails, the function asserts and throws a std::bad_alloc
66 * exception if exceptions are enabled.
67 *
68 * @param size The size of the memory block to allocate.
69 * @return A pointer to the allocated memory block.
70 */
71 static void * malloc (size_t size)
72 {
73 assert (size);
74 void * result = pvPortMalloc (size);
75
76 if (result == nullptr)
77 {
78 assert (0);
79
80#if defined(__cpp_exceptions) && (__cpp_exceptions == 1)
81 if (result == nullptr) throw std::bad_alloc ();
82#endif
83 }
84#if (MEMODY_MANAGER_DIAG > 0)
85 diagnostic.allocated++;
86 diagnostic.active++;
87#endif
88 return result;
89 }

91 /**
92 * @brief Frees memory using the FreeRTOS memory management functions.
93 *
94 * This function frees memory using the FreeRTOS function vPortFree.
95 * It asserts that the memory block is not null before freeing it.
96 *
97 * @param block A pointer to the memory block to free.
98 */
99 static void free (void * block)
100 {
101 assert (block);
102 vPortFree (block);
103#if (MEMODY_MANAGER_DIAG > 0)
104 diagnostic.deallocated++;
105 diagnostic.active--;
106#endif
107 }

109 /**

115 *
116 * @return The size of the free heap memory in bytes.

120 return xPortGetFreeHeapSize ();
121 }
