87 /// \brief Set bits (flags) within an RTOS event group.
88 ///
89 /// \details
90 /// Set bits (flags) within an RTOS event group. This function cannot be called from an interrupt.
91 /// `xEventGroupSetBitsFromISR()` is a version that can be called from an interrupt.
92 /// Setting bits in an event group will automatically unblock tasks that are blocked waiting for the bits.The
93 /// RTOS source file `FreeRTOS/source/event_groups.c` must be included in the build for the
94 /// xEventGroupSetBits() function to be available.
95 ///
96 /// \return
97 /// The value of the event group at the time the call to `xEventGroupSetBits()` returns.
98 /// There are two reasons why the returned value might have the bits specifiedby the `uxBitsToSet`
99 /// parameter cleared: If setting a bit results in a task that was waiting for the bit leaving the blocked state
100 /// then it is possible the bit will have been cleared automatically (see the `xClearBitOnExit` parameter of
101 /// `xEventGroupWaitBits()`). Any unblocked (or otherwise Ready state) task that has a priority above that of the
102 /// task that called `xEventGroupSetBits ()` will execute and may change the event group value before the call to
103 /// `xEventGroupSetBits ()` returns.
104 ///
105 /// \sa [xEventGroupSetBits()](https://www.freertos.org/xEventGroupSetBits.html)
106 EventBits setBits (const EventBits & bitsToSet)
107 {
108 assert (isValid () && "The EventGroup has not been created");
109 return EventBits (xEventGroupSetBits (_handle, bitsToSet.to_ulong ()));
110 }
111
112 /// \brief Set bits (flags) within an RTOS event group form an interrupt service routine (ISR).
113 ///
114 /// \details
115 /// Function that calls ```BaseType_t xEventGroupSetBitsFromISR( EventGroupHandle_t xEventGroup, const
116 /// EventBits_t uxBitsToSet, BaseType_t *pxHigherPriorityTaskWoken )```
117 /// \see [xEventGroupSetBitsFromISR()](https://www.freertos.org/xEventGroupSetBitsFromISR.html)
118 /// \bug Calling xTimerPendFunctionCallFromISR()
119 bool setBitsFromISR (const EventBits & bitsToSet)
120 {
121 assert (isValid () && "The EventGroup has not been created");
122 BaseType_t woken (pdFALSE);
123 if (xEventGroupSetBitsFromISR (_handle, bitsToSet.to_ulong (), &woken) == pdPASS)
124 {
125 portYIELD_FROM_ISR (woken);
126 return true;
127 }
128 else
129 {
130 assert (0);
131 return false;
132 }
133 }

194 /// \brief Wait for a bits
195 ///
196 /// \param bitsToWait
197 /// A bitwise value that indicates the bit or bits to test inside the event group
198 /// \param clearOnExit

224 /// \brief Sync bits
225 ///
226 /// \details
227 /// Atomically set bits (flags) within an RTOS event group, then wait for a combination of bits to be set within
228 /// the same event group. This functionality is typically used to synchronise multiple tasks (often called a
229 /// task rendezvous), where each task has to wait for the other tasks to reach a synchronisation point before
230 /// proceeding.
231 /// \param bitsToSet
232 /// The bit or bits to set in the event group before determining if (and possibly waiting
233 /// for), all the bits specified by the uxBitsToWait parameter are set. For example, set uxBitsToSet to 0x04 to
234 /// set bit 2 within the event group.
235 /// \param bitsToWaitFor
236 /// A bitwise value that indicates the bit or bits to test inside the event group. For
237 /// example, set uxBitsToWaitFor to 0x05 to wait for bits 0 and bit 2. Set uxBitsToWaitFor to 0x07 to wait for
238 /// bit 0 and bit 1 and bit 2. Etc.
239 /// \param ticksToWait
240 /// The maximum amount of time (specified in 'ticks') to wait for all the bits specifiedby the uxBitsToWaitFor parameter value to become set.
241 ///
242 /// \return The value of the event group at the time either the bits being waited for became set, or the block
243 /// time expired. Test the return value to know which bits were set.
244 /// If xEventGroupSync() returned because its timeout expired then not all the bits being waited for will be set.
245 /// If xEventGroupSync() returned because all the bits it was waiting for were set then the returned value is
246 /// the event group value before any bits were automatically cleared.
247 /// \see [xEventGroupSync()](https://www.freertos.org/xEventGroupSync.html)
248 ///
249 /// <b>Example Usage</b>
250 /// \include EventGroups/sync.cpp
251 EventBits syncBits (const EventBits & bitsToSet, const EventBits & bitsToWaitFor, TickType ticksToWait = portMAX_DELAY) const
252 {
253 assert (isValid () && "The EventGroup has not been created");
254 return EventBits (xEventGroupSync (_handle,
255 bitsToSet.to_ulong (),
256 bitsToWaitFor.to_ulong (),
257 ticksToWait));
258 }
