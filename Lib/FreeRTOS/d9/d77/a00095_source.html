43 /// \brief Timer constructor - creates the timer and starts it
44 /// \param name A human readable text name that is assigned to the timer. This is done purely to assist
45 /// debugging. The RTOS kernel itself only ever references a timer by its handle, and never by its name.
46 /// \param func reference to `std::function<void ()>`
47 /// \param period Period -The period of the timer. The period is specified in ticks, and the macro
48 /// pdMS_TO_TICKS() can be used to convert a time specified in milliseconds to a time specified in ticks. For
49 /// example, if the timer must expire after 100 ticks, then simply set xTimerPeriod to 100. Alternatively, if
50 /// the timer must expire after 500ms, then set xTimerPeriod to pdMS_TO_TICKS( 500 ). pdMS_TO_TICKS() can only
51 /// be used if configTICK_RATE_HZ is less than or equal to 1000. The timer period must be greater than 0.
52 /// \param autoReload If uxAutoReload is set to pdTRUE, then the timer will expire repeatedly with a frequency
53 /// set by the xTimerPeriod parameter. If uxAutoReload is set to pdFALSE, then the timer will be a one-shot and
54 /// enter the dormant state after it expires.
55 /// \param autostart if autostart is \b true after the timer is created, it will be started

62 }

93 static Timer * run (const std::string & name, const std::function<void ()> & func, uint32_t period = 1, bool autoReload = true)
94 {
95 Timer * instance = new Timer ();
96 assert (instance);
97 if (instance->create (name, func, period, autoReload) == false) assert (0);
98 if (instance->start () == false) assert (0);
99 return instance;
100 }

106 /// \brief Creating a timer
107 /// \param name A human readable text name that is assigned to the timer. This is done purely to assist
108 /// debugging. The RTOS kernel itself only ever references a timer by its handle, and never by its name.
109 /// \param func reference to `std::function<void ()>`
110 /// \param period Period -The period of the timer. The period is specified in ticks, and the macro
111 /// pdMS_TO_TICKS() can be used to convert a time specified in milliseconds to a time specified in ticks. For
112 /// example, if the timer must expire after 100 ticks, then simply set xTimerPeriod to 100. Alternatively, if
113 /// the timer must expire after 500ms, then set xTimerPeriod to pdMS_TO_TICKS( 500 ). pdMS_TO_TICKS() can only
114 /// be used if configTICK_RATE_HZ is less than or equal to 1000. The timer period must be greater than 0.
115 /// \param autoReload If uxAutoReload is set to pdTRUE, then the timer will expire repeatedly with a frequency
116 /// set by the xTimerPeriod parameter. If uxAutoReload is set to pdFALSE, then the timer will be a one-shot and
117 /// enter the dormant state after it expires.
118 /// \return bool - returns \b true on successful timer creation
119 virtual bool create (const std::string & name, const std::function<void ()> & func, uint32_t period = 1, bool autoReload = true)
120 {
121 auto fn = [] (Handle handle) {
122 assert (handle);
123 auto sender = reinterpret_cast<Timer *> (pvTimerGetTimerID (handle));
124 assert (sender);
125 if (sender->_function != nullptr) sender->_function ();
126 };
127
128 assert (func);
129 assert (_handle == nullptr); // has already been created
130
131 _function = func;
132 _handle = xTimerCreate (name.c_str (), period, autoReload ? pdTRUE : pdFALSE, reinterpret_cast<void *> (this), fn);
133 assert (_handle != nullptr);
134 return (_handle != nullptr);
135 }

169 /// \brief Starts the timer
170 /// \see https://www.freertos.org/FreeRTOS-timers-xTimerStart.html
171 /// \param blockTime Specifies the time, in ticks, that the calling task should be held in the Blocked state to
172 /// wait for the start command to be successfully sent to the timer command queue, should the queue already be
173 /// full when xTimerStart() was called. xBlockTime is ignored if xTimerStart() is called before the RTOS
174 /// scheduler is started.
175 /// \return Returns \b true will be returned if the command was successfully sent to the timer command queue.
176 /// \return Returns \b false will be returned if the start command could not be sent to the timer command queue even
177 /// after xBlockTime ticks had passed.
178 virtual bool start (uint32_t blockTime = DEFAULT_BLOCK_TIME)
179 {
180 assert (_handle && "Timer not created");
181 return (xTimerStart (_handle, blockTime) == pdPASS);
182 }
