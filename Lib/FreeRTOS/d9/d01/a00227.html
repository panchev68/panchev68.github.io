Karnel

This example demonstrates how to use the Karnel class to interact with the FreeRTOS kernel.

/**
* @file Karnel.cpp
* @brief Demonstrates various FreeRTOS kernel operations.
*
* This file contains a main function that demonstrates the following FreeRTOS kernel operations:
* - Retrieving and printing the current tick count.
* - Retrieving and printing the number of tasks in the scheduler.
* - If the macro INCLUDE_xTaskGetSchedulerState is set to 1, retrieving and printing the current state of the scheduler.
* - Entering and exiting a critical section.
* - Disabling and enabling interrupts.
*
* @date 11.01.2025
*/
#include "Karnel.hpp"
#include <FreeRTOS.hpp>
#include <iostream>
/**
* @brief Main function demonstrating various FreeRTOS kernel operations.
*
* This function performs the following operations:
* - Retrieves and prints the current tick count.
* - Retrieves and prints the number of tasks in the scheduler.
* - If the macro INCLUDE_xTaskGetSchedulerState is set to 1, retrieves and prints the current state of the scheduler.
* - Enters and exits a critical section.
* - Disables and enables interrupts.
*
* @return int Returns 0 upon successful completion.
*/
int main () {
// Get the current tick count
TickType_t tickCount = Lib::FreeRTOS::Karnel::getTickCount ();
std::cout << "Current tick count: " << tickCount << std::endl;
// Get the number of tasks in the scheduler
UBaseType_t numberOfTasks = Lib::FreeRTOS::Karnel::getNumberOfTasks ();
std::cout << "Number of tasks: " << numberOfTasks << std::endl;
#if (INCLUDE_xTaskGetSchedulerState == 1)
// Get the current state of the scheduler
Lib::FreeRTOS::Karnel::SchedulerState schedulerState = Lib::FreeRTOS::Karnel::getSchedulerState ();
std::cout << "Scheduler state: " << static_cast<BaseType_t> (schedulerState) << std::endl;
#endif
// Enter and exit a critical section
Lib::FreeRTOS::Karnel::enterCritical ();
// Critical section code here
Lib::FreeRTOS::Karnel::exitCritical ();
// Enable and disable interrupts
Lib::FreeRTOS::Karnel::disableInterrupts ();
// Code that should not be interrupted
Lib::FreeRTOS::Karnel::enableInterrupts ();
return 0;
}
