#include <FreeRTOS.hpp>
/**
* @brief Example function demonstrating the usage of a FreeRTOS co-routine.
*
* This function creates a co-routine using a lambda function. The lambda function
* contains an infinite loop with a delay of 100 ticks.
*
* The lambda function is defined as follows:
* - @param hnd: The handle to the co-routine.
* - @param index: The index of the co-routine.
*
* The co-routine is created with the following parameters:
* - func: The lambda function defined above.
* - 0: The priority of the co-routine.
* - 0: The index of the co-routine.
*/
void example () {
auto func = [] (CoRoutineHandle_t hnd, UBaseType_t index) {
crSTART (hnd);
for (;;) {
crDELAY (hnd, 100);
}
crEND ();
};
Lib::FreeRTOS::CoRoutine coRoutine (func, 0, 0);
}
