MemoryManager

Example usage of the MemoryManager class to allocate and free memory.

Example usage of the MemoryManager class to allocate and free memory.This example demonstrates how to use the MemoryManager class to allocate and free memory using FreeRTOS memory management functions.

#include "MemoryManager.hpp"
#include <iostream>
int main() {
// Allocate a block of memory
size_t size = 128;
void * memory = Lib::FreeRTOS::MemoryManager::malloc(size);
if (memory == nullptr) {
std::cerr << "Failed to allocate memory" << std::endl;
return -1;
}
// Use the allocated memory
// (For demonstration purposes, we'll just fill it with zeros)
memset(memory, 0, size);
// Print the amount of free heap memory
size_t freeHeapSize = Lib::FreeRTOS::MemoryManager::getFreeHeapSize();
std::cout << "Free heap size: " << freeHeapSize << " bytes" << std::endl;
// Free the allocated memory
Lib::FreeRTOS::MemoryManager::free(memory);
// Print the amount of free heap memory after freeing
freeHeapSize = Lib::FreeRTOS::MemoryManager::getFreeHeapSize();
std::cout << "Free heap size after freeing: " << freeHeapSize << " bytes" << std::endl;
return 0;
}
