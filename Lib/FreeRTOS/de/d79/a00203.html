TaskHelperExample

Example usage of the TaskHelper class.

Example usage of the TaskHelper class.This example demonstrates how to use the TaskHelper class to interact with FreeRTOS tasks.

#include "TaskHelper.hpp"
#include <cstdio>
#include <string>
void exampleTaskFunction(void *pvParameters) {
// Get the handle of the current task
TaskHelper::Handle currentTaskHandle = TaskHelper::getCurrentTaskHandler();
// Print the name of the current task
std::string taskName = TaskHelper::getName(currentTaskHandle);
printf("Task name: %s\n", taskName.c_str());
// Set the priority of the current task
TaskHelper::setPriority(currentTaskHandle, 2);
// Get the current tick count
uint32_t tickCount = TaskHelper::getTickCount();
printf("Current tick count: %u\n", tickCount);
// Delay the task for 1000 milliseconds
TaskHelper::delay(1000);
// Suspend the current task
TaskHelper::suspend(currentTaskHandle);
}
int main() {
// Create a task using FreeRTOS API
xTaskCreate(exampleTaskFunction, "ExampleTask", configMINIMAL_STACK_SIZE, nullptr, 1, nullptr);
// Start the scheduler
vTaskStartScheduler();
// The program should never reach here
for (;;);
}
