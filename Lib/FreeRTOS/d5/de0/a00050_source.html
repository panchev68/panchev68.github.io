89 /// \brief Set bits (flags) within an RTOS event group.
90 /// \details
91 /// Set bits (flags) within an RTOS event group. This function cannot be called from an interrupt.
92 /// `xEventGroupSetBitsFromISR()` is a version that can be called from an interrupt.
93 /// Setting bits in an event group will automatically unblock tasks that are blocked waiting for the bits.The
94 /// RTOS source file `FreeRTOS/source/event_groups.c` must be included in the build for the
95 /// xEventGroupSetBits() function to be available.
96 /// \return
97 /// The value of the event group at the time the call to `xEventGroupSetBits()` returns.
98 /// There are two reasons why the returned value might have the bits specifiedby the `uxBitsToSet`
99 /// parameter cleared: If setting a bit results in a task that was waiting for the bit leaving the blocked state
100 /// then it is possible the bit will have been cleared automatically (see the `xClearBitOnExit` parameter of
101 /// `xEventGroupWaitBits()`). Any unblocked (or otherwise Ready state) task that has a priority above that of the
102 /// task that called `xEventGroupSetBits ()` will execute and may change the event group value before the call to
103 /// `xEventGroupSetBits ()` returns.
104 /// \sa [xEventGroupSetBits()](https://www.freertos.org/xEventGroupSetBits.html)
105 EventBits setBits (const EventBits & bitsToSet)
106 {
107 assert (isValid () && "The EventGroup has not been created");
108 return EventBits (xEventGroupSetBits (_handle, bitsToSet.to_ulong ()));
109 }
110
111 /// \brief Set bits (flags) within an RTOS event group form an interrupt service routine (ISR).
112 /// \details
113 /// Function that calls ```BaseType_t xEventGroupSetBitsFromISR( EventGroupHandle_t xEventGroup, const
114 /// EventBits_t uxBitsToSet, BaseType_t *pxHigherPriorityTaskWoken )```
115 /// \see [xEventGroupSetBitsFromISR()](https://www.freertos.org/xEventGroupSetBitsFromISR.html)
116 /// \bug Calling xTimerPendFunctionCallFromISR()
117 bool setBitsFromISR (const EventBits & bitsToSet)
118 {
119 assert (isValid () && "The EventGroup has not been created");
120 BaseType_t woken (pdFALSE);
121 if (xEventGroupSetBitsFromISR (_handle, bitsToSet.to_ulong (), &woken) == pdPASS)
122 {
123 portYIELD_FROM_ISR (woken);
124 return true;
125 }
126 else
127 {
128 assert (0);
129 return false;
130 }
131 }

188 /// \brief Wait for a bits
189 /// \param bitsToWait
190 /// A bitwise value that indicates the bit or bits to test inside the event group
191 /// \param clearOnExit
192 /// is set to pdTRUE then any bits set in the value passed as the uxBitsToWaitFor
193 /// parameter will be cleared in the event group before
194 /// \param waitForAllBytes
195 /// is used to create either a logical AND test (where all bits must be set) or a logical OR test (where one or
196 /// more bits must be set) as follows
197 /// \param ticksToWait
198 /// The maximum amount of time (specified in 'ticks') to wait for one/all (depending on the xWaitForAllBits

216 /// \brief Sync bits
217 /// \details
218 /// Atomically set bits (flags) within an RTOS event group, then wait for a combination of bits to be set within
219 /// the same event group. This functionality is typically used to synchronise multiple tasks (often called a
220 /// task rendezvous), where each task has to wait for the other tasks to reach a synchronisation point before
221 /// proceeding.
222 /// \param bitsToSet
223 /// The bit or bits to set in the event group before determining if (and possibly waiting
224 /// for), all the bits specified by the uxBitsToWait parameter are set. For example, set uxBitsToSet to 0x04 to
225 /// set bit 2 within the event group.
226 /// \param bitsToWaitFor
227 /// A bitwise value that indicates the bit or bits to test inside the event group. For
228 /// example, set uxBitsToWaitFor to 0x05 to wait for bits 0 and bit 2. Set uxBitsToWaitFor to 0x07 to wait for
229 /// bit 0 and bit 1 and bit 2. Etc.
230 /// \param ticksToWait
231 /// The maximum amount of time (specified in 'ticks') to wait for all the bits specifiedby the uxBitsToWaitFor parameter value to become set.
232 /// \return The value of the event group at the time either the bits being waited for became set, or the block
233 /// time expired. Test the return value to know which bits were set.
234 /// If xEventGroupSync() returned because its timeout expired then not all the bits being waited for will be set.
235 /// If xEventGroupSync() returned because all the bits it was waiting for were set then the returned value is
236 /// the event group value before any bits were automatically cleared.
237 /// \see [xEventGroupSync()](https://www.freertos.org/xEventGroupSync.html)
238 ///
239 /// <b>Example Usage</b>
240 /// \include EventGroups/sync.cpp
241 EventBits syncBits (const EventBits & bitsToSet, const EventBits & bitsToWaitFor, TickType ticksToWait = portMAX_DELAY) const
242 {
243 assert (isValid () && "The EventGroup has not been created");
244 return EventBits (xEventGroupSync (_handle,
245 bitsToSet.to_ulong (),
246 bitsToWaitFor.to_ulong (),
247 ticksToWait));
248 }
