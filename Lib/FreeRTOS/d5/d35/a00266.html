StreamBuffer

This is an example of how to use the StreamBuffer class.

/**
* @file StreamBuffer.cpp
* @brief Example demonstrating the usage of a stream buffer in FreeRTOS.
*
* This file contains an example of how to create, send, receive, and release
* a stream buffer using the FreeRTOS library.
*
* @author Bobby
* @date 11.01.2025
*/
#include <StreamBuffer.hpp>
#include <cstdio>
#include <cstring>
/**
* @brief Main function demonstrating the usage of a stream buffer in FreeRTOS.
*
* This function performs the following steps:
* 1. Creates a stream buffer with a buffer size of 1024 bytes and a trigger level of 1 byte.
* 2. Sends a message "Hello, StreamBuffer!" to the stream buffer.
* 3. Receives the message from the stream buffer and prints it.
* 4. Releases the stream buffer.
*
* @return int Returns 0 on success, -1 on failure.
*/
int main () {
// Create a StreamBuffer object
Lib::FreeRTOS::StreamBuffer streamBuffer;
// Create a stream buffer with a buffer size of 1024 bytes and a trigger level of 1 byte
if (streamBuffer.create (1024, 1)) {
// Send a message to the stream buffer
const char * message = "Hello, StreamBuffer!";
streamBuffer.send (message, strlen (message), portMAX_DELAY);
// Receive a message from the stream buffer
char buffer [1024];
size_t bytesReceived = streamBuffer.receive (buffer, sizeof (buffer), portMAX_DELAY);
// Null-terminate the received message
buffer [bytesReceived] = '\0';
// Print the received message
printf ("Received message: %s\n", buffer);
// Release the stream buffer
streamBuffer.release ();
} else {
printf ("Failed to create stream buffer\n");
}
}
