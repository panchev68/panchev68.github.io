Task

This example demonstrates how to use the Task class to create and manage FreeRTOS tasks.

/**
* @file Task.cpp
* @brief Example of creating and managing a FreeRTOS task.
*
* This file contains an example of how to create, manage, and destroy a FreeRTOS task using the FreeRTOS C++ wrapper.
* The example demonstrates task creation, suspension, resumption, priority setting, and destruction.
*
* @details
* The task function prints messages indicating its running state and delays for 1000 milliseconds in a loop.
* The main function creates a task with a stack size of 512 and priority 1, performs various operations on the task,
* and finally destroys the task.
*
* @note
* Ensure that the FreeRTOS C++ wrapper library is included and properly configured in your project.
*/
#include "FreeRTOS.hpp"
void taskFunction () {
// Task code here
while (true) {
// Print a message indicating the task is running
printf ("Task is running\n");
// Delay for 1000 milliseconds
vTaskDelay (pdMS_TO_TICKS (1000));
// Print a message indicating the task is complete
printf ("Task is complete\n");
}
}
/**
* @brief Entry point of the program.
*
* This function creates a new FreeRTOS task with a stack size of 512 and priority 1,
* checks if the task was created successfully, and performs various operations on the task
* such as suspending, resuming, setting priority, and getting priority. Finally, it destroys
* the task when done.
*
* @return int Returns 0 upon successful completion.
*/
int main () {
// Create a new FreeRTOS task with a stack size of 512 and priority 1
auto task = Lib::FreeRTOS::Task<512, 1>::run (taskFunction);
// Check if the task was created successfully
if (task != nullptr) {
// Perform various operations on the task
task->suspend ();
task->resume ();
task->setPriority (3);
uint32_t priority = task->getPriority ();
// Print the task priority
task->destroy ();
}
return 0;
}
