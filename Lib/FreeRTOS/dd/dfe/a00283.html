Task

This example demonstrates how to use the Task class to create and manage FreeRTOS tasks.

/**
* @file Task.cpp
* @brief Example demonstrating the usage of the Task class in FreeRTOS.
*
* This file contains an example task function and a main function to demonstrate
* how to create and run a task using the Task class in FreeRTOS. The example task
* function prints messages indicating its running state, delays for 1000 milliseconds,
* and then prints a message indicating its completion.
*
* The main function creates a task configuration, runs the task, and starts the
* FreeRTOS scheduler.
*
* @note This example assumes the presence of the FreeRTOS library and the Task class
* with the appropriate methods and configurations.
*/
#include "Task.hpp"
#include <cstdio>
/**
* @brief Example task function to demonstrate the usage of the Task class.
*
* This function prints a message indicating the task is running, delays for 1000 milliseconds,
* and then prints a message indicating the task is completing.
*/
void exampleTaskFunction () {
// Print a message indicating the task is running
printf ("Example task is running\n");
// Simulate some work by delaying for 1000 milliseconds
Lib::FreeRTOS::TaskHelper::delay (1000);
// Print a message indicating the task is completing
printf ("Example task is completing\n");
}
/**
* @brief Main function to demonstrate the usage of the Task class.
*
* This function creates a new task using the Task class and runs it.
* The task prints a message indicating it is running, delays for 1000 milliseconds,
* and then prints a message indicating it is completing.
*
* @return int Returns 0 upon successful completion.
*/
int main () {
// Create a task configuration
Lib::FreeRTOS::Task::Config config{
"ExampleTask",
exampleTaskFunction,
configMINIMAL_STACK_SIZE,
DEFAULT_TASK_PRIORITY};
// Create and run the task
Lib::FreeRTOS::Task * task = Lib::FreeRTOS::Task::run (config);
if (task == nullptr) {
printf ("Failed to create task\n");
return -1;
}
// Start the scheduler
vTaskStartScheduler ();
// The program should never reach here
for (;;);
}
