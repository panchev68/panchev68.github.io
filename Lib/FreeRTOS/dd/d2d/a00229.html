MemoryManager

This is an example of how to use the MemoryManager class to allocate and free memory using FreeRTOS.

/**
* @file MemoryManager.cpp
*
* @brief MemoryManager class example
*
* @version 1.0
* @date 11.01.2025
*/
#include "MessageBuffer.hpp"
#include <iostream>
/**
* @brief Main function demonstrating the usage of a message buffer in FreeRTOS.
*
* This function performs the following steps:
* 1. Creates a message buffer with a size of 100 bytes.
* 2. Sends a message "Hello, FreeRTOS!" to the message buffer.
* 3. Receives the message from the message buffer and prints it.
* 4. Releases the message buffer.
*
* @return int Returns 0 on success, -1 on failure.
*/
int main () {
// Create a message buffer with a size of 100 bytes
Lib::FreeRTOS::MessageBuffer messageBuffer;
if (!messageBuffer.create (100)) {
std::cerr << "Failed to create message buffer" << std::endl;
return -1;
}
// Send a message to the message buffer
const char * message = "Hello, FreeRTOS!";
size_t messageLength = strlen (message) + 1; // Include null terminator
if (messageBuffer.send (message, messageLength, 100) != messageLength) {
std::cerr << "Failed to send message" << std::endl;
return -1;
}
// Receive a message from the message buffer
char receivedMessage [100];
size_t receivedLength = messageBuffer.receive (receivedMessage, sizeof (receivedMessage), 100);
if (receivedLength > 0) {
std::cout << "Received message: " << receivedMessage << std::endl;
} else {
std::cerr << "Failed to receive message" << std::endl;
return -1;
}
// Release the message buffer
messageBuffer.release ();
return 0;
}
