MessageBuffer

This is an example of how to use the MessageBuffer class to create and manage a FreeRTOS message buffer.

/**
* MessageBuffer.cpp
*
* Example demonstrating the use of FreeRTOS message buffers.
*
* This file contains the implementation of a sender task and a receiver task
* that communicate using a FreeRTOS message buffer.
*
* Date: 11.01.2025
*/
#include "MessageBuffer.hpp"
#include "FreeRTOS.h"
#include "task.h"
/**
* @brief Task function to send a message to a message buffer periodically.
*
* This task sends a predefined message to the provided message buffer every second.
*
* @param pvParameters Pointer to the message buffer object (Lib::FreeRTOS::MessageBuffer).
*/
void senderTask (void * pvParameters) {
Lib::FreeRTOS::MessageBuffer * messageBuffer = static_cast<Lib::FreeRTOS::MessageBuffer *> (pvParameters);
const char * message = "Hello, FreeRTOS!";
while (true) {
messageBuffer->send (message, strlen (message) + 1, portMAX_DELAY);
vTaskDelay (pdMS_TO_TICKS (1000)); // Delay for 1 second
}
}
/**
* @brief Task function to receive messages from a message buffer.
*
* This function runs indefinitely, receiving messages from the provided
* message buffer and processing them. The received messages are stored
* in a local buffer and processed if any bytes are received.
*
* @param pvParameters Pointer to the message buffer object (Lib::FreeRTOS::MessageBuffer).
*/
void receiverTask (void * pvParameters) {
Lib::FreeRTOS::MessageBuffer * messageBuffer = static_cast<Lib::FreeRTOS::MessageBuffer *> (pvParameters);
char buffer [50];
while (true) {
size_t receivedBytes = messageBuffer->receive (buffer, sizeof (buffer), portMAX_DELAY);
if (receivedBytes > 0) {
// Process the received message
}
}
}
/**
* @brief Entry point of the application.
*
* This function initializes a message buffer and creates two tasks: a sender task and a receiver task.
* If the message buffer is successfully created, the tasks are created with the specified stack size and priority.
* Finally, the FreeRTOS scheduler is started.
*
* @return int Returns 0 upon completion.
*/
int main () {
Lib::FreeRTOS::MessageBuffer messageBuffer;
if (messageBuffer.create (100)) {
xTaskCreate (senderTask, "Sender", configMINIMAL_STACK_SIZE, &messageBuffer, tskIDLE_PRIORITY + 1, nullptr);
xTaskCreate (receiverTask, "Receiver", configMINIMAL_STACK_SIZE, &messageBuffer, tskIDLE_PRIORITY + 1, nullptr);
vTaskStartScheduler ();
}
return 0;
}
