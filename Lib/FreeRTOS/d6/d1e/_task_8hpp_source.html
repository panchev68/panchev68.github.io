125 bool create (const std::string & name, const std::function<void ()> & func, size_t stackDept = configMINIMAL_STACK_SIZE, int32_t priority = DEFAULT_TASK_PRIORITY)
126 {
127 auto taskFn = [] (void * arg) {
128 assert (arg);
129 auto task = reinterpret_cast<Task *> (arg);
130 assert (task->_function != nullptr);
131 if (task->_function != nullptr) task->_function ();
132 task->release ();
133 };
134
135 assert ((handle == nullptr) && "Task is already created");
136 assert (func);
137 _function = func;
138
139 // create a native task
140 auto retvar = (xTaskCreate (taskFn, name.c_str (), stackDept, this, NORMAL_TASK_PRIORITY + priority, &handle) == pdPASS);
141 assert (handle);
142 return retvar;
143 }
