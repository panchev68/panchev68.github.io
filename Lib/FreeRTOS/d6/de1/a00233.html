Example usage of the StreamBuffer class for sending and receiving data between tasks in FreeRTOS.

Example usage of the StreamBuffer class for sending and receiving data between tasks in FreeRTOS.

#include "RTOS.hpp"
#include <cstdint>
RTOS::StreamBuffer stream;
void senderTask(void *) {
stream.create(128, 16);
uint8_t data[32] = {1, 2, 3, 4};
stream.send(data, sizeof(data), 10);
}
void receiverTask(void *) {
stream.create(128, 16);
uint8_t buffer[32] = {};
if (stream.receive(buffer, sizeof(buffer), 10) > 0) {
// Process received data
}
}
