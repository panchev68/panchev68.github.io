QueueSet

This is an example of how to use the QueueSet class.

/**
* @file Queueset.cpp
* @brief Example usage of FreeRTOS QueueSet in C++.
*
* This file demonstrates how to create and use a QueueSet with two queues
* in FreeRTOS using C++. The example includes creating the queues, adding
* them to a QueueSet, selecting a queue from the QueueSet, and handling
* the selected queue.
*
* @note This example assumes the existence of a FreeRTOS C++ wrapper library.
*/
#include "FreeRTOS.hpp"
/**
* @brief Demonstrates the usage of a QueueSet in FreeRTOS.
*
* This function creates two queues and adds them to a QueueSet. It then waits
* indefinitely for one of the queues to have data and processes the queue that
* has data available. Finally, it cleans up the created queues.
*
* @note This example assumes that the FreeRTOS namespace and relevant classes
* (Queue, QueueSet) are available and properly implemented.
*
* @details
* - Two queues (queue1 and queue2) are created.
* - A QueueSet is created to hold the two queues.
* - The function waits indefinitely for one of the queues to have data.
* - Depending on which queue has data, the function processes the respective queue.
* - The queues are destroyed at the end of the function.
*
* @warning If the QueueSet creation fails, the function asserts and returns immediately.
*/
void exampleQueueSetUsage () {
using namespace FreeRTOS;
// Create two queues
Queue<uint32_t> queue1;
Queue<uint32_t> queue2;
// Create a queue set that can hold 2 queues
QueueSet queueSet;
if (!queueSet.create (2, {queue1, queue2})) {
assert (0 && "Error creating QueueSet");
return;
}
auto selectedMember = queueSet.select (portMAX_DELAY);
if (selectedMember == queue1) {
// Handle queue1
} else if (selectedMember == queue2) {
// Handle queue2
}
// Clean up
queue1.destroy ();
queue2.destroy ();
}
