Timer

This example demonstrates how to create, start, and stop a FreeRTOS timer using the Timer class. It shows how to set up a timer with a callback function and manage its lifecycle.

/**
* @file Timer.cpp
* @brief Example demonstrating the usage of a FreeRTOS timer.
*
* This file contains an example of how to create, start, and stop a FreeRTOS timer.
* The timer callback function prints a message to the console when the timer expires.
*
* @author Bobby
* @date 11.01.2025
*/
#include "Timer.hpp"
#include <iostream>
/**
* @brief Timer callback function to be executed when the timer expires.
* This function prints a message to the console.
*/
void timerCallback () {
std::cout << "Timer expired!" << std::endl;
}
/**
* @brief Main function to demonstrate the usage of a FreeRTOS timer.
*
* This function creates a FreeRTOS timer with a period of 1000 ticks (1 second)
* and auto-reload enabled. It then waits for 5000 milliseconds (5 seconds) to
* allow the timer to expire a few times. After that, it stops and releases the timer.
*
* @return int Returns 0 upon successful completion.
*/
int main () {
// Create a timer with a period of 1000 ticks (1 second) and auto-reload enabled
Lib::FreeRTOS::Timer timer ("ExampleTimer", timerCallback, 1000, true, true);
// Wait for some time to let the timer expire a few times
vTaskDelay (pdMS_TO_TICKS (5000));
// Stop the timer
timer.stop ();
// Release the timer
timer.release ();
return 0;
}
