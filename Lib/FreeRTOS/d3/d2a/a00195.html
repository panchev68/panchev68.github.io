Mutex

Example usage of the Mutex class to interact with FreeRTOS mutexes.

Example usage of the Mutex class to interact with FreeRTOS mutexes.This example demonstrates how to use the Mutex class to create, take, and give a mutex using FreeRTOS.

#include "Mutex.hpp"
#include <iostream>
int main() {
// Create a mutex
Lib::FreeRTOS::Mutex mutex;
if (!mutex.create()) {
std::cerr << "Failed to create mutex" << std::endl;
return -1;
}
// Take the mutex
if (!mutex.take(100)) {
std::cerr << "Failed to take mutex" << std::endl;
return -1;
}
// Critical section
std::cout << "Mutex taken, entering critical section" << std::endl;
// Give the mutex
if (!mutex.give()) {
std::cerr << "Failed to give mutex" << std::endl;
return -1;
}
std::cout << "Mutex given, exiting critical section" << std::endl;
// Release the mutex
mutex.release();
return 0;
}
