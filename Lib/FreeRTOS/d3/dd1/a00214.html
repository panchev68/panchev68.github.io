QueueSet

This is an example of how to use the QueueSet class.

/**
* @file Queueset.cpp
* @brief Example demonstrating the use of FreeRTOS QueueSet with multiple queues.
*
* This example creates a queue set with a length of 2 and adds two queues to it.
* It then sends a message to one of the queues, selects a member from the queue set,
* and reads the message from the appropriate queue. Finally, it releases the queues
* and the queue set.
*
* @author Bobby
* @date 11.01.2025
*/
#include "QueueSet.hpp"
#include "Queue.hpp"
#include <iostream>
/**
* @brief Main function demonstrating the usage of FreeRTOS queue sets.
*
* This function performs the following steps:
* 1. Creates a queue set with a length of 2.
* 2. Creates two queues and adds them to the queue set.
* 3. Sends a message to one of the queues.
* 4. Selects a member from the queue set.
* 5. Checks which queue received the message and reads it.
* 6. Prints the received message to the standard output.
* 7. Releases the queues and the queue set.
*
* @return int Returns 0 on success, -1 on failure.
*/
int main () {
// Create a queue set with a length of 2
Lib::FreeRTOS::QueueSet queueSet;
if (!queueSet.create (2, {})) {
std::cerr << "Failed to create queue set" << std::endl;
return -1;
}
// Create two queues to add to the queue set
Lib::FreeRTOS::Queue<int, 5> queue1;
Lib::FreeRTOS::Queue<int, 5> queue2;
if (!queue1.create () || !queue2.create ()) {
std::cerr << "Failed to create queues" << std::endl;
return -1;
}
// Add the queues to the queue set
if (!queueSet.add (queue1.getHandle ()) || !queueSet.add (queue2.getHandle ())) {
std::cerr << "Failed to add queues to queue set" << std::endl;
return -1;
}
// Send a message to one of the queues
int message = 42;
if (!queue1.send (message, 100)) {
std::cerr << "Failed to send message to queue" << std::endl;
return -1;
}
// Select a member from the queue set
auto selectedMember = queueSet.select (100);
if (selectedMember == nullptr) {
std::cerr << "Failed to select a member from the queue set" << std::endl;
return -1;
}
// Check which queue received the message and read it
int receivedMessage;
if (selectedMember == queue1.getHandle ()) {
if (!queue1.receive (receivedMessage, 100)) {
std::cerr << "Failed to receive message from queue1" << std::endl;
return -1;
}
} else if (selectedMember == queue2.getHandle ()) {
if (!queue2.receive (receivedMessage, 100)) {
std::cerr << "Failed to receive message from queue2" << std::endl;
return -1;
}
}
std::cout << "Received message: " << receivedMessage << std::endl;
// Release the queues and queue set
queue1.release ();
queue2.release ();
queueSet.release ();
return 0;
}
