31 /**

56 /**
57 * @brief Creates a new message buffer using dynamically allocated memory.
58 *
59 * This function creates a new message buffer using dynamically allocated memory.
60 * The message buffer will be able to hold messages of the specified size.
61 *
62 * @param nsize The size of the message buffer in bytes.
63 * @return true if the message buffer was created successfully, false otherwise.
64 */
65 bool create (size_t nsize)
66 {
67 assert (nsize);
68 assert (_handle == nullptr && "MessageBuffer is already created");
69 _handle = xMessageBufferCreate (nsize + sizeof (size_t));
70 if (_handle == nullptr)
71 {
72 assert (0 && "Error creating MessageBuffer");
73 return false;
74 }
75 return true;
76 }

78 /**
79 * @brief Releases the message buffer.
80 *
81 * This function deletes the message buffer if it has been allocated.
82 * It checks if the message buffer handle is not null, and if so,
83 * it deletes the message buffer and sets the handle to null.
84 */
85 void release ()
86 {
87 if (_handle != nullptr)
88 {
89 vMessageBufferDelete (_handle);
90 _handle = nullptr;
91 }

94 /**
95 * @brief Sends a message to the message buffer.
96 *
97 * This function sends a message to the message buffer. The message is copied
98 * from the memory pointed to by `src` and has a length of `len` bytes. The
99 * function will block for up to `ticksToWait` ticks if there is not enough
100 * space in the buffer to hold the message.
101 *
102 * @param src Pointer to the message to be sent.
103 * @param len Length of the message in bytes.
104 * @param ticksToWait Maximum number of ticks to wait for space to become
105 * available in the buffer.
106 * @return The number of bytes actually sent to the buffer, which will be
107 * equal to `len` if the message was sent successfully, or 0 if the message
108 * could not be sent within the specified wait time.
109 *
110 * @note This function asserts that the message buffer handle is not null,
111 * that the source pointer is valid, and that the message length is greater
112 * than 0.
113 */
114 size_t send (const void * src, size_t len, uint32_t ticksToWait)
115 {
116 assert ((_handle != nullptr) && "MessageBuffer is not created");
117 assert (src);
118 assert (len > 0);
119 return xMessageBufferSend (_handle, src, len, ticksToWait);
120 }

122 /**
123 * @brief Sends a message to the message buffer from an ISR.
124 *
125 * This function sends a message to the message buffer from an interrupt
126 * service routine (ISR). The message is copied from the memory pointed to
127 * by `src` and has a length of `len` bytes. The function will not block
128 * and will return immediately if there is not enough space in the buffer
129 * to hold the message.
130 *
131 * @param src Pointer to the message to be sent.
132 * @param len Length of the message in bytes.
133 * @return The number of bytes actually sent to the buffer, which will be
134 * equal to `len` if the message was sent successfully, or 0 if the message
135 * could not be sent.
136 *
137 * @note This function asserts that the message buffer handle is not null,
138 * that the source pointer is valid, and that the message length is greater
139 * than 0.
140 */
141 size_t sendFromISR (const void * src, size_t len)
142 {
143 assert ((_handle != nullptr) && "MessageBuffer is not created");
144 assert (src);
145 assert (len > 0);
146 long woken (0);
147 auto result = xMessageBufferSendFromISR (_handle, src, len, &woken);
148 assert (result == len);
149 portYIELD_FROM_ISR (woken);
150 return result;
151 }

159 * @param dst Pointer to the buffer where the received message will be stored.
160 * @param len The length of the buffer pointed to by dst.

163 * @return The number of bytes received, or 0 if no message was received within the specified time.
164 *

166 */
167 size_t receive (void * dst, size_t len, uint32_t ticksToWait)

169 assert ((_handle != nullptr) && "MessageBuffer is not created");
