/**
* @file Timeout.cpp
* @brief Demonstrates the usage of the Timeout class in FreeRTOS.
*
* This file contains an example of how to create, set, check, and disable a timeout using the Timeout class.
* It also shows how to manipulate the timeout duration and compare the timeout status.
*
* @author Bobby
* @date 11.01.2025
*/
#include "Timeout.hpp"
#include <cstdint>
/**
* @brief Main function to demonstrate the usage of the Timeout class.
*
* This function demonstrates how to create, set, check, and disable a timeout using the Timeout class.
*/
void main () {
// Create a Timeout object with a duration of 1000 ticks
Lib::FreeRTOS::Timeout timeout (1000);
// Set the timeout duration to 2000 ticks
timeout.set (2000);
// Check if the timeout has elapsed
if (timeout.check ()) {
// Timeout has elapsed
} else {
// Timeout has not yet elapsed
}
// Disable the timeout
timeout.disable ();
// Check if the timeout is disabled
if (timeout.isDisable ()) {
// Timeout is disabled
}
// Add 500 ticks to the current timeout duration
timeout += 500;
// Subtract 300 ticks from the current timeout duration
timeout -= 300;
// Assign a new duration to the timeout
timeout = 1500;
// Compare the result of the check() function with a boolean value
if (timeout == true) {
// Timeout has elapsed
} else {
// Timeout has not yet elapsed
}
// Convert the Timeout object to a uint32_t
uint32_t duration = timeout;
}
