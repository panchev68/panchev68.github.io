EventGroup

This is an example of how to use the EventGroup class to create and manage a FreeRTOS event group.

/**
* @file EventGroup.cpp
*
* @brief EventGroup class example
*
* @version 1.0
* @date 11.01.2025
*/
#pragma once
#include <FreeRTOS.hpp>
#include <iostream>
/**
* @brief Main function to demonstrate the usage of the EventGroup class.
*
* This function demonstrates how to create, set, get, wait for, and clear bits in an event group.
*/
int main () {
// Create an EventGroup object
FreeRTOS::EventGroup eventGroup;
// Create the event group
if (eventGroup.create ()) {
std::cout << "Event group created successfully." << std::endl;
} else {
std::cerr << "Failed to create event group." << std::endl;
return -1;
}
// Set some bits in the event group
FreeRTOS::EventBits bitsToSet;
bitsToSet.set (0); // Set bit 0
bitsToSet.set (1); // Set bit 1
eventGroup.setBits (bitsToSet);
std::cout << "Bits set in event group." << std::endl;
// Get the current bits in the event group
FreeRTOS::EventBits currentBits = eventGroup.getBits ();
std::cout << "Current bits in event group: " << currentBits << std::endl;
// Wait for specific bits to be set
FreeRTOS::EventBits bitsToWaitFor;
bitsToWaitFor.set (0); // Wait for bit 0
FreeRTOS::EventBits resultBits = eventGroup.waitForBits (bitsToWaitFor, true, true, 1000);
std::cout << "Bits after waiting: " << resultBits << std::endl;
// Clear some bits in the event group
FreeRTOS::EventBits bitsToClear;
bitsToClear.set (1); // Clear bit 1
eventGroup.clearBits (bitsToClear);
std::cout << "Bits cleared in event group." << std::endl;
// Release the event group
eventGroup.destroy ();
std::cout << "Event group released." << std::endl;
return 0;
}
