46 /// \brief Delete an event group that was previously created
47 /// \details
48 /// Delete an event group that was previously created using a call to xEventGroupCreate().
49 /// Tasks that are blocked on the event group being deleted will be unblocked, and report an event group value of 0.

64 /// \brief Read bits within an RTOS event group, optionally entering the Blocked state (with a timeout) to wait for a bit or group of bits to become set.
65 /// \param bitsToWait A bitwise value that indicates the bit or bits to test inside the event group
66 /// \param clearOnExit is set to pdTRUE then any bits set in the value passed as the uxBitsToWaitFor parameter will be cleared in the event group before
67 /// \param waitForAllBytes is used to create either a logical AND test (where all bits must be set) or a logical OR test (where one or more bits must be set) as follows
68 /// \param ticksToWait The maximum amount of time (specified in 'ticks') to wait for one/all (depending on the xWaitForAllBits value) of the bits specified by uxBitsToWaitFor to become set.
69 /// \see https://www.freertos.org/xEventGroupWaitBits.html
70 uint32_t waitBits (uint32_t bitsToWait, bool clearOnExit, bool waitForAllBytes, uint32_t ticksToWait) const
71 {
72 assert (_handle && "EventGroup is not created");
73 return xEventGroupWaitBits (_handle, bitsToWait, (clearOnExit ? pdTRUE : pdFALSE), (waitForAllBytes ? pdTRUE : pdFALSE), ticksToWait);
74 }

76 /// \brief wait for a bits

78 /// \param ticksToWait
79 /// \return bool
80 uint32_t wait (uint32_t bitsToWait, uint32_t ticksToWait) const

140 /// \brief Atomically set bits within an RTOS event group, then wait for a combination of bits to be set within the same event group.
141 /// This functionality is typically used to synchronise multiple tasks (often called a task rendezvous),
142 /// where each task has to wait for the other tasks to reach a synchronisation point before proceeding.
143 /// \param bitsToSet The bit or bits to set in the event group before determining if (and possibly waiting for), all the bits specified by the uxBitsToWait parameter are set. For example, set uxBitsToSet to 0x04 to set bit 2 within the event group.
144 /// \param bitsToWaitFor A bitwise value that indicates the bit or bits to test inside the event group. For example, set uxBitsToWaitFor to 0x05 to wait for bits 0 and bit 2. Set uxBitsToWaitFor to 0x07 to wait for bit 0 and bit 1 and bit 2. Etc.
145 /// \param ticksToWait The maximum amount of time (specified in 'ticks') to wait for all the bits specified by the uxBitsToWaitFor parameter value to become set.
146 /// \return The value of the event group at the time either the bits being waited for became set, or the block time expired. Test the return value to know which bits were set.
147 /// If xEventGroupSync() returned because its timeout expired then not all the bits being waited for will be set.
148 /// If xEventGroupSync() returned because all the bits it was waiting for were set then the returned value is the event group value before any bits were automatically cleared.
149 /// \see [xEventGroupSync()](https://www.freertos.org/xEventGroupSync.html)
150 uint32_t sync (uint32_t bitsToSet, uint32_t bitsToWaitFor, uint32_t ticksToWait) const
151 {
152 assert ((_handle != nullptr) && "EventGroup is not created");
153 return xEventGroupSync (_handle, bitsToSet, bitsToWaitFor, ticksToWait);
154 }
