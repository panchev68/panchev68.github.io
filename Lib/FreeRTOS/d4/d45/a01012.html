xCoRoutineCreate
void create ()
{
auto func = [] (CoRoutineHandle_t hnd, UBaseType_t index) {
crSTART (hnd);
for (;;)
{
crDELAY (hnd, 100);
}
// Must end every co-routine with a call to crEND();
crEND ();
};
Core::FreeRTOS::CoRoutine CoRoutine (func, 0, 0);
}
