85 /**
86 * @brief Sends a message to the stream buffer.
87 *
88 * This function sends a message to the stream buffer. The message is copied
89 * from the memory pointed to by `src` and has a length of `len` bytes. The
90 * function will block for up to `ticksToWait` ticks if there is not enough
91 * space in the buffer to hold the message.
92 *
93 * @param src Pointer to the message to be sent.
94 * @param len Length of the message in bytes.
95 * @param ticksToWait Maximum number of ticks to wait for space to become
96 * available in the buffer.
97 * @return The number of bytes actually sent to the buffer, which will be
98 * equal to `len` if the message was sent successfully, or 0 if the message
99 * could not be sent within the specified wait time.
100 *
101 * @note This function asserts that the stream buffer handle is not null,
102 * that the source pointer is valid, and that the message length is greater
103 * than 0.
104 */
105 size_t send (const void * src, size_t len, uint32_t ticksToWait)
106 {
107 assert ((_handle != nullptr) && "Stream Buffer is not created");
108 assert (src);
109 assert (len);
110 size_t result = 0;
111 if (result = xStreamBufferSend (_handle, src, len, ticksToWait) != len) assert (0);
112 return result;
113 }

115 /**
116 * @brief Sends a message to the stream buffer from an ISR.
117 *
118 * This function sends a message to the stream buffer from an interrupt
119 * service routine (ISR). The message is copied from the memory pointed to
120 * by `src` and has a length of `len` bytes. The function will not block
121 * and will return immediately if there is not enough space in the buffer
122 * to hold the message.
123 *
124 * @param src Pointer to the message to be sent.
125 * @param len Length of the message in bytes.
126 * @return The number of bytes actually sent to the buffer, which will be
127 * equal to `len` if the message was sent successfully, or 0 if the message
128 * could not be sent.
129 *
130 * @note This function asserts that the stream buffer handle is not null,
131 * that the source pointer is valid, and that the message length is greater
132 * than 0.
133 */
134 size_t sendFromISR (const void * src, size_t len)
135 {
136 assert ((_handle != nullptr) && "Stream Buffer is not created");
137 assert (src);
138 assert (len);
139 long woken (0);
140 auto result = (xStreamBufferSendFromISR (_handle, src, len, &woken));
141 assert (result == len);
142 portYIELD_FROM_ISR (woken);
143 return result;
144 }
