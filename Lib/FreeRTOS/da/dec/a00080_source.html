48 /// \brief Write to a stream buffer from a task.
49 /// \param src - A pointer to the buffer that holds the bytes to be copied into the stream buffer
50 /// \param len - The maximum number of bytes to copy from src into the stream buffer.
51 /// \param ticksToWait - The maximum amount of time the task should remain in the Blocked state to wait for enough space to become available in the stream buffer
52 /// \return size_t
53 /// \see https://www.freertos.org/xStreamBufferSend.html
54 size_t send (const void * src, size_t len, uint32_t ticksToWait)
55 {
56 assert ((_handle != nullptr) && "Stream Buffer is not created");
57 assert (src);
58 assert (len);
59 size_t result = 0;
60 if (result = xStreamBufferSend (_handle, src, len, ticksToWait) != len) assert (0);
61 return result;
62 }

64 /// \brief Write to a stream buffer from a task from ISR.
65 /// \param src - A pointer to the buffer that holds the bytes to be copied into the stream buffer
66 /// \param len - The maximum number of bytes to copy from src into the stream buffer.
67 /// \return size_t
68 size_t sendFromISR (const void * src, size_t len)
69 {
70 assert ((_handle != nullptr) && "Stream Buffer is not created");
71 assert (src);
72 assert (len);
73 long woken (0);
74 auto result = (xStreamBufferSendFromISR (_handle, src, len, &woken));
75 assert (result == len);
76 portYIELD_FROM_ISR (woken);
77 return result;
78 }
