29 /// \brief Creates a new stream buffer
30 /// \param bufferSize The total number of bytes the stream buffer will be able to hold at any one time.
31 /// \param triggerLevel The number of bytes that must be in the stream buffer before a task that is blocked on
32 /// the stream buffer to wait for data is moved out of the blocked state. For example, if a task is blocked on a
33 /// read of an empty stream buffer that has a trigger level of 1, then the task will be unblocked when a single
34 /// byte is written to the buffer or the task's block time expires. As another example, if a task is blocked on
35 /// a read of an empty stream buffer that has a trigger level of 10, then the task will not be unblocked until
36 /// the stream buffer contains at least 10 bytes or the task's block time expires. If a reading task's block
37 /// time expires before the trigger level is reached then the task will still receive as many bytes as are
38 /// actually available. Setting a trigger level of 0 will result in a trigger level of 1 being used. It is not
39 /// valid to specify a trigger level that is greater than the buffer size.
40 /// \return if \b false is returned, the stream buffer cannot be created because there is insufficient heap memory
41 /// available for FreeRTOS to allocate the stream buffer data structures and storage area. If a \b true value
42 /// is returned, the stream buffer has been created successfully - the returned value should be stored as the
43 /// handle to the created stream buffer.
44 /// \see [xStreamBufferCreate](https://www.freertos.org/xStreamBufferCreate.html)
45 bool create (size_t bufferSize, size_t triggerLevel)
46 {
47 assert (_handle == nullptr);
48 _handle = xStreamBufferCreate (bufferSize, triggerLevel);
49 assert ((_handle != nullptr) && "Error creating Stream Buffer");
50 return (_handle != nullptr);
51 }
52
53 /// \brief Releases the stream buffer
54 void release ()
55 {

57 {
58 vStreamBufferDelete (_handle);
59 _handle = nullptr;
60 }
61 }

63 /// \brief Write to a stream buffer from a task.
64 /// \param src - A pointer to the buffer that holds the bytes to be copied into the stream buffer
65 /// \param len - The maximum number of bytes to copy from src into the stream buffer.
66 /// \param ticksToWait - The maximum amount of time the task should remain in the Blocked state to wait for
67 /// enough space to become available in the stream buffer
68 /// \return The number of bytes written to the stream buffer. If a task times out before it can write all
69 /// `len` into the buffer it will still write as many bytes as possible.
70 /// \see https://www.freertos.org/xStreamBufferSend.html
71 size_t send (const void * src, size_t len, uint32_t ticksToWait)
72 {
73 assert ((_handle != nullptr) && "Stream Buffer is not created");
74 assert (src);
75 assert (len);
76 size_t result = 0;
77 if (result = xStreamBufferSend (_handle, src, len, ticksToWait) != len) assert (0);
78 return result;
79 }

81 /// \brief Write to a stream buffer from a task from ISR.
82 /// \param src - A pointer to the buffer that holds the bytes to be copied into the stream buffer
83 /// \param len - The maximum number of bytes to copy from src into the stream buffer.
84 /// \return size_t
85 size_t sendFromISR (const void * src, size_t len)
86 {
87 assert ((_handle != nullptr) && "Stream Buffer is not created");
88 assert (src);
89 assert (len);
90 long woken (0);
91 auto result = (xStreamBufferSendFromISR (_handle, src, len, &woken));
92 assert (result == len);
93 portYIELD_FROM_ISR (woken);
94 return result;
95 }
