12 /**
13 * @class TaskHelper
14 * @brief A class providing helper functions for FreeRTOS tasks.
15 *
16 * This class provides a set of static functions that allow access to FreeRTOS tasks.
17 * The functions include resuming and suspending tasks, setting and getting task priorities,
18 * and handling task notifications.
19 */
20 class TaskHelper
21 {
22 public:

24 /**
25 * @brief Type for task handle.
26 *
27 * This type alias is used to simplify the usage of TaskHandle_t within the code.
28 * It represents a handle to a task in FreeRTOS.
29 */
30 using Handle = TaskHandle_t;

32 /**
33 * @brief Resumes a task that has been suspended.
34 *
35 * This function resumes a task that has been suspended using vTaskSuspend().
36 *
37 * @param handle The handle of the task to resume.
38 */
39 static void resume (Handle handle)
40 {
41 assert (handle);
42 vTaskResume (handle);
43 }
44
45 /**
46 * @brief Resumes a task from an ISR.
47 *
48 * This function resumes a task from an Interrupt Service Routine (ISR).
49 *
50 * @param handle The handle of the task to resume.
51 * @return pdPASS if the task was successfully resumed, pdFAIL otherwise.
52 */
53 static long resumeFromISR (Handle handle)
54 {
55 assert (handle);
56 return xTaskResumeFromISR (handle);
57 }

59 /**
60 * @brief Suspends a task.

66 static void suspend (Handle handle)

102 /**
103 * @brief Notifies a task with the specified action and value.
104 *
105 * This function notifies a task with the specified action and value.
106 *
107 * @param handle The handle of the task to notify.
108 * @param action The action to take when notifying the task.
109 * @param value The value to send to the task.

112 static uint32_t notifyTake (uint32_t clearCountOnExit, uint32_t ticksToWait)

117 /**
118 * @brief Waits for a task notification.
119 *
120 * This function waits for a task notification with the specified parameters.
121 *
122 * @param bitsToClearOnEntry Bits to clear in the task's notification value on entry.
123 * @param bitsToClearOnExit Bits to clear in the task's notification value on exit.
124 * @param notificationValue Pointer to a variable to receive the task's notification value.
125 * @param ticksToWait The maximum amount of time (in tick periods) to wait for a notification.
126 * @return true if the notification was received before the timeout period expired, false otherwise.
127 */
128 static bool notifyWait (uint32_t bitsToClearOnEntry, uint32_t bitsToClearOnExit, uint32_t * notificationValue, uint32_t ticksToWait)
129 {
130 return xTaskNotifyWait (bitsToClearOnEntry, bitsToClearOnExit, notificationValue, ticksToWait) == pdTRUE;
131 }

244 /**
245 * @brief Delays the execution of the current task for a specified time.

247 * This function suspends the execution of the current task for the specified

258 /**
259 * @brief Delays a task until a specified time.
260 *
261 * This function delays the calling task until the time specified by the
262 * combination of the previous wake time and the time increment. It is
263 * typically used to create periodic tasks.

277 /**

288 /**
289 * @brief Resumes the FreeRTOS scheduler.
290 *
291 * This function resumes the FreeRTOS scheduler by calling xTaskResumeAll().
292 * It returns true if the scheduler was successfully resumed, otherwise false.
293 *
294 * @return true if the scheduler was successfully resumed, false otherwise.
295 */
296 static bool resumeScheduler ()

311 /**
312 * @brief Enters a critical section by disabling interrupts.
313 *
314 * This function calls the FreeRTOS macro `taskENTER_CRITICAL()` to disable interrupts
315 * and enter a critical section. Critical sections are used to protect code that must
316 * not be interrupted, ensuring atomic access to shared resources.
317 *
318 * @note It is important to pair this function with a corresponding call to
319 * `endCriticalSection()` to re-enable interrupts and exit the critical section.
320 */
321 static void beginCriticalSection ()
322 {
323 taskENTER_CRITICAL ();
324 }
325
326 /**
327 * @brief Ends a critical section.
328 *
329 * This function exits a critical section that was previously entered.
330 * It should be used to mark the end of a critical section where
331 * interrupts were disabled to protect shared resources.
332 *
333 * @note This function relies on the FreeRTOS macro taskEXIT_CRITICAL().
334 */
335 static void endCriticalSection ()
336 {
337 taskEXIT_CRITICAL ();
338 }
