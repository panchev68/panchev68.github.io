10 /// \brief Counting Semaphore class \ingroup SemaphoreCounting
11 /// \details Counting semaphore can be used to control the access to the resource.
12 /// To obtain control of a resource, a task must first obtain a semaphore.
13 /// Thus decrementing the semaphore count value. When the count value reaches zero there are no free resources.
14 /// When a task finishes with the resource it ‘gives’ the semaphore back and thus incrementing the semaphore count value.
15 /// \see [FreeRTOS Counting Semaphores](https://www.freertos.org/Real-time-embedded-RTOS-Counting-Semaphores.html)
16 class SemaphoreCounting : SemaphoreBase
17 {
18 public:
19
20 /// \brief Semaphore Counting constructor
21 /// \param maxCount
22 /// \param initCount
23 SemaphoreCounting (uint32_t maxCount, uint32_t initCount) : _maxCount (maxCount), _initCount (initCount)
24 {}
25
26 /// \brief SemaphoreCounting destructor
27 ~SemaphoreCounting () = default;
28
29 /// \brief Creates a counting semaphore
30 /// \param maxCount - The maximum count value that can be reached. When the semaphore reaches this value it can no longer be 'given'
31 /// \param initCount - The count value assigned to the semaphore when it is created.
32 /// \return bool
33 /// \see [xSemaphoreCreateCounting](https://www.freertos.org/CreateCounting.html)
34 bool create () override
35 {
36 assert (handle == nullptr);
37 handle = xSemaphoreCreateCounting (_maxCount, _initCount);
38 assert (handle);
39 return (handle != nullptr);
40 }
41
42 /// \brief Returns the count of a semaphore
43 /// \return uint32_t

45 uint32_t getCount ()
46 {
47 assert (handle);

49 }
