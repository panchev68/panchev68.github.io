17 * @brief A class representing a counting semaphore in FreeRTOS.
18 *
19 * This class provides a C++ interface for creating and managing counting semaphores in FreeRTOS.
20 * Counting semaphores are used to control access to a shared resource by multiple tasks.
21 */
22 class SemaphoreCounting : SemaphoreBase
23 {
24 public:
25
26 /**

28 *
29 * This constructor initializes the maximum and initial count for the counting semaphore.

36
37 /**
38 * @brief Destructor for the SemaphoreCounting class.

40 ~SemaphoreCounting () = default;

42 /**
43 * @brief Creates a new counting semaphore using dynamically allocated memory.
44 *
45 * This function creates a new counting semaphore using dynamically allocated memory.
46 * The counting semaphore will have the specified maximum and initial count values.
47 *
48 * @return true if the counting semaphore was successfully created, false otherwise.
49 */
50 bool create () override

108 /**
109 * @brief Take the semaphore from an ISR context.
110 *
111 * This function attempts to take the semaphore from within an Interrupt Service Routine (ISR).
112 * It overrides the base class implementation to provide specific functionality for counting semaphores.

114 * @return true if the semaphore was successfully taken, false otherwise.
115 */
116 bool takeFromISR () override
117 {
118 return SemaphoreBase::takeFromISR ();
119 }
120
121 /**
122 * @brief Gives the semaphore from an ISR context.
123 *
124 * This function attempts to release the semaphore from within an Interrupt Service Routine (ISR).
125 * If the semaphore is successfully given, it will also check if a higher priority task was woken
126 * by the operation and yield to it if necessary.
127 *
128 * @return true if the semaphore was successfully given, false otherwise.
129 *
130 * @note This function should only be called from an ISR context.
131 * @note The semaphore must be created before calling this function.
132 */
133 bool giveFromISR () override
134 {
135 return SemaphoreBase::giveFromISR ();
136 }
137
138 /**
139 * @brief Checks if the semaphore has been successfully created.
140 *
141 * This function overrides the base class implementation to determine
142 * if the semaphore object has been created. It calls the base class method
143 * SemaphoreBase::isCreated() to perform the check.
144 *
145 * @return true if the semaphore has been created, false otherwise.
146 */
147 bool isCreated () const override
148 {
149 return SemaphoreBase::isCreated ();
150 }
151
152 private:
153
154 /**
155 * @brief Maximum count value for the semaphore.
156 *
157 * This constant represents the maximum number of times the semaphore can be
158 * given (released) before it reaches its maximum count. It is used to control
159 * the maximum number of available resources that the semaphore can manage.
160 */
161 const uint32_t _maxCount;
162 /**
163 * @brief Initial count for the semaphore.
164 *
165 * This constant represents the initial count value for the semaphore.

167 */
168 const uint32_t _initCount;
