15 *
16 * This class provides a C++ interface for creating and managing timeouts in FreeRTOS.
17 * Timeouts are used to trigger an event after a specified duration has elapsed.
18 */
19 class Timeout
20 {
21 public:

23 /**

26 Timeout () : _duration (0)

29 /**
30 * @brief Constructor for the Timeout class.

39 /**
40 * @brief Destructor for the Timeout class.

42 ~Timeout () = default;

44 /**
45 * @brief Sets the duration of the timeout.
46 *
47 * This function sets the duration of the timeout to the specified value.
48 *
49 * @param duration The duration of the timeout in ticks.
50 */
51 void set (uint32_t duration)
52 {
53 _startTicks = TickCounter::get ();
54 _duration = duration;
55 _isSet = true;
56 }
57
58 /**
59 * @brief Checks if the timeout has elapsed.
60 *
61 * This function checks if the timeout has elapsed. It returns true if the timeout has elapsed,
62 * or if the timeout has not been set. It returns false if the timeout has not yet elapsed.
63 *
64 * @return true if the timeout has elapsed, false otherwise.
65 */
66 bool check () const
67 {
68 if (isDisable ()) return false;
69 if (_isSet == false) return true;
70 return (TickCounter::get () >= (_startTicks + _duration));
71 }
72
73 /**
74 * @brief Assignment operator to set the timeout duration.
75 *
76 * This operator allows assigning a duration value to a Timeout object.
77 *
78 * @param duration The duration value to set for the timeout.
79 * @return A reference to the updated Timeout object.
80 */
81 const Timeout & operator= (uint32_t duration)
82 {
83 set (duration);
84 return *this;
85 }
86
87 /**
88 * @brief Equality operator to compare the result of the check() function with a boolean value.
89 *
90 * @param value The boolean value to compare against the result of check().
91 * @return true if the result of check() is equal to the provided boolean value.
92 * @return false otherwise.
93 */
94 bool operator== (bool value) const
95 {
96 return check () == value;
97 }
98
99 /**
100 * @brief Adds a specified duration to the current duration if the timeout is not disabled.
101 *
102 * This operator overload allows adding a given duration to the existing duration
103 * of the timeout object. The addition only occurs if the timeout is not disabled.
104 *
105 * @param duration The duration to be added to the current duration.
106 */
107 void operator+= (uint32_t duration)
108 {
109 if (isDisable () == false)
110 {
111 _duration += duration;
112 }
113 }
114
115 /**
116 * @brief Subtracts a specified duration from the current duration.
117 *
118 * This operator overload allows you to subtract a given duration from the
119 * current duration of the object. The subtraction only occurs if the
120 * object is not disabled.
121 *
122 * @param duration The duration to subtract from the current duration.
123 */
124 void operator-= (uint32_t duration)
125 {
126 if (isDisable () == false)
127 {
128 _duration -= duration;
129 }
130 }
131
132 /**
133 * @brief Conversion operator to uint32_t.
134 *
135 * This operator allows an instance of the class to be implicitly converted to a uint32_t.
136 * It returns the duration stored in the instance.
137 *
138 * @return The duration as a uint32_t.
139 */
140 operator uint32_t () const

142 return _duration;
143 }
144
145 /**
146 * @brief Disables the timeout.
147 *
148 * This function disables the timeout by setting the duration to portMAX_DELAY.
149 */
150 void disable ()
151 {
152 _duration = portMAX_DELAY;
153 }
154
155 /**
156 * @brief Checks if the timeout is disabled.
157 *
158 * This function checks if the timeout is disabled.
159 *
160 * @return true if the timeout is disabled, false otherwise.
161 */
162 bool isDisable () const
163 {
164 return (_duration == portMAX_DELAY);
165 }

172 };
