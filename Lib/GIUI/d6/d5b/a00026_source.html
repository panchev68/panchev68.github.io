23 /**
24 * @brief Font header structure
25 * @details This structure defines the header of a font data array.
26 * It includes the character width, character height, first character,
27 * and number of characters in the font.
28 */
29 __packed struct FontHeader
30 {
31 uint8_t charWidth;
32 uint8_t charHeight;
33 uint8_t firstChar;
34 uint8_t numberOfChars;
35 };
36
37 public:
38
39 /**
40 * @brief Font constructor
41 *
42 * Constructs a Font object from a raw font data array.
43 *
44 * @param rawFont A pointer to the raw font data array
45 */
46 Font (const uint8_t * rawFont)
47 {
48 assert (rawFont);
49 auto header = reinterpret_cast<const FontHeader *> (rawFont);
50
51 _charWidth = header->charWidth;
52 _charHeight = header->charHeight;
53 _firstChar = header->firstChar;
54 _numberOfChars = header->numberOfChars;
55 _isNonProportional = (header->charWidth == 0);
56
57 if (_isNonProportional)
58 {
59 _charWidths = rawFont + sizeof (FontHeader);
60 _faceArray = _charWidths + header->numberOfChars;
61 }
62 else _faceArray = rawFont + sizeof (FontHeader);
63 }
64
65 /**
66 * @brief Font destructor
67 *
68 * Destroys the Font object.
69 */
70 ~Font () = default;
71
72 /**
73 * @brief Get the character at the specified index.
74 *
75 * Returns the character at the specified index in the font data array.
76 *
77 * @param ch The index of the character to retrieve
78 * @return The character at the specified index
79 */
80 const Char at (uint8_t ch) const
81 {
82 size_t offset = 0;
83 auto width = (_isNonProportional) ? _charWidths [ch] : _charWidth;
84 auto height = _charHeight;
85
86 if ((ch < _firstChar) || (ch > _firstChar + _numberOfChars)) ch = '.';
87 ch -= _firstChar;
88 if (_isNonProportional)
89 {
90 assert (_charWidths);
91 for (auto i = 0; i < ch; i++)
92 {
93 width = _charWidths [i];
94 offset += (1 + ((width - 1) / 8)) * height;
95 }
96 width = _charWidths [ch];
97 }
98 else offset += (1 + ((width - 1) / 8)) * height * ch;
99 return Char (_faceArray + offset, width, height);
100 }
101
102 /**
103 * @brief Get the character at the specified index.
104 *
105 * Returns the character at the specified index in the font data array.
106 *
107 * @param ch The index of the character to retrieve
108 * @return The character at the specified index
109 */
110 const Char operator[] (uint8_t ch) const { return at (ch); }
111
112 /**
113 * @brief Retrieves the height of the font characters.
114 *
115 * @return The height of the font characters in pixels.
116 */
117 size_t getHeight () const { return _charHeight; }
118
119 private:
120

122 size_t _charHeight; ///< The height of the font characters
123 uint8_t _firstChar; ///< The first character in the font data array
124 uint8_t _numberOfChars; ///< The number of characters in the font data array
125 const uint8_t * _faceArray; ///< A pointer to the font face data array
126 const uint8_t * _charWidths; ///< A pointer to the character width data array
127 bool _isNonProportional; ///< A flag indicating whether the font is non-proportional
128 };
