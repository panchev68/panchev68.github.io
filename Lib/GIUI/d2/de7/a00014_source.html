41 /**
42 * @brief Converts RGB color values to a 16-bit color format.
43 *
44 * This function takes red, green, and blue color values (each ranging from 0 to 255)
45 * and converts them into a 16-bit color format. The conversion is done by truncating
46 * the least significant bits of each color component and combining them into a single
47 * 16-bit value.
48 *
49 * @param r The red color component (0-255).
50 * @param g The green color component (0-255).
51 * @param b The blue color component (0-255).
52 * @return A 16-bit color value representing the RGB color.
53 */
54 constexpr static auto rgb (uint8_t r, uint8_t g, uint8_t b) -> uint16_t { return (((r & 0xf8) << 8) | ((g & 0xfc) << 3) | (b >> 3)); }
