Example of a standalone CAN driver

This example demonstrates how to use the CANBUS driver in a standalone application.

file: Standalone.cpp
/**
* @file Standalone.cpp
* @brief Example implementation of CAN bus communication using FreeRTOS.
*
* This file contains the implementation of a CAN bus driver and an OBD2 service
* that utilizes FreeRTOS tasks for handling CAN message transmission and reception.
*
* The CAN1Port class is derived from the Canbus::Driver class and is used to
* initialize the CAN hardware.
*
* The OBD2::Service class provides methods to open the CAN bus, set up filters,
* and create FreeRTOS tasks for handling incoming and outgoing CAN messages.
*
* The startup function initializes the OBD2 service.
*
* @note This example is designed to work with the IAR Embedded Workbench for ARM (EWARM).
*
* @version 2.00
* @date 08.01.2025
*/
#include "Lib/Canbus/ServiceBase.hpp"
#include "can.h"
#include <FreeRTOS.hpp>
#include <Lib/HAL/CAN/Port.hpp>
#include <memory>
namespace HAL {
/**
* @brief CAN1 port object.
*/
Lib::HAL::CAN::Port can1Port (&hcan1);
} // namespace HAL
namespace OBD2 {
class Service {
constexpr static inline uint32_t OBD2_SENDER_ID = 0x7df; // The OBD2 sender ID.
constexpr static inline uint32_t OBD2_RECEIVER_ID = 0x7e8; // The OBD2 receiver ID.
public:
/**
* @brief Opens the CAN bus service.
*
* This function initializes and starts the CAN bus driver, sets up the CAN filter,
* and creates the receive and transmit tasks for handling CAN messages.
*
* @return true if the service was successfully opened and started, false otherwise.
*/
bool Service::open () {
if (_driver->open () == false) {
return false;
}
auto filter = _driver->createFilter ();
filter.FilterFIFOAssignment = CAN_FilterFIFO0;
filter.FilterBank = 14;
filter.FilterMode = CAN_FILTERMODE_IDMASK;
filter.FilterScale = CAN_FILTERSCALE_32BIT;
filter.FilterActivation = CAN_FILTER_ENABLE;
filter.SlaveStartFilterBank = 14;
filter.setId (OBD2_RECEIVER_ID);
_driver->setFilter (filter);
if (_driver->start () == false) {
return false;
}
if (_taskRxMessage.create ("OBD2_RX", [this] () { _taskFnRxMessage (); }) == false) {
assert (0);
return false;
}
if (_taskTxMessage.create ("OBD2_TX", [this] () { _taskFnTxMessage (); }) == false) {
assert (0);
return false;
}
}
private:
/**
* @brief A task object for handling received messages in FreeRTOS.
*
* This task is responsible for processing incoming messages in the CAN bus
* system. It is part of the FreeRTOS framework and is used to manage the
* execution of the message handling routine.
*/
FreeRTOS::Task _taskRxMessage;
/**
* @brief A task object for transmitting messages in FreeRTOS.
*
* This object represents a FreeRTOS task responsible for handling
* the transmission of messages. It is part of the FreeRTOS namespace
* and is used to manage the task's lifecycle and execution.
*/
FreeRTOS::Task _taskTxMessage;
void _taskFnRxMessage ();
void _taskFnTxMessage ();
// TODO Not complete
Canbus::Driver * _driver;
};
inline void Service::_taskFnRxMessage () {
Canbus::Message canMessage;
for (;;) {
_driver->receive (canMessage, portMAX_DELAY);
}
}
inline void Service::_taskFnTxMessage () {
uint8_t data [8];
for (;;) {
if (_driver->send (OBD2_SENDER_ID, &data, sizeof (data)) == false) {
assert (0);
}
vTaskDelay (100);
}
}
inline Service service;
} // namespace OBD2
void startup () {
if (OBD2::service.open () == false) { assert (0); }
}
