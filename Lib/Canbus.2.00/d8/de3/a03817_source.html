117 /** Handle task loop callback. */
118 void update() override {
119 using State = TaskFuncData::State;
120 using DeliveryStatus = TaskFuncData::DeliveryStatus;
121
122 auto & state = taskData.state;
123 auto & txMessageBuffer = taskData.txMessageBuffer;
124 auto & timeoutConnection = taskData.timeoutConnection;
125 auto & transmitDeliveryStatus = taskData.transmitDeliveryStatus;
126 auto & isError = taskData.isError;
127 auto & rxMessage = taskData.rxMessage;
128 auto & timeoutTransaction = taskData.timeoutTransaction;
129
130 auto isConnected = [this]() { return connection.state == Connection::State::CONNECTED; };
131 auto disconnect = [&timeoutConnection] { timeoutConnection.set(0); };
132 auto connect = [this] { connection.state = Connection::State::CONNECTED; };
133 auto refreshConnectionTimeout = [&timeoutConnection]() { timeoutConnection.set(ApplicationBase::CONNECTION_TIMEOUT); };
134
135 while (true) {
136 switch (state) {
137 case State::BEGIN_TRANSACTION:
138 if (!timeoutTransaction.isExpired()) break;
139 timeoutTransaction.set(25);
140 state = State::SEND_MESSAGE;
141 continue;
142 case State::SEND_MESSAGE:
143 if (isConnected()) connection.info.transmitPeriod.beginMeasurement();
144 handleOutgoingMessage(txMessageBuffer);
145 if (queueTransmitDeliveryStatus.receive(transmitDeliveryStatus, 0))
146 connection.info.deletedTransmitStatus++;
147 if (!sendData(txMessageBuffer)) {
148 assert(0);
149 state = State::ERROR;
150 break;
151 }
152 state = State::WAIT_FOR_SENDING_COMPLETED;
153 continue;
154 case State::WAIT_FOR_SENDING_COMPLETED:
155 if (queueTransmitDeliveryStatus.receive(transmitDeliveryStatus, MAX_RESPONSE_TIME / 2)) {
156 if (transmitDeliveryStatus == DeliveryStatus::Success) {
157 isError = false;
158 } else {
159 isError = true;
160 if (transmitDeliveryStatus == DeliveryStatus::NoACKReceived) {
161 disconnect();
162 vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
163 } else {
164 connection.info.unsuccessTransmitStatus++;
165 }
166 }
167 } else {
168 connection.info.noTransmitStatus++;
169 isError = true;
170 }
171 if (isError) {
172 state = State::ERROR;
173 break;
174 }
175 if (queueRxMessage.receive(rxMessage, 0))
176 connection.info.deletedMessages++;
177 state = State::WAITING_FOR_RECEIVE_INCOMING_MESSAGE;
178 continue;
179 case State::WAITING_FOR_RECEIVE_INCOMING_MESSAGE:
180 if (queueRxMessage.receive(rxMessage, ApplicationBase::MAX_RESPONSE_TIME)) {
181 isError = false;
182 if (handleIncomingMessage(rxMessage)) {
183 if (!isConnected()) connect();
184 connection.info.transmitPeriod.endMeasurement();
185 outputsUpdatedSucceessfulCallback();
186 refreshConnectionTimeout();
187 }
188 } else {
189 isError = true;
190 if (isConnected()) {
191 connection.info.retriesCounter++;
192 } else {
193 vTaskDelay(ApplicationBase::PAUSE_BETWEEN_MESSAGES_IN_DISCONNECTED);
194 }
195 }
196 if (isError) {
197 state = State::ERROR;
198 break;
199 }
200 state = State::END_TRANSACTION;
201 continue;
202 case State::END_TRANSACTION:
203 if (timeoutConnection.isExpired()) {
204 if (isConnected()) {
205 connection.info.signalStrength = 0;
206 connection.info.transmitPeriod.reset();
207 connection.info.reconnectionCounter++;
208 connection.state = Connection::State::DISCONNECTED;
209 }
210 }
211 state = State::BEGIN_TRANSACTION;
212 break;
213 case State::ERROR:
214 state = State::END_TRANSACTION;
215 break;
216 }
217 break;
218 }
219 }
