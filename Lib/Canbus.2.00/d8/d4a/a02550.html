M24C16

Example usage of the M24C16 EEPROM driver. Demonstrates basic read/write and connection management for the M24C16 device.

Example Usage
#include "EEPROM/I2C/M24C16.hpp"
#include <cstring> // For std::memcmp
using namespace Lib::BSP::EEPROM::I2C;
int main() {
M24C16::Config config;
// Initialize config fields as required by your platform
M24C16 eeprom(config);
if (!eeprom.open()) return -1;
uint8_t writeData[8]{1, 2, 3, 4, 5, 6, 7, 8};
uint32_t address{0x0000};
if (!eeprom.write(address, writeData, sizeof(writeData))) {
eeprom.close();
return -1;
}
uint8_t readData[8]{};
if (!eeprom.read(address, readData, sizeof(readData))) {
eeprom.close();
return -1;
}
if (std::memcmp(writeData, readData, sizeof(writeData)) == 0) {
// Success
} else {
// Data mismatch
}
eeprom.close();
return 0;
}
