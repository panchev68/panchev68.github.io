10 /** Register addresses for ILI9341 display controller.
11 * Enumerates all command and control register addresses for the ILI9341 display controller.
12 */
13 enum Registers : uint8_t {
14 NOP = 0x00, /**< No operation. */
15 SWRESET = 0x01, /**< Software reset. */
16 RDDID = 0x04, /**< Read display ID. */

18 RDMODE = 0x0A, /**< Read display power mode. */
19 RDMADCTL = 0x0B, /**< Read display MADCTL. */
20 RDPIXFMT = 0x0C, /**< Read display pixel format. */
21 RDIMGFMT = 0x0D, /**< Read display image format. */

24 SLPOUT = 0x11, /**< Exit sleep mode. */
25 PTLON = 0x12, /**< Partial mode on. */
26 NORON = 0x13, /**< Normal display mode on. */
27 INVOFF = 0x20, /**< Display inversion off. */
28 INVON = 0x21, /**< Display inversion on. */
29 GAMMASET = 0x26, /**< Gamma set. */
30 DISPOFF = 0x28, /**< Display off. */
31 DISPON = 0x29, /**< Display on. */
32 CASET = 0x2A, /**< Column address set. */
33 PASET = 0x2B, /**< Page address set. */
34 RAMWR = 0x2C, /**< Memory write. */
35 CLRSET = 0x2D, /**< Color set. */

53 PWCTR3 = 0xC2, /**< Power control 3. */

66 DRVTCTRA = 0xE8, /**< Driver timing control A. */
67 DRVTCTRB = 0xEA, /**< Driver timing control B. */
68 POWONSEQCTRL = 0xED, /**< Power on sequence control. */
69 ENBLABLE3G = 0xF2, /**< Enable 3G. */
70 PUMPCRTL = 0xF7 /**< Pump ratio control. */
