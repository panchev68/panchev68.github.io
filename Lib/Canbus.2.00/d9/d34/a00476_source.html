21 /** Moving average integrator for fixed-size window.
22 *
23 * The Integrator class template computes a moving average over a fixed-size window of arithmetic values.
24 * It uses a circular buffer to store the most recent values and provides methods to add values, compute
25 * the average, reset the window, and check if the window is full. The class is type-safe and suitable for
26 * embedded systems and real-time signal processing.
27 * @tparam T_VALUE Arithmetic type for values (e.g., int, float, double).
28 * @tparam T_WINDOW_SIZE Size of the moving average window (must be > 0).
29 */
30 template <typename T_VALUE, size_t T_WINDOW_SIZE> class Integrator {
31 public:
32 static_assert(T_WINDOW_SIZE > 0, "Window size must be greater than 0");
33 static_assert(std::is_arithmetic_v<T_VALUE>, "T_VALUE must be an arithmetic type");
34

38 /** Adds a value to the moving average window.
39 * @param value Value to add to the window.
40 */
41 void addValue(T_VALUE value) {
42 assert(T_WINDOW_SIZE > 0);
43 values[index] = value;
44 index = (index + 1 < T_WINDOW_SIZE) ? (index + 1) : 0;
45 if (index == 0) full = true;
46 }
47
48 /** Computes the moving average of the window.
49 * @return Moving average of the current window values.
50 */
51 T_VALUE computeResult() const {
52 assert(full || index > 0);
53 const size_t count = full ? T_WINDOW_SIZE : index;
54 const auto sum = std::accumulate(values.begin(), values.begin() + count, T_VALUE{0});
55 if (count == 0) return T_VALUE{0}; /* WARNING: Division by zero avoided. */
56 return sum / static_cast<T_VALUE>(count);
57 }

59 /** Adds a value and computes the moving average.
60 * @param value Value to add and process.
61 * @return Moving average after adding the value.
62 */
63 T_VALUE processValue(T_VALUE value) {
64 addValue(value);
65 return computeResult();
66 }
67
68 /** Adds a value and computes the moving average (function call operator).
69 * @param value Value to add and process.
70 * @return Moving average after adding the value.
71 */
72 T_VALUE operator()(T_VALUE value) { return processValue(value); }
73
74 /** Resets the moving average window. */
75 void resetWindow() {
76 index = 0;
77 full = false;
78 values.fill(T_VALUE{0});
79 }
80
81 /** Checks if the moving average window is full.
82 * @return `true` if the window is full, `false` otherwise.
83 */
84 bool isWindowFull() const { return full; }
85
86 private:

89 bool full{false}; /*!< Indicates if the window is full. */
