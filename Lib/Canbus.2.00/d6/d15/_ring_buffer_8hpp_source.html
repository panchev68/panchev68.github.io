17 /** Fixed-size ring buffer for FIFO storage.
18 *
19 * The RingBuffer class implements a circular buffer of fixed capacity for storing elements of type T.
20 * It provides efficient push and pop operations, as well as methods to check if the buffer is full or
21 * empty, retrieve the number of stored elements, and clear the buffer. This class is suitable for
22 * embedded and real-time systems where bounded, non-blocking data queues are required.
23 * @tparam T Type of elements stored in the buffer.
24 * @tparam T_SIZE Capacity of the buffer.
25 */
26 template <typename T, size_t T_SIZE> class RingBuffer {
27 public:

36 /** Pushes an element into the buffer.
37 * @param value Element to be added to the buffer.
38 * @return `true` if the element was added, `false` if the buffer is full.
39 */
40 bool push(const T & value) {
41 if (isFull())
42 return false;
43 buffer[writePos] = value;
44 writePos = (writePos + 1) % CAPACITY;
45 ++count;
46 return true;
47 }
48
49 /** Pops an element from the buffer.
50 * @param[out] value Element removed from the buffer.
51 * @return `true` if an element was removed, `false` if the buffer is empty.
52 */
53 bool pop(T & value) {
54 if (isEmpty())
55 return false;
56 value = buffer[readPos];
57 readPos = (readPos + 1) % CAPACITY;
58 --count;
59 return true;
60 }
61
62 /** Clears the buffer and resets its state. */
63 void clear() {
64 readPos = 0;
65 writePos = 0;
66 count = 0;
67 }
68
69 /** Returns the number of elements currently stored in the buffer.
70 * @return Number of elements in the buffer.
71 */
72 size_t available() const { return count; }
73
74 /** Checks if the buffer is empty.
75 * @return `true` if the buffer is empty, `false` otherwise.
76 */
77 bool isEmpty() const { return count == 0; }
78
79 /** Checks if the buffer is full.
80 * @return `true` if the buffer is full, `false` otherwise.
81 */
82 bool isFull() const { return count == CAPACITY; }
83
84 private:
