28 * Delays execution for the specified number of microseconds using the DWT cycle counter.

31 explicit DelayUs(uint32_t delay) {
32 constexpr uint32_t TICKS_PER_US = 1000000; /* Number of microseconds per second. */
33 const uint32_t hclkFreq = HAL_RCC_GetHCLKFreq(); /* Get system clock frequency in Hz. */
34 const uint32_t ticksPerUs = hclkFreq / TICKS_PER_US; /* Calculate ticks per microsecond. */
35 CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk; /* Enable DWT and trace. */
36 DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk; /* Enable cycle counter. */
37 DWT->CYCCNT = 0; /* Reset cycle counter. */
38 const uint32_t targetTicks = delay * ticksPerUs; /* Calculate target tick count for delay. */
39 const uint32_t startTicks = DWT->CYCCNT; /* Record starting tick count. */
40 while ((DWT->CYCCNT - startTicks) < targetTicks) {
41 /* Busy wait for the required number of cycles. */
42 }
43 }
