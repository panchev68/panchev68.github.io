154 /** Calculates and updates the connection quality.
155 * @return Updated quality state.
156 */
157 State update() {
158 int32_t result = 100;
159 int32_t value;
160 if (connection->state == Connection::State::CONNECTED) {
161 if (timeout.isExpired()) {
162 timeout.set(RESET_CYCLE_PERIOD);
163 lastRetriesCounter = connection->info.retriesCounter;
164 if (mode == Mode::RUNTIME) {
165 if (retriesSubst > 0) retriesSubst--;
166 if (transmitTimeSubst > 0) transmitTimeSubst--;
167 if (rssiSubst > 0) rssiSubst--;
168 }
169 }
170 if (connection->info.retriesCounter != lastRetriesCounter) {
171 if (connection->info.retriesCounter > lastRetriesCounter) {
172 retriesSubst += 50;
173 if (retriesSubst >= 100) retriesSubst = 100;
174 }
175 lastRetriesCounter = connection->info.retriesCounter;
176 }
177 if (lastTransmitPeriod != connection->info.transmitPeriod.getRawValue()) {
178 lastTransmitPeriod = connection->info.transmitPeriod.getRawValue();
179 if (connection->info.transmitPeriod > TRANSMIT_TIME_THRESHOLD) {
180 transmitTimeSubst += 5;
181 if (transmitTimeSubst > 100) transmitTimeSubst = 100;
182 }
183 }
184 value = connection->info.signalStrength;
185 value = ((lastRssi * 2) + value) / 3;
186 lastRssi = static_cast<uint8_t>(value);
187 if (value < RSSI_MIN_THRESHOLD) rssiSubst = (RSSI_MIN_THRESHOLD - value);
188 result -= retriesSubst;
189 result -= transmitTimeSubst;
190 result -= rssiSubst;
191 if (result < 0) result = 0;
192 percents = static_cast<uint8_t>(result);
193 if (percents > 90) state = State::AMAZING;
194 else if (percents > 75) state = State::VERY_GOOD;
195 else if (percents > 50) state = State::OKAY;
196 else if (percents > 30) state = State::NOT_GOOD;
197 else state = State::UNUSABLE;
198 if (onUpdate) onUpdate(this);
199 } else {
200 state = State::DISCONNECTED;
201 percents = 0;
202 }
203 return state;
204 }
205
206 private:
207 uint8_t percents; /*!< Quality in percents */
208 uint32_t lastRetriesCounter; /*!< Last value of retries counter */
209 uint32_t lastTransmitPeriod; /*!< Last transmit period */
210 int32_t rssiSubst; /*!< rssi subst */
211 int32_t retriesSubst; /*!< retries subst */
212 int32_t transmitTimeSubst; /*!< transmit time subst */
213 Mode mode; /*!< Connection Mode (runtime, scanner) */
214 uint8_t lastRssi; /*!< last rssi value */
215 RTOS::Timeout timeout; /*!< connection timeout */
216 State state; /*!< quality state */
217 Connection * connection; /*!< pointer to connection instance */
218
219 /** Sets the quality mode.
220 * @param mode Quality mode.
221 */
222 void setMode(Mode mode = Mode::RUNTIME) {
223 this->mode = mode;
224 clear();
225 }
226 };

301 /** Begins a measurement. */
302 void beginMeasurement() {
303 if (isStarted) return;
304 isStarted = true;
305 startTicks = RTOS::TickCounter::get();
306 }
307
308 /** Ends a measurement. */
309 void endMeasurement() {
310 if (!isStarted) return;
311 isStarted = false;
312 auto ticks = RTOS::TickCounter::get();
313 auto measuredValue = (ticks - startTicks);
314 if (value == 0) value = measuredValue;
315 if (std::abs(static_cast<int32_t>(measuredValue - rawValue)) >= 5) value = measuredValue;
316 else value = ((value * 2) + measuredValue) / 3;
317 rawValue = measuredValue;
318 if ((measuredValue < minimum) || (minimum == 0)) minimum = measuredValue;
319 if ((measuredValue > maximum) || (maximum == 0)) maximum = measuredValue;
320 }
321
322 /** Stops a measurement. */
323 void stopMeasurement() { isStarted = false; }
324
325 /** Resets the period counter. */
326 void reset() {
327 value = 0;
328 clearMinMax();
329 }

332 * @return Period value.
333 */
334 operator uint32_t() const { return value; }

336 private:
