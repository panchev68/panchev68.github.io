88 if (__HAL_RCC_GET_FLAG(RCC_FLAG_PINRST)) return State::EXTERNAL_RESET_PIN_RESET;
89 if (__HAL_RCC_GET_FLAG(RCC_FLAG_BORRST)) return State::BROWNOUT_RESET;
