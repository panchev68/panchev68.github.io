95 /** Assigns data from memory to a vector.
96 * @param[out] dst Destination vector.
97 * @param src Source memory pointer.
98 * @param length Number of bytes to assign.
99 */
100 static void assign(std::vector<uint8_t> & dst, const void * src, size_t length) {
101 if (src && length)
102 dst.assign(static_cast<const uint8_t *>(src), static_cast<const uint8_t *>(src) + length);
103 }
104
105 /** Appends data from memory to a vector.
106 * @param[out] dst Destination vector.
107 * @param src Source memory pointer.
108 * @param length Number of bytes to append.
109 */
110 static void append(std::vector<uint8_t> & dst, const void * src, size_t length) {
111 if (src && length)
112 dst.insert(dst.end(), static_cast<const uint8_t *>(src), static_cast<const uint8_t *>(src) + length);
113 }

115 /** Appends a value to a vector.
116 * @tparam T Type of the value.
117 * @param[out] dst Destination vector.
118 * @param value Value to append.
119 */
120 template <typename T>
121 static void append(std::vector<uint8_t> & dst, T value) {
122 for (size_t i = 0; i < sizeof(T); ++i)
123 dst.push_back(static_cast<uint8_t>(value >> (8 * (sizeof(T) - 1 - i))));
124 }
