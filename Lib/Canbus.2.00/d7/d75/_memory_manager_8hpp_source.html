52 /** Allocates memory from FreeRTOS heap.
53 * @param size Number of bytes to allocate.
54 * @return Pointer to allocated memory or nullptr if failed.
55 * @warning Returns nullptr if size is zero or exceeds heap.
56 */
57 static void * allocate(size_t size) {
58 if (!isValidSize(size)) {
59#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
60 diagnostic.failedAllocations++;
61#endif
62 assert(false);
63 return nullptr;
64 }
65 void * ptr = pvPortMalloc(size);
66 if (!ptr) {
67 handleAllocationFailure();
68#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
69 diagnostic.failedAllocations++;
70#endif
71 return nullptr;
72 }
73#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
74 diagnostic.allocated++;
75 diagnostic.active++;
76 diagnostic.currentUsage += size;
77 if (diagnostic.currentUsage > diagnostic.peakUsage) diagnostic.peakUsage = diagnostic.currentUsage;
78#endif
79 return ptr;
80 }
