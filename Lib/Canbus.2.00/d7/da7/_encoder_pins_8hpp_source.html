63 /** Reads debounced button state.
64 * @param debounce Debounce period in ms.
65 * @return `true` if pressed, `false` otherwise.
66 */
67 bool getButtonState(uint32_t debounce = 20) {
68 bool state = button.read();
69 for (uint32_t i = debounce; debounce > 0 && i > 0; --i) {
70 if (button.read() != state) {
71 state = button.read();
72 i = debounce;
73 }
74 vTaskDelay(1);
75 }
76 return !state;
77 }

79 /** Handles pin change event.
80 * @param pinNumber Pin number that triggered the event.
81 * @return `true` if handled, `false` otherwise.
82 */
83 bool handlePinChangedEvent(uint16_t pinNumber) {
84 static bool direction = false;
85 static bool ch1Changed = false;
86 static bool ch2Changed = false;
87 if (pinNumber == button.getPinNumber()) {
88 write(button.read() ? EventState::BUT_PULL : EventState::BUT_PUSH);
89 return true;
90 }
91 if (pinNumber == channel1.getPinNumber()) {
92 ch1Changed = true;
93 if (ch2Changed) {
94 ch2Changed = false;
95 write(direction == channel1.read() ? EventState::CCW : EventState::CW);
96 }
97 return true;
98 }
99 if (pinNumber == channel2.getPinNumber()) {
100 ch2Changed = true;
101 if (ch1Changed) {
102 ch1Changed = false;
103 direction = channel2.read();
104 }
105 return true;
106 }
107 return false;
108 }

133 /** Reads an event from the queue with timeout.
134 * @param timeout Timeout in ms.
135 * @return Event state.
136 */
137 EventState read(uint32_t timeout) const {
138 EventState result;
139 return isOpen() && queue.receive(result, timeout) ? result : EventState::NONE;
140 }
141
142 /** Writes an event state to the queue.
143 * @param state Event state.
144 */
145 void write(EventState state) {
146 if (!isOpen()) return;
147 if (state != lastState) {
148 lastState = state;
149 dividerCounter = 0;
150 if (state == EventState::BUT_PUSH)
151 buttonStartTime = xTaskGetTickCount();
152 else if (state == EventState::BUT_PULL)
153 buttonHoldPeriod = xTaskGetTickCount() - buttonStartTime;
154 }
155 if ((state == EventState::CCW || state == EventState::CW) && (++dividerCounter > pulseDivider)) {
156 dividerCounter = 0;
157 if (isReverse)
158 state = (state == EventState::CCW) ? EventState::CW : EventState::CCW;
159 queue.sendToBackFromISR(state);
160 }
161 }
