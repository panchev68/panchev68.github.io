namespace OBD2 { class Service { constexpr static inline uint32_t OBD2_SENDER_ID = 0x7df; constexpr static inline uint32_t OBD2_RECEIVER_ID = 0x7e8;

public: 

/// \brief Open OBD II Service
/// \return bool
bool Service::open ()
{
    // Open the driver...

    if (_driver->open () == false)
    {
        return false;
    }

    // Set the Filter...

    auto filter                 = _driver->createFilter ();
    filter.FilterFIFOAssignment = CAN_FilterFIFO0;
    filter.FilterBank           = 14;
    filter.FilterMode           = CAN_FILTERMODE_IDMASK;
    filter.FilterScale          = CAN_FILTERSCALE_32BIT;
    filter.FilterActivation     = CAN_FILTER_ENABLE;
    filter.SlaveStartFilterBank = 14;
    filter.setId (OBD2_RECEIVER_ID);

    _driver->setFilter (filter);

    // Start driver...
    if (_driver->start () == false)
    {
        return false;
    }

    // Create Service Tasks...
    if (_taskRxMessage.create ("OBD2_RX", [this] () { _taskFnRxMessage (); }) == false)
    {
        assert (0);
        return false;
    }

    if (_taskTxMessage.create ("OBD2_TX", [this] () { _taskFnTxMessage (); }) == false)
    {
        assert (0);
        return false;
    }
}

private: 

FreeRTOS::Task _taskRxMessage;
FreeRTOS::Task _taskTxMessage;
void _taskFnRxMessage ();
void _taskFnTxMessage ();
Canbus::Driver * _driver;

};
