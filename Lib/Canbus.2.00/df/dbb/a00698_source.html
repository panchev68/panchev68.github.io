139 void scan()
140 {
141 RxMessage msg; // rx message
142 std::array<Node, NUM_OF_CANBUS_CLIENT_NODES> nodes; // remote nodes array
143
144 // scan for remote nodes
145 for (auto i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; i++)
146 {
147 auto &node = nodes.at(i);
148 node.activity = false;
149 auto id = i + 1;
150
151 // send message with empty payload ...
152 if (send(id) == true)
153 {
154 // receive incoming message...
155 if (ServiceBase::receive(msg, RECEIVE_TIMEOUT))
156 {
157 if (msg.getId() == id)
158 node.activity = true;
159 }
160 }
161 }
162 postScanCallback(nodes);
163 }
