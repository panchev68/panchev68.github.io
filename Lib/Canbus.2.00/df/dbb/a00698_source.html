141 void scan ()
142 {
143 RxMessage msg; // rx message
144 std::array<Node, NUM_OF_CANBUS_CLIENT_NODES> nodes; // remote nodes array
145
146 // scan for remote nodes
147 for (auto i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; i++)
148 {
149 auto & node = nodes.at (i);
150 node.activity = false;
151 auto id = i + 1;
152
153 // send message with empty payload ...
154 if (send (id) == true)
155 {
156 // receive incoming message...
157 if (ServiceBase::receive (msg, RECEIVE_TIMEOUT))
158 {
159 if (msg.getId () == id)
160 node.activity = true;
161 }
162 }
163 }
164 handleScanCompletedEvent (nodes);
165 }
