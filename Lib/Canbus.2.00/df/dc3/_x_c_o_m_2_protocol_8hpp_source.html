53 /** Build a protocol frame from the given Frame object.
54 * @param frameIn Frame object containing command and payload.
55 * @return Encoded frame as a vector of bytes.
56 */
57 static std::vector<uint8_t> build(const Frame & frameIn) {
58 std::vector<uint8_t> frameRaw{START_DELIMITER};
59 frameRaw.push_back(static_cast<uint8_t>(frameIn.payload.size()));
60 frameRaw.push_back(frameIn.command);
61 frameRaw.insert(frameRaw.end(), frameIn.payload.begin(), frameIn.payload.end());
62
63 auto crc = calcChkSum(frameRaw);
64 frameRaw.push_back(static_cast<uint8_t>(crc & 0xFF));
65 frameRaw.push_back(static_cast<uint8_t>(crc >> 8));
66
67 std::vector<uint8_t> frameEscaped{START_DELIMITER};
68 for (size_t index = 1; index < frameRaw.size(); ++index) {
69 auto value = frameRaw[index];
70 if (isEsc(value)) {
71 frameEscaped.push_back(ESC_SUBST);
72 value ^= PRECODE_MASK;
73 }
74 frameEscaped.push_back(value);
75 }
76 return frameEscaped;
77 }

79 /** Parse an encoded frame and extract its contents.
80 * @param frameIn Input frame as a vector of bytes.
81 * @param[out] frameOut Output Frame object to store parsed command and payload.
82 * @return `true` if parsing was successful, `false` otherwise.
83 */
84 static bool parse(const std::vector<uint8_t> & frameIn, Frame & frameOut) {
85 if (frameIn.size() < sizeof(FrameHeader)) return false;
86 if (frameIn.at(offsetof(FrameHeader, startDelimiter)) != START_DELIMITER) return false;
87
88 std::vector<uint8_t> frameDecoded{START_DELIMITER};
89 bool isEscape = false;
90 for (auto it = frameIn.begin() + 1; it != frameIn.end(); ++it) {
91 auto value = *it;
92 if (isEscape) {
93 value ^= PRECODE_MASK;
94 isEscape = false;
95 } else if (value == ESC_SUBST) {
96 isEscape = true;
97 continue;
98 }
99 frameDecoded.push_back(value);
100 }
101
102 if (frameDecoded.size() < sizeof(FrameHeader) + 2) return false;
103 auto payloadLength = frameDecoded.at(offsetof(FrameHeader, lenght));
104 auto payloadOffset = sizeof(FrameHeader);
105 auto crcOffset = payloadLength + payloadOffset;
106 if (crcOffset != frameDecoded.size() - 2) return false;
107
108 auto crcCalc = calcChkSum(frameDecoded);
109 auto crcFrame = (static_cast<uint16_t>(frameDecoded.at(crcOffset + 1)) << 8) | frameDecoded.at(crcOffset);
110 if (crcCalc != crcFrame) return false;
111
112 frameOut.command = frameDecoded.at(offsetof(FrameHeader, command));
113 frameOut.payload.clear();
114 for (size_t i = 0, j = payloadOffset; i < payloadLength; ++i, ++j) {
115 frameOut.payload.push_back(frameDecoded.at(j));
116 }
117 if (frameIn.size() > 300) return false;
118 return true;
119 }
