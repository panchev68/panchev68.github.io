67 bool receive(std::vector<uint8_t> & buffer, uint32_t timeout) {
68 static QueueMessage msg;
69 if (queueMessage.receive(msg, timeout)) {
70 buffer.assign(msg.data.begin(), msg.data.begin() + msg.size);

76 /** Send a remote AT command with parameters.
77 * @param command AT Command String.
78 * @param address Destination address.

84 sendFrame.atCommand = command;
85 sendFrame.frameId = 0x17;
86 sendFrame.dstAddress = address;

89 queueMessage.reset();

112 /** Handle received frame event.
113 * @param frame Reference to the received frame.
114 */
115 void handleReceivedFrameEvent(const Frame0x97 & frame) {
116 static QueueMessage msg;
117 if ((frame.frameId == 0x17) && (sendFrame.atCommand == frame.atCmd)) {
118 if ((frame.status != 0) && (frame.status != 4)) assert(false);
119 else {
120 msg.size = frame.parameters.size();
121 assert(msg.size < msg.DATA_SIZE);
122 std::memcpy(msg.data.data(), frame.parameters.data(), msg.size);
123 if (!queueMessage.send(msg, SEND_TIMEOUT)) assert(false);
124 }
125 }
