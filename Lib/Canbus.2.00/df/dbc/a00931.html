/**
* @file Server.cpp
* @brief Implementation of the Canbus Server class.
*
* This file contains the implementation of the Canbus::Server class, which
* inherits from Lib::Canbus::ServerBase. The Server class is responsible for
* managing CAN bus communication using FreeRTOS for task management.
*
* @note This file is part of the Canbus library examples.
*
* @version 2.00
* @date 2023-10-01
*
*/
#pragma once
#include <Canbus/ServerBase.hpp>
#include <FreeRTOS.hpp>
namespace Canbus {
/**
* @brief Canbus Server class.
*
* This class is responsible for managing CAN bus communication using FreeRTOS
* for task management.
*
*/
class Server : Lib::Canbus::ServerBase {
/**
* @brief QueueDataRxMessage struct.
*
* This struct defines the data structure for received messages in the
* Server class.
*
*/
struct QueueDataRxMessage {
uint8_t id; ///< The message ID.
uint64_t data [8]; ///< The message data.
size_t len; //< The message length.
};
FreeRTOS::Queue<QueueDataRxMessage> _queueMessageReceived;
public:
/**
* @brief Default constructor for the Server class.
*
* This constructor initializes the Server object with default values.
*
*/
Server ();
/**
* @brief Destructor for the Server class.
*
* This destructor releases the Server object and its resources.
*
*/
~Server ();
/**
* @brief Starts the server.
*
* This function starts the server by creating the task and initializing
* the CAN bus communication.
*
* @return true if the server was successfully started, false otherwise.
*/
bool start () override;
/**
* @brief Stops the server.
*
* This function stops the server by deleting the task and releasing the
* CAN bus communication resources.
*
* @return true if the server was successfully stopped, false otherwise.
*/
bool stop () override;
/**
* @brief Sends a message with the specified ID.
*
* This function sends a message with the given ID and waits for the specified
* timeout period.
*
* @param id The identifier of the message to be sent.
* @param timeout The timeout period to wait for the message to be sent. Default is SEND_TIMEOUT.
* @return true if the message was sent successfully, false otherwise.
*/
bool send (uint8_t id, uint32_t timeout = SEND_TIMEOUT) override;
bool send (uint8_t id, const void * data, size_t len, uint32_t timeout = SEND_TIMEOUT) override;
/**
* @brief Sends a message with the specified ID and value.
*
* This function sends a message with the given ID and value, and waits for the specified
* timeout period.
*
* @param id The identifier of the message to be sent.
* @param value The value of the message to be sent.
* @param timeout The timeout period to wait for the message to be sent. Default is SEND_TIMEOUT.
* @return true if the message was sent successfully, false otherwise.
*/
bool send (uint8_t id, uint64_t value, uint32_t timeout = SEND_TIMEOUT) override;
/**
* @brief Retrieves the diagnostic information.
*
* This function retrieves the diagnostic information for the Server object.
*
* @return const Diagnostic& A constant reference to the Diagnostic object.
*/
void handleScanCompletedEvent (const Nodes & nodes) override;
/**
* @brief Pure virtual function that defines the task to be executed.
*
* This function must be overridden by any derived class to implement
* the specific task functionality.
*/
void taskFn () override;
};
} // namespace Canbus

/// \file
/// \brief Canbus Server class
/// \date 31.01.2024
#include "Server.hpp"
#include "can.h"
using ServerBase = Lib::Canbus::ServerBase;
namespace Canbus
{
/// \brief Data of Rx Message Queue
struct Server::QueueDataRxMessage
{
uint8_t id;
uint64_t data [8];
size_t len;
};
// Service constructor
Server::Server () : ServerBase (&hcan1)
{
}
// Server destructor
Server::~Server ()
{
}
// Start the Server
bool Server::start ()
{
_queueMessageReceived.create (1, sizeof (Server::QueueDataRxMessage));
return ServerBase::start ();
}
// Stop the Server
bool Server::stop ()
{
return ServerBase::stop ();
}
// Sent the Echo (empty message)
bool Server::send (uint8_t id, uint32_t timeout)
{
return ServerBase::send (id, timeout);
}
// Send the memory buffer
bool Server::send (uint8_t id, const void * data, size_t len, uint32_t timeout)
{
return ServerBase::send (id, data, len, timeout);
}
// Send `uint64_t` value
bool Server::send (uint8_t id, uint64_t value, uint32_t timeout)
{
return ServerBase::send (id, value, timeout);
}
// Post Scan Callback
void Server::postScanCallback (const std::array<Node, NUM_OF_CANBUS_CLIENT_NODES> & nodes)
{
if (nodes.at (0).activity == true)
{
__NOP ();
}
}
// Task Function
void Server::taskFn ()
{
FreeRTOS::Timeout timeout;
ServerBase::RxMessage rxMessage;
if (send (0x01, 0xaaaa, 100) == false) assert (0);
while (1)
{
if (ServerBase::receive (rxMessage, 1) == true)
{
if (rxMessage.getLen () == 8)
{
__NOP ();
}
}
if (timeout.check ())
{
timeout.set (100);
for (auto i = 1; i < 8; i++)
{
if (send (i) == false)
{
assert (0);
}
}
}
}
}
} // namespace Canbus
