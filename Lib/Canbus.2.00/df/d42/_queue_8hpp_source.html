1/** Queue abstraction for FreeRTOS queue management.
2 * This file provides a C++17 template class for FreeRTOS queue creation, control, and resource management.
3 * The Queue class encapsulates queue lifecycle, ownership, and basic operations for sending, receiving,
4 * peeking, and overwriting queue elements in both task and ISR contexts. It supports move semantics,
5 * safe destruction, and provides APIs for registry management. The class ensures proper resource management
6 * and error handling using RAII principles.
7 * @file Queue.hpp
8 */

29 /** Queue abstraction for FreeRTOS queue management.
30 * The Queue class provides RAII management for FreeRTOS queues, supporting creation, destruction,
31 * element send/receive/peek/overwrite, and registry operations. It ensures safe resource handling
32 * and move semantics.
33 * @tparam T Element type (must be trivially copyable).
34 * @tparam QUEUE_LENGTH Number of elements in the queue.
35 */
36 template <typename T, size_t QUEUE_LENGTH = 1> class Queue {
37 static_assert(std::is_trivially_copyable<T>::value, "Queue element type must be trivially copyable");
38 static_assert(QUEUE_LENGTH > 0, "Queue length must be greater than zero");
