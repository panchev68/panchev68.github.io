209 /**
210 * @brief Scans the CAN bus for active nodes.
211 *
212 * This function iterates through all possible CAN bus client nodes, sending a
213 * message to each node and checking for a response. If a response is received
214 * from a node, it is marked as active. Once the scan is complete, the
215 * handleScanCompletedEvent function is called with the results.
216 *
217 * @details
218 * - The function creates an RxMessage object to store received messages and a
219 * Nodes object to keep track of active nodes.
220 * - It iterates through all possible nodes (from 0 to NUM_OF_CANBUS_CLIENT_NODES - 1).
221 * - For each node, it sends a message and waits for a response within a
222 * specified timeout (RECEIVE_TIMEOUT).
223 * - If a response is received and the message ID matches the node ID, the node
224 * is marked as active.
225 * - After scanning all nodes, the handleScanCompletedEvent function is called
226 * with the Nodes object containing the scan results.
227 */
228 void scan () {
229 RxMessage msg;
230 Nodes nodes;
231
232 for (auto i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; i++) {
233 auto bit (1 << i);
234
235 nodes.reset (bit);
236 auto id = i + 1;
237
238 if (send (id) == true) {
239 if (ServiceBase::receive (msg, RECEIVE_TIMEOUT)) {
240 if (msg.getId () == id)
241 nodes.set (bit);
242 }
243 }
244 }
245 handleScanCompletedEvent (nodes);
246 }
