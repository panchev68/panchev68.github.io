73 /** Starts the tick service with the specified interval.
74 * @param intervalMs Tick interval in milliseconds (default: 1).
75 * @return `true` if started successfully, `false` otherwise.
76 * @pre intervalMs > 0 && intervalMs <= 1000
77 */
78 bool start(uint32_t intervalMs = 1) {
79 assert(intervalMs > 0);
80 assert(intervalMs <= 1000);
81 if (isRunning()) return true;
82 tickInterval = intervalMs;
83 auto taskFunction = [this]() {
84 while (true) {
85 auto startTime = TickCounter::get();
86 {
87 MutexGuard guard(mutex);
88 for (auto * client : clients)
89 if (client) client->handleTickEvent();
90 }
91 auto elapsed = TickCounter::getElapsedTicks(startTime, TickCounter::get());
92 auto targetTicks = TickCounter::msToTicks(tickInterval);
93 if (elapsed < targetTicks) Task::delayTicks(targetTicks - elapsed);
94 }
95 };
96 auto handle = task.start(taskFunction);
97 return handle != nullptr;
98 }

169 /** Registers a client for tick events.
170 * @param client Pointer to the client to register.
171 * @return `true` if registration succeeded, `false` otherwise.
172 * @pre client != nullptr

174 bool registerClient(ClientBase * client) {
175 assert(client != nullptr);
176 if (!client) return false;
177 MutexGuard guard(mutex);
178 if (std::find(clients.begin(), clients.end(), client) == clients.end()) {
179 clients.push_back(client);
180 return true;
181 }
182 return false;
183 }

185 /** Unregisters a client from tick events.
186 * @param client Pointer to the client to unregister.
187 * @return `true` if unregistration succeeded, `false` otherwise.
188 * @pre client != nullptr
189 */
190 bool unregisterClient(ClientBase * client) {
191 assert(client != nullptr);
192 if (!client) return false;
193 MutexGuard guard(mutex);
194 auto it = std::find(clients.begin(), clients.end(), client);
195 if (it != clients.end()) {
196 clients.erase(it);
197 return true;
198 }
