64 /** SPI full-duplex transfer.
65 * @param txData Source buffer.
66 * @param txSize Source length.
67 * @param rxData Destination buffer.
68 * @param rxSize Destination length.
69 * @return `true` if successful, `false` otherwise.
70 * @pre txData and rxData must be valid pointers if txSize/rxSize > 0.
71 * @post Data is transferred; rxData contains received bytes.
72 */
73 bool readWrite(const void * txData, size_t txSize, void * rxData, size_t rxSize) {
74 size_t totalSize = txSize + rxSize;
75 auto tempTxBuffer = std::make_unique<uint8_t[]>(totalSize);
76 auto tempRxBuffer = std::make_unique<uint8_t[]>(totalSize);
77 std::memset(tempTxBuffer.get(), 0xFF, totalSize);
78 if (txData && txSize) std::memcpy(tempTxBuffer.get(), txData, txSize);
79 bool ok = spi.sendReceive(tempTxBuffer.get(), tempRxBuffer.get(), totalSize, SPI_TIMEOUT);
80 if (ok && rxData && rxSize) std::memcpy(rxData, tempRxBuffer.get() + txSize, rxSize);
81 assert(ok);
82 taskYIELD();
83 return ok;
84 }

86 /** Writes data to device.
87 * @param srcData Source data pointer.
88 * @param dataSize Data length.
89 * @return `true` if successful, `false` otherwise.
90 * @pre srcData must be valid pointer and dataSize > 0.
91 * @post Data is written to device.
92 */
93 bool write(const void * srcData, size_t dataSize) {
94 assert(srcData && dataSize);
95 bool ok = spi.send(srcData, dataSize, 10);
96 assert(ok);
