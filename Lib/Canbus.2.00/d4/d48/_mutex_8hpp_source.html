23 /** C++ RAII wrapper for FreeRTOS mutex.
24 * Manages lifecycle, safe creation, destruction, locking, unlocking, and status queries.
25 * Supports both normal and recursive mutex types for STM32/FreeRTOS systems.
26 */
27 class Mutex {
28 public:
29 static constexpr uint32_t NO_WAIT = 0; /*!< No wait timeout for lock operations. */
30 static constexpr uint32_t WAIT_FOREVER = portMAX_DELAY; /*!< Infinite wait timeout for lock operations. */
31
32 /** Mutex type enumeration.
33 * Specifies whether the mutex is normal or recursive.
34 */
35 enum class MutexType : uint8_t {
36 NORMAL = 0, /*!< Standard mutex type. */
37 RECURSIVE = 1 /*!< Recursive mutex type. */
38 };

40 /** Scoped lock guard for Mutex.
41 * Acquires and releases mutex lock automatically using RAII.
42 * Prevents deadlocks and resource leaks by unlocking on destruction.
43 */
44 class LockGuard {
45 public:
46 /** Constructor of LockGuard class.
47 * Acquires the mutex lock with optional timeout.
48 * @param mutex Mutex to lock.
49 * @param timeout Timeout for lock operation.
50 */
51 explicit LockGuard(Mutex & mutex, uint32_t timeout = WAIT_FOREVER)
52 : mutex(mutex), lockAcquired(mutex.lock(timeout)) {}
53
54 /** Destructor of LockGuard class.
55 * Releases the mutex lock if acquired.
56 */
57 ~LockGuard() {
58 if (lockAcquired) mutex.unlock();
59 }
60
61 /** Deleted copy constructor of LockGuard class. */
62 LockGuard(const LockGuard &) = delete;
63 /** Deleted copy assignment operator of LockGuard class. */
64 LockGuard & operator=(const LockGuard &) = delete;
65 /** Deleted move constructor of LockGuard class. */
66 LockGuard(LockGuard &&) = delete;
67 /** Deleted move assignment operator of LockGuard class. */
68 LockGuard & operator=(LockGuard &&) = delete;
69
70 /** Checks if lock was acquired.
71 * @return `true` if lock acquired, `false` otherwise.
72 */
73 bool isLocked() const { return lockAcquired; }
74
75 /** Conversion to bool for lock status.
76 * @return `true` if lock acquired, `false` otherwise.
77 */
78 explicit operator bool() const { return lockAcquired; }
79
80 private:
81 Mutex & mutex; /*!< Reference to associated mutex. */
82 const bool lockAcquired; /*!< Indicates if lock was acquired. */
83 };

96 /** Creates the mutex.
97 * Allocates a FreeRTOS mutex of the specified type.
98 * @return `true` if created, `false` otherwise.
99 * @pre Mutex not already created.
100 * @post Mutex is created and handle is valid if return is `true`.
101 */
102 bool create() {
103 if (handle) return false;
104 if (mutexType == MutexType::RECURSIVE) {
105#if configUSE_RECURSIVE_MUTEXES == 1
106 handle = xSemaphoreCreateRecursiveMutex();
107#else
108 assert(false);
109 handle = nullptr;
110#endif
111 } else {
112#if configUSE_MUTEXES == 1
113 handle = xSemaphoreCreateMutex();
114#else
115 assert(false);
116 handle = nullptr;
117#endif
118 }
119 assert(handle != nullptr);
120 return handle != nullptr;
121 }
122
123 /** Destroys the mutex.
124 * Deletes the FreeRTOS mutex and releases associated resources.
125 * @post Mutex is destroyed and handle is null.
126 */
127 void destroy() {
128 if (handle) {
129 vSemaphoreDelete(handle);
130 handle = nullptr;
131 }
132 }

160 /** Locks the mutex with optional timeout.
161 * Acquires the mutex lock, blocking for up to timeout ticks.
162 * @param timeout Timeout in FreeRTOS ticks.
163 * @return `true` if lock acquired, `false` otherwise.
164 * @pre Mutex is created.
165 * @post Mutex is locked if return is `true`.
166 */
167 bool lock(uint32_t timeout = WAIT_FOREVER) {
168 if (!handle) {
169 assert(false);
170 return false;
171 }
172 const TickType_t ticks = (timeout == WAIT_FOREVER) ? portMAX_DELAY : pdMS_TO_TICKS(timeout);
173 if (mutexType == MutexType::RECURSIVE) {
174#if configUSE_RECURSIVE_MUTEXES == 1
175 return xSemaphoreTakeRecursive(handle, ticks) == pdTRUE;
176#else
177 assert(false);
178 return false;
179#endif
180 } else {
181#if configUSE_MUTEXES == 1
182 return xSemaphoreTake(handle, ticks) == pdTRUE;
183#else
184 assert(false);
185 return false;
186#endif
187 }
188 }
189
190 /** Attempts to lock the mutex without waiting.
191 * Acquires the mutex lock if available, returns immediately.
192 * @return `true` if lock acquired, `false` otherwise.
193 * @pre Mutex is created.
194 * @post Mutex is locked if return is `true`.
195 */
196 bool tryLock() { return lock(NO_WAIT); }
197
198 /** Unlocks the mutex.
199 * Releases the mutex lock.
200 * @pre Mutex is locked.
201 * @post Mutex is unlocked.
202 */
203 void unlock() {
204 if (!handle) return;
205 if (mutexType == MutexType::RECURSIVE) {
206#if configUSE_RECURSIVE_MUTEXES == 1
207 xSemaphoreGiveRecursive(handle);
208#else
209 assert(false);
210#endif
211 } else {
212#if configUSE_MUTEXES == 1
213 xSemaphoreGive(handle);
214#else
215 assert(false);
216#endif
217 }
218 }
219
220 /** Checks if mutex is valid.
221 * Determines if the mutex handle is valid.
222 * @return `true` if valid, `false` otherwise.
223 */
224 bool isValid() const { return handle != nullptr; }
225
226 /** Gets native FreeRTOS mutex handle.
227 * Returns the underlying FreeRTOS semaphore handle.
228 * @return Native FreeRTOS handle.
229 */
230 SemaphoreHandle_t getNativeHandle() const { return handle; }
231
232 /** Gets mutex type.
233 * Returns the type of mutex (normal or recursive).
234 * @return Mutex type.
235 */
236 MutexType getType() const { return mutexType; }
237
238 private:
239 MutexType mutexType; /*!< Mutex type (normal or recursive). */
240 SemaphoreHandle_t handle; /*!< FreeRTOS semaphore handle. */
241 };
