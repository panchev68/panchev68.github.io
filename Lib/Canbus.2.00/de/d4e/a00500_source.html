35 /** Swaps the byte order of a memory buffer.
36 * @param src Pointer to the source buffer.
37 * @param dst Pointer to the destination buffer.
38 * @param size Length of the buffer in bytes.
39 */
40 inline void swapBytes(const void * src, void * dst, size_t size) {
41 const auto * srcPtr = static_cast<const uint8_t *>(src);
42 auto * dstPtr = static_cast<uint8_t *>(dst);
43 for (size_t index = 0; index < size; ++index) dstPtr[index] = srcPtr[size - 1 - index];
44 }
