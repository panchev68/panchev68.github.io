17 /** Kalman filter for generic numeric types.
18 * The Kalman class implements a simple Kalman filter for smoothing and estimating sensor values.
19 * It supports configuration of process noise, sensor noise, and initial error, and provides
20 * methods for filtering, resetting, and parameter adjustment. Internal calculations use double

27 /** Configuration parameters for Kalman filter.
28 * The Config struct holds process noise, sensor noise, estimated error, and initial value.
29 * It allows fine-tuning of filter behavior for different sensor and signal characteristics.

31 struct Config {
32 double processNoise = 0.00001; /*!< Process noise covariance. */
33 double sensorNoise = 0.004; /*!< Sensor noise covariance. */
34 double estimatedError = 1.0; /*!< Initial estimation error. */

70 /** Filter a new value and return the estimated value.
71 * @param value New sensor or input value to filter.
72 * @return Filtered value estimate.
73 * @pre Value must be convertible to double.
74 * @post Internal state is updated with new filtered value.
75 */
76 virtual T operator()(T value) {
77 estimatedError += processNoise;
78 const double gain = estimatedError / (estimatedError + sensorNoise);
79 filteredValue += gain * (static_cast<double>(value) - filteredValue);
80 estimatedError *= (1.0 - gain);
81 return static_cast<T>(filteredValue);
82 }
