124 /** Gets current input value.
125 * @return Current input value.

130 * @return Number of input changes detected.

134 /** Pre-update callback for SPI configuration. */
135 void preUpdateCallback() override {
136 auto handle = spiPort.getHandle();
137 if (!handle) return;
138 HAL_SPI_DeInit(handle);
139 handle->Init.BaudRatePrescaler = SPI_PRESCALER;
140 handle->Init.CLKPolarity = SPI_POLARITY_LOW;
141 handle->Init.CLKPhase = SPI_PHASE_1EDGE;
142 spiPort.init();
143 }
144
145 /** Updates input value from SPI.
146 * Reads SPI data, updates value, and triggers change callback.
147 */
148 void update() override {
149 nssPin.write(0);
150 if (!spiPort.receive(buffer.get(), bufferSize, INPUT_SPI_TIMEOUT)) {
151 nssPin.write(1);
152 return;
153 }
154 nssPin.write(1);
155 RTOS::Task::yield();
156 uint64_t newValue = 0;
157 for (size_t i = 0; i < pinCount; ++i) {
158 const auto & map = mappingFields[i];
159 if ((buffer[map.port] & map.mask) != 0) newValue |= (1ULL << i);
160 }
161 if (value != newValue) {
162 value = newValue;
163 ++changesCounter;
164 if (onValueChanged) onValueChanged(value);
165 }
166 }
167
168 /** Gets SPI handle.
169 * @return SPI handle pointer.
170 */
171 SPI_HandleTypeDef * getSpiHandle() const override { return spiPort.getHandle(); }
172
173 /** Callback for input value change. */
174 std::function<void(const uint64_t &)> onValueChanged;
175 };

281 handle->Init.CLKPolarity = SPI_POLARITY_LOW;
282 handle->Init.CLKPhase = SPI_PHASE_2EDGE;
283 spiPort.init();
284 }

286 /** Updates output value and status.
287 * Handles SPI transfer, buffer update, and status change detection.
288 */
289 void update() override {
290 bool needTransfer = false;
291 if (isValueChanged) {
292 isValueChanged = false;
293 updateTxBuffer();
294 timeout.set(UPDATE_TIMEOUT_MS);
295 ++changesCounter;
296 needTransfer = true;
297 }
298 if (!needTransfer && timeout.isExpired()) {
299 timeout.set(UPDATE_TIMEOUT_MS);
300 needTransfer = true;
301 }
302 if (!needTransfer) return;
303 performSpiTransfer();
304 processRxBuffer();
305 }
306
307 /** Callback for output value change. */
308 std::function<void(const uint64_t &)> onValueChanged;
309 /** Callback for output status change. */
310 std::function<void(const uint64_t &)> onStatusChanged;
311 /** Callback for SPI error event. */
312 std::function<void()> onSpiError;
313
314 private:
315 /* Updates transmit buffer based on output value. */
316 void updateTxBuffer() {
317 auto tx = txBuffer.get() + 1;
318 for (size_t i = 0; i < pinCount; ++i) {
319 const auto & map = mappingFields[i];
320 auto & val = tx[map.port + offsetof(MC33879::TxPort, value)];
321 if ((value & (1ULL << i)) != 0) val |= map.mask;
322 else val &= static_cast<uint8_t>(~map.mask);
323 }
324 }
325
326 /* Performs SPI transfer for output update. */
327 void performSpiTransfer() {
328 txBuffer[0] = diagCounter;
329 nssPin.write(0);
330 const bool spiSuccess = spiPort.sendReceive(txBuffer.get(), rxBuffer.get(), bufferSize, OUTPUT_SPI_TIMEOUT);
331 nssPin.write(1);
332 RTOS::Task::yield();
333 if (!spiSuccess) return;
334 if (diagCounter == 0) oePin.write(1);
335 if (++diagCounter == 0) diagCounter = 1;
336 }
337
338 /* Processes received SPI buffer and updates output status. */
339 void processRxBuffer() {
340 if ((rxBuffer[bufferSize - 1] != txBuffer[0]) && diagCounter > 0 && !isSpiError) {
341 isSpiError = true;
342 if (onSpiError) onSpiError();
343 }
344 uint64_t newStatus = 0;
345 for (size_t i = 0; i < pinCount; ++i) {
346 const auto & map = mappingFields[i];
347 if ((rxBuffer[map.port + offsetof(MC33879::RxPort, status)] & map.mask) != 0) newStatus |= (1ULL << i);
348 }
349 if (status != newStatus) {
350 status = newStatus;
351 if (onStatusChanged) onStatusChanged(status);
352 }
353 }
354 };

381 virtual void setup() {

383 outputs.setup();
384 const bool useSequentialMode = (inputs.getSpiHandle() == outputs.getSpiHandle());
385 if (useSequentialMode) {

401 virtual void handlePostUpdateEvent() {}

403 private:
404 /* RTOS task function for update loop. */
405 template <UpdateMode MODE> void taskFn() {
406 if constexpr (MODE == UpdateMode::SEQUENTIAL) {
407 while (true) {
408 handlePreUpdateEvent();
409 inputs.preUpdateCallback();
410 inputs.update();
411 outputs.preUpdateCallback();
412 outputs.update();
413 handlePostUpdateEvent();
414 RTOS::Task::delay(TASK_LOOP_DELAY);
415 }
416 } else {
417 inputs.preUpdateCallback();
418 outputs.preUpdateCallback();
419 while (true) {
420 handlePreUpdateEvent();
421 inputs.update();
422 outputs.update();
423 handlePostUpdateEvent();
424 RTOS::Task::delay(TASK_LOOP_DELAY);
425 }
426 }
427 }
