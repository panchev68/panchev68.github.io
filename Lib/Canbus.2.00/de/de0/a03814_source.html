114 /** Scans the CAN bus for active client nodes.
115 * Sends a message to each possible client node and checks for a response.
116 * Calls handleScanCompletedEvent with the discovered nodes.
117 */
118 void scan() {
119 RxMessage msg;
120 Nodes nodes;
121 for (uint8_t i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; ++i) {
122 nodes.reset(i);
123 uint8_t id = i + 1;
124 if (send(id)) {
125 if (ServiceBase::receive(msg, RECEIVE_TIMEOUT)) {
126 if (msg.getId() == id) nodes.set(i);
127 }
128 }
129 }
130 handleScanCompletedEvent(nodes);
131 }
