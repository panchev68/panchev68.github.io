67 static uint32_t stringToU32(const std::string & versionString) {
68 std::array<uint8_t, 2> versionArray = {0, 0};
69 size_t begin = 0;
70 for (size_t i = 0; i < versionArray.size(); ++i) {
71 size_t found = versionString.find('.', begin);
72 if (found == std::string::npos) found = versionString.size();
73 versionArray[i] = static_cast<uint8_t>(std::stoul(versionString.substr(begin, found - begin)));
74 begin = found + 1;
75 if (found == versionString.size()) break;
76 }
77 return (static_cast<uint32_t>(versionArray[0]) << 8) | versionArray[1];
