47 /** Starts the service with the given identifier.
48 * Initializes filter, opens port, and starts the task.
49 * @param serviceId Service identifier.
50 * @return `true` if started successfully, `false` otherwise.
51 */
52 bool start(uint8_t serviceId) {
53 id = serviceId;
54 Filter filter{serviceId};
55 filter.setId(0);
56 filter.setMask(0);
57 filter.setBank(0);
58 if (!queueRxMessage.create() || !PortBase::setFilter(filter) || !PortBase::open()) {
59 assert(!"ServiceBase initialization failed");
60 return false;
61 }
62 return task.start([this] { taskFn(); });
63 }

72 }

88 /** Sends a message with data.

95 bool send(uint8_t messageId, const void * data, size_t length, uint32_t timeout = SEND_TIMEOUT) override {

104 /** Sends a 64-bit value.

110 bool send(uint8_t messageId, uint64_t value, uint32_t timeout = SEND_TIMEOUT) {
