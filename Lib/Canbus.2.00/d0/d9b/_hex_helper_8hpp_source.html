39 static constexpr uint32_t toBcd(uint16_t value) {
40 uint32_t result = 0;
41 uint8_t index = 0;
42 while (value) {
43 /* Extract the least significant decimal digit and place it in BCD format. */
44 result |= ((value % 10) << ((index % 2) ? 4 : 0)) << (index / 2 * 8);
45 value /= 10;
46 ++index;
47 }
