24 /** STM32 RTC peripheral device abstraction.
25 * Provides thread-safe access to the STM32 RTC peripheral, supporting date/time read/write and backup register
26 * operations. Integrates with RTOS mutex and backup SRAM domain for safe concurrent access.
27 */

41 /** Initialize RTC device and enable backup SRAM. */
42 void setup() {
43 assert(!mutex.isCreated() /* Mutex already created */);
44 if (!mutex.create()) assert(0 /* Failed to create mutex */);
45 BKPSRAM::Domain::getInstance().enable();
46 }

48 /** Release device resources. */
49 void release() { mutex.destroy(); }

51 /** Read current date and time from RTC.
52 * @return DateTime structure with current date and time.
53 */
54 DateTime read() {
55 RTC_TimeTypeDef time{};
56 RTC_DateTypeDef date{};
57 MutexGuard guard(mutex);
58 if (HAL_RTC_GetTime(getHandle(), &time, FORMAT_BIN) != HAL_OK) assert(0 /* Failed to get RTC time */);
59 if (HAL_RTC_GetDate(getHandle(), &date, FORMAT_BIN) != HAL_OK) assert(0 /* Failed to get RTC date */);
60 return {{date.Date, date.Month, date.Year + DateTime::Date::FIRST_YEAR},
61 {time.Hours, time.Minutes, time.Seconds}};
62 }
63
64 /** Write date and time to RTC.
65 * @param dateTime DateTime structure to write.
66 */
67 void write(const DateTime & dateTime) {
68 RTC_TimeTypeDef time{};
69 RTC_DateTypeDef date{};
70 time.Hours = dateTime.time.hours;
71 time.Minutes = dateTime.time.minutes;
72 time.Seconds = dateTime.time.seconds;
73 date.Date = dateTime.date.day;
74 date.Month = dateTime.date.month;
75 date.Year = dateTime.date.year - DateTime::Date::FIRST_YEAR;
76 MutexGuard guard(mutex);
77 if (HAL_RTC_SetTime(getHandle(), &time, FORMAT_BIN) != HAL_OK) assert(0 /* Failed to set RTC time */);
78 if (HAL_RTC_SetDate(getHandle(), &date, FORMAT_BIN) != HAL_OK) assert(0 /* Failed to set RTC date */);
79 }
80
81 /** Read RTC backup register.
82 * @param index Register index.
83 * @param[out] value Output value.
84 * @return `true` if successful, `false` if index out of range.
85 */
86 bool readRegister(uint32_t index, uint32_t & value) {
87 if (index >= NUMBER_OF_REGISTERS) return false;
88 MutexGuard guard(mutex);
89 value = HAL_RTCEx_BKUPRead(getHandle(), index);
90 return true;
91 }
92
93 /** Write RTC backup register.
94 * @param index Register index.
95 * @param value Value to write.
96 * @return `true` if successful, `false` if index out of range.
97 */
98 bool writeRegister(uint32_t index, uint32_t value) {
99 if (index >= NUMBER_OF_REGISTERS) return false;
100 MutexGuard guard(mutex);
101 HAL_RTCEx_BKUPWrite(getHandle(), index, value);
102 return true;
103 }
104
105 private:
106 RTOS::Mutex mutex; /**< RTOS mutex for thread safety. */
