39 /** Dimmer for smooth backlight fading and sleep-based dimming.
40 * Manages gradual transitions between brightness levels and automatic dimming after inactivity.

42 class Dimmer {

49 explicit Dimmer(Backlight * parent)

55 /** Restart dimmer to maximum value and reset sleep timeout.
56 * @note Only restarts if enabled.

58 void restart() {
59 if (!isEnabled) return;
60 parent->value = maxValue;
61 timeoutSleep.set(config.sleepTime);
62 }

65 void start() {
66 isEnabled = true;
67 restart();

74 Backlight * parent; /*!< Pointer to parent Backlight instance. */

76 RTOS::Timeout timeout; /*!< Timeout for fade step. */
77 RTOS::Timeout timeoutSleep; /*!< Timeout for sleep dimming. */
78 Config & config; /*!< Reference to Backlight configuration. */
79 uint8_t value; /*!< Current dimmer value. */
80 uint8_t minValue; /*!< Minimum dimmer value. */
81 uint8_t maxValue; /*!< Maximum dimmer value. */
82
83 /** Handle periodic tick event for fading and sleep dimming.
84 * Updates brightness smoothly and triggers sleep dimming when timeout expires.
85 */
86 void handleTickEvent() {
87 if (timeout.isExpired()) {
88 timeout.set(config.fadeStep);
89 if (value < parent->value) *parent->timerCCR = ++value;
90 else if (value > parent->value) *parent->timerCCR = --value;
91 }
92 if (isEnabled && timeoutSleep.isExpired()) {
93 timeoutSleep.disable();
94 parent->set(minValue);
95 }
96 }
97 };

99 /** Constructor of Backlight class.

103 explicit Backlight(const Config & config)

139 /** Resolve timer CCR register address for given channel.
140 * @param config Backlight configuration.
141 * @return Pointer to timer CCR register.
142 * @pre config.timChannel must be a valid TIM_CHANNEL_x constant.
143 * @post Returns non-null pointer for valid channel, nullptr otherwise.
144 */
145 volatile static uint32_t * resolveTimerCCR(const Config & config) {
146 switch (config.timChannel) {
147 case TIM_CHANNEL_1:
148 return &config.timHandle->Instance->CCR1;
149 case TIM_CHANNEL_2:
150 return &config.timHandle->Instance->CCR2;
151 case TIM_CHANNEL_3:
152 return &config.timHandle->Instance->CCR3;
153 case TIM_CHANNEL_4:
154 return &config.timHandle->Instance->CCR4;
155 default:
156 assert(0);
157 return nullptr;
158 }
159 }
