145 /** Memory pool size optimization for different allocation patterns.
146 *
147 * Computes optimal pool size with cache alignment for given element size and count.
148 * @tparam ELEMENT_SIZE Size of each element.
149 * @tparam COUNT Number of elements.
150 * @return Optimal pool size in bytes.
151 */
152 template <size_t ELEMENT_SIZE, size_t COUNT> constexpr size_t optimal_pool_size() {
153 constexpr size_t alignedSize = (ELEMENT_SIZE + CACHE_LINE_SIZE - 1) & ~(CACHE_LINE_SIZE - 1);
154 return alignedSize * COUNT;

157 /** Stack size optimization based on optimization level.
158 *
159 * Computes optimal stack size for tasks based on base size and optimization level.
160 * @tparam BASE_SIZE Base stack size in bytes.
161 * @return Optimized stack size in bytes.
162 */
163 template <size_t BASE_SIZE> constexpr size_t optimal_stack_size() {
164 switch (CURRENT_LEVEL) {
165 case Level::SPACE:
166 return BASE_SIZE;
167 case Level::BALANCED:
168 return BASE_SIZE + (BASE_SIZE / 4);
169 case Level::SPEED:
170 return BASE_SIZE + (BASE_SIZE / 2);
171 case Level::REALTIME:
172 return BASE_SIZE * 2;
173 default:
174 return BASE_SIZE;
175 }
176 }

178 /** Buffer size optimization for queue and message passing.
179 *
180 * Computes optimal buffer count for a given type and requested count.
181 * @tparam T Buffer element type.
182 * @tparam REQUESTED_COUNT Requested buffer count.
183 * @return Optimized buffer count.
184 */
185 template <typename T, size_t REQUESTED_COUNT> constexpr size_t optimal_buffer_count() {
186 constexpr size_t elementSize = sizeof(T);
187 if constexpr (elementSize <= 8)
188 return REQUESTED_COUNT * 2;
189 else if constexpr (elementSize <= 64)
190 return REQUESTED_COUNT;
191 else
192 return (REQUESTED_COUNT + 1) / 2;
193 }
