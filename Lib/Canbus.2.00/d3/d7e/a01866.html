#pragma once
#include "Components/ServiceBase.hpp"
#include "Project\DAL\CDataLayer.hpp"
#include "can.h"
extern DAL::CDataLayer DataLayer;
namespace Canbus
{
class Server : Lib::Canbus::ServiceBase
{
public:
Server () : ServiceBase (0, &hcan1) {}
~Server () = default;
bool start () override
{
return ServiceBase::start ();
}
bool stop () override
{
return ServiceBase ::stop ();
}
private:
void incomingMsgCallback (uint32_t id, const uint8_t * data, size_t len) override
{
if (len == sizeof (uint64_t))
{
auto index = id - 1;
if (index < DataLayer.Nodes.size ())
{
DataLayer.Nodes.at (index).Inputs = *reinterpret_cast<const uint64_t *> (data);
}
}
}
void handleMessageReceivedEvent (const RxMessage & msg) override
{
for (auto i = 0; i < DataLayer.Nodes.size (); i++)
{
if (DataLayer.Nodes.at (i).Outputs != DataLayer.Nodes.at (i).PrevOutputs)
{
auto id = i + 1;
sendMessage (id, DataLayer.Nodes.at (i).Outputs);
DataLayer.Nodes.at (i).PrevOutputs = DataLayer.Nodes.at (i).Outputs;
}
}
}
void postScanCallback (const std::array<XNode, 8> nodes) override
{
for (auto i = 0; i < nodes.size (); i++)
{
DataLayer.Nodes.at (i).IsActive = nodes.at (i).getActivity ();
}
}
};
} // namespace Canbus

В сървърният клас се налага да се предефинират наколко callback -а от ServiceBase . Те служат за синхронизация на данните от DAL слоят с отдаличените възли

  • messageReceivedCallback - той се извиква при приемане на пакет от CAN интерфейса
  • outgoingMsgCallback - извиква се през 1 мс от задачата на ServiceBase . В него се проверва промените в DAL Layer и ако има промяна се изпращат към отдаличеният
  • postScanCallback - служи за начална синхронизация на състоянието на IsActive
