1/** Base class for I2C EEPROM drivers.
2 * Provides an interface for I2C EEPROM operations, including reading and writing data.
3 * This class encapsulates common logic for I2C EEPROM access, including address translation,
4 * page-wise writes, and device readiness checks. It supports both 8-bit and 16-bit memory addressing
5 * and ensures robust operation via assertions and error codes. The design is suitable for STM32-based
6 * embedded systems and integrates with RTOS tick services for timing and delays.
7 * @file DriverBase.hpp

18 /** Base class for I2C EEPROM drivers.
19 * Provides common functionality for interacting with I2C EEPROM devices.
20 */
21 class DriverBase {
22 public:
23 /** Configuration structure for DriverBase. */
24 struct Config {
25 I2CPort * port; /*!< Pointer to the I2C port interface. */
26 uint8_t addr = 0; /*!< I2C device address. */
27 };
28
29 static constexpr uint32_t I2C_TIMEOUT = 100; /*!< Timeout for I2C operations (ms). */
30
31 /** Constructor of DriverBase class.
32 * @param config Configuration structure.
33 * @param pageSize Size of the EEPROM page.
34 * @param chipSize Total size of the EEPROM chip.
35 */
36 explicit DriverBase(const Config & config, uint8_t pageSize, size_t chipSize)
37 : i2cPort(config.port),
38 deviceAddr(static_cast<uint8_t>((config.addr & 0x0F) | 0xA0)),
39 pageSize(pageSize),
40 chipSize(chipSize),
41 isReady(false) {
42 assert(i2cPort);
43 assert(pageSize > 0);
44 assert(chipSize > 0);
45 }
46
47 /** Destructor of DriverBase class. */
48 virtual ~DriverBase() = default;
49
50 /** Opens the I2C connection.
51 * @return `true` if successful, `false` otherwise.
52 */
53 virtual bool open() {
54 if (!i2cPort->init()) return false;
55 isReady = i2cPort->isReady(deviceAddr);
56 assert(isReady);
57 return isReady;
58 }
59
60 /** Closes the I2C connection.
61 * @return `true` if successful, `false` otherwise.
62 */
63 virtual bool close() {
64 isReady = false;
65 i2cPort->deinit();
66 return true;
67 }
68
69 /** Reads data from the EEPROM.
70 * @param addr Memory address to read from.
71 * @param buffer Pointer to the buffer to store read data.
72 * @param bufferSize Size of the buffer.
73 * @return `true` if successful, `false` otherwise.
74 */
75 virtual bool read(uint32_t addr, void * buffer, size_t bufferSize) = 0;
76
77 /** Writes data to the EEPROM.
78 * @param addr Memory address to write to.
79 * @param buffer Pointer to the buffer containing data to write.
80 * @param bufferSize Size of the buffer.
81 * @return `true` if successful, `false` otherwise.
82 */
83 virtual bool write(uint32_t addr, const void * buffer, size_t bufferSize) = 0;
84
85 /** Gets the total size of the EEPROM chip.
86 * @return Size of the EEPROM chip.
87 */
88 virtual size_t getChipSize() const { return chipSize; }
89
90 protected:
91 /** Reads data from the EEPROM using 8-bit memory addressing.
92 * @param addr Memory address to read from.
93 * @param buffer Pointer to the buffer to store read data.
94 * @param size Size of the buffer.
95 * @return `true` if successful, `false` otherwise.
96 */
97 virtual bool read8(uint32_t addr, void * buffer, size_t size) {
98 if (!isReady) {
99 assert(false);
100 return false;
101 }
102 return i2cPort->receive(deviceAddr, I2C_MEMADD_SIZE_8BIT, addr, buffer, size, I2C_TIMEOUT);
103 }
104
105 /** Reads data from the EEPROM using 16-bit memory addressing.
106 * @param addr Memory address to read from.
107 * @param buffer Pointer to the buffer to store read data.
108 * @param size Size of the buffer.
109 * @return `true` if successful, `false` otherwise.
110 */
111 virtual bool read16(uint32_t addr, void * buffer, size_t size) {
112 if (!isReady) {
113 assert(false);
114 return false;
115 }
116 return i2cPort->receive(deviceAddr, I2C_MEMADD_SIZE_16BIT, addr, buffer, size, I2C_TIMEOUT);
117 }
118
119 /** Writes data to the EEPROM using 8-bit memory addressing.
120 * @param addr Memory address to write to.
121 * @param buffer Pointer to the buffer containing data to write.
122 * @param size Size of the buffer.
123 * @return `true` if successful, `false` otherwise.
124 */
125 bool write8(uint32_t addr, const void * buffer, size_t size) {
126 return write(addr, buffer, size, I2C_MEMADD_SIZE_8BIT);
127 }
128
129 /** Writes data to the EEPROM using 16-bit memory addressing.
130 * @param addr Memory address to write to.
131 * @param buffer Pointer to the buffer containing data to write.
132 * @param size Size of the buffer.
133 * @return `true` if successful, `false` otherwise.
134 */
135 bool write16(uint32_t addr, const void * buffer, size_t size) {
136 return write(addr, buffer, size, I2C_MEMADD_SIZE_16BIT);
137 }
138
139 private:
140 /** Internal helper for writing data to the EEPROM.
141 * @param addr Memory address to write to.
142 * @param buffer Pointer to the buffer containing data to write.
143 * @param size Size of the buffer.
144 * @param memAddSize Memory address size (8-bit or 16-bit).
145 * @return `true` if successful, `false` otherwise.
146 */
147 bool write(uint32_t addr, const void * buffer, size_t size, uint16_t memAddSize) {
148 if (!isReady || (addr + size) > chipSize) {
149 assert(false);
150 return false;
151 }
152 auto * dataPtr = static_cast<uint8_t *>(const_cast<void *>(buffer));
153 size_t remaining = size;
154 uint32_t currentAddr = addr;
155
156 while (remaining > 0) {
157 size_t pageOffset = currentAddr % pageSize;
158 size_t writeLen = std::min(remaining, pageSize - pageOffset);
159 if (!i2cPort->send(deviceAddr, memAddSize, currentAddr, dataPtr, writeLen, I2C_TIMEOUT)) {
160 assert(false);
161 return false;
162 }
163 vTaskDelay(5);
164 currentAddr += writeLen;
165 dataPtr += writeLen;
166 remaining -= writeLen;
167 }
168 return true;
169 }
170
171 I2CPort * i2cPort; /*!< Pointer to the I2C port interface. */
172 uint8_t deviceAddr; /*!< I2C device address. */
173 const uint8_t pageSize; /*!< Size of the EEPROM page. */
174 const size_t chipSize; /*!< Total size of the EEPROM chip. */
175 bool isReady; /*!< Indicates if the device is ready. */
176 };
