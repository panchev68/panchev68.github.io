39 /** Starts the DTH component.
40 * @return `true` if started successfully, `false` otherwise.
41 */
42 virtual bool start() {
43 if (!semaphoreCplt.create()) assert(0 /* Semaphore creation failed */);
44 if (HAL_TIM_RegisterCallback(config.timerHandle.handle, HAL_TIM_IC_CAPTURE_CB_ID, isrCallback) != HAL_OK)
45 assert(0 /* Callback registration failed */);
46 if (RTOS::Task::run({"DTH", [this] { taskFn(); }}) == nullptr) {
47 assert(0 /* Task creation failed */);
48 return false;
49 }
50 return true;
51 }
52
53 /** DTH task function. */
54 virtual void taskFn() = 0;

56 /** Sends startup impulse to DTH sensor.
57 * @param impulseDuration Impulse duration in ms.
58 */
59 void startupImpulse(uint32_t impulseDuration) {
60 configurePin(GPIO_MODE_OUTPUT_PP, GPIO_PULLUP, GPIO_SPEED_FREQ_LOW);
61 HAL_GPIO_WritePin(config.pin.getPort(), config.pin.getPinNumber(), GPIO_PIN_RESET);
62 vTaskDelay(impulseDuration);
63 HAL_GPIO_WritePin(config.pin.getPort(), config.pin.getPinNumber(), GPIO_PIN_SET);
64 configurePin(GPIO_MODE_AF_PP, GPIO_PULLUP, GPIO_SPEED_FREQ_LOW, GPIO_AF1_TIM1);
65 }
66
67 /** Ends data reception. */
68 void endReceivedData() {
69 configurePin(GPIO_MODE_OUTPUT_PP, GPIO_PULLUP, GPIO_SPEED_FREQ_LOW);
70 HAL_GPIO_WritePin(config.pin.getPort(), config.pin.getPinNumber(), GPIO_PIN_SET);
71 }

73 /** Begins DMA data reception.
74 * @param data Data buffer.
75 * @param length Buffer length.
76 * @return `true` if started successfully, `false` otherwise.
77 */
78 bool beginReceiveData(uint32_t * data, size_t length) {
79 auto result = HAL_TIM_IC_Start_DMA(config.timerHandle.handle, config.timerHandle.channel, data, length);
80 assert(result == HAL_OK /* DMA start failed */);
81 return result == HAL_OK;
82 }

84 /** Waits for data reception completion.
85 * @param timeout Timeout in ms.

88 bool waitForComplete(uint32_t timeout) { return semaphoreCplt.take(timeout); }

95 /** Timer ISR callback.
96 * @param timerHandle Timer handle pointer.
97 */
98 static void isrCallback(TIM_HandleTypeDef * timerHandle) {
99 assert(instance /* Instance must be set */);
100 if (timerHandle == instance->config.timerHandle.handle) {
101 if (!instance->semaphoreCplt.giveFromISR()) assert(0 /* Semaphore give failed */);
102 }
103 }
