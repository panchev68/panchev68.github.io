67 virtual bool open() override {

69 RTOS::delay(50);

164 /** USB CDC control callback.
165 * @param command CDC command.
166 * @param buffer Data buffer.
167 * @param length Buffer length.
168 * @return USBD_OK on success.
169 */
170 int8_t controlCallback(uint8_t command, uint8_t * buffer, uint16_t length) {
171 __packed struct LineCoding {
172 uint32_t baudRate; /*!< Baud rate. */
173 uint8_t charFormat; /*!< Character format. */
174 uint8_t parityType; /*!< Parity type. */
175 uint8_t dataBits; /*!< Number of data bits. */
176 };
177
178 static LineCoding lineCoding{115200, 0, 0, 8};
179 switch (command) {
180 case CDC_SET_LINE_CODING:
181 memcpy(&lineCoding, buffer, sizeof(LineCoding));
182 SerialPort::setBaudRate(lineCoding.baudRate);
183 break;
184 case CDC_GET_LINE_CODING:
185 memcpy(buffer, &lineCoding, sizeof(LineCoding));
186 break;
187 default: break;
188 }
189 return USBD_OK;
190 }
191
192 /** USB CDC receive callback.
193 * @param buffer Received data buffer.
194 * @param length Pointer to received data length.
195 * @return USBD_OK if successful, USBD_FAIL otherwise.
196 */
197 int8_t receiveCallback(uint8_t * buffer, uint32_t * length) {
198 return SerialPort::send(buffer, static_cast<size_t>(*length)) ? USBD_OK : USBD_FAIL;
199 }
