174 /** Optimized IP string parsing helper.
175 * @param str String to parse.
176 * @param[out] address Output array (nullptr for validation only).
177 * @return `true` if parsing was successful, `false` otherwise.
178 */
179 static bool parseIpString(const std::string & str, std::array<uint8_t, 4> * address) {
180 if (str.empty() || str.size() > 15 || str.find(':') != std::string::npos) return false;
181 std::array<uint8_t, 4> temp{};
182 size_t pos = 0;
183 for (size_t i = 0; i < 4; ++i) {
184 if (pos >= str.size()) return false;
185 const size_t found = (i < 3) ? str.find('.', pos) : std::string::npos;
186 if (i < 3 && (found == std::string::npos || found == pos)) return false;
187 uint32_t value = 0;
188 const size_t end = (i < 3) ? found : str.size();
189 for (size_t j = pos; j < end; ++j) {
190 const char c = str[j];
191 if (c < '0' || c > '9') return false;
192 value = value * 10 + static_cast<uint32_t>(c - '0');
193 if (value > 255) return false;
194 }
195 if (pos == end) return false;
196 temp[i] = static_cast<uint8_t>(value);
197 pos = (i < 3) ? found + 1 : str.size();
198 }
199 if (str.find('.', pos) != std::string::npos) return false;
200 if (address) *address = temp;
201 return true;
202 }
