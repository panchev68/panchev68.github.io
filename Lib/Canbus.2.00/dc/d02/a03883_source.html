35 /** Adds a device to the manager.

39 virtual bool addDevice(DevicePtr device) {
40 if (!device) return false;
41 HandlePtr handle = device->getHandle();
42 if (!handle) return false;
43 auto result = deviceMap.emplace(handle, device);

45 }

47 /** Removes a device from the manager.
48 * @param device Pointer to the device.
49 * @return `true` if the device was removed successfully, `false` otherwise.
50 */
51 virtual bool removeDevice(DevicePtr device) {
52 if (!device) return false;
53 auto it = std::find_if(deviceMap.begin(), deviceMap.end(),
54 [device](const auto & entry) { return entry.second == device; });
55 if (it == deviceMap.end()) return false;
56 deviceMap.erase(it);
57 return true;
58 }

60 /** Retrieves a device by its handle.
61 * @param handle Pointer to the device handle.
62 * @return Pointer to the device if found, `nullptr` otherwise.
63 */
64 virtual DevicePtr findDevice(HandlePtr handle) {
65 auto it = deviceMap.find(handle);
66 return it != deviceMap.end() ? it->second : nullptr;
67 }

69 private:
