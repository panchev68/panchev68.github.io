228 void scan () {
229 RxMessage msg;
230 Nodes nodes;
231
232 for (auto i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; i++) {
233 auto bit (1 << i);
234
235 nodes.reset (bit);
236 auto id = i + 1;
237
238 if (send (id) == true) {
239 if (ServiceBase::receive (msg, RECEIVE_TIMEOUT)) {
240 if (msg.getId () == id)
241 nodes.set (bit);
242 }
243 }
244 }
