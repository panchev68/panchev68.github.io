143 void scan ()
144 {
145 RxMessage msg; // rx message
146 Nodes nodes;
147
148 // scan for remote nodes
149 for (auto i = 0; i < NUM_OF_CANBUS_CLIENT_NODES; i++)
150 {
151 auto bit (1 << i);
152
153 nodes.reset (bit);
154 auto id = i + 1;
155
156 // send message with empty payload ...
157 if (send (id) == true)
158 {
159 // receive incoming message...
160 if (ServiceBase::receive (msg, RECEIVE_TIMEOUT))
161 {
162 if (msg.getId () == id)
163 nodes.set (bit);
164 }
165 }
166 }
167 handleScanCompletedEvent (nodes);
168 }
