109 /** Service update callback. */
110 void update() override {
111 static std::vector<uint8_t> response;
112 using Convert = Core::Frame::Helper::Convert;
113 if (!pause && state == State::IDLE && timoutUpdate.isExpired()) {
114 timoutUpdate.set(50);
115 auto mask = 1u << (tickCounter++ % 10);
116 readDynamicParameters(mask);
117 }
118 switch (state) {
119 case State::IDLE: break;
120 case State::SET_PARAMETERS_REMOTELY:
121 state = (remotePanId == UINT64_MAX) ? State::GET_REMOTE_PAN_ID : State::SET_CHANNEL_REMOTELY;
122 break;
123 case State::GET_REMOTE_PAN_ID:
124 if (newData.isChannelChanged && newData.channel != data.channel) {
125 if (!localAtCommand->send({"ND"})) assert(false);
126 state = State::GET_REMOTE_PAN_ID_RESPONSE;
127 } else state = State::END;
128 break;
129 case State::GET_REMOTE_PAN_ID_RESPONSE:
130 if (!localAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
131 else remotePanId = Convert::to<uint64_t>(response.data() + 2);
132 state = State::SET_CHANNEL_REMOTELY;
133 break;
134 case State::SET_CHANNEL_REMOTELY:
135 if (newData.isChannelChanged && newData.channel != data.channel) {
136 if (!remoteAtCommand->send({"CH"}, remotePanId, {newData.channel})) assert(false);
137 else state = State::SET_CHANNEL_REMOTELY_RESPONSE;
138 } else state = State::END;
139 break;
140 case State::SET_CHANNEL_REMOTELY_RESPONSE:
141 if (!remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
142 state = State::WRITE_CHANGES_REMOTELY;
143 break;
144 case State::WRITE_CHANGES_REMOTELY:
145 if (!remoteAtCommand->send({"WR"}, true)) assert(false);
146 state = State::WRITE_CHANGES_REMOTELY_RESPONSE;
147 break;
148 case State::WRITE_CHANGES_REMOTELY_RESPONSE:
149 if (!remoteAtCommand->receive(response, TIMEOUT_REMOTE_AT_COMMAND)) assert(false);
150 state = State::SET_PARAMETERS_LOCAL;
151 break;
152 case State::SET_PARAMETERS_LOCAL:
153 hasWrite = false;
154 state = State::SET_CHANNEL_LOCALLY;
155 break;
156 case State::SET_CHANNEL_LOCALLY:
157 if (newData.isChannelChanged && newData.channel != data.channel) {
158 if (!localAtCommand->send({"CH"}, {newData.channel})) assert(false);
159 else {
160 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
161 data.channel = newData.channel;
162 hasWrite = true;
163 }
164 }
165 state = State::SET_PAN_ID_LOCALLY;
166 break;
167 case State::SET_PAN_ID_LOCALLY:
168 if (newData.isPanIdChanged && newData.panId != data.panId) {
169 if (!localAtCommand->send({"ID"}, {static_cast<uint8_t>(newData.panId >> 8), static_cast<uint8_t>(newData.panId & 0xff)})) assert(false);
170 else {
171 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
172 data.panId = newData.panId;
173 hasWrite = true;
174 }
175 }
176 state = State::WRITE_CHANGES_LOCALLY;
177 break;
178 case State::WRITE_CHANGES_LOCALLY:
179 if (hasWrite) {
180 if (localAtCommand->send({"WR"})) {
181 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
182 }
183 state = State::APPLY_CHANGES_LOCALLY;
184 } else state = State::END;
185 break;
186 case State::APPLY_CHANGES_LOCALLY:
187 if (localAtCommand->send({"AC"})) {
188 if (!localAtCommand->receive(response, TIMEOUT_LOCAL_AT_COMMAND)) assert(false);
189 }
190 connection->quality.clear();
191 state = State::END;
192 break;
193 case State::END:
194 newData.isChannelChanged = false;
195 newData.isPanIdChanged = false;
196 state = State::IDLE;
197 break;
198 }
199 }

489 /** Extracts an unsigned integer value from a vector.
490 * @tparam T Unsigned integer type (uint8_t, uint16_t, uint32_t).
491 * @param buffer Source vector.
492 * @return The extracted value.
493 */
494 template <typename T>
495 T get(const std::vector<uint8_t> & buffer) const {
496 static_assert(std::is_unsigned<T>::value, "T must be an unsigned integer type");
497 assert(buffer.size() == sizeof(T));
498 T value = 0;
499 for (size_t i = 0; i < sizeof(T); ++i) value = (value << 8) | buffer[i];
500 return value;
501 }
