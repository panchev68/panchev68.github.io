76 void handleTickEvent() override {
77 if (!timeout.isExpired()) return;
78 timeout.set(updatePeriod);
79 uint16_t adcValue = 0;
80 if (!adcPort.read(adcValue, 0)) return;
81 adcValue = integrator.processValue(adcValue);
82 if (!integrator.isWindowFull()) {
83 adcPort.start();
84 return;
85 }
86 float newVoltage = interpolator.interpolate(adcValue);
87 if (newVoltage != currentVoltage) {
88 currentVoltage = newVoltage;
89 handleValueChanged(newVoltage);
90 }
