84 /** Task function for periodic updates and error state checks. */
85 void handleTickEvent() override {
86 if (!timeout.isExpired()) return;
87 timeout.set(UPDATE_PERIOD);
88
89 bool hasError = false;
90 uint32_t clearedErrorCount = 0;
91
92 for (auto & error : errorVector) {
93 if (error.checkForStateChanged()) {
94 auto state = error.getState();
95 if (state > Error::State::NO_ERROR) hasError = true;
96 else if (state == Error::State::CLEARED) clearedErrorCount++;
97 }
98 }
99
100 if (hasError) {
101 if (statusChangedHook) statusChangedHook(Status::STATUS_CHANGED);
102 } else if (clearedErrorCount != lastClearedErrorCount) {
103 lastClearedErrorCount = clearedErrorCount;
104 if (statusChangedHook) statusChangedHook(Status::ALL_ERRORS_CLEARED);
105 }
106 }
