132 /**
133 * @brief Prints the given string onto the canvas using the specified font and alignment.
134 *
135 * This function renders each character of the provided string onto the canvas.
136 * It calculates the dimensions of the string using the specified font and adjusts
137 * the position based on the alignment settings. It handles special characters like
138 * newline ('\n') and carriage return ('\r') appropriately.
139 *
140 * @param string The string to be printed on the canvas.
141 */
142 void print (const std::string & string)
143 {
144 auto stringDimensions = [&string] (const Font & font) -> Dimension {
145 Dimension result{0, 0};
146 result.height = font.getHeight ();
147 for (auto ch : string)
148 {
149 if (ch < ' ') return result;
150 result.width += font.at (ch).width + SPACE_AFTER_CHAR;
151 }
152 return result;
153 };
154
155 assert (_font);
156 assert (_canvas);
157
158 if (_align != Align::NONE)
159 {
160 _position = _align (*_canvas, stringDimensions (*_font), _position);
161 }
162 for (auto & ch : string)
163 {
164 auto gc = _font->at (ch);
165 if (ch == '\n') { _position.y += gc.height; }
166 else if (ch == '\r') { _position.x = 0; }
167 else
168 {
169 _canvas->render (gc.getFace (), _colors, _position, {gc.width, gc.height});
170 _position.x += gc.width + SPACE_AFTER_CHAR;
171 }
172 }
173 }

179 Position _position; ///< The position of the text on the canvas.
180 Colors _colors; ///< The colors used for rendering the text.
181 Align _align = Align::NONE; ///< The alignment of the text on the canvas.
182 };
