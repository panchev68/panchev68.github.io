113 /// \brief Add the device to the Device Manager
114 /// \param device - pointer to device
115 /// \return true if success
116 bool add (PortTransparentBase * device) override
117 {
118 // add device to the device manager container
119 if (DeviceManagerBase::add (device) == false)
120 {
121 assert (0);
122 return false;
123 }
124 // bind the ISR callbacks
125
126 ADC_HandleTypeDef * handle = *device;
127 if (handle != nullptr)
128 {
129 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_CONVERSION_COMPLETE_CB_ID, eventHandler<HAL_ADC_CONVERSION_COMPLETE_CB_ID>) != HAL_OK) assert (0);
130 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_CONVERSION_HALF_CB_ID, eventHandler<HAL_ADC_CONVERSION_HALF_CB_ID>) != HAL_OK) assert (0);
131 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_ERROR_CB_ID, eventHandler<HAL_ADC_ERROR_CB_ID>) != HAL_OK) assert (0);
132 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID, eventHandler<HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID>) != HAL_OK) assert (0);
133 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_LEVEL_OUT_OF_WINDOW_1_CB_ID, eventHandler<HAL_ADC_LEVEL_OUT_OF_WINDOW_1_CB_ID>) != HAL_OK) assert (0);
134 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_MSPINIT_CB_ID, eventHandler<HAL_ADC_MSPINIT_CB_ID>) != HAL_OK) assert (0);
135 if (HAL_ADC_RegisterCallback (handle, HAL_ADC_MSPDEINIT_CB_ID, eventHandler<HAL_ADC_MSPDEINIT_CB_ID>) != HAL_OK) assert (0);
136 }
137
138 return true;
139 }

141 /// \brief Remove the device from the Device Manager
142 /// \param device - pointer to device
143 /// \return true if success
144 bool remove (PortTransparentBase * device) override
145 {
146 assert (device);
147 ADC_HandleTypeDef * handle = *device;
148
149 if (DeviceManagerBase::remove (device) == true)
150 {
151 // un-bind ISR callbacks
152 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_CONVERSION_COMPLETE_CB_ID) != HAL_OK) assert (0);
153 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_CONVERSION_HALF_CB_ID) != HAL_OK) assert (0);
154 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_ERROR_CB_ID) != HAL_OK) assert (0);
155 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID) != HAL_OK) assert (0);
156 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_LEVEL_OUT_OF_WINDOW_1_CB_ID) != HAL_OK) assert (0);
157 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_MSPDEINIT_CB_ID) != HAL_OK) assert (0);
158 if (HAL_ADC_UnRegisterCallback (handle, HAL_ADC_MSPINIT_CB_ID) != HAL_OK) assert (0);
159 return true;
160 }
161 return false;
162 }
