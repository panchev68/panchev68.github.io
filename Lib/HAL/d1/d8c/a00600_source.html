198 /// \brief Waits for complete handler
199 /// \param bit
200 /// \param timeout
201 /// \return true if success
202 template <EventBit T_BIT> bool waitForCompleteHandler (uint32_t timeout)
203 {
204 assert (_isOpen); // errorL port not opened
205 if (timeout == 0) return true;
206 auto result = _eventGroup.waitForBits (T_BIT, true, false, timeout).to_ulong ();
207 assert (result & T_BIT);
208 return (result & T_BIT);
209 }

286 /// \brief Handle ISR eventHandler
287 /// \param id - eventHandler ID
288 /// \pre ISR Function
289 void eventHandler (CallbackID id)
290 {
291 assert (_isOpen == true);
292
293 if (id == CallbackID::HAL_SPI_TX_COMPLETE_CB_ID) handleTxCompleteEvent ();
294 else if (id == CallbackID::HAL_SPI_RX_COMPLETE_CB_ID) handleRxCompleteEvent ();
295 else if (id == CallbackID::HAL_SPI_TX_RX_COMPLETE_CB_ID) handleTxRxCompleteEvent ();
296 else if (id == CallbackID::HAL_SPI_ABORT_CB_ID) handleAbortCompleteEvent ();
297
298 else if (id == CallbackID::HAL_SPI_RX_HALF_COMPLETE_CB_ID) handleRxHalfCompleteEvent ();
299 else if (id == CallbackID::HAL_SPI_TX_HALF_COMPLETE_CB_ID) handleTxHalfCompleteEvent ();
300 else if (id == CallbackID::HAL_SPI_TX_RX_HALF_COMPLETE_CB_ID) handleTxRxHalfCompleteEvent ();
301 }
