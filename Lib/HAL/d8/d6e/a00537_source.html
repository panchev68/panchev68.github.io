54 /// \brief Open The CAN port
55 virtual bool open ()
56 {
57 _queueRxMessage.create (); // queue for received messages
58 _queueTxMessage.create (); // queue of transmit messages
59 _semaphoreError.create (); // semaphore for errors
60 _queueSet.create (17, {_queueRxMessage, _queueTxMessage, _semaphoreError}); // queset
61
62 _semaphoreTxComplete.create (); // semaphore tx completed
63 _semaphoreTxCompleted.create (); // semaphore tx is completed
64
65 // activate notification
66 if (HAL_CAN_ActivateNotification (getHandle (), (CAN_IT_TX_MAILBOX_EMPTY | CAN_IT_RX_FIFO0_MSG_PENDING | CAN_IT_RX_FIFO0_FULL | CAN_IT_RX_FIFO0_OVERRUN)) != HAL_OK)
67 {
68 assert (0);
69 return false;
70 }
71 // start CAN port
72 if (HAL_CAN_Start (getHandle ()) != HAL_OK)
73 {
74 assert (0);
75 return false;
76 }
77
78 // starts the task that processes the requests from the corresponding port
79 return _task.create ({"CAN port", [this] { taskFn (); }, 128, 0});
80 }

90 {
91 assert (0);
92 }
93 // stop CAN port...
94 if (HAL_CAN_Stop (getHandle ()) != HAL_OK)
95 {
96 assert (0);
97 return false;

99 // release resources...
100 _queueRxMessage.release ();
101 _semaphoreTxComplete.release ();
102 _semaphoreTxCompleted.release ();
103 _semaphoreError.release ();
104 _queueTxMessage.release ();

112 /// \brief Emit ISR callback routine
113 /// \param id [ID of ISR callback](https://www.disca.upv.es/aperles/arm_cortex_m3/llibre/st/STM32L486xx_User_Manual/group__can__exported__types.html#gacabc6dbb856a1840b90d703738491759)
114 /// \remark called from `DeviceManager::eventHandler`
115 virtual void eventHandler (CallbackID id)
116 {
117 RxMessage rxMessage;
118
119 switch (id)
120 {
121 case CallbackID::HAL_CAN_ERROR_CB_ID: // Errors callback
122 _semaphoreError.giveFromISR ();
123 break;
124
125 case CallbackID::HAL_CAN_TX_MAILBOX0_COMPLETE_CB_ID: // CAN Tx Mailbox 0 complete callback
126 _semaphoreTxComplete.giveFromISR ();
127 break;
128
129 case CallbackID::HAL_CAN_TX_MAILBOX1_COMPLETE_CB_ID: // CAN Tx Mailbox 1 complete callback
130 _semaphoreTxComplete.giveFromISR ();
131 break;
132
133 case CallbackID::HAL_CAN_TX_MAILBOX2_COMPLETE_CB_ID: // CAN Tx Mailbox 2 complete callback
134 _semaphoreTxComplete.giveFromISR ();
135 break;
136
137 case CallbackID::HAL_CAN_RX_FIFO0_MSG_PENDING_CB_ID: // CAN Rx FIFO 0 message pending callback
138
139 if (HAL_CAN_GetRxMessage (getHandle (), CAN_RX_FIFO0, &rxMessage, rxMessage.getData ()) == HAL_OK)
140 {
141 _queueRxMessage.sendToBackFromISR (rxMessage);
142 }
143 break;
144
145 case CallbackID::HAL_CAN_RX_FIFO1_MSG_PENDING_CB_ID: // CAN Rx FIFO 1 message pending callback
146
147 if (HAL_CAN_GetRxMessage (getHandle (), CAN_RX_FIFO1, &rxMessage, rxMessage.getData ()) == HAL_OK)
148 {
149 _queueRxMessage.sendToBackFromISR (rxMessage); // sent to taskFn
150 }
151
152 break;
153 default:
154 assert (0 && "Unsupported Callback ID");
155 break;
156 }
157 }

276 /// Task pending function
277 void taskFn ()
278 {
279 // Raw send function(low level)
280 // \see [example](https://controllerstech.com/can-protocol-in-stm32/)
281 auto sendFunc = [this] (const TxMessage & msg, uint32_t & txMailbox) {
282 // request to send a message
283 // https://www.disca.upv.es/aperles/arm_cortex_m3/llibre/st/STM32F439xx_User_Manual/group__can__exported__functions__group3.html#ga82cd04bf8478c081247d933efeb19082
284 if (HAL_CAN_AddTxMessage (getHandle (), &msg, (uint8_t *) msg.getData (), &txMailbox) != HAL_OK)
285 {
286 // assert (0); // TODO : Test only
287 return false;
288 }
289 return true;
290 };
291
292 while (true)
293 {
294 // wait for event
295 auto selected = _queueSet.select (portMAX_DELAY);
296
297 // check for a received message
298 if (selected == _queueRxMessage)
299 {
300 // process of incoming message
301 if (_queueRxMessage.receive (_rxMessage, 0))
302 {
303 handleReceivedMessageEvent (_rxMessage);
304 }
305 }
306 else if (selected == _queueTxMessage) // send the message
307 {
308 // process of outgoing message
309 if (_queueTxMessage.receive (_txMessage, 0) == false)
310 {
311 assert (0);
312 }
313 else
314 {
315 preSendCallback ();
316
317 if (sendFunc (_txMessage, _txMailbox) == true)
318 {
319 // wait for a tx completed (invike from HAL_CAN_TX_MAILBOX0_COMPLETE_CB_ID)
320 if (_semaphoreTxComplete.take (100) == false)
321 {
322 // assert (0); // TODO Test Only
323 }
324 else _semaphoreTxCompleted.give (); // notify send function
325 }
326 postSendCallback ();
327 }
328 }
329 else if (selected == _semaphoreError) // error process
330 {
331 // process of error event
332 handleErrorEvent (HAL_CAN_GetError (getHandle ()));
333 }
334 }
335 }
