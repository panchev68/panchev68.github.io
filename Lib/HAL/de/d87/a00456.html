`XDeviceManager` наследява CDeviceManagerBase със съответният тип на `Handle`- cpp CDeviceManagerBase<ADC_HandleTypeDef>

example:

```cpp class XDeviceManager : CDeviceManagerBase<ADC_HandleTypeDef> { public: bool add (CDeviceBase<ADC_HandleTypeDef> * device) override { CDeviceManagerBase::add (device);

// bind ISR callbacks auto hnd = device->getHandle (); if (hnd != nullptr) { if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_COMPLETE_CB_ID, _callbackConvertionComplete) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_HALF_CB_ID, _callbackConvertionHalf) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_ERROR_CB_ID, _callbackError) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID, _callbackINJConvertionComplete) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_LEVEL_OUT_OF_WINDOW_1_CB_ID, _callbackLevelOutWindow1) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPDEINIT_CB_ID, _callbackMSPDeinit) != HAL_OK) assert (0); if (HAL_ADC_RegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPINIT_CB_ID, _callbackMSPInit) != HAL_OK) assert (0); } return true; }

/// Remove the deice from the Device Manager ///

Parameters
device- pointer to device ///
Returns
true if success bool remove (CDeviceBase<ADC_HandleTypeDef> * device) override { assert (device); auto hnd = device->getHandle ();

if (CDeviceManagerBase::remove (device) == true) { // un-bind ISR callbacks if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_COMPLETE_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_HALF_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_ERROR_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_LEVEL_OUT_OF_WINDOW_1_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPDEINIT_CB_ID) != HAL_OK) assert (0); if (HAL_ADC_UnRegisterCallback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPINIT_CB_ID) != HAL_OK) assert (0); return true; } return false; }

/// Get the Singleton Instance ///

Returns
XDeviceManager & static XDeviceManager & getInstance ()

{ static XDeviceManager instance; return instance; }

private: 

/// \brief XDeviceManager Constructor
XDeviceManager () = default;

static void _callback (ADC_HandleTypeDef * hnd, HAL_ADC_CallbackIDTypeDef id)
{
    CADCTransparentBase * ptr = reinterpret_cast<CADCTransparentBase *> (getInstance ().getDevice (hnd));

    if (ptr != nullptr) ptr->eventHandler (id);
}

static void _callbackConvertionComplete (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_COMPLETE_CB_ID);
}

static void _callbackConvertionHalf (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_CONVERSION_HALF_CB_ID);
}

static void _callbackError (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_ERROR_CB_ID);
}

static void _callbackINJConvertionComplete (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_INJ_CONVERSION_COMPLETE_CB_ID);
}

static void _callbackLevelOutWindow1 (ADC_HandleTypeDef * hnd)
{

}

static void _callbackMSPDeinit (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPDEINIT_CB_ID);
}

static void _callbackMSPInit (ADC_HandleTypeDef * hnd)
{
    _callback (hnd, HAL_ADC_CallbackIDTypeDef::HAL_ADC_MSPINIT_CB_ID);
}

};
```
