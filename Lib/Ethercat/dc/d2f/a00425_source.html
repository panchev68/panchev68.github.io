37#include "esc.h"
38#include "esc_eep.h"
39
40#include <string.h>
41
42static uint8_t eep_buf[8];
43
47void EEP_process (void)
48{
49 eep_stat_t stat;
50
51 /* check for eeprom event */
52 if ((ESCvar.ALevent & ESCREG_ALEVENT_EEP) == 0) {
53 return;
54 }
55
56 while (1) {
57 /* read eeprom status */
58 ESC_read (ESCREG_EECONTSTAT, &stat, sizeof (eep_stat_t));
59 stat.contstat.reg = etohs(stat.contstat.reg);
60 stat.addr = etohl(stat.addr);
61
62 /* check busy flag, exit if job finished */
63 if (!stat.contstat.bits.busy) {
64 return;
65 }
66
67 /* clear error bits */
68 stat.contstat.bits.csumErr = 0;
69 stat.contstat.bits.eeLoading = 0;
70 stat.contstat.bits.ackErr = 0;
71 stat.contstat.bits.wrErr = 0;
72
73 /* process commands */
74 switch (stat.contstat.bits.cmdReg) {
75 case EEP_CMD_IDLE:
76 break;
77
78 case EEP_CMD_READ:
79 case EEP_CMD_RELOAD:
80 /* handle read request */
81 if (EEP_read (stat.addr * sizeof(uint16_t), eep_buf, EEP_READ_SIZE) != 0) {
82 stat.contstat.bits.ackErr = 1;
83 } else {
84 ESC_write (ESCREG_EEDATA, eep_buf, EEP_READ_SIZE);
85 }
86 break;
87
88 case EEP_CMD_WRITE:
89 /* handle write request */
90 ESC_read (ESCREG_EEDATA, eep_buf, EEP_WRITE_SIZE);
91 if (EEP_write (stat.addr * sizeof(uint16_t), eep_buf, EEP_WRITE_SIZE) != 0) {
92 stat.contstat.bits.ackErr = 1;
93 }
94 break;
95
96 default:
97 stat.contstat.bits.ackErr = 1;
98 }
99
100 /* acknowledge command */
101 stat.contstat.reg = htoes(stat.contstat.reg);
102 ESC_write (ESCREG_EECONTSTAT, &stat.contstat.reg, sizeof(uint16_t));
103 }
104}
