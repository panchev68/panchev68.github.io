164uint8_t ESC_checkmbx (uint8_t state)
165{
166 _ESCsm2 *SM;
167 ESC_read (ESCREG_SM0, (void *) &ESCvar.SM[0], sizeof (ESCvar.SM[0]));
168 ESC_read (ESCREG_SM1, (void *) &ESCvar.SM[1], sizeof (ESCvar.SM[1]));
169 SM = (_ESCsm2 *) & ESCvar.SM[0];
170 if ((etohs (SM->PSA) != MBX0_sma) || (etohs (SM->Length) != MBX0_sml)
171 || (SM->Command != MBX0_smc) || (ESCvar.SM[0].ECsm == 0))
172 {
173 ESCvar.SMtestresult = SMRESULT_ERRSM0;
174 ESC_SMdisable (0);
175 ESC_SMdisable (1);
176 return (uint8_t) (ESCinit | ESCerror); //fail state change
177 }
178 SM = (_ESCsm2 *) & ESCvar.SM[1];
179 if ((etohs (SM->PSA) != MBX1_sma) || (etohs (SM->Length) != MBX1_sml)
180 || (SM->Command != MBX1_smc) || (ESCvar.SM[1].ECsm == 0))
181 {
182 ESCvar.SMtestresult = SMRESULT_ERRSM1;
183 ESC_SMdisable (0);
184 ESC_SMdisable (1);
185 return ESCinit | ESCerror; //fail state change
186 }
187 return state;
188}

199{
200 ESC_SMenable (0);
201 ESC_SMenable (1);
202 ESC_SMstatus (0);
203 ESC_SMstatus (1);

205 {
206 ESC_ALerror (ALERR_INVALIDMBXCONFIG);
207 MBXrun = 0;
208 }
209 else
210 {
211 ESCvar.toggle = ESCvar.SM[1].ECrep; //sync repeat request toggle state
212 MBXrun = 1;
213 }
214 return state;

308void ESC_readmbx (void)
309{
310 _MBX *MB = &MBX[0];
311 uint16_t length;
312
313 if (ESCvar.ALstatus == ESCboot)
314 {
315 ESC_read (MBX0_sma_b, MB, MBXHSIZE);
316 length = etohs (MB->header.length);
317
318 if (length > (MBX0_sml_b - MBXHSIZE))
319 {
320 length = MBX0_sml_b - MBXHSIZE;
321 }
322 ESC_read (MBX0_sma_b + MBXHSIZE, &(MB->b[0]), length);
323 if (length + MBXHSIZE < MBX0_sml_b)
324 {
325 ESC_read (MBX0_sme_b, &length, 1);
326 }
327 }
328 else
329 {
330 ESC_read (MBX0_sma, MB, MBXHSIZE);
331 length = etohs (MB->header.length);
332
333 if (length > (MBX0_sml - MBXHSIZE))
334 {
335 length = MBX0_sml - MBXHSIZE;
336 }
337 ESC_read (MBX0_sma + MBXHSIZE, &(MB->b[0]), length);
338 if (length + MBXHSIZE < MBX0_sml)
339 {
340 ESC_read (MBX0_sme, &length, 1);
341 }
342 }
343 MBXcontrol[0].state = MBXstate_inclaim;
344}
351void ESC_writembx (uint8_t n)
352{
353 _MBX *MB = &MBX[n];
354 uint8_t dummy = 0;
355 uint16_t length;
356 length = etohs (MB->header.length);
357 if (ESCvar.ALstatus == ESCboot)
358 {
359 if (length > (MBX1_sml_b - MBXHSIZE))
360 {
361 length = MBX1_sml_b - MBXHSIZE;
362 }
363 ESC_write (MBX1_sma_b, MB, MBXHSIZE + length);
364 if (length + MBXHSIZE < MBX1_sml_b)
365 {
366 ESC_write (MBX1_sme_b, &dummy, 1);
367 }
368 }
369 else
370 {
371 if (length > (MBX1_sml - MBXHSIZE))
372 {
373 length = MBX1_sml - MBXHSIZE;
374 }
375 ESC_write (MBX1_sma, MB, MBXHSIZE + length);
376 if (length + MBXHSIZE < MBX1_sml)
377 {
378 ESC_write (MBX1_sme, &dummy, 1);
379 }
380 }
381 ESCvar.mbxfree = 0;

406uint8_t ESC_claimbuffer (void)
407{
408 _MBX *MB;
409 uint8_t n = MBXBUFFERS - 1;
410 while ((n > 0) && (MBXcontrol[n].state))
411 {
412 n--;
413 }
414 if (n)
415 {
416 MBXcontrol[n].state = MBXstate_outclaim;
417 MB = &MBX[n];
418 ESCvar.mbxcnt = (++ESCvar.mbxcnt) & 0x07;
419 if (ESCvar.mbxcnt == 0)
420 {
421 ESCvar.mbxcnt = 1;
422 }
423 MB->header.address = htoes (0x0000); // destination is master
424 MB->header.channel = 0;
425 MB->header.priority = 0;
426 MB->header.mbxcnt = ESCvar.mbxcnt;
427 ESCvar.txcue++;
428 }
429 return n;
430}

436uint8_t ESC_outreqbuffer (void)

438 uint8_t n = MBXBUFFERS - 1;

451void MBX_error (uint16_t error)
452{
453 uint8_t MBXout;
454 _MBXerr *mbxerr;
455 MBXout = ESC_claimbuffer ();
456 if (MBXout)
457 {
458 mbxerr = (_MBXerr *) & MBX[MBXout];
459 mbxerr->mbxheader.length = htoes ((uint16_t) 0x04);
460 mbxerr->mbxheader.mbxtype = MBXERR;
461 mbxerr->type = htoes ((uint16_t) 0x01);
462 mbxerr->detail = htoes (error);
463 MBXcontrol[MBXout].state = MBXstate_outreq;
464 }

473uint8_t ESC_mbxprocess (void)
474{
475 uint8_t mbxhandle = 0;
476 _MBX *MB = &MBX[0];
477
478 if (!MBXrun)
479 {
480 /* nothing to do */
481 return 0;
482 }
483
484 /* SM0/1 access or SMn change event */
485 if (ESCvar.ALevent & ESCREG_ALEVENT_SM_MASK)
486 {
487 ESC_SMstatus (0);
488 ESC_SMstatus (1);
489 }
490
491 /* outmbx read by master */
492 if (ESCvar.mbxoutpost && ESCvar.SM[1].IntR)
493 {
494 ESC_ackmbxread ();
495 /* dispose old backup */
496 if (ESCvar.mbxbackup)
497 {
498 MBXcontrol[ESCvar.mbxbackup].state = MBXstate_idle;
499 }
500 /* if still to do */
501 if (MBXcontrol[ESCvar.mbxoutpost].state == MBXstate_again)
502 {
503 ESC_writembx (ESCvar.mbxoutpost);
504 }
505 /* create new backup */
506 MBXcontrol[ESCvar.mbxoutpost].state = MBXstate_backup;
507 ESCvar.mbxbackup = ESCvar.mbxoutpost;
508 ESCvar.mbxoutpost = 0;
509 /* Do we have any ongoing protocol transfers, return 1 */
510 if(ESCvar.xoe > 0)
511 {
512 return 1;
513 }
514 return 0;
515 }
516
517 /* repeat request */
518 if (ESCvar.SM[1].ECrep != ESCvar.toggle)
519 {
520 if (ESCvar.mbxoutpost || ESCvar.mbxbackup)
521 {
522 /* if outmbx empty */
523 if (!ESCvar.mbxoutpost)
524 {
525 /* use backup mbx */
526 ESC_writembx (ESCvar.mbxbackup);
527 }
528 else
529 {
530 /* reset mailbox */
531 ESC_SMdisable (1);
532 /* have to resend later */
533 MBXcontrol[ESCvar.mbxoutpost].state = MBXstate_again;
534 /* activate mailbox */
535 ESC_SMenable (1);
536 /* use backup mbx */
537 ESC_writembx (ESCvar.mbxbackup);
538 }
539 ESCvar.toggle = ESCvar.SM[1].ECrep;
540 ESCvar.SM[1].PDIrep = ESCvar.toggle;
541 ESC_SMwritepdi (1);
542 }
543 return 0;
544 }
545
546 /* if the outmailbox is free check if we have something to send */
547 if (ESCvar.txcue && (ESCvar.mbxfree || !ESCvar.SM[1].MBXstat))
548 {
549 /* check out request mbx */
550 mbxhandle = ESC_outreqbuffer ();
551 /* outmbx empty and outreq mbx available */
552 if (mbxhandle)
553 {
554 ESC_writembx (mbxhandle);
555 /* change state */
556 MBXcontrol[mbxhandle].state = MBXstate_outpost;
557 ESCvar.mbxoutpost = mbxhandle;
558 if (ESCvar.txcue)
559 {
560 ESCvar.txcue--;
561 }
562 }
563 }
564
565 /* read mailbox if full and no xoe in progress */
566 if (ESCvar.SM[0].MBXstat && !MBXcontrol[0].state && !ESCvar.mbxoutpost
567 && !ESCvar.xoe)
568 {
569 ESC_readmbx ();
570 ESCvar.SM[0].MBXstat = 0;
571 if (etohs (MB->header.length) == 0)
572 {
573 MBX_error (MBXERR_INVALIDHEADER);
574 /* drop mailbox */
575 MBXcontrol[0].state = MBXstate_idle;
576 }
577 if ((MB->header.mbxcnt != 0) && (MB->header.mbxcnt == ESCvar.mbxincnt))
578 {
579 /* drop mailbox */
580 MBXcontrol[0].state = MBXstate_idle;
581 }
582 ESCvar.mbxincnt = MB->header.mbxcnt;
583 return 1;
584 }
585
586 /* ack changes in non used SM */
587 if (ESCvar.ALevent & ESCREG_ALEVENT_SMCHANGE)
588 {
589 ESC_SMack (4);
590 ESC_SMack (5);
591 ESC_SMack (6);
592 ESC_SMack (7);
593 }
594
595 return 0;
596}
600void ESC_xoeprocess (void)
601{
602 _MBXh *mbh;
603 if (!MBXrun)
604 {
605 return;
606 }
607 if (!ESCvar.xoe && (MBXcontrol[0].state == MBXstate_inclaim))
608 {
609 mbh = (_MBXh *) & MBX[0];
610 if ((mbh->mbxtype == 0) || (etohs (mbh->length) == 0))
611 {
612 MBX_error (MBXERR_INVALIDHEADER);
613 }
614 else
615 {
616 MBX_error (MBXERR_UNSUPPORTEDPROTOCOL);
617 }
618 /* mailbox type not supported, drop mailbox */
619 MBXcontrol[0].state = MBXstate_idle;
620 }
621}

629uint8_t ESC_checkSM23 (uint8_t state)
630{
631 _ESCsm2 *SM;
632 ESC_read (ESCREG_SM2, (void *) &ESCvar.SM[2], sizeof (ESCvar.SM[2]));
633 ESC_read (ESCREG_SM3, (void *) &ESCvar.SM[3], sizeof (ESCvar.SM[3]));
634 SM = (_ESCsm2 *) & ESCvar.SM[2];
635 if ((etohs (SM->PSA) != SM2_sma) || (etohs (SM->Length) != SM2_sml)
636 || (SM->Command != SM2_smc) || !(SM->ActESC & SM2_act))
637 {
638 ESCvar.SMtestresult = SMRESULT_ERRSM2;
639 /* fail state change */
640 return (ESCpreop | ESCerror);
641 }
642 SM = (_ESCsm2 *) & ESCvar.SM[3];
643 if ((etohs (SM->PSA) != SM3_sma) || (etohs (SM->Length) != SM3_sml)
644 || (SM->Command != SM3_smc) || !(SM->ActESC & SM3_act))
645 {
646 ESCvar.SMtestresult = SMRESULT_ERRSM3;
647 /* fail state change */
648 return (ESCpreop | ESCerror);
649 }
650 return state;
651}

660uint8_t ESC_startinput (uint8_t state)
661{
662 state = ESC_checkSM23 (state);
663 if (state != (ESCpreop | ESCerror))
664 {
665 ESC_SMenable (3);
666 App.state = APPSTATE_INPUT;
667 }
668 else
669 {
670 ESC_SMdisable (2);
671 ESC_SMdisable (3);
672 if (ESCvar.SMtestresult & SMRESULT_ERRSM3)
673 {
674 ESC_ALerror (ALERR_INVALIDINPUTSM);
675 }
676 else
677 {
678 ESC_ALerror (ALERR_INVALIDOUTPUTSM);
679 }
680 }
681 return state;
682}

725void ESC_state (void)
726{
727 uint8_t ac, an, as, ax, ax23;
728 uint8_t handle_smchanged = 0;
729
730 /* Do we have a state change request pending */
731 if (ESCvar.ALevent & ESCREG_ALEVENT_CONTROL)
732 {
733 ESC_read (ESCREG_ALCONTROL, (void *) &ESCvar.ALcontrol,
734 sizeof (ESCvar.ALcontrol));
735 ESCvar.ALcontrol = etohs (ESCvar.ALcontrol);
736 }
737 /* Have at least on Sync Manager changed */
738 else if (ESCvar.ALevent & ESCREG_ALEVENT_SMCHANGE)
739 {
740 handle_smchanged = 1;
741 }
742 else
743 {
744 /* nothing to do */
745 return;
746 }
747 /* Mask state request bits + Error ACK */
748 ac = ESCvar.ALcontrol & ESCREG_AL_STATEMASK;
749 as = ESCvar.ALstatus & ESCREG_AL_STATEMASK;
750 an = as;
751 if (((ac & ESCerror) || (ac == ESCinit)))
752 {
753 /* if error bit confirmed reset */
754 ac &= ESCREG_AL_ERRACKMASK;
755 an &= ESCREG_AL_ERRACKMASK;
756 }
757 /* Enter SM changed handling for all steps but Init and Boot when Mailboxes
758 * is up and running
759 */
760 if (handle_smchanged && (as & ESCREG_AL_ALLBUTINITMASK) &&
761 !(as == ESCboot) && MBXrun)
762 {
763 /* Validate Sync Managers, reading the Activation register will
764 * acknowledge the SyncManager Activation event making us enter
765 * this execution path.
766 */
767 ax = ESC_checkmbx (as);
768 ax23 = ESC_checkSM23 (as);
769 if ((an & ESCerror) && !(ac & ESCerror))
770 {
771 /* if in error then stay there */
772 return;
773 }
774 /* Have we been forced to step down to INIT we will stop mailboxes,
775 * update AL Status Code and exit ESC_state
776 */
777 if (ax == (ESCinit | ESCerror))
778 {
779 /* If we have activated Inputs and Outputs we need to disable them */
780 if (App.state)
781 {
782 ESC_stopoutput ();
783 ESC_stopinput ();
784 }
785 /* Stop mailboxes and update ALStatus code */
786 ESC_stopmbx ();
787 ESC_ALerror (ALERR_INVALIDMBXCONFIG);
788 MBXrun = 0;
789 ESC_ALstatus (ax);
790 return;
791 }
792 /* Have we been forced to step down to PREOP we will stop inputs
793 * and outputs, update AL Status Code and exit ESC_state
794 */
795 if ((App.state) && (ax23 == (ESCpreop | ESCerror)))
796 {
797 ESC_stopoutput ();
798 ESC_stopinput ();
799 if (ESCvar.SMtestresult & SMRESULT_ERRSM3)
800 {
801 ESC_ALerror (ALERR_INVALIDINPUTSM);
802 }
803 else
804 {
805 ESC_ALerror (ALERR_INVALIDOUTPUTSM);
806 }
807 ESC_ALstatus (ax23);
808 return;
809 }
810 }
811
812 /* Error state not acked, leave original */
813 if ((an & ESCerror) && !(ac & ESCerror))
814 {
815 return;
816 }
817
818 /* Mask high bits ALcommand, low bits ALstatus */
819 as = (ac << 4) | (as & 0x0f);
820
821 /* Call post state change hook case it have been configured */
822 if ((esc_cfg != NULL) && esc_cfg->pre_state_change_hook)
823 {
824 esc_cfg->pre_state_change_hook (&as, &an);
825 }
826
827 /* Switch through the state change requested via AlControl from
828 * current state read in AL status
829 */
830 switch (as)
831 {
832 case INIT_TO_INIT:
833 case PREOP_TO_PREOP:
834 case OP_TO_OP:
835 {
836 break;
837 }
838 case INIT_TO_PREOP:
839 {
840 /* get station address */
841 ESC_address ();
842 an = ESC_startmbx (ac);
843 break;
844 }
845 case INIT_TO_BOOT:
846 case BOOT_TO_BOOT:
847 {
848 /* get station address */
849 ESC_address ();
850 an = ESC_startmbxboot (ac);
851 break;
852 }
853 case INIT_TO_SAFEOP:
854 case INIT_TO_OP:
855 {
856 an = ESCinit | ESCerror;
857 ESC_ALerror (ALERR_INVALIDSTATECHANGE);
858 break;
859 }
860 case OP_TO_INIT:
861 {
862 ESC_stopoutput ();
863 }
864 case SAFEOP_TO_INIT:
865 {
866 ESC_stopinput ();
867 }
868 case PREOP_TO_INIT:
869 {
870 ESC_stopmbx ();
871 an = ESCinit;
872 break;
873 }
874 case BOOT_TO_INIT:
875 {
876 ESC_stopmbx ();
877 an = ESCinit;
878 break;
879 }
880 case PREOP_TO_BOOT:
881 case BOOT_TO_PREOP:
882 case BOOT_TO_SAFEOP:
883 case BOOT_TO_OP:
884 {
885 an = ESCpreop | ESCerror;
886 ESC_ALerror (ALERR_INVALIDSTATECHANGE);
887 break;
888 }
889 case PREOP_TO_SAFEOP:
890 case SAFEOP_TO_SAFEOP:
891 {
892 SM2_sml = sizeRXPDO ();
893 SM3_sml = sizeTXPDO ();
894 an = ESC_startinput (ac);
895 if (an == ac)
896 {
897 ESC_SMenable (2);
898 }
899 break;
900 }
901 case PREOP_TO_OP:
902 {
903 an = ESCpreop | ESCerror;
904 ESC_ALerror (ALERR_INVALIDSTATECHANGE);
905 break;
906 }
907 case OP_TO_PREOP:
908 {
909 ESC_stopoutput ();
910 }
911 case SAFEOP_TO_PREOP:
912 {
913 ESC_stopinput ();

916 }
917 case SAFEOP_TO_BOOT:

920 ESC_ALerror (ALERR_INVALIDSTATECHANGE);
921 break;
922 }

924 {

926 break;
927 }
928 case OP_TO_BOOT:
929 {
930 an = ESCsafeop | ESCerror;
931 ESC_ALerror (ALERR_INVALIDSTATECHANGE);
932 ESC_stopoutput ();
933 break;
934 }
935 case OP_TO_SAFEOP:
936 {
937 an = ESCsafeop;
938 ESC_stopoutput ();
939 break;
940 }
941 default:
942 {
943 if (an == ESCop)
944 {
945 ESC_stopoutput ();
946 an = ESCsafeop;
947 }
948 if (as == ESCsafeop)
949 {
950 ESC_stopinput ();
951 }
952 an |= ESCerror;
953 ESC_ALerror (ALERR_UNKNOWNSTATE);
954 break;
955 }
956 }
957
958 /* Call post state change hook case it have been configured */
959 if ((esc_cfg != NULL) && esc_cfg->post_state_change_hook)
960 {
961 esc_cfg->post_state_change_hook (&as, &an);
962 }
963
964 if (!(an & ESCerror) && (ESCvar.ALerror))
965 {
966 /* clear error */
967 ESC_ALerror (ALERR_NONE);
968 }
969
970 ESC_ALstatus (an);
971
