Example usage of the JaggedArray class template.

This example demonstrates how to use the JaggedArray class template to create a jagged array of integers. The example below shows how to add rows and elements to the jagged array and access them using the subscript operator.

#include "JaggedArray.hpp"
#include <iostream>
int main() {
using namespace Lib::Helper;
// Create a jagged array of integers
Array<int> jaggedArray;
// Create some rows
Row<int> row1;
Row<int> row2;
// Add elements to the rows
int a = 1, b = 2, c = 3, d = 4;
row1.add(a);
row1.add(b);
row2.add(c);
row2.add(d);
// Add rows to the jagged array
jaggedArray.add(row1);
jaggedArray.add(row2);
// Access and print elements
for (size_t i = 0; i < jaggedArray.size(); ++i) {
Row<int>& row = jaggedArray[i];
for (size_t j = 0; j < row.size(); ++j) {
std::cout << row[j] << " ";
}
std::cout << std::endl;
}
return 0;
}
