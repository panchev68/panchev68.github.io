107 inline long int std::atol (const char * str) {
108 int res = 0; // Initialize result
109 int sign = 1; // Initialize sign as positive
110 int i = 0; // Initialize index of first digit
111
112 if (str [0] == '-') // If number is negative, then update sign
113 {
114 sign = -1;
115 i++; // Also update index of first digit
116 }
117
118 // Iterate through all digits and update the result
119 for (; str [i] != '\0'; ++i) {
120 res = res * 10 + str [i] - '0';
121 }
122 return sign * res; // Return result with sign
123 }

125 inline double strtod (const char * string, char ** endPtr) {
126 int sign, expSign = FALSE;
127 double fraction, dblExp, *d;
128 register const char * p;
129 register int c;
130 int exp = 0; // Exponent read from "EX" field.
131
132 // Exponent that derives from the fractional part. Under normal circumstatnces, it is the negative of the
133 // number of digits in F. However, if I is very long, the last digits of I get dropped (otherwise a long I with
134 // a large negative exponent could cause an unnecessary overflow on I alone). In this case, fracExp is
135 // incremented one for each dropped digit.
136 int fracExp = 0;
137
138 int mantSize; // Number of digits in mantissa.
139 int decPt; // Number of mantissa digits BEFORE decimal point.
140 const char * pExp; // Temporarily holds location of exponent in string.
141
142 /*
143 * Strip off leading blanks and check for a sign.
144 */
145
146 p = string;
147 while (isspace (*p)) {
148 p += 1;
149 }
150 if (*p == '-') {
151 sign = TRUE;
152 p += 1;
153 }
154 else {
155 if (*p == '+') {
156 p += 1;
157 }
158 sign = FALSE;
159 }
160
161 // Count the number of digits in the mantissa (including the decimal point), and also locate the decimal point.
162
163 decPt = -1;
164 for (mantSize = 0;; mantSize += 1) {
165 c = *p;
166 if (!isdigit (c)) {
167 if ((c != '.') || (decPt >= 0)) {
168 break;
169 }
170 decPt = mantSize;
171 }
172 p += 1;
173 }
174
175 // Now suck up the digits in the mantissa. Use two integers to collect 9 digits each (this is faster than
176 // using floating-point). If the mantissa has more than 18 digits, ignore the extras, since they can't affect
177 // the value anyway.
178
179 pExp = p;
180 p -= mantSize;
181 if (decPt < 0) {
182 decPt = mantSize;
183 }
184 else {
185 mantSize -= 1; // One of the digits was the point.
186 }
187 if (mantSize > 18) {
188 fracExp = decPt - 18;
189 mantSize = 18;
190 }
191 else {
192 fracExp = decPt - mantSize;
193 }
194 if (mantSize == 0) {
195 fraction = 0.0;
196 p = string;
197 goto done;
198 }
199 else {
200 int frac1, frac2;
201 frac1 = 0;
202 for (; mantSize > 9; mantSize -= 1) {
203 c = *p;
204 p += 1;
205 if (c == '.') {
206 c = *p;
207 p += 1;
208 }
209 frac1 = 10 * frac1 + (c - '0');
210 }
211 frac2 = 0;
212 for (; mantSize > 0; mantSize -= 1) {
213 c = *p;
214 p += 1;
215 if (c == '.') {
216 c = *p;
217 p += 1;
218 }
219 frac2 = 10 * frac2 + (c - '0');
220 }
221 fraction = (1.0e9 * frac1) + frac2;
222 }
223
224 /*
225 * Skim off the exponent.
226 */
227
228 p = pExp;
229 if ((*p == 'E') || (*p == 'e')) {
230 p += 1;
231 if (*p == '-') {
232 expSign = TRUE;
233 p += 1;
234 }
235 else {
236 if (*p == '+') {
237 p += 1;
238 }
239 expSign = FALSE;
240 }
241 while (isdigit (*p)) {
242 exp = exp * 10 + (*p - '0');
243 p += 1;
244 }
245 }
246 if (expSign) {
247 exp = fracExp - exp;
248 }
249 else {
250 exp = fracExp + exp;
251 }
252
253 /*
254 * Generate a floating-point number that represents the exponent.
255 * Do this by processing the exponent one bit at a time to combine
256 * many powers of 2 of 10. Then combine the exponent with the
257 * fraction.
258 */
259
260 if (exp < 0) {
261 expSign = TRUE;
262 exp = -exp;
263 }
264 else {
265 expSign = FALSE;
266 }
267 if (exp > maxExponent) {
268 exp = maxExponent;
269 errno = ERANGE;
270 }
271 dblExp = 1.0;
272 for (d = powersOf10; exp != 0; exp >>= 1, d += 1) {
273 if (exp & 01) {
274 dblExp *= *d;
275 }
276 }
277 if (expSign) {
278 fraction /= dblExp;
279 }
280 else {
281 fraction *= dblExp;
282 }
283
284 done:
285 if (endPtr != NULL) {
286 *endPtr = (char *) p;
287 }
288
289 if (sign) {
290 return -fraction;
291 }
292 return fraction;
293 }
