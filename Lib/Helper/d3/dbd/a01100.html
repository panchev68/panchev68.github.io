Example usage of the RingBuffer class.

This example demonstrates how to use the RingBuffer class to create a ring buffer of integers. The example below shows how to push and pop elements from the buffer.

#include <iostream>
#include "RingBuffer.hpp"
int main() {
// Create a ring buffer of integers with a size of 5
Lib::Helper::RingBuffer<int, 5> buffer;
// Push some elements onto the buffer
for (int i = 0; i < 5; ++i) {
buffer.push(i);
}
// Pop elements from the buffer
int value;
while (buffer.pop(value)) {
std::cout << "Popped value: " << value << std::endl;
}
return 0;
}
