This example shows how to calculate crc16 checksum of block

#include <Helper/crc16.hpp>
#include <vector>
std::vector<uint8_t> vector;
void example ()
{
vector = {1,2,3,4,5 6}
{
auto result = crc16(vector.data(), vector.size());
}
// --- or ---
{
auto result = crc16(vector);
}
}
