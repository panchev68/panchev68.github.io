397 static DayOfWeek dayOfWeek (const Date & date)
398 {
399 auto fm = [date] () {
400 int fmonth, leap;
401 if ((date.year % 100 == 0) && (date.year % 400 != 0)) leap = 0;
402 else if (date.year % 4 == 0) leap = 1;
403 else leap = 0;
404
405 fmonth = 3 + (2 - leap) * ((date.month + 2) / (2 * date.month)) + (5 * date.month + date.month / 9) / 2;
406
407 return fmonth % 7;
408 };
409
410 int yy = date.year % 100;
411 int century = date.year / 100;
412 uint8_t res = (static_cast<uint32_t> (1.25 * yy + fm () + date.day - 2 * (century % 4)) % 7);
413
414 (res >= 2) ? res -= 2 : res += 5;
415 res++;
416
417 return static_cast<DayOfWeek> (res);
418 }
