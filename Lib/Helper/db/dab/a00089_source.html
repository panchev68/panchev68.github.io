364 inline double Sunset::calcEquationOfTime (double t) const
365 {
366 double epsilon = calcObliquityCorrection (t);
367 double l0 = calcGeomMeanLongSun (t);
368 double e = calcEccentricityEarthOrbit (t);
369 double m = calcGeomMeanAnomalySun (t);
370 double y = tan (degToRad (epsilon) / 2.0);
371
372 y *= y;
373
374 double sin2l0 = sin (2.0 * degToRad (l0));
375 double sinm = sin (degToRad (m));
376 double cos2l0 = cos (2.0 * degToRad (l0));
377 double sin4l0 = sin (4.0 * degToRad (l0));
378 double sin2m = sin (2.0 * degToRad (m));
379 double Etime = y * sin2l0 - 2.0 * e * sinm + 4.0 * e * y * sinm * cos2l0 - 0.5 * y * y * sin4l0 - 1.25 * e * e * sin2m;
380 return radToDeg (Etime) * 4.0;
381 }

435 inline double Sunset::calcJD (int y, int m, int d) const
436 {
437 if (m <= 2)
438 {
439 y -= 1;
440 m += 12;
441 }
442 double A = floor (y / 100);
443 double B = 2.0 - A + floor (A / 4);
444
445 double JD = floor (365.25 * (y + 4716)) + floor (30.6001 * (m + 1)) + d + B - 1524.5;
446 return JD;
447 }

455 inline double Sunset::calcSunEqOfCenter (double t) const
456 {
457 double m = calcGeomMeanAnomalySun (t);
458 double mrad = degToRad (m);
459 double sinm = sin (mrad);
460 double sin2m = sin (mrad + mrad);
461 double sin3m = sin (mrad + mrad + mrad);
462 double C = sinm * (1.914602 - t * (0.004817 + 0.000014 * t)) + sin2m * (0.019993 - 0.000101 * t) + sin3m * 0.000289;
463
464 return C;
465 }

467 inline double Sunset::calcAbsSunrise (double offset) const
468 {
469 double t = calcTimeJulianCent (_julianDate);
470 double eqTime = calcEquationOfTime (t);
471 double solarDec = calcSunDeclination (t);
472 double hourAngle = calcHourAngleSunrise (_latitude, solarDec, offset);
473 double delta = _longitude + radToDeg (hourAngle);
474 double timeDiff = 4 * delta;
475 double timeUTC = 720 - timeDiff - eqTime;
476 double newt = calcTimeJulianCent (calcJDFromJulianCent (t) + timeUTC / 1440.0);
477
478 eqTime = calcEquationOfTime (newt);
479 solarDec = calcSunDeclination (newt);
480
481 hourAngle = calcHourAngleSunrise (_latitude, solarDec, offset);
482 delta = _longitude + radToDeg (hourAngle);
483 timeDiff = 4 * delta;
484 timeUTC = 720 - timeDiff - eqTime;
485 return timeUTC;
