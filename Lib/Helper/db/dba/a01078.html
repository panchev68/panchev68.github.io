Example usage of the ContainerBase class template.

This example demonstrates how to use the ContainerBase class template to create a container for integers and perform basic operations.

#include "ContainerBase.hpp"
#include <iostream>
int main() {
Lib::Helper::ContainerBase<int> container;
// Add elements to the container
container.add(1);
container.add(2);
container.add(3);
// Iterate and print elements
container.each([](int &item) {
std::cout << item << std::endl;
});
// Access elements using subscript operator
std::cout << "Element at index 1: " << container[1] << std::endl;
// Remove an element
container.remove(2);
// Clear the container
container.clear();
return 0;
}
