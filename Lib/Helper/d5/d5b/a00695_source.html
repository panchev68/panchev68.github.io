104 inline void swapBytes (const void * src, void * dst, size_t len)
105 {
106 auto ps = reinterpret_cast<const uint8_t *> (src);
107 auto pd = reinterpret_cast<uint8_t *> (dst);
108 for (uint8_t i = 0, j = len - 1; i < len; i++, j--) { pd [j] = ps [i]; }
109 }
