499 static DayOfWeek dayOfWeek (const Date & date) {
500 // Lambda function to calculate the day of the week
501 auto fm = [date] () {
502 int fmonth, leap;
503 if ((date.year % 100 == 0) && (date.year % 400 != 0)) leap = 0; //
504 else if (date.year % 4 == 0) leap = 1;
505 else leap = 0;
506
507 fmonth = 3 + (2 - leap) * ((date.month + 2) / (2 * date.month)) + (5 * date.month + date.month / 9) / 2;
508
509 return fmonth % 7; // Return the day of the week
510 };
511
512 int yy = date.year % 100;
513 int century = date.year / 100;
514 uint8_t res = (static_cast<uint32_t> (1.25 * yy + fm () + date.day - 2 * (century % 4)) % 7);
515
516 (res >= 2) ? res -= 2 : res += 5;
517 res++;
518
519 return static_cast<DayOfWeek> (res); // Return the day of the week
520 }
