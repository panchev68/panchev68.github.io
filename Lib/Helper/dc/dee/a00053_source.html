537 static DayOfWeek dayOfWeek (const Date & date)
538 {
539 auto fm = [date] () {
540 int fmonth, leap;
541 if ((date.year % 100 == 0) && (date.year % 400 != 0)) leap = 0;
542 else if (date.year % 4 == 0) leap = 1;
543 else leap = 0;
544
545 fmonth = 3 + (2 - leap) * ((date.month + 2) / (2 * date.month)) + (5 * date.month + date.month / 9) / 2;
546
547 return fmonth % 7;
548 };
549
550 int yy = date.year % 100;
551 int century = date.year / 100;
552 uint8_t res = (static_cast<uint32_t> (1.25 * yy + fm () + date.day - 2 * (century % 4)) % 7);
553
554 (res >= 2) ? res -= 2 : res += 5;
555 res++;
556
557 return static_cast<DayOfWeek> (res);
558 }
