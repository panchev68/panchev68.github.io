Example usage of the Container class template.
#include <iostream>
using namespace Lib::Helper;
class MyClass {
public:
void display() {
std::cout << "MyClass object" << std::endl;
}
};
void processFunction(MyClass* obj, void* arg) {
obj->display();
}
int main() {
Container<MyClass> container;
MyClass obj1, obj2;
container.add(&obj1);
container.add(&obj2);
container.process(processFunction);
std::cout << "Container size: " << container.getSize() << std::endl;
return 0;
}
