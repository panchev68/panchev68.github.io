Example usage of the TimeZone class.

This example demonstrates how to use the TimeZone class to convert between local time and UTC time, and to check if a given time is in Daylight Saving Time (DST). The example below shows how to create a time zone object, set the time change rules, and use the conversion functions to convert between local time and UTC time.

#include <iostream>
#include "TimeZone.hpp"
int main() {
// Create a time zone object for Pacific Standard Time (PST)
TimeZone pst("PST", -480); // UTC-8:00
// Set the time change rules for DST
TimeZone::TimeChangeRule startRule = {"DST Start", 2, 0, 3, 2, 60}; // Second Sunday in March at 2:00 AM
TimeZone::TimeChangeRule endRule = {"DST End", 1, 0, 11, 2, 0}; // First Sunday in November at 2:00 AM
std::vector<TimeZone::TimeChangeRule> rules = {startRule, endRule};
pst.setRules(rules);
// Convert local time to UTC time
int localTime = 780; // 1:00 PM
std::string utcTime = pst.toUTC(localTime);
std::cout << "Local time: 1:00 PM, UTC time: " << utcTime << std::endl;
// Check if the local time is in DST
bool isDST = pst.locIsDST(localTime, startRule);
std::cout << "Is DST: " << (isDST ? "Yes" : "No") << std::endl;
return 0;
}
