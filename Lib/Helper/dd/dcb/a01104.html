Example usage of the StringHelper class.

This example demonstrates how to use the StringHelper class to convert strings to uppercase and lowercase, and to trim whitespace from the start and end of a string.

#include <iostream>
#include "StringHelper.hpp"
int main() {
std::string str = " Hello, World! ";
// Convert the string to uppercase
std::string upper = Lib::Helper::StringHelper::toUpperCase(str);
std::cout << "Uppercase: " << upper << std::endl;
// Convert the string to lowercase
std::string lower = Lib::Helper::StringHelper::toLowerCase(str);
std::cout << "Lowercase: " << lower << std::endl;
// Trim whitespace from the string
std::string trimmed = Lib::Helper::StringHelper::trim(str);
std::cout << "Trimmed: " << trimmed << std::endl;
return 0;
}
