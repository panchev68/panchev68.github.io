501 static DayOfWeek dayOfWeek (const Date & date) {
502 // Lambda function to calculate the day of the week
503 auto fm = [date] () {
504 int fmonth, leap;
505 if ((date.year % 100 == 0) && (date.year % 400 != 0)) leap = 0; //
506 else if (date.year % 4 == 0) leap = 1;
507 else leap = 0;
508
509 fmonth = 3 + (2 - leap) * ((date.month + 2) / (2 * date.month)) + (5 * date.month + date.month / 9) / 2;
510
511 return fmonth % 7; // Return the day of the week
512 };
513
514 int yy = date.year % 100;
515 int century = date.year / 100;
516 uint8_t res = (static_cast<uint32_t> (1.25 * yy + fm () + date.day - 2 * (century % 4)) % 7);
517
518 (res >= 2) ? res -= 2 : res += 5;
519 res++;
520
521 return static_cast<DayOfWeek> (res); // Return the day of the week
522 }
