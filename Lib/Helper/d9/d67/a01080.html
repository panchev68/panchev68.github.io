Example usage of the Hex class.
#include <iostream>
#include "Hex.hpp"
int main() {
uint8_t value8 = 0x2A;
uint16_t value16 = 0x1234;
uint64_t value64 = 0x123456789ABCDEF0;
hex<uint8_t> hex8(value8);
hex<uint16_t> hex16(value16);
hex<uint64_t> hex64(value64);
std::cout << "Hex8: " << hex8.toString() << std::endl;
std::cout << "Hex16: " << hex16.toString() << std::endl;
std::cout << "Hex64: " << hex64.toString() << std::endl;
return 0;
}
