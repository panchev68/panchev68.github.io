Example usage of the Percent utility class.

This example demonstrates how to use the Percent utility class to convert percentage values to dBm values.

#include "Percent.hpp"
int main() {
uint8_t percent = 75;
int8_t dBm = Lib::Helper::Convert::Percent::toDBm(percent);
// dBm should be -62
return 0;
}
