Example usage of the Sunset class

This example demonstrates how to use the Sunset class to calculate sunrise, sunset, and moon phases based on geographical coordinates and date. The example below shows how to calculate the sunrise, sunset, and moon phase for a specific location and date.

#include "Sunset.hpp"
#include <iostream>
int main() {
// Create a Sunset object with latitude, longitude, and time zone offset
Lib::Helper::Sunset sunset(37.7749, -122.4194, -8); // San Francisco, UTC-8
// Set the current date
sunset.setCurrentDate(2023, 10, 15);
// Calculate and print the official sunrise and sunset times
double sunrise = sunset.calcSunrise();
double sunsetTime = sunset.calcSunset();
Lib::Helper::DateTime::Time sunriseTime = Lib::Helper::Sunset::toTime(sunrise);
Lib::Helper::DateTime::Time sunsetTimeObj = Lib::Helper::Sunset::toTime(sunsetTime);
std::cout << "Sunrise: " << sunriseTime.hours << ":" << sunriseTime.minutes << std::endl;
std::cout << "Sunset: " << sunsetTimeObj.hours << ":" << sunsetTimeObj.minutes << std::endl;
// Calculate and print the moon phase
int moonPhase = sunset.moonPhase();
std::cout << "Moon Phase: " << moonPhase << std::endl;
return 0;
}
