Example usage of the DateTime class.

This example demonstrates the usage of the DateTime class to create a DateTime object, print its string representation, check if it is valid, pack it into a 32-bit integer, and unpack the integer back into a DateTime object.

#include <iostream>
int main() {
// Create a DateTime object
DateTime dt(Date(25, 12, 2023), Time(10, 30, 45));
// Print the DateTime object
std::cout << "DateTime: " << dt.toString() << std::endl;
// Check if the DateTime object is valid
if (dt.isValid()) {
std::cout << "The DateTime object is valid." << std::endl;
} else {
std::cout << "The DateTime object is invalid." << std::endl;
}
// Pack the DateTime object into a 32-bit integer
uint32_t packed = DateTime::pack(dt);
std::cout << "Packed DateTime: " << packed << std::endl;
// Unpack the 32-bit integer back into a DateTime object
DateTime unpacked = DateTime::unpack(packed);
std::cout << "Unpacked DateTime: " << unpacked.toString() << std::endl;
return 0;
}
