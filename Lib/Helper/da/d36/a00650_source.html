390 static DayOfWeek dayOfWeek (const Date & date)
391 {
392 auto fm = [date] () {
393 int fmonth, leap;
394 if ((date.year % 100 == 0) && (date.year % 400 != 0)) leap = 0;
395 else if (date.year % 4 == 0) leap = 1;
396 else leap = 0;
397
398 fmonth = 3 + (2 - leap) * ((date.month + 2) / (2 * date.month)) + (5 * date.month + date.month / 9) / 2;
399
400 return fmonth % 7;
401 };
402
403 int yy = date.year % 100;
404 int century = date.year / 100;
405 uint8_t res = (static_cast<uint32_t> (1.25 * yy + fm () + date.day - 2 * (century % 4)) % 7);
406
407 (res >= 2) ? res -= 2 : res += 5;
408 res++;
409
410 return static_cast<DayOfWeek> (res);
411 }
