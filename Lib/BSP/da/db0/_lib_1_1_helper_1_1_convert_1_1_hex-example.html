Lib::Helper::Convert::Hex

Example usage of Hex conversion utilities.

Demonstrates conversion of various unsigned integer types to hexadecimal and BCD string representations.

Example usage:

#include <iostream>
#include "Lib/Helper/Convert/Hex.hpp"
int main() {
uint8_t value8 = 0x2A;
uint16_t value16 = 0x1234;
uint32_t value32 = 0xABCDEF12;
uint64_t value64 = 0x123456789ABCDEF0ULL;
std::cout << "Hex<uint8_t>::toString: " << Lib::Helper::Convert::Hex<uint8_t>::toString(value8) << std::endl;
std::cout << "Hex<uint16_t>::toString: " << Lib::Helper::Convert::Hex<uint16_t>::toString(value16) << std::endl;
std::cout << "Hex<uint32_t>::toString: " << Lib::Helper::Convert::Hex<uint32_t>::toString(value32) << std::endl;
std::cout << "Hex<uint64_t>::toString: " << Lib::Helper::Convert::Hex<uint64_t>::toString(value64) << std::endl;
std::cout << "Hex<uint8_t>::toBcdString: " << Lib::Helper::Convert::Hex<uint8_t>::toBcdString(value8) << std::endl;
std::cout << "Hex<uint16_t>::toBcdString: " << Lib::Helper::Convert::Hex<uint16_t>::toBcdString(value16) << std::endl;
std::cout << "Hex<uint32_t>::toBcdString: " << Lib::Helper::Convert::Hex<uint32_t>::toBcdString(value32) << std::endl;
return 0;
}
