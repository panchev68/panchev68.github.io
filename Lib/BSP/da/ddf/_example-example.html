template<size_t N>
auto createOptimalQueue() {
if constexpr (RTOS::Optimization::should_inline_function<N>()) {
// Use inline implementation for small queues
return Queue<uint8_t, N>();
} else {
// Use standard implementation for large queues
return Queue<uint8_t, N>();
}
}
