71 /** Helper to parse build date and time macros to string.
72 * @return Build date string.
73 */
74 static std::string buildDateString() {
75 auto getDigit = [](const char * ptr, uint32_t offset) -> uint8_t {
76 assert(ptr);
77 return static_cast<uint8_t>(ptr[offset] - '0');
78 };
79 Lib::Helper::DateTime dt;
80 dt.time.hours = 10 * getDigit(__TIME__, 0) + getDigit(__TIME__, 1);
81 dt.time.minutes = 10 * getDigit(__TIME__, 3) + getDigit(__TIME__, 4);
82 dt.time.seconds = 10 * getDigit(__TIME__, 6) + getDigit(__TIME__, 7);
83 dt.date.year = 1000 * getDigit(__DATE__, 7) + 100 * getDigit(__DATE__, 8) +
84 10 * getDigit(__DATE__, 9) + getDigit(__DATE__, 10);
85
86 // Month parsing
87 uint8_t month = 0;
88 switch (__DATE__[0]) {
89 case 'J':
90 month = (__DATE__[1] == 'a') ? 1 : (__DATE__[2] == 'n' ? 6 : 7);
91 break;
92 case 'F': month = 2; break;
93 case 'M': month = (__DATE__[2] == 'r') ? 3 : 5; break;
94 case 'A': month = (__DATE__[1] == 'p') ? 4 : 8; break;
95 case 'S': month = 9; break;
96 case 'O': month = 10; break;
97 case 'N': month = 11; break;
98 case 'D': month = 12; break;
99 default: month = 0; break;
100 }
101 dt.date.month = month;
102
103 // Day parsing
104 dt.date.day = 10 * ((__DATE__[4] == ' ') ? 0 : getDigit(__DATE__, 4)) + getDigit(__DATE__, 5);
105
106 return dt.date.toString();
107 }
