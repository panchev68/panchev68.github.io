42 const std::string getBuildDate() const {
43 auto getDigit = [](const char * ptr, uint32_t offset) -> uint8_t {
44 assert(ptr);
45 return static_cast<uint8_t>(ptr[offset] - '0');
46 };
47 Lib::Helper::DateTime dt;
48 dt.time.hours = 10 * getDigit(__TIME__, 0) + getDigit(__TIME__, 1);
49 dt.time.minutes = 10 * getDigit(__TIME__, 3) + getDigit(__TIME__, 4);
50 dt.time.seconds = 10 * getDigit(__TIME__, 6) + getDigit(__TIME__, 7);
51 dt.date.year = 1000 * getDigit(__DATE__, 7) + 100 * getDigit(__DATE__, 8) +
52 10 * getDigit(__DATE__, 9) + getDigit(__DATE__, 10);
53 uint8_t monthIndex = 0;
54 if (__DATE__[0] == 'J') monthIndex = (__DATE__[1] == 'a') ? 0 : (__DATE__[2] == 'n' ? 5 : 6);
55 else if (__DATE__[0] == 'F') monthIndex = 1;
56 else if (__DATE__[0] == 'M') monthIndex = (__DATE__[2] == 'r') ? 2 : 4;
57 else if (__DATE__[0] == 'A') monthIndex = (__DATE__[1] == 'p') ? 3 : 7;
58 else if (__DATE__[0] == 'S') monthIndex = 8;
59 else if (__DATE__[0] == 'O') monthIndex = 9;
60 else if (__DATE__[0] == 'N') monthIndex = 10;
61 else if (__DATE__[0] == 'D') monthIndex = 11;
62 dt.date.month = MONTH_MAP[monthIndex];
63 dt.date.day = 10 * (__DATE__[4] == ' ' ? 0 : getDigit(__DATE__, 4)) + getDigit(__DATE__, 5);
64 return dt.date.toString();
