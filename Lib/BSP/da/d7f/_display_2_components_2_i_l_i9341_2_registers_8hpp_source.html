14enum Registers : uint8_t {

35 RAMWR = 0x2C, /*!< Memory write. */

50 INVCTR = 0xB4, /*!< Display inversion control. */
51 DFUNCTR = 0xB6, /*!< Display function control. */
52 PWCTR1 = 0xC0, /*!< Power control 1. */
53 PWCTR2 = 0xC1, /*!< Power control 2. */
54 PWCTR3 = 0xC2, /*!< Power control 3. */
55 PWCTR4 = 0xC3, /*!< Power control 4. */
56 PWCTR5 = 0xC4, /*!< Power control 5. */
57 VMCTR1 = 0xC5, /*!< VCOM control 1. */
58 VMCTR2 = 0xC7, /*!< VCOM control 2. */
59 POWCTRA = 0xCB, /*!< Power control A. */
60 POWCTRB = 0xCF, /*!< Power control B. */
61 RDID1 = 0xDA, /*!< Read ID1. */
62 RDID2 = 0xDB, /*!< Read ID2. */
63 RDID3 = 0xDC, /*!< Read ID3. */
64 RDID4 = 0xDD, /*!< Read ID4. */
65 GMCTRP1 = 0xE0, /*!< Positive gamma correction. */
66 GMCTRN1 = 0xE1, /*!< Negative gamma correction. */
67 DRVTCTRA = 0xE8, /*!< Driver timing control A. */
68 DRVTCTRB = 0xEA, /*!< Driver timing control B. */
69 POWONSEQCTRL = 0xED, /*!< Power on sequence control. */
70 ENBLABLE3G = 0xF2, /*!< Enable 3G. */
71 PUMPCRTL = 0xF7 /*!< Pump ratio control. */
72};
