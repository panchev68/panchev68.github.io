Lib::XCOM::Protocol

Example usage of Protocol class for building and parsing XCOM frames.

This example demonstrates how to build a protocol frame using the Protocol class and then parse it back to extract the command and payload.

Example usage:

#include "Lib/XCOM/Protocol.hpp"
#include <iostream>
int main() {
Lib::XCOM::Protocol::Frame frame;
frame.command = 0xA5;
frame.payload = {0x01, 0x02, 0x03};
// Build frame
std::vector<uint8_t> encodedFrame = Lib::XCOM::Protocol::build(frame);
// Parse frame
Lib::XCOM::Protocol::Frame parsedFrame;
bool success = Lib::XCOM::Protocol::parse(encodedFrame, parsedFrame);
// Output results
std::cout << "Parse success: " << success << std::endl;
std::cout << "Command: " << static_cast<int>(parsedFrame.command) << std::endl;
std::cout << "Payload:";
for (auto value : parsedFrame.payload) std::cout << " " << static_cast<int>(value);
std::cout << std::endl;
return 0;
}
