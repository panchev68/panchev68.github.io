19 /** Base class for user-defined protocol parsing and message building.
20 *
21 * The UserProtocol class provides an abstract interface for parsing incoming messages and building
22 * outgoing messages according to user-defined protocol commands. It defines a set of protocol commands
23 * and callback methods for handling specific protocol actions. Derived classes can override these
24 * callbacks to implement custom protocol logic. The class ensures efficient message handling and
25 * type safety for Digi XBee API communication.
26 */
27 class UserProtocol {
28 public:
29 /** User-defined protocol commands. */
30 enum Command : uint8_t {
31 INPUTS = 0, /*!< Command for inputs update. */
32 GET_PARAMETERS = 1, /*!< Command for getting parameters. */
33 OUTPUTS = 10, /*!< Command for outputs update. */
34 SET_PARAMETERS = 11 /*!< Command for setting parameters. */
35 };

44 /** Parses the incoming message.
45 * @param buffer Reference to vector containing the received data.
46 * @return `true` if handled, `false` otherwise.
47 * @warning Asserts if buffer is empty or command is invalid.
48 */
49 virtual bool parse(const std::vector<uint8_t> & buffer) {
50 size_t length = buffer.size();
51 if (length == 0) return false;
52 auto command = buffer.at(0);
53 length -= 1;
54 const uint8_t * data = (length > 0) ? (buffer.data() + 1) : nullptr;
55 switch (static_cast<Command>(command)) {
56 case INPUTS:
57 handleInputsUpdate(data, length);
58 return true;
59 case GET_PARAMETERS:
60 handleGetParameters(data, length);
61 return false;
62 case OUTPUTS:
63 handleOutputsUpdate(data, length);
64 return true;
65 case SET_PARAMETERS:
66 handleSetParameters(data, length);
67 return false;
68 default:
69 assert(false);
70 return false;
71 }
72 }
73
74 /** Builds the outgoing message.
75 * @param[out] buffer Result buffer.
76 * @param cmd User command.
77 * @param data Pointer to data (payload).
78 * @param len Data length.
79 * @warning Asserts if data is nullptr when len > 0.
80 */
81 virtual void build(std::vector<uint8_t> & buffer, Command cmd, const void * data, size_t len) {
82 buffer.clear();
83 buffer.push_back(static_cast<uint8_t>(cmd));
84 if (len == 0) return;
85 assert(data != nullptr);
86 auto src = static_cast<const uint8_t *>(data);
87 for (size_t i = 0; i < len; ++i) buffer.push_back(src[i]);
88 }
89
90 /** Handles inputs update.
91 * @param data Pointer to input data.
92 * @param len Data length.
93 */
94 virtual void handleInputsUpdate(const uint8_t * data, size_t len) {}
95
96 /** Handles get parameters.
97 * @param data Pointer to parameter data.
98 * @param len Data length.
99 */
100 virtual void handleGetParameters(const uint8_t * data, size_t len) {}
101
102 /** Handles set parameters.
103 * @param data Pointer to parameter data.
104 * @param len Data length.
105 */
106 virtual void handleSetParameters(const uint8_t * data, size_t len) {}
107
108 /** Handles outputs update.
109 * @param data Pointer to output data.
110 * @param len Data length.
111 */
112 virtual void handleOutputsUpdate(const uint8_t * data, size_t len) {}
113 };
