141 void handleReceivedCharEvent(uint8_t data) {
142 constexpr size_t MAX_FRAME_SIZE = 65;
143 static RTOS::Timeout timeout;
144 static int32_t remain = -1;
145 static size_t payloadLen = 0;
146 static bool inProgress = false;
147
148 if (inProgress && timeout.isExpired()) {
149 diagnostic.ignoredMessagesByTimeout++;
150 rxFrame.clear();
151 remain = -1;
152 inProgress = false;
153 }
154
155 size_t pos = rxFrame.getSize();
156 if (pos == offsetof(Frame::Header, Frame::Header::startDelimiter)) {
157 if (data == Protocol::START_DELIMITER) {
158 if (inProgress) assert(!"Message Overlap");
159 emitBeginMessageReceive();
160 inProgress = true;
161 rxFrame.clear();
162 remain = -1;
163 rxFrame.append(data);
164 timeout.set(5);
165 }
166 timeout.set(5);
167 return;
168 }
169 rxFrame.append(data);
170 if (pos == offsetof(Frame::Header, Frame::Header::lengthMSB)) {
171 payloadLen = static_cast<size_t>(data) << 8;
172 } else if (pos == offsetof(Frame::Header, Frame::Header::lengthLSB)) {
173 payloadLen |= data;
174 if (payloadLen >= MAX_FRAME_SIZE) {
175 diagnostic.maxMsgLenghtErrors++;
176 rxFrame.clear();
177 emitEndMessageReceived();
178 inProgress = false;
179 remain = -1;
180 return;
181 }
182 remain = payloadLen + sizeof(uint8_t);
183 } else {
184 if (remain > 0) --remain;
185 if (remain == 0) {
186 if (Protocol::isValid(rxFrame)) {
187 Protocol::processMessage(rxFrame);
188 } else {
189 diagnostic.invalidMessages++;
190 assert(!"Invalid message");
191 }
192 rxFrame.clear();
193 emitEndMessageReceived();
194 inProgress = false;
195 remain = -1;
196 }
197 }
198 }
