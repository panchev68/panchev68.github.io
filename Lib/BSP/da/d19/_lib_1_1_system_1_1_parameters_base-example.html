Lib::System::ParametersBase

Example usage of ParametersBase for system parameter management.

Demonstrates how to implement a custom storage backend and use ParametersBase to manage persistent system parameters with CRC validation and default restoration.

Example usage:

#include <Lib/System/ParametersBase.hpp>
struct MyParams {
uint32_t value;
uint8_t flags;
};
class MyStorage : public Lib::System::ParametersBase::StorageBase {
public:
bool open() override { return true; }
bool close() override { return true; }
bool read(void * buffer, size_t size) override {
// Simulate reading from persistent storage
std::memset(buffer, 0, size);
return true;
}
bool write(const void * buffer, size_t size) override {
// Simulate writing to persistent storage
return true;
}
};
class MyParamsManager : public Lib::System::ParametersBase {
public:
MyParamsManager(StorageBase * storage, MyParams * params)
: ParametersBase(storage, params, sizeof(MyParams)), params(params) {}
protected:
void restoreByDefault() override {
params->value = 42;
params->flags = 0xFF;
}
private:
MyParams * params;
};
int main() {
MyParams params;
MyStorage storage;
MyParamsManager manager(&storage, &params);
manager.setup();
// params now contains either loaded or default values
return 0;
}
