126 /** Calculates the moon phase index for a given epoch time.
127 * @param epochTime Epoch time in seconds.
128 * @return Moon phase index (0-29).
129 */
130 int calculateMoonPhase(int epochTime) const {
131 static constexpr int MOON_EPOCH{614100}; /*!< Reference epoch for moon phase calculation. */
132 static constexpr int LUNAR_CYCLE_SECONDS{2551443}; /*!< Duration of lunar cycle in seconds. */
133 int phase{(epochTime - MOON_EPOCH) % LUNAR_CYCLE_SECONDS};
134 int result{static_cast<int>(std::floor(phase / (24.0 * 3600.0))) + 1};
135 return (result == 30) ? 0 : result;
136 }

185 /** Calculates the Julian date for a given calendar date.
186 * @param year Year value.
187 * @param month Month value.
188 * @param day Day value.
189 * @return Julian date for the specified date.
190 */
191 double calculateJulianDate(int year, int month, int day) const {
192 if (month <= 2) {
193 year -= 1;
194 month += 12;
195 }
196 double A{std::floor(year / 100.0)};
197 double B{2.0 - A + std::floor(A / 4.0)};
198 return std::floor(365.25 * (year + 4716)) + std::floor(30.6001 * (month + 1)) + day + B - 1524.5;
199 }

201 /** Calculates the absolute sun time (sunrise or sunset) for a given sun angle.
202 * @param angle Sun angle for event calculation.
203 * @param isSunrise True for sunrise, false for sunset.
204 * @return Sun time in minutes from midnight UTC.
205 */
206 double calculateAbsoluteSunTime(double angle, bool isSunrise) const {
207 double t{calculateTimeJulianCenturies(julianDate)};
208 double eqTime{calculateEquationOfTime(t)};
209 double solarDec{calculateSunDeclination(t)};
210 double hourAngle{isSunrise ? calculateHourAngleSunrise(geoLatitude, solarDec, angle) : calculateHourAngleSunset(geoLatitude, solarDec, angle)};
211 double delta{geoLongitude + radiansToDegrees(hourAngle)};
212 double timeDiff{4.0 * delta};
213 double timeUTC{720.0 - timeDiff - eqTime};
214 double newt{calculateTimeJulianCenturies(calculateJulianDateFromCenturies(t) + timeUTC / 1440.0)};
215 double newSolarDec{calculateSunDeclination(newt)};
216 double newHourAngle{isSunrise ? calculateHourAngleSunrise(geoLatitude, newSolarDec, angle) : calculateHourAngleSunset(geoLatitude, newSolarDec, angle)};
217 double newDelta{geoLongitude + radiansToDegrees(newHourAngle)};
218 return 720.0 - 4.0 * newDelta - calculateEquationOfTime(newt);
219 }

227 /** Calculates the hour angle for sunrise for a given latitude, declination, and sun angle.
228 * @param latitude Geographic latitude in degrees.
229 * @param solarDeclination Solar declination in degrees.
230 * @param angle Sun angle for event calculation.
231 * @return Hour angle for sunrise in radians.
232 */
233 static double calculateHourAngleSunrise(double latitude, double solarDeclination, double angle) {
234 double latRad{degreesToRadians(latitude)};
235 double declRad{degreesToRadians(solarDeclination)};
236 double cosH{std::cos(degreesToRadians(90.0 + angle)) / (std::cos(latRad) * std::cos(declRad)) -
237 std::tan(latRad) * std::tan(declRad)};
238 return std::acos(std::min(std::max(cosH, -1.0), 1.0));
239 }
240
241 /** Calculates the hour angle for sunset for a given latitude, declination, and sun angle.
242 * @param latitude Geographic latitude in degrees.
243 * @param solarDeclination Solar declination in degrees.
244 * @param angle Sun angle for event calculation.
245 * @return Hour angle for sunset in radians.
246 */
247 static double calculateHourAngleSunset(double latitude, double solarDeclination, double angle) {
248 double latRad{degreesToRadians(latitude)};
249 double declRad{degreesToRadians(solarDeclination)};
250 double cosH{std::cos(degreesToRadians(90.0 + angle)) / (std::cos(latRad) * std::cos(declRad)) -
251 std::tan(latRad) * std::tan(declRad)};
252 return -std::acos(std::min(std::max(cosH, -1.0), 1.0));
253 }

263 /** Calculates the equation of time for a given Julian centuries value.
264 * @param julianCenturies Julian centuries since J2000.0.
265 * @return Equation of time in minutes.
266 */
267 double calculateEquationOfTime(double julianCenturies) const {
268 double epsilon{23.439292 - 0.000013 * julianCenturies};
269 double l0{std::fmod(280.46646 + 36000.76983 * julianCenturies, 360.0)};
270 double e{0.016708634 - 0.000042037 * julianCenturies};
271 double m{std::fmod(357.52911 + 35999.05029 * julianCenturies, 360.0)};
272 double y{std::pow(std::tan(degreesToRadians(epsilon) / 2.0), 2)};
273 double eqTime{
274 y * std::sin(2.0 * degreesToRadians(l0)) - 2.0 * e * std::sin(degreesToRadians(m)) +
275 4.0 * e * y * std::sin(degreesToRadians(m)) * std::cos(2.0 * degreesToRadians(l0)) -
276 0.5 * y * y * std::sin(4.0 * degreesToRadians(l0)) - 1.25 * e * e * std::sin(2.0 * degreesToRadians(m))};
277 return radiansToDegrees(eqTime) * 4.0;
278 }

280 /** Calculates the solar declination for a given Julian centuries value.
281 * @param julianCenturies Julian centuries since J2000.0.
282 * @return Solar declination in degrees.
283 */
284 double calculateSunDeclination(double julianCenturies) const {
285 double epsilon{23.439292 - 0.000013 * julianCenturies};
286 double l0{std::fmod(280.46646 + 36000.76983 * julianCenturies, 360.0)};
287 double m{std::fmod(357.52911 + 35999.05029 * julianCenturies, 360.0)};
288 double lambda{
289 l0 +
290 (1.914602 - 0.004817 * julianCenturies - 0.000014 * julianCenturies * julianCenturies) *
291 std::sin(degreesToRadians(m)) +
292 (0.019993 - 0.000101 * julianCenturies) * std::sin(2.0 * degreesToRadians(m)) +
293 0.000289 * std::sin(3.0 * degreesToRadians(m))};
294 return radiansToDegrees(std::asin(std::sin(degreesToRadians(epsilon)) * std::sin(degreesToRadians(lambda))));
295 }
