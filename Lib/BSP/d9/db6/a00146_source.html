189 void preProcessing ()
190 {
191 for (auto i = 0; i < T_PIN_NUM; i++)
192 {
193 auto port = _mapFields [i].port;
194 auto mask = _mapFields [i].mask;
195
196 auto & outReg = _txBuffer [port + MC33879::TxPort::OUTPUTS + 1];
197 auto & ctrlReg = _txBuffer [port + MC33879::TxPort::CONTROL + 1];
198
199 if ((grouping.getValue () & BitMask::at (i)) != 0)
200 {
201 outReg |= mask; // Activate Output
202 ctrlReg |= mask; // If an output is activated it remains active in the 33879 status register
203 }
204 else
205 {
206 outReg &= ~mask; // Deactivate Output
207 }
208 }
209 _txBuffer [0] = _diagCounter; // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
210 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
211 }
212
215 bool postProcessing ()
216 {
217 // Last Byte or received data must be equal to _diagCounter
218
219 if ((_rxBuffer.at (BUFFER_SIZE - 1) != _txBuffer.at (0)) && (_txBuffer.at (0) != 0)) // verifies that the ID (_diagCounter) of the received frame coincides with the last sent
220 {
221 // error: SPI Interface ...
222 if (_isSpiError == false)
223 {
224 _isSpiError = true;
225 _spiErronrsCounter++;
226 emitErrorEvent (true); // error is activated
227 assert (0);
228 }
229 return false;
230 }
231 else
232 {
233 // success : No Error in SPI Interface
234 if (_isSpiError == true)
235 {
236 _isSpiError = false;
237 emitErrorEvent (false); // error is cleared
238 }
239 }
240
241 // Processing of MC33879 Status after received ...
242
243 for (auto i = 0; i < T_PIN_NUM; i++)
244 {
245 uint64_t value = 0;
246 for (auto i = 0; i < T_PIN_NUM; i++)
247 {
248 auto & map_index = _mapFields [i];
249 if ((_rxBuffer [map_index.port + MC33879::RxPort::STATUS + 1] & map_index.mask) != 0)
250 {
251 value |= BitMask::at (i);
252 }
253 }
254 _statusRegister = value;
255 }
256 return true;
257 }

345 void taskFn ()
346 {
347 uint64_t value;
348
349 // out_mask - contains a mask for the whole port, for example: 00000'1111'1111'1111'1111 for 24-bit
350 for (;;)
351 {
352 _queueValueChanged.receive (value, portMAX_DELAY); // wait for Outputs changed
353 _outputsChangedCounter++;
354
355 if (_parameters.enable == true) // if Grouping is enable
356 {
357 value &= _valueMaskAnd; // clear unnecessary bits
358
359 // detect different bits ...
360
361 uint64_t diffBits = value ^ _result;
362
363 uint64_t diffActiveBits = 0; // number of active different bits (is 1)
364 auto diffBitCounter = 0; // different bits counter
365
366 // count the different bits...
367
368 for (auto i = 0; i < T_PIN_NUM; i++)
369 {
370 auto mask = BitMask::at (i);
371 // is changed is active
372 if ((diffBits & mask) && (value & mask))
373 {
374 diffActiveBits |= mask;
375 diffBitCounter++;
376 }
377 }
378
379 // process ...
380
381 if (diffBitCounter >= _parameters.outputsNumber) // NUMBER_OF_SWITCH_OUTPUTS
382 {
383 uint32_t count = 0;
384 uint64_t group_bit_mask = 0;
385
386 for (auto i = 0; i < T_PIN_NUM; i++)
387 {
388 group_bit_mask |= BitMask::at (i);
389
390 if ((diffActiveBits & BitMask::at (i)) != 0)
391 {
392 if ((++count) >= _parameters.outputsNumber)
393 {
394 count = 0;
395
396 _result &= value;
397 _result |= (value & group_bit_mask);
398
399 vTaskDelay (_parameters.switchPeriod);
400 }
401 }
402 }
403 }
404 }
405
406 _result = value;
407 }
408 }
