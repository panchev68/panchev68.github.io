197 auto checkPressed = [this] () {
198 bool pinState = (_nirq.read () == 0);

200 if (_debountTime == 0) return pinState;

215 auto readAdc = [this] (uint8_t cmd) {
216 auto writeByte = [this] (uint8_t data) -> void {
217 for (auto i = 0; i < 8; i++)
218 {
219 _si.write ((data & 0x80) != 0);
220 __NOP ();
221 data <<= 1;
222 _sck.write (0);
223 __NOP ();
224 _sck.write (1);
225 __NOP ();
226 }

229 uint32_t num = 0;
230 _si.write (0);
231 _nss.write (0);
232 writeByte (cmd);

239 for (auto i = 0; i < 16; i++)
