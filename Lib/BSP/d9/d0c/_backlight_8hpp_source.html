79 /** Handle tick event for dimming and sleep. */
80 void handleTickEvent() {
81 if (timeout.isExpired()) {
82 timeout.set(config.fadeStep);
83 if (value < parent->value) *parent->timerCCR = ++value;
84 else if (value > parent->value) *parent->timerCCR = --value;
85 }
86 if (isEnabled && timeoutSleep.isExpired()) {
87 timeoutSleep.disable();
88 parent->set(minValue);
89 }
90 }

135 volatile static uint32_t * resolveTimerCCR(const Config & config) {
136 switch (config.timChannel) {
137 case TIM_CHANNEL_1: return &config.timHandle->Instance->CCR1;
138 case TIM_CHANNEL_2: return &config.timHandle->Instance->CCR2;
139 case TIM_CHANNEL_3: return &config.timHandle->Instance->CCR3;
140 case TIM_CHANNEL_4: return &config.timHandle->Instance->CCR4;
141 default: assert(0); return nullptr;
142 }
