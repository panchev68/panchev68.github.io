78 uint8_t minValue; /*!< Minimum dimmer value. */
79 uint8_t maxValue; /*!< Maximum dimmer value. */
80
81 /** Handle tick event for dimming and sleep. */
82 void handleTickEvent() {
83 if (fadeTimeout.isExpired()) {
84 fadeTimeout.set(config.fadeStep);
85 if (currentValue < parent->targetValue) {
86 *parent->timerCCR = ++currentValue;
87 } else if (currentValue > parent->targetValue) {
88 *parent->timerCCR = --currentValue;
89 }
90 }
91 if (isEnabled && sleepTimeout.isExpired()) {
92 sleepTimeout.disable();
93 parent->set(minValue);
94 }
95 }
96 };

104public:

151private:
152 /** Resolve timer CCR register pointer from config.
153 * @param config Backlight configuration.
154 * @return Pointer to timer CCR register.
155 */
156 volatile static uint32_t * resolveTimerCCR(const Config & config) {
157 assert(config.timHandle);
158 assert(config.timHandle->Instance);
159 switch (config.timChannel) {
160 case TIM_CHANNEL_1: return &config.timHandle->Instance->CCR1;
161 case TIM_CHANNEL_2: return &config.timHandle->Instance->CCR2;
162 case TIM_CHANNEL_3: return &config.timHandle->Instance->CCR3;
163 case TIM_CHANNEL_4: return &config.timHandle->Instance->CCR4;
164 default:
165 assert(false);
166 return nullptr;
167 }
168 }
