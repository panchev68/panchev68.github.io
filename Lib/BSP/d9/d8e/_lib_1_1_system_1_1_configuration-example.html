Lib::System::Configuration

Example usage of Configuration class for OTP memory management.

Demonstrates reading, writing, and validating configuration data using the Configuration class.

Example usage:
#include "Lib/System/Configuration.hpp"
int main() {
// Create a Configuration object at offset 0
Lib::System::Configuration config(0);
// Prepare configuration bitset
Lib::System::Configuration::Bitset bits;
bits.set(0, true); // Set bit 0
bits.set(5, true); // Set bit 5
// Write configuration to OTP
bool writeResult = config.write(bits);
// Read configuration from OTP
Lib::System::Configuration::Bitset readBits = config.read();
// Check if configuration is empty
bool isEmpty = config.isEmpty();
// Check if configuration is valid
bool isValid = config.isValid();
return 0;
}
