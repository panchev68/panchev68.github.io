Lib::Helper::DataHandle

Example usage of DataHandle.

Demonstrates how to create and use a DataHandle for a raw buffer and a typed element.

Example usage:

#include "Lib/Helper/DataHandle.hpp"
#include <cstdint>
#include <cstdio>
int main() {
uint8_t buffer[16] = {0};
Lib::Helper::DataHandle<> handle(buffer, sizeof(buffer));
printf("Buffer size: %zu\n", handle.getSize());
handle.getData()[0] = 42;
printf("First byte: %u\n", handle.getData()[0]);
uint32_t value = 123456;
Lib::Helper::DataHandle<uint32_t> valueHandle(value);
printf("Value: %u\n", *valueHandle.getData());
return 0;
}
