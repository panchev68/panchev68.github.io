Lib::Helper::DateTime

Example usage of DateTime, Date, and Time classes.

Demonstrates creation, validation, formatting, packing, and unpacking of date/time values.

Example usage:

#include "Lib/Helper/DateTime.hpp"
#include <iostream>
int main() {
Date date(15, 6, 2024); // 15 June 2024
Time time(14, 30, 45); // 14:30:45
DateTime dateTime(date, time);
std::cout << "Date: " << date.toString() << std::endl;
std::cout << "Time: " << time.toString() << std::endl;
std::cout << "DateTime: " << dateTime.toString() << std::endl;
std::cout << "Is valid: " << (dateTime.isValid() ? "true" : "false") << std::endl;
std::cout << "Day of week: " << Date::getDayOfWeekShortString(date) << std::endl;
uint32_t packed = DateTime::pack(dateTime);
std::cout << "Packed value: " << packed << std::endl;
DateTime unpacked = DateTime::unpack(packed);
std::cout << "Unpacked: " << unpacked.toString() << std::endl;
return 0;
}
