2 * This file provides a C++17 wrapper class for managing FreeRTOS stream buffers, including creation,
3 * destruction, data transmission, and reception. The StreamBuffer class encapsulates the underlying
4 * FreeRTOS stream buffer handle and offers safe, type-checked methods for buffer operations. It
5 * supports move semantics, error checking, and provides utility methods for buffer status and
6 * utilization. This abstraction simplifies stream buffer usage in embedded applications, ensuring
7 * resource management and API consistency.
8 * @file StreamBuffer.hpp

58 /** Create a stream buffer.
59 * @param bufferSize Size of the buffer in bytes.
60 * @param triggerLevel Minimum bytes to trigger a receive.
61 * @return `true` if creation was successful, `false` otherwise.
62 * @pre bufferSize > 0, triggerLevel > 0, triggerLevel <= bufferSize.
63 * @post handle is valid if creation succeeds.
64 */
65 bool create(size_t bufferSize, size_t triggerLevel) {
66 assert(handle == nullptr);
67 assert(bufferSize > 0);
68 assert(triggerLevel > 0);
69 assert(triggerLevel <= bufferSize);
70 if (handle || bufferSize == 0 || triggerLevel == 0 || triggerLevel > bufferSize) return false;
71 handle = xStreamBufferCreate(bufferSize, triggerLevel);
72 return handle != nullptr;
73 }

75 /** Destroy the stream buffer and release resources. */

80 }

83 /** Send data to the stream buffer.
84 * @param source Pointer to source data.
85 * @param length Number of bytes to send.
86 * @param timeout Maximum ticks to wait.
87 * @return Number of bytes sent.

90 size_t send(const void * source, size_t length, TickType_t timeout = 0) {

111 /** Send data to the stream buffer from ISR context.
112 * @param source Pointer to source data.
