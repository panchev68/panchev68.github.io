111 /** Client task function. */
112 void taskFn() {
113 Socket::Error error;
114 size_t length;
115 DataHandle<> incomingData;
116 DataHandle<> outgoingData;
117
118 setConnectionState(ConnectionState::INIT);
119 initCallback();
120
121 while (true) {
122 FreeRTOS::delay(1);
123 while (!readLinkState()) {
124 if (getConnectionState() == ConnectionState::CONNECTED) {
125 setConnectionState(ConnectionState::DISCONNECTED);
126 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
127 }
128 FreeRTOS::delay(100);
129 }
130 if (getConnectionState() != ConnectionState::CONNECTED) {
131 if ((error = sockets[0].open(Socket::Protocol::TCP, remote.port, 0)) == Socket::Error::OK) {
132 if ((error = sockets[0].connect(remote)) == Socket::Error::OK) {
133 setConnectionState(ConnectionState::CONNECTED);
134 } else {
135 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
136 FreeRTOS::delay(100);
137 }
138 } else assert(0);
139 } else {
140 if ((length = sockets[0].getReceivedSize()) > 0) {
141 updateRxLedCallback(true);
142 diagnostic.rxMessageCounter++;
143 incomingData(new uint8_t[length], length);
144 if ((error = sockets[0].receive(incomingData, length)) == Socket::Error::OK) {
145 handleIncomingMessage(incomingData);
146 } else {
147 if (error != Socket::Error::ERR_DATALEN) {
148 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
149 }
150 }
151 delete incomingData.data();
152 }
153 if (handleOutgoingMessage(outgoingData)) {
154 if ((error = sockets[0].send(outgoingData, length)) == Socket::Error::OK) {
155 updateTxLedCallback(true);
156 diagnostic.txMessageCounter++;
157 } else {
158 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
159 setConnectionState(ConnectionState::DISCONNECTED);
160 }
161 }
162 }
163 taskYIELD();
164 handleApplicationProcess();
165 }
166 }
