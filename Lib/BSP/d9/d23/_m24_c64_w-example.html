Example usage of the M24C64W EEPROM I2C driver for STM32 BSP.

Example usage of the M24C64W EEPROM I2C driver for STM32 BSP.Demonstrates initialization, open, read, write, and close operations with a sample buffer.

Example usage:
#include "Lib/BSP/EEPROM/I2C/M24C64W.hpp"
#include <array>
#include <cstdint>
#include <iostream>
int main() {
Lib::BSP::EEPROM::I2C::M24C64W::Config config{};
Lib::BSP::EEPROM::I2C::M24C64W eeprom(config);
if (!eeprom.open()) {
// Failed to open EEPROM
return 1;
}
std::array<uint8_t, 32> writeBuffer{};
for (size_t i = 0; i < writeBuffer.size(); ++i) writeBuffer[i] = static_cast<uint8_t>(i);
if (!eeprom.write(0, writeBuffer.data(), writeBuffer.size())) {
// Failed to write
eeprom.close();
return 2;
}
std::array<uint8_t, 32> readBuffer{};
if (!eeprom.read(0, readBuffer.data(), readBuffer.size())) {
// Failed to read
eeprom.close();
return 3;
}
for (size_t i = 0; i < readBuffer.size(); ++i)
std::cout << static_cast<int>(readBuffer[i]) << " ";
std::cout << std::endl;
eeprom.close();
return 0;
}
