204 void taskFn () override
205 {
206 auto process = [this] (uint16_t * data, size_t len) -> bool {
207 // Lambda function to calculate checksum of the received data
208 auto calcChksum = [] (const uint8_t * data) -> uint8_t {
209 assert (data);
210 return (data [0] + data [1] + data [2] + data [3]) & 0xfe;
211 };
212
213 // Lambda function to extract raw temperature and humidity values from the received data
214 auto getRaw = [] (const uint8_t * data) -> int16_t {
215 assert (data);
216 return data [0] << 8 | data [1];
217 };
218
219 // Buffer to store the received data
220 uint8_t buf [5] = {0};
221
222 // Process the received data
223 uint32_t buf_pos = 0; // Buffer position
224 uint32_t bit_pos = 0; // Bit position
225 auto last_value = data [0]; // Last value
226
227 // Iterate over the received data
228 for (auto i = 0; i < len; i++)
229 {
230 auto sub = last_value;
231 last_value = data [i];
232 data [i] -= sub;
233
234 if (i >= 3)
235 {
236 auto j = i - 3;
237 if (data [i] > TIMER_THRESHOLD)
238 {
239 buf_pos = (j) / 8;
240 bit_pos = (j % 8);
241 buf [buf_pos] |= Base::BIT_MASK_8MSB.at (bit_pos);
242 }
243 }
244 }
245 // Extract the checksum and temperature/humidity values
246 const uint8_t & frame_chksum = buf [4];
247 // Calculate the checksum
248 auto chksum = calcChksum (buf);
249
250 // Update diagnostic information
251 // Check if the checksum is valid
252
253 diagnostic.frameChkSum = frame_chksum;
254 diagnostic.calculatedChkSum = chksum;
255
256 // Check if the checksum is valid
257 if (frame_chksum != chksum)
258 {
259 diagnostic.invalidFrameChkSum++;
260 return false;
261 }
262
263 bool error = false;
264
265 // Extract the raw temperature and humidity values
266 auto temperature = getRaw (&buf [2]);
267
268 auto humidity = getRaw (&buf [0]);
269
270 // Validate the extracted values against predefined thresholds
271
272 temperature = ((temperature & 0x8000) ? -(temperature & 0x7fff) : temperature);
273
274 if ((temperature <= MIN_TEMPERATURE) || (temperature >= MAX_TEMPERATURE))
275 {
276 diagnostic.invalidTemperature++;
277 error = true;
278 }
279
280 // Validate the extracted values against predefined thresholds
281
282 if ((humidity <= MIN_HUMIDITY) || (temperature >= MAX_HUMIDITY))
283 {
284 diagnostic.invalidHumidity++;
285 error = true;
286 }
287 // Return false if an error occurred
288 if (error == true) return false;
289
290 // Update the internal state if the values have changed
291 if (_rawTemperature != temperature)
292 {
293 _rawTemperature = temperature;
294 _temperature = (static_cast<float> (_rawTemperature)) / 10.0F;
295 _isChanged = true;
296 }
297 // Update the internal state if the values have changed
298 if (_rawHumidity != humidity)
299 {
300 _rawHumidity = humidity;
301 _humidity = (static_cast<float> (_rawHumidity)) / 10.0F;
302 _isChanged = true;
303 }
304 return true;
305 };
306
307 for (;;)
308 {
309 // Perform a startup impulse and begin receiving data from the sensor
310
311 // Delay for 500ms
312 vTaskDelay (500);
313
314 // Generate a startup impulse
315 startupImpulse (1);
316
317 // Begin receiving data from the sensor
318 beginReceiveData ((uint32_t *) _rawBuffer.data (), _rawBuffer.size ());
319
320 // Wait for the data to be received
321 auto res = waitForComplete (1000);
322
323 // End receiving data from the sensor
324 endReceivedData ();
325
326 // Process the received data
327 if (res == true) process (_rawBuffer.data (), _rawBuffer.size ());
328 else diagnostic.receivedTimeout++;
329 }
330 }
331 };
