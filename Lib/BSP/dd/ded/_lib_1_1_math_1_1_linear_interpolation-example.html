Lib::Math::LinearInterpolation

Example of using LinearInterpolation for float values.

Demonstrates how to create a linear interpolation table, add points, and interpolate values.

Example usage:
#include "Lib/Math/Interpolation/LinearInterpolation.hpp"
#include <cstdio>
int main() {
Lib::Math::LinearInterpolationF interp;
interp.addPoint(0.0f, 0.0f);
interp.addPoint(1.0f, 10.0f);
interp.addPoint(2.0f, 20.0f);
interp.addPoint(3.0f, 30.0f);
float x = 1.5f;
float y = interp.interpolate(x);
std::printf("Interpolated value at x=%.2f: %.2f\n", x, y);
x = 0.0f;
y = interp.interpolate(x);
std::printf("Interpolated value at x=%.2f: %.2f\n", x, y);
x = 3.0f;
y = interp.interpolate(x);
std::printf("Interpolated value at x=%.2f: %.2f\n", x, y);
x = 2.25f;
y = interp.interpolate(x);
std::printf("Interpolated value at x=%.2f: %.2f\n", x, y);
return 0;
}
