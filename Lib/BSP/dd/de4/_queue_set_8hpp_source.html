26/** QueueSet class.
27 * C++17 wrapper for FreeRTOS queue set management, including creation, destruction, and member operations.
28 */
29class QueueSet {
30public:
31 using Handle = QueueSetHandle_t; /**< Queue set handle type. */
32 using Member = QueueSetMemberHandle_t; /**< Queue set member handle type. */
33
34 /** Constructor of the QueueSet class. */
35 QueueSet() : handle(nullptr) {}
36
37 /** Destructor of the QueueSet class. */
38 ~QueueSet() {
39 if (handle) {
40 destroy();
41 assert(handle == nullptr);
42 }
43 }
44
45 /** Move constructor of the QueueSet class.
46 * @param other Source QueueSet to move from.
47 */
48 QueueSet(QueueSet && other) : handle(other.handle) { other.handle = nullptr; }
49
50 /** Move assignment operator of the QueueSet class.
51 * @param other Source QueueSet to move from.
52 * @return Reference to this QueueSet.
53 */
54 QueueSet & operator=(QueueSet && other) {
55 if (this != &other) {
56 if (handle) { destroy(); }
57 handle = other.handle;
58 other.handle = nullptr;
59 }
60 return *this;
61 }
62
63 /** Creates a queue set.
64 * @param length Number of members in the set.
65 * @return `true` if creation was successful, `false` otherwise.
66 */
67 bool create(UBaseType_t length) {
68 assert(handle == nullptr);
69 if (handle) return false;
70 handle = xQueueCreateSet(length);
71 return handle != nullptr;
72 }
73
74 /** Destroys the queue set.
75 * @return `true` if destruction was successful, `false` otherwise.
76 */
77 bool destroy() {
78 if (!handle) return false;
79 vQueueDelete(handle);
80 handle = nullptr;
81 return true;
82 }
83
84 /** Adds a member to the queue set.
85 * @param member Queue or semaphore handle to add.
86 * @return `true` if addition was successful, `false` otherwise.
87 */
88 bool addToSet(Member member) const {
89 assert(handle != nullptr);
90 assert(member != nullptr);
91 return xQueueAddToSet(member, handle) == pdPASS;
92 }
93
94 /** Removes a member from the queue set.
95 * @param member Queue or semaphore handle to remove.
96 * @return `true` if removal was successful, `false` otherwise.
97 */
98 bool removeFromSet(Member member) const {
99 assert(handle != nullptr);
100 assert(member != nullptr);
101 return xQueueRemoveFromSet(member, handle) == pdPASS;
102 }
103
104 /** Selects a member from the queue set with timeout.
105 * @param timeout Maximum ticks to wait.
106 * @return Handle of the ready member, or nullptr if timeout.
107 */
108 Member selectFromSet(TickType_t timeout) const {
109 assert(handle != nullptr);
110 return xQueueSelectFromSet(handle, timeout);
111 }
112
113 /** Selects a member from the queue set without waiting.
114 * @return Handle of the ready member, or nullptr if none.
115 */
116 Member selectFromSetNow() const { return selectFromSet(0); }
117
118 /** Selects a member from the queue set, waiting forever.
119 * @return Handle of the ready member.
120 */
121 Member selectFromSetForever() const { return selectFromSet(portMAX_DELAY); }
122
123 /** Selects a member from the queue set from ISR context.
124 * @return Handle of the ready member, or nullptr if none.
125 */
126 Member selectFromSetFromISR() const {
127 assert(handle != nullptr);
128 return xQueueSelectFromSetFromISR(handle);
129 }
130
131 /** Checks if the queue set is created.
132 * @return `true` if created, `false` otherwise.
133 */
134 bool isCreated() const { return handle != nullptr; }
135
136 /** Gets the native queue set handle.
137 * @return Queue set handle.
138 */
139 Handle getHandle() const { return handle; }
140
141private:
142 Handle handle; /**< Internal queue set handle. */
143
144 friend QueueSet createQueueSet(UBaseType_t length);
145
146 QueueSet(const QueueSet &) = delete;
147 QueueSet & operator=(const QueueSet &) = delete;
148};
