448 * ## Example usage of RTOS::Kernel for task management, timing, and heap diagnostics.

450 * This example demonstrates how to use the `RTOS::Kernel` class to manage tasks, perform delays,

464 * RTOS::Kernel::MemoryInfo memInfo = RTOS::Kernel::getMemoryInfo();
465 * std::printf("Initial heap: total=%u, free=%u, minFree=%u\n",
466 * static_cast<unsigned>(memInfo.totalHeap),
467 * static_cast<unsigned>(memInfo.freeHeap),
468 * static_cast<unsigned>(memInfo.minimumFreeHeap));
469 *
470 * // Delay for 500 ms
471 * RTOS::Kernel::delayMs(500);
472 *
473 * // Enter critical section
474 * RTOS::Kernel::withCriticalSection([] {
475 * std::printf("In critical section\n");
476 * });
477 *
478 * // Print tick count and uptime
479 * std::printf("Tick count: %u\n", static_cast<unsigned>(RTOS::Kernel::getTickCount()));

482 * // Print scheduler state

486 * // Print heap usage percent
487 * std::printf("Heap usage: %u%%\n", RTOS::Kernel::getHeapUsagePercent());
488 *
489 * // End task
490 * vTaskDelete(nullptr);
491 * }

493 * int main() {
494 * // Start FreeRTOS scheduler with one task
