Lib::Helper::swapBytes

Example of using swapBytes for various integer types and memory buffers.

This example demonstrates how to use the swapBytes utility functions to swap the byte order of 16-bit, 32-bit, and 64-bit signed and unsigned integers, as well as how to swap the byte order of a raw memory buffer. The output shows the original and swapped values in hexadecimal format.

Example usage:
#include <iostream>
#include <iomanip>
#include "Lib/Helper/SwapBytes.hpp"
int main() {
using Lib::Helper::swapBytes;
uint16_t u16 = 0x1234;
int16_t i16 = -0x1234;
uint32_t u32 = 0x12345678;
int32_t i32 = -0x12345678;
uint64_t u64 = 0x0123456789ABCDEFULL;
int64_t i64 = -0x0123456789ABCDEFLL;
std::cout << std::hex << std::showbase;
std::cout << "Original uint16_t: " << u16 << ", Swapped: " << swapBytes(u16) << std::endl;
std::cout << "Original int16_t: " << static_cast<uint16_t>(i16) << ", Swapped: "
<< static_cast<uint16_t>(swapBytes(i16)) << std::endl;
std::cout << "Original uint32_t: " << u32 << ", Swapped: " << swapBytes(u32) << std::endl;
std::cout << "Original int32_t: " << static_cast<uint32_t>(i32) << ", Swapped: "
<< static_cast<uint32_t>(swapBytes(i32)) << std::endl;
std::cout << "Original uint64_t: " << u64 << ", Swapped: " << swapBytes(u64) << std::endl;
std::cout << "Original int64_t: " << static_cast<uint64_t>(i64) << ", Swapped: "
<< static_cast<uint64_t>(swapBytes(i64)) << std::endl;
// Swapping a memory buffer
uint8_t srcBuffer[4] = {0x11, 0x22, 0x33, 0x44};
uint8_t dstBuffer[4] = {};
swapBytes(srcBuffer, dstBuffer, 4);
std::cout << "Original buffer: ";
for (size_t i = 0; i < 4; ++i) std::cout << "0x" << static_cast<int>(srcBuffer[i]) << " ";
std::cout << std::endl << "Swapped buffer: ";
for (size_t i = 0; i < 4; ++i) std::cout << "0x" << static_cast<int>(dstBuffer[i]) << " ";
std::cout << std::endl;
return 0;
}
