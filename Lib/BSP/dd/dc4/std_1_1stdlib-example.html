std::stdlib

Example usage of utility conversion functions.

Demonstrates how to use atol, strtod, itoa, and trip for string and value conversions.

Example usage:

#include "Lib/Helper/stdlib.hpp"
int main() {
// Convert string to long
const char * str = "-12345";
long value = std::atol(str);
// Convert string to double
const char * doubleStr = "3.14159";
char * endPtr = nullptr;
double dValue = std::strtod(doubleStr, &endPtr);
// Convert integer to string
char buffer[16];
char * result = std::itoa(255, buffer, sizeof(buffer), 16);
// Limit a value to a range
int limited = std::trip(42, 0, 10);
return 0;
}
