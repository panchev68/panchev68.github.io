27 /** LocalAtCommand class for Digi XBee modules.
28 *
29 * Manages sending and receiving local AT commands, handling response events,
30 * and maintaining open/close state for command transactions. The class provides
31 * queue-based message handling, event-driven response processing, and robust
32 * lifecycle management for Digi XBee AT command operations.
33 */
34 class LocalAtCommand {
35 public:
36 using Command = AtCommandString;
37
38 static constexpr size_t MAX_RESPONSE_SIZE = 32; /*!< Maximum response size. */
39 static constexpr uint32_t SEND_TIMEOUT = 200; /*!< Send timeout in ms. */
40
41 /** Constructor of LocalAtCommand class.
42 * @param module Pointer to the Digi XBee module.
43 */
44 explicit LocalAtCommand(Module * module) : module(module) { assert(module); }
45
46 /** Destructor of LocalAtCommand class. */
47 ~LocalAtCommand() = default;

49 /** Opens the Local AT command service. */
50 void open() {
51 assert(!isOpen);
52 if (queueMessage.isCreated()) {
53 /* The service is already open. */
54 assert(false);
55 return;
56 }
57 assert(module->responseEvents.onLocalAtCommand == nullptr);
58 module->responseEvents.onLocalAtCommand = [this](const Frame0x88 & frame) { handleReceivedData(frame); };
59 if (!queueMessage.create()) {
60 /* Queue creation failed. */
61 assert(false);
62 return;
63 }
64 isOpen = true;
65 }
66
67 /** Closes the Local AT command service. */
68 void close() {
69 isOpen = false;
70 module->responseEvents.onLocalAtCommand = nullptr;
71 queueMessage.destroy();
72 }
73
74 /** Checks if the Local AT command service is open.
75 * @return `true` if open, `false` otherwise.
76 */
77 bool isOpenState() const { return isOpen; }
78
79 /** Receives an AT command response into the provided buffer.
80 * @param buffer Destination buffer for received data.
81 * @param timeout Timeout duration in milliseconds.
82 * @return `true` if data was received, `false` otherwise.
83 */
84 bool receive(std::vector<uint8_t> & buffer, uint32_t timeout) {
85 QueueMessage message;
86 buffer.clear();
87 if (!queueMessage.receive(message, timeout)) return false;
88 buffer.assign(message.data.begin(), message.data.begin() + message.size);
89 return true;
90 }
91
92 /** Sends an AT command with parameters.
93 * @param command AT command string.
94 * @param params Parameters to send.
95 * @return `true` if send succeeded, `false` otherwise.
96 */
97 bool send(const Command & command, const std::initializer_list<uint8_t> & params) {
98 sendFrame.atCommand = command;
99 sendFrame.frameId = 0x08;
100 sendFrame.parameters = params;
101 queueMessage.reset();
102 return module->send(sendFrame, SEND_TIMEOUT);
103 }
104
105 /** Sends an AT command without parameters.
106 * @param command AT command string.
107 * @return `true` if send succeeded, `false` otherwise.
108 */
109 bool send(const Command & command) {
110 sendFrame.atCommand = command;
111 sendFrame.frameId = 0x08;
112 sendFrame.parameters.clear();
113 queueMessage.reset();
114 return module->send(sendFrame, SEND_TIMEOUT);
115 }
116
117 private:
118 RTOS::Queue<QueueMessage> queueMessage; /*!< FreeRTOS queue for messages. */
119 Module * module; /*!< Pointer to Digi XBee module. */
120 bool isOpen = false; /*!< Open state flag. */
121 Frame0x08 sendFrame; /*!< Frame for sending AT commands. */
122
123 /** Callback for receiving AT command responses.
124 * @param frame Received Frame0x88 response.
125 */
126 void handleReceivedData(const Frame0x88 & frame) {
127 static QueueMessage message;
128 assert(frame.frameId == 0x08);
129 if (frame.frameId == 0x08 && sendFrame.atCommand == frame.atCommand) {
130 message.size = static_cast<uint8_t>(frame.data.size());
131 assert(message.size < message.DATA_SIZE);
132 std::copy(frame.data.begin(), frame.data.end(), message.data.begin());
133 queueMessage.send(message, SEND_TIMEOUT);
134 } else {
135 /* Invalid frame ID or AT command. */
136 assert(false);
137 }
138 }
139 };
