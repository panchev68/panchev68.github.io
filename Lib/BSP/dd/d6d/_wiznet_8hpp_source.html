121 /** TCP client main task loop. */
122 void taskFn() {
123 Socket::Error errorCode;
124 size_t messageSize;
125 DataHandle<> receivedDataHandle;
126 DataHandle<> outgoingDataHandle;
127
128 setConnectionState(ConnectionState::INIT);
129 initCallback();
130
131 while (true) {
132 FreeRTOS::delay(1);
133 while (!readLinkState()) {
134 if (getConnectionState() == ConnectionState::CONNECTED) {
135 setConnectionState(ConnectionState::DISCONNECTED);
136 if (sockets[0].close() != Socket::Error::OK) assert(0);
137 }
138 FreeRTOS::delay(100);
139 }
140 if (getConnectionState() != ConnectionState::CONNECTED) {
141 if (sockets[0].open(Socket::Protocol::TCP, remote.port, 0) == Socket::Error::OK) {
142 if (sockets[0].connect(remote) == Socket::Error::OK) setConnectionState(ConnectionState::CONNECTED);
143 else {
144 if (sockets[0].close() != Socket::Error::OK) assert(0);
145 FreeRTOS::delay(100);
146 }
147 } else assert(0);
148 } else {
149 if ((messageSize = sockets[0].getReceivedSize()) > 0) {
150 updateRxLedCallback(true);
151 diagnostic.rxMessageCounter++;
152 receivedDataHandle(new uint8_t[messageSize], messageSize);
153 errorCode = sockets[0].receive(receivedDataHandle, messageSize);
154 if (errorCode == Socket::Error::OK) handleIncomingMessage(receivedDataHandle);
155 else if (errorCode != Socket::Error::ERR_DATALEN) {
156 if (sockets[0].close() != Socket::Error::OK) assert(0);
157 }
158 delete receivedDataHandle.data();
159 }
160 if (handleOutgoingMessage(outgoingDataHandle)) {
161 if (sockets[0].send(outgoingDataHandle, messageSize) == Socket::Error::OK) {
162 updateTxLedCallback(true);
163 diagnostic.txMessageCounter++;
164 } else {
165 if (sockets[0].close() != Socket::Error::OK) assert(0);
166 setConnectionState(ConnectionState::DISCONNECTED);
167 }
168 }
169 }
170 taskYIELD();
171 applicationLoopCallback();
172 }
173 }
