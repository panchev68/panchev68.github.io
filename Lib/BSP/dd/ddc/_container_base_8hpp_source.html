15/** Generic value container base for object management.
16 * @tparam T Type of the managed values.
17 */
18template <typename T>
19class ContainerBase {
20public:
21 /** Adds a value to the container if not already present.
22 * @param value Value to add.
23 * @return `true` if the value was added, `false` otherwise.
24 */
25 virtual bool add(const T & value) {
26 if (std::find(items.begin(), items.end(), value) != items.end()) return false;
27 items.emplace_back(value);
28 return true;
29 }
30
31 /** Removes a value from the container.
32 * @param value Value to remove.
33 * @return `true` if the value was removed, `false` otherwise.
34 */
35 virtual bool remove(const T & value) {
36 auto it = std::find(items.begin(), items.end(), value);
37 if (it == items.end()) return false;
38 items.erase(it);
39 return true;
40 }
41
42 /** Clears all values from the container. */
43 virtual void clear() { items.clear(); }
44
45 /** Accesses a value by index.
46 * @param index Index of the value.
47 * @return Value at the specified index.
48 */
49 virtual T operator[](size_t index) const {
50 if (index >= items.size()) assert(false);
51 return items.at(index);
52 }
53
54 /** Iterates over all values with a callback function.
55 * @param func Function to process each value.
56 */
57 virtual void each(const std::function<void(T &)> & func) {
58 if (!func) return;
59 for (auto & item : items) func(item);
60 }
61
62 /** Returns the number of values in the container.
63 * @return Number of values.
64 */
65 constexpr size_t size() const { return items.size(); }
66
67 /** Checks if the container is empty.
68 * @return `true` if empty, `false` otherwise.
69 */
70 constexpr bool isEmpty() const { return items.empty(); }
71
72 /** Returns a reference to the underlying vector.
73 * @return Reference to the vector of values.
74 */
75 std::vector<T> & getVector() { return items; }
76
77protected:
78 /** Constructor of ContainerBase class. */
79 constexpr ContainerBase() = default;
80
81 /** Destructor of ContainerBase class. */
82 virtual ~ContainerBase() = default;
83
84private:
85 std::vector<T> items; /*!< Vector of managed values. */
86};
