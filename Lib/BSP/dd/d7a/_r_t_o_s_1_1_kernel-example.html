RTOS::Kernel

Example usage of RTOS::Kernel for task management, timing, and heap diagnostics.

This example demonstrates how to use the RTOS::Kernel class to manage tasks, perform delays, enter critical sections, and retrieve heap and scheduler information in a FreeRTOS-based system.

Example usage:

The example creates a simple task, demonstrates delays, critical section usage, and prints heap and scheduler information to the standard output.

#include <cstdio>
#include "Lib/RTOS/Kernel.hpp"
void exampleTask(void *param) {
// Print initial heap info
RTOS::Kernel::MemoryInfo memInfo = RTOS::Kernel::getMemoryInfo();
std::printf("Initial heap: total=%u, free=%u, minFree=%u\n",
static_cast<unsigned>(memInfo.totalHeap),
static_cast<unsigned>(memInfo.freeHeap),
static_cast<unsigned>(memInfo.minimumFreeHeap));
// Delay for 500 ms
RTOS::Kernel::delayMs(500);
// Enter critical section
RTOS::Kernel::withCriticalSection([] {
std::printf("In critical section\n");
});
// Print tick count and uptime
std::printf("Tick count: %u\n", static_cast<unsigned>(RTOS::Kernel::getTickCount()));
std::printf("Uptime (ms): %u\n", RTOS::Kernel::getUptimeMs());
// Print scheduler state
auto state = RTOS::Kernel::getSchedulerState();
std::printf("Scheduler state: %d\n", static_cast<int>(state));
// Print heap usage percent
std::printf("Heap usage: %u%%\n", RTOS::Kernel::getHeapUsagePercent());
// End task
vTaskDelete(nullptr);
}
int main() {
// Start FreeRTOS scheduler with one task
xTaskCreate(exampleTask, "Example", 256, nullptr, 1, nullptr);
RTOS::Kernel::startScheduler();
// Should never reach here
return 0;
}
