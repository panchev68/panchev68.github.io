1/** GPIO pin handle abstraction.
2 * Defines the Handle class for referencing a specific GPIO port and pin. Used for type-safe identification and
3 * manipulation of GPIO pins in the GPIO subsystem.
4 * @file Handle.hpp
5 */
6#pragma once
7
8#include <cstdint> /*!< uint16_t type. */
9
10#if defined(HAL_GPIO_MODULE_ENABLED) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)

12namespace Lib::HAL::GPIO {
13
14/** Represents a GPIO port and pin.
15 * Provides construction, access, and comparison for a GPIO port/pin pair. Intended for use as a handle or key in GPIO
16 * APIs and device managers.
17 */
18class Handle {
19public:
20 /** Constructor of Handle class. */
21 Handle() : port{nullptr}, pinNumber{0} {}
22
23 /** Constructor of Handle class.
24 * @param port GPIO port pointer.
25 * @param pinNumber Pin number (0-15).
26 */
27 Handle(GPIO_TypeDef * port, uint16_t pinNumber) : port{port}, pinNumber{pinNumber} {}
28
29 /** Destructor of Handle class. */
30 virtual ~Handle() = default;
31
32 /** Equality operator of Handle class.
33 * @param ref Handle to compare.
34 * @return `true` if port and pin match, `false` otherwise.
35 */
36 bool operator==(const Handle & ref) const { return port == ref.port && pinNumber == ref.pinNumber; }
37
38 /** Gets the GPIO port pointer.
39 * @return GPIO port pointer.
40 */
41 GPIO_TypeDef * getPort() const { return port; }
42
43 /** Gets the pin number.
44 * @return Pin number (0-15).
45 */
46 uint16_t getPinNumber() const { return pinNumber; }
47
48 /** Gets pointer to the pin number.
49 * @return Pointer to pin number member.
50 */
51 uint16_t * getPointerToPinNumber() { return &pinNumber; }
52
53private:
54 GPIO_TypeDef * port; /*!< GPIO port pointer. */
55 uint16_t pinNumber; /*!< Pin number (0-15). */
56};
57
58} // namespace Lib::HAL::GPIO
59#endif
