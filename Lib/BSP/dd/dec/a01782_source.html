77 __packed struct Command
78 {
79 uint8_t len;
80 uint8_t cmd;
81 uint8_t data;
82 };
83
84 _nss.write (1); // de-select spi
85 _spi.open ();
86 _spi.bindTxCompleteEvent (([this] () { transferCpltCallback (); }));
87
88 setBacklight (0); // Switch off backlight
89 vTaskDelay (100); // startup deflay before reset
90 _reset (); // reset the display controller
91 _nss.write (0); // the TFT display is permanently selected
92
93 // const uint8_t * initSequence = (_IsTFTDisplay == true) ? _InitDataTFT : initData;
94
95 const uint8_t * initSequence = initData;
96 for (;;)
97 {
98 auto * row = reinterpret_cast<const Command *> (initSequence);
99 if (row->len == 0) break;
100 _writeReg (row->cmd);
101 initSequence = &row->data;
102 for (auto i = 0; i < row->len - 1; i++) _writeData (initSequence [i]);
103 initSequence += row->len - 1;
104 }
105 vTaskDelay (120);
106 _writeReg (Registers::DISPON);
107 vTaskDelay (150);
108 setBacklight (100); // in not use postInitCallback
109 postInitCallback (); // switch backlight and display rotation
110
111 return true;
112 }

114 /// \brief Transnit block to the display
115 /// \param data - pointer to buffer
116 /// \param x - X coordinate
117 /// \param y - Y coordinate
118 /// \param width - buffer width
119 /// \param height - height per buffer
120 /// \return The length of the copied data in bytes
121 void transmitBlock (const void * src, uint32_t x, uint32_t y, size_t width, size_t height) override
122 {
123 auto data = reinterpret_cast<const uint16_t *> (src);
124 auto sendBuffer = [this, &data, &x, &y, &width, &height] () -> size_t {
125 auto writeCommand = [this] (uint8_t reg, uint32_t begin, uint32_t end) {
126 std::array<uint8_t, 4> data{HI_BYTE (begin), LO_BYTE (begin), HI_BYTE (end), LO_BYTE (end)};
127 _writeReg (reg);
128 if (HAL_SPI_Transmit (_spi.getHandle (), data.data (), data.size (), 100) != HAL_OK) assert (0);
129 };
130
131 auto spiHandle = _spi.getHandle ();
132 spiHandle->Init.DataSize = SPI_DATASIZE_8BIT;
133
134 if (HAL_SPI_Init (spiHandle) != HAL_OK) assert (0);
135
136 // Sets the position
137 writeCommand (Registers::CASET, x, x + width - 1);
138 writeCommand (Registers::PASET, y, y + height - 1);
139 _writeReg (Registers::RAMWR);
140
141 spiHandle->Init.DataSize = SPI_DATASIZE_16BIT;
142 if (HAL_SPI_Init (spiHandle) != HAL_OK) assert (0);
143 return width * height;
144 };
145
146 DriverBase::setTransmitActive (1); // notifies TouchGFX that a data transfer has started
147 auto size = sendBuffer ();
148 assert (size <= 0xffff);
149 if (_spi.send (data, size) == false) assert (0); // 0 - without transfer completed
150 }
