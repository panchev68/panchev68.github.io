15/** FindPeaks class.
16 * Class for detecting rising and falling peaks in a stream of samples using a fixed-size window.
17 */
18class FindPeaks {
19public:
20 /** Window size constant. */
21 static constexpr size_t WINDOW_SIZE = 3;
22
23 /** Constructor of FindPeaks class. */
24 FindPeaks() = default;
25
26 /** Destructor of FindPeaks class. */
27 ~FindPeaks() = default;
28
29 /** Resets the peak finder state. */
30 void reset() {
31 risingPeak = PeakInfo{0, 0};
32 fallingPeak = PeakInfo{0, 0};
33 peakValue = 0;
34 sampleCount = 0;
35 window.fill(0);
36 }
37
38 /** Adds a sample to the window and updates peak detection.
39 * @param sample The input sample value.
40 */
41 void addSample(int sample) {
42 if (sampleCount < WINDOW_SIZE) {
43 window[sampleCount++] = sample;
44 if (sampleCount < WINDOW_SIZE) return;
45 } else {
46 std::move(window.begin() + 1, window.end(), window.begin());
47 window[WINDOW_SIZE - 1] = sample;
48 }
49
50 int diff = window[2] - window[0];
51 peakValue = diff + peakValue / 2;
52
53 if (peakValue > risingPeak.value || risingPeak.position == 0)
54 risingPeak = PeakInfo{peakValue, sampleCount - 1};
55 if (peakValue < fallingPeak.value || fallingPeak.position == 0)
56 fallingPeak = PeakInfo{peakValue, sampleCount - 1};
57 }
58
59 /** Gets the position of the rising peak.
60 * @return The position of the rising peak.
61 */
62 size_t getRisingPeakPosition() const { return risingPeak.position; }
63
64 /** Gets the position of the falling peak.
65 * @return The position of the falling peak.
66 */
67 size_t getFallingPeakPosition() const { return fallingPeak.position; }
68
69private:
70 /** PeakInfo struct.
71 * Structure holding peak value and position.
72 */
73 struct PeakInfo {
74 int value; /*!< Peak value. */
75 size_t position; /*!< Peak position. */
76 };
77
78 std::array<int, WINDOW_SIZE> window{0}; /*!< Sliding window for samples. */
79 int peakValue = 0; /*!< Current peak value. */
80 PeakInfo risingPeak{0, 0}; /*!< Rising peak info. */
81 PeakInfo fallingPeak{0, 0}; /*!< Falling peak info. */
82 size_t sampleCount = 0; /*!< Number of samples processed. */
83};
