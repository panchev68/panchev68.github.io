142 /** Open socket with protocol, port, and flags.
143 * @param protocol Protocol type.
144 * @param port Port number.
145 * @param flags Socket flags.
146 * @return Error code.
147 * @pre protocol, port, and flags must be valid.
148 * @post Socket opened if successful.
149 */
150 virtual Error open(uint8_t protocol, Port port, Flag flags) {
151 const auto result = ::socket(number, protocol, port.get(), flags);
152 return (result == number) ? Error::OK : static_cast<Error>(result);
153 }

193 /** Receive incoming message.
194 * @param buffer Data handle for received data.
195 * @param[out] receivedLength Length of received data.
196 * @return Error code.
197 * @pre buffer must be valid.
198 * @post Data received if successful.
199 */
200 virtual Error receive(DataHandle<> & buffer, size_t & receivedLength) {
201 const auto result = ::recv(number, buffer.data(), buffer.size());
202 receivedLength = (result > 0) ? static_cast<size_t>(result) : 0;
203 return (result > 0) ? Error::OK : static_cast<Error>(result);
204 }
205
206 /** Get received size from hardware.
207 * @return Number of bytes available to read.

209 size_t getReceivedSize() const { return static_cast<size_t>(getSn_RX_RSR(number)); }
210
211 /** Send data to connected peer.
212 * @param buffer Data handle to send.
213 * @param[out] sentLength Number of bytes sent.
214 * @return Error code.
215 * @pre buffer must be valid.
216 * @post Data sent if successful.
217 */
218 virtual Error send(const DataHandle<> & buffer, size_t & sentLength) {
219 const auto result = ::send(number, buffer.data(), buffer.size());
220 sentLength = (result > 0) ? static_cast<size_t>(result) : 0;
221 return (result > 0) ? Error::OK : static_cast<Error>(result);
222 }
223
224 /** Receive data from remote node.
225 * @param buffer Data handle for received data.
226 * @param remoteNode Node to receive from.
227 * @return Number of bytes received.
228 * @pre buffer and remoteNode must be valid.
229 * @post Data received if successful.
230 */
231 virtual size_t receiveFrom(DataHandle<> & buffer, Node & remoteNode) {
232 const auto result = ::recvfrom(number, buffer.data(), buffer.size(), const_cast<uint8_t *>(remoteNode.ip.get().data()), &remoteNode.port.get());
233 return (result > 0) ? static_cast<size_t>(result) : 0;
234 }

236 /** Send data to remote node.

242 */
243 virtual size_t sendTo(const DataHandle<> & buffer, const Node & remoteNode) {
244 const auto result = ::sendto(number, buffer.data(), buffer.size(), const_cast<uint8_t *>(remoteNode.ip.get().data()), remoteNode.port.get());
