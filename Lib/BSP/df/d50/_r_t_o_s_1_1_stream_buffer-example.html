RTOS::StreamBuffer

Example usage of StreamBuffer.

Demonstrates basic creation, sending, and receiving with RTOS::StreamBuffer.

Example usage:

#include "Lib/RTOS/StreamBuffer.hpp"
void exampleStreamBufferUsage() {
RTOS::StreamBuffer buffer;
constexpr size_t bufferSize = 128;
constexpr size_t triggerLevel = 16;
if (buffer.create(bufferSize, triggerLevel)) {
uint8_t dataToSend[8] = {1,2,3,4,5,6,7,8};
buffer.send(dataToSend, sizeof(dataToSend));
uint8_t receiveBuffer[8] = {};
buffer.receive(receiveBuffer, sizeof(receiveBuffer));
buffer.destroy();
}
}
