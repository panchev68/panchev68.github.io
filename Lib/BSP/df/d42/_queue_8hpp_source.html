27/** RTOS queue wrapper class.
28 * Provides a type-safe, fixed-capacity queue abstraction over FreeRTOS queues, supporting both dynamic
29 * and static allocation, ISR-safe operations, and utility methods for queue management.
30 * @tparam T Element type (must be trivially copyable).
31 * @tparam QUEUE_LENGTH Number of elements in the queue.
32 */
33template <typename T, size_t QUEUE_LENGTH = 1> class Queue {
34 static_assert(std::is_trivially_copyable<T>::value, "Queue element type must be trivially copyable");
35 static_assert(QUEUE_LENGTH > 0, "Queue length must be greater than zero");
36
37public:
38 using ElementType = T; /*!< Element type for the queue. */
39 using Handle = QueueHandle_t; /*!< Underlying FreeRTOS queue handle type. */
40 static constexpr size_t LENGTH = QUEUE_LENGTH; /*!< Number of elements in the queue. */
41 static constexpr size_t ELEMENT_SIZE = sizeof(T); /*!< Size of each element in bytes. */
42
43 /** Constructor of the Queue class. */
44 Queue() : handle(nullptr) {}
45
46 /** Destructor of the Queue class. */
47 ~Queue() { destroy(); }
48
49 Queue(const Queue &) = delete;
50 Queue & operator=(const Queue &) = delete;
51
52 /** Move constructor of the Queue class.
53 * @param other Source queue to move from.
54 */
55 Queue(Queue && other) : handle(other.handle) { other.handle = nullptr; }
56
57 /** Move assignment operator of the Queue class.
58 * @param other Source queue to move from.
59 * @return Reference to this queue.
60 */
61 Queue & operator=(Queue && other) {
62 if (this != &other) {
63 destroy();
64 handle = other.handle;
65 other.handle = nullptr;
66 }
67 return *this;
68 }

70 /** Create the queue dynamically.
71 * @return `true` if the queue was created successfully, `false` otherwise.
72 */
73 bool create() {
74 if (handle) {
75 assert(false);
76 return false;
77 }
78 handle = xQueueCreate(LENGTH, ELEMENT_SIZE);
79 if (!handle) {
80 assert(false);
81 return false;
82 }
83 return true;
84 }

123 /** Receive an item from the queue.
124 * @param[out] item Reference to store the received item.
125 * @param maxTicksToWait Maximum number of ticks to wait.
126 * @return `true` if an item was received, `false` otherwise.
127 */
128 bool receive(T & item, TickType_t maxTicksToWait = portMAX_DELAY) const {
129 return handle && xQueueReceive(handle, &item, maxTicksToWait) == pdPASS;
130 }
131
132 /** Receive an item from the queue in ISR context.
133 * @param[out] item Reference to store the received item.
134 * @return `true` if an item was received, `false` otherwise.
135 */
136 bool receiveFromISR(T & item) const {
137 if (!handle) return false;
138 BaseType_t woken = pdFALSE;
139 BaseType_t result = xQueueReceiveFromISR(handle, &item, &woken);
140 if (result == pdPASS) {
141 portYIELD_FROM_ISR(woken);
142 return true;
143 }
144 return false;
145 }

255 /** Check if the queue is full.
