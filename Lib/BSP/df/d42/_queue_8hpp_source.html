29/** RAII wrapper for FreeRTOS queues.
30 * Provides type-safe operations for queue management, including creation, destruction,
31 * sending, receiving, and querying. Supports both dynamic and static allocation.
32 * @tparam T_ELEMENT_TYPE Element type, must be trivially copyable.
33 * @tparam QUEUE_LENGTH Maximum number of elements in the queue.
34 */
35template <typename T_ELEMENT_TYPE, size_t QUEUE_LENGTH = 1>
36class Queue {
37 static_assert(std::is_trivially_copyable<T_ELEMENT_TYPE>::value, "Element type must be trivially copyable");
38 static_assert(QUEUE_LENGTH > 0, "Queue length must be greater than zero");
