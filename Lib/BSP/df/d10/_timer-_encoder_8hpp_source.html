21/** Timer specialization for encoder mode.

41 /** Opens the encoder timer.
42 * @return `true` if the timer was opened successfully, `false` otherwise.
43 */
44 bool open() override {
45 if (!Base::open()) return false;
46 if (HAL_TIM_Encoder_Start_IT(getHandle(), TIM_CHANNEL_ALL) != HAL_OK) return false;
47 getHandle()->Instance->ARR = MAX_COUNTER_VALUE;
48 return true;
49 }
50
51 /** Closes the encoder timer.
52 * @return `true` if the timer was closed successfully, `false` otherwise.

54 bool close() override {
55 if (!Base::close()) return false;

57 return true;
58 }
59
60private:
61 /** Handles timer events for encoder mode.
62 * @param id Callback event identifier.
63 */
64 void eventHandler(HAL_TIM_CallbackIDTypeDef id) override {
65 if (id == HAL_TIM_IC_CAPTURE_CB_ID && onValueChanged)
66 onValueChanged(static_cast<uint32_t>(__HAL_TIM_GET_COUNTER(getHandle())));
67 }
