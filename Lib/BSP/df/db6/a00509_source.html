191 T_VALUE interpolate(T_INPUT x) {
192 assert(pointCount >= 2);
193 if (!isSorted) sortPoints();
194 if (!isComputed) computeSplineCoefficients();
195 if (x <= points[0].x) return points[0].y;
196 if (x >= points[pointCount - 1].x) return points[pointCount - 1].y;
197 size_t index = findInterval(x);
198 return evaluateSplineSegment(segments[index], x - points[index].x);

205 T_VALUE getDerivative(T_INPUT x) {
206 assert(pointCount >= 2);
207 if (!isSorted) sortPoints();
208 if (!isComputed) computeSplineCoefficients();
209 if (x <= points[0].x) return segments[0].b;
210 if (x >= points[pointCount - 1].x) {
211 size_t lastIndex = pointCount - 2;
212 T_VALUE dx = points[pointCount - 1].x - points[lastIndex].x;
213 return segments[lastIndex].b + 2 * segments[lastIndex].c * dx + 3 * segments[lastIndex].d * dx * dx;
214 }
215 size_t index = findInterval(x);
216 T_VALUE dx = x - points[index].x;
217 return segments[index].b + 2 * segments[index].c * dx + 3 * segments[index].d * dx * dx;

230 void sortPoints() {
231 if (isSorted || pointCount <= 1) return;
232 for (size_t i = 0; i < pointCount - 1; ++i)
233 for (size_t j = 0; j < pointCount - i - 1; ++j)
234 if (points[j].x > points[j + 1].x) {
235 auto temp = points[j];
236 points[j] = points[j + 1];
237 points[j + 1] = temp;
238 }
239 isSorted = true;
240 isComputed = false;
241 }

243private:
245 void computeSplineCoefficients() {
246 if (isComputed || pointCount < 2) return;
247 if (pointCount == 2) {
248 T_VALUE slope = (points[1].y - points[0].y) / (points[1].x - points[0].x);
249 segments[0] = SplineSegment(points[0].y, slope, T_VALUE{}, T_VALUE{});
250 isComputed = true;
251 return;
252 }
253 size_t n = pointCount - 1;
254 std::array<T_VALUE, MAX_POINTS> h, alpha, l, mu, z, c;
255 for (size_t i = 0; i < n; ++i) {
256 h[i] = points[i + 1].x - points[i].x;
257 assert(h[i] > T_VALUE{});
258 }
259 for (size_t i = 1; i < n; ++i)
260 alpha[i] = (3 / h[i]) * (points[i + 1].y - points[i].y) - (3 / h[i - 1]) * (points[i].y - points[i - 1].y);
261 if (boundaryType == BoundaryType::NATURAL) {
262 l[0] = T_VALUE{1};
263 mu[0] = T_VALUE{};
264 z[0] = T_VALUE{};
265 l[n] = T_VALUE{1};
266 z[n] = T_VALUE{};
267 c[n] = T_VALUE{};
268 } else {
269 l[0] = 2 * h[0];
270 mu[0] = T_VALUE{0.5};
271 alpha[0] = (3 / h[0]) * (points[1].y - points[0].y) - 3 * leftDerivative;
272 z[0] = alpha[0] / l[0];
273 l[n] = h[n - 1] * (2 - mu[n - 1]);
274 alpha[n] = 3 * rightDerivative - (3 / h[n - 1]) * (points[n].y - points[n - 1].y);
275 z[n] = (alpha[n] - h[n - 1] * z[n - 1]) / l[n];
276 c[n] = z[n];
277 }
278 for (size_t i = 1; i < n; ++i) {
279 l[i] = 2 * (points[i + 1].x - points[i - 1].x) - h[i - 1] * mu[i - 1];
280 mu[i] = h[i] / l[i];
281 z[i] = (alpha[i] - h[i - 1] * z[i - 1]) / l[i];
282 }
283 for (size_t j = n - 1; j > 0; --j) c[j] = z[j] - mu[j] * c[j + 1];
284 c[0] = boundaryType == BoundaryType::NATURAL ? T_VALUE{} : z[0] - mu[0] * c[1];
285 for (size_t j = 0; j < n; ++j) {
286 T_VALUE a = points[j].y;
287 T_VALUE b = (points[j + 1].y - points[j].y) / h[j] - h[j] * (c[j + 1] + 2 * c[j]) / 3;
288 T_VALUE d = (c[j + 1] - c[j]) / (3 * h[j]);
289 segments[j] = SplineSegment(a, b, c[j], d);
290 }
291 isComputed = true;
292 }

298 size_t findInterval(T_INPUT x) const {
299 assert(pointCount >= 2);
300 assert(isSorted);
301 size_t left = 0, right = pointCount - 1;
302 while (right - left > 1) {
303 size_t middle = left + (right - left) / 2;
304 if (points[middle].x <= x) left = middle;
305 else right = middle;
306 }
307 return left;
308 }

315 static constexpr T_VALUE evaluateSplineSegment(const SplineSegment & segment, T_VALUE dx) {
316 return segment.a + segment.b * dx + segment.c * dx * dx + segment.d * dx * dx * dx;
317 }
