196 inline bool Base::resetChip ()
197 {
198 FreeRTOS::Timeout timeout;
199
200 _pinReset.write (0);
201 vTaskDelay (50);
202 _pinReset.write (1);
203 timeout.set (1000);
204 while (1)
205 {
206 if (verifyChipID () == true)
207 {
208 _state = State::READY;
209 vTaskDelay (100);
210 return true;
211 }
212 else if (timeout.check ())
213 {
214 _state = State::ERROR;
215 return false;
216 }
217 }
218 }

268 inline void Base::readCSR (uint16_t addr, void * dst, size_t len)
269 {
270 UU32 param = {0};
271 UU16 waddr;
272 waddr.val = addr;
273
274 param.v [0] = waddr.bytes.lb;
275 param.v [1] = waddr.bytes.hb;
276 param.v [2] = len;
277 param.v [3] = Registers::ESC_READ_BYTE;
278
279 writeU32 (Registers::ESC_CSR_CMD_REG, param.val);
280
281 do
282 {
283 param.val = readU32 (Registers::ESC_CSR_CMD_REG);
284 } while (param.v [3] & Registers::ESC_CSR_BUSY);
285
286 param.val = readU32 (Registers::ESC_CSR_DATA_REG);
287 memcpy (dst, param.v, len);
288 }

290 inline void Base::writeCSR (uint16_t addr, const uint8_t * src, uint8_t len)
291 {
292 Helper::UU32 param = {0};
293 Helper::UU16 waddr;
294
295 memcpy (param.v, src, len);
296
297 writeU32 (Registers::ESC_CSR_DATA_REG, param.val);
298 waddr.val = addr;
299 param.v [0] = waddr.bytes.lb;
300 param.v [1] = waddr.bytes.hb;
301 param.v [2] = len;
302 param.v [3] = Registers::ESC_WRITE_BYTE;
303
304 writeU32 (0x304, param.val);
305
306 do
307 {
308 param.val = readU32 (0x304);
309 } while (param.v [3] & Registers::ESC_CSR_BUSY);
310 }

312 inline void Base::readPRAM (uint16_t addr, uint8_t * dst, uint16_t len)
313 {
314 Helper::UU64 param = {0};
315 uint8_t i = 0, part_len, byte_pos;
316 uint8_t available;
317
318 // Reset/Abort any previous commands.
319 param.val = Registers::PRAM_RW_ABORT_MASK;
320
321 writeU32 (Registers::PRAM_READ_CMD_REG, param.val);
322
323 // The host should not modify this field unless the PRAM Read Busy
324 // (PRAM_READ_BUSY) bit is a 0.
325 do
326 {
327 param.val = readU32 (Registers::PRAM_READ_CMD_REG);
328 } while ((param.v [3] & Registers::PRAM_RW_BUSY_8B));
329
330 // Write addr and Length Register (PRAM_READ_ADDR_LEN) with the
331 // starting uint8_t addr and length) and Set PRAM Read Busy (PRAM_READ_BUSY)
332 // bit(-LAN9252 Process RAM Read Command Register) to start read operatrion
333 param.w [0] = addr;
334 param.w [1] = len;
335 param.w [2] = 0x0;
336 param.w [3] = 0x8000;
337
338 writeBuffer (Registers::PRAM_READ_ADDR_LEN_REG, (uint8_t *) &param.val, 8);
339
340 // Read PRAM Read Data Available (PRAM_READ_AVAIL) bit is set
341 do
342 {
343 param.val = readU32 (Registers::PRAM_READ_CMD_REG);
344 } while (!(param.v [0] & Registers::IS_PRAM_SPACE_AVBL_MASK));
345
346 available = param.v [1] & Registers::PRAM_SPACE_AVBL_COUNT_MASK;
347
348 // Fifo registers are aliased addr. In indexed it will read indexed
349 // data reg 0x04, but it will point to reg 0 In other modes read 0x04
350 // FIFO register since all registers are aliased
351
352 // get the uint8_t lenth for first read
353 // Auto increment is supported in SPIO
354 param.val = readU32 (Registers::PRAM_READ_FIFO_REG);
355 available--;
356 byte_pos = (addr & 0x03);
357 part_len = (4 - byte_pos) > len ? len : (4 - byte_pos);
358 memcpy (dst + i, &param.v [byte_pos], part_len);
359 len -= part_len;
360 i += part_len;
361
362 Interface::select ();
363 Interface::write (Registers::CMD_FAST_READ);
364 sendAddr (Registers::PRAM_READ_FIFO_REG);
365 Interface::write (Registers::CMD_FAST_READ_DUMMY);
366
367 while (len)
368 {
369 param.val = readBurstMode ();
370
371 part_len = len > 4 ? 4 : len;
372 memcpy ((dst + i), &param, part_len);
373
374 i += part_len;
375 len -= part_len;
376 available--;
377 }
378
379 Interface::deselect ();
380 }

382 inline void Base::writePRAM (const uint8_t * src, uint16_t addr, uint16_t len)
383 {
384 Helper::UU64 param = {0};
385
386 uint8_t i = 0, part_len, byte_pos, available;
387
388 // Reset or Abort any previous commands.
389 param.val = Registers::PRAM_RW_ABORT_MASK;
390
391 writeU32 (Registers::PRAM_WRITE_CMD_REG, param.val);
392
393 // Make sure there is no previous write
394 // is pending (PRAM Write Busy) bit is a 0
395 do
396 {
397 param.val = readU32 (Registers::PRAM_WRITE_CMD_REG);
398 } while ((param.v [3] & Registers::PRAM_RW_BUSY_8B));
399
400 // Write addr and Length Register (ECAT_PRAM_WR_ADDR_LEN) with the
401 // starting uint8_t addr and length) and write to the LAN9252 Process
402 // RAM Write Command Register (ECAT_PRAM_WR_CMD) with the PRAM Write Busy
403 // (PRAM_WRITE_BUSY) bit set
404
405 param.w [0] = addr;
406 param.w [1] = len;
407 param.w [2] = 0x0;
408 param.w [3] = 0x8000;
409
410 writeBuffer (Registers::PRAM_WRITE_ADDR_LEN_REG, (uint8_t *) &param.val, 8);
411
412 // Read PRAM write Data Available (PRAM_READ_AVAIL) bit is set
413 do
414 {
415 param.val = readU32 (Registers::PRAM_WRITE_CMD_REG);
416 } while (!(param.v [0] & Registers::IS_PRAM_SPACE_AVBL_MASK));
417
418 // Check write data available len
419 available = param.v [1] & Registers::PRAM_SPACE_AVBL_COUNT_MASK;
420
421 // Write data to Write FIFO get the byte lenth for first read
422 byte_pos = (addr & 0x03);
423
424 part_len = (4 - byte_pos) > len ? len : (4 - byte_pos);
425
426 param.val = 0;
427 memcpy (&param.v [byte_pos], src + i, part_len);
428
429 writeU32 (Registers::PRAM_WRITE_FIFO_REG, param.val);
430
431 available--;
432 len -= part_len;
433 i += part_len;
434
435 Interface::select ();
436 Interface::write (Registers::CMD_SERIAL_WRITE); // Auto increment mode
437 sendAddr (Registers::PRAM_WRITE_FIFO_REG); // Write Command
438
439 while (len)
440 {
441 part_len = len > 4 ? 4 : len;
442 param.val = 0;
443 memcpy (&param, (src + i), part_len);
444
445 writeBurstMode (param.val);
446 i += part_len;
447 len -= part_len;
448 available--;
449 }
450
451 Interface::deselect ();
452 }

466 inline uint32_t Base::readBurstMode ()
467 {
468 Helper::UU32 retvar;
469 uint8_t readBytes [4] = {0};
470
471 Interface::readWrite (NULL, 0x00, readBytes, 4);
472
473 // Read Bytes
474 retvar.bytes.lb = readBytes [0];
475 retvar.bytes.hb = readBytes [1];
476 retvar.bytes.ub = readBytes [2];
477 retvar.bytes.mb = readBytes [3];
478
479 return retvar.val;
480 }
