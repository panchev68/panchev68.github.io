Lib::Helper::instance

Example usage of singleton instance helpers.

This example demonstrates how to use Instance, InstanceWithArg, and getInstance to create and access singleton objects.

Example usage:

#include "Lib/Helper/instance.hpp"
struct Config {
int value;
};
class MySingleton {
public:
explicit MySingleton(const Config & config) : value(config.value) {}
int getValue() const { return value; }
private:
int value;
};
int main() {
// Using Instance
Lib::Helper::Instance<MySingleton> singletonInstance;
// Using InstanceWithArg
Config config{42};
Lib::Helper::InstanceWithArg<MySingleton, Config> singletonWithArg(config);
// Using getInstance
MySingleton & ref1 = Lib::Helper::getInstance<MySingleton>();
MySingleton & ref2 = Lib::Helper::getInstance<MySingleton, Config>(config);
return 0;
}
