Lib::Helper::TimeZone

Example usage of TimeZone class for time zone conversion.

This example demonstrates how to create a TimeZone object, set its properties, and convert local time to UTC minutes using the provided method. The output shows the UTC minutes for a given local time.

Example usage:
#include "Lib/Helper/TimeZone.hpp"
#include <iostream>
int main() {
// Create a TimeZone object for Eastern European Time (EET, UTC+2)
Lib::Helper::TimeZone tz("EET", 120);
// Print the time zone name and offset
std::cout << "Time zone: " << tz.getName() << std::endl;
std::cout << "Offset (minutes): " << tz.getOffsetMinutes() << std::endl;
// Set a new name and offset
tz.setName("Eastern European Summer Time");
tz.setOffsetMinutes(180); // UTC+3 for daylight saving
std::cout << "Updated time zone: " << tz.getName() << std::endl;
std::cout << "Updated offset (minutes): " << tz.getOffsetMinutes() << std::endl;
// Convert a local time (2024-07-01 15:30) to UTC minutes
int year = 2024;
int month = 7;
int day = 1;
int hour = 15;
int minute = 30;
time_t offsetSeconds = tz.getOffsetMinutes() * 60;
int utcMinutes = tz.convertLocalToUtcMinutes(year, month, day, hour, minute, offsetSeconds);
std::cout << "UTC minutes: " << utcMinutes << std::endl;
return 0;
}
