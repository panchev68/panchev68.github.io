Lib::Helper::FindPeaks

Example usage of FindPeaks class.

Example usage of FindPeaks class.

Example usage:
#include "Lib/Helper/FindPeaks.hpp"
int main() {
Lib::Helper::FindPeaks peakFinder;
peakFinder.reset();
int samples[] = {1, 3, 7, 2, 5, 0, 4};
for (int sample : samples) {
peakFinder.addSample(sample);
}
size_t risingPos = peakFinder.getRisingPeakPosition();
size_t fallingPos = peakFinder.getFallingPeakPosition();
// Use risingPos and fallingPos as needed
return 0;
}
