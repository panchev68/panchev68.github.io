190 void preProcessing ()
191 {
192 for (auto i = 0; i < T_PIN_NUM; i++)
193 {
194 auto port = _mapFields [i].port;
195 auto mask = _mapFields [i].mask;
196
197 auto & outReg = _txBuffer [port + MC33879::TxPort::OUTPUTS + 1];
198 auto & ctrlReg = _txBuffer [port + MC33879::TxPort::CONTROL + 1];
199
200 if ((grouping.getValue () & BitMask::at (i)) != 0)
201 {
202 outReg |= mask; // Activate Output
203 ctrlReg |= mask; // If an output is activated it remains active in the 33879 status register
204 }
205 else
206 {
207 outReg &= ~mask; // Deactivate Output
208 }
209 }
210 _txBuffer [0] = _diagCounter; // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
211 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
212 }
213
216 bool postProcessing ()
217 {
218 // Last Byte or received data must be equal to _diagCounter
219
220 if ((_rxBuffer.at (BUFFER_SIZE - 1) != _txBuffer.at (0)) && (_txBuffer.at (0) != 0)) // verifies that the ID (_diagCounter) of the received frame coincides with the last sent
221 {
222 // error: SPI Interface ...
223 if (_isSpiError == false)
224 {
225 _isSpiError = true;
226 _spiErronrsCounter++;
227 emitErrorEvent (true); // error is activated
228 assert (0);
229 }
230 return false;
231 }
232 else
233 {
234 // success : No Error in SPI Interface
235 if (_isSpiError == true)
236 {
237 _isSpiError = false;
238 emitErrorEvent (false); // error is cleared
239 }
240 }
241
242 // Processing of MC33879 Status after received ...
243
244 for (auto i = 0; i < T_PIN_NUM; i++)
245 {
246 uint64_t value = 0;
247 for (auto i = 0; i < T_PIN_NUM; i++)
248 {
249 auto & map_index = _mapFields [i];
250 if ((_rxBuffer [map_index.port + MC33879::RxPort::STATUS + 1] & map_index.mask) != 0)
251 {
252 value |= BitMask::at (i);
253 }
254 }
255 _statusRegister = value;
256 }
257 return true;
258 }

343 uint64_t _result;

346 void taskFn ()
347 {
348 uint64_t value;
349
350 // out_mask - contains a mask for the whole port, for example: 00000'1111'1111'1111'1111 for 24-bit
351 for (;;)
352 {
353 _queueValueChanged.receive (value, portMAX_DELAY); // wait for Outputs changed
354 _outputsChangedCounter++;
355
356 if (_parameters.enable == true) // if Grouping is enable
357 {
358 value &= _valueMaskAnd; // clear unnecessary bits
359
360 // detect different bits ...
361
362 uint64_t diffBits = value ^ _result;
363
364 uint64_t diffActiveBits = 0; // number of active different bits (is 1)
365 auto diffBitCounter = 0; // different bits counter
366
367 // count the different bits...
368
369 for (auto i = 0; i < T_PIN_NUM; i++)
370 {
371 auto mask = BitMask::at (i);
372 // is changed is active
373 if ((diffBits & mask) && (value & mask))
374 {
375 diffActiveBits |= mask;
376 diffBitCounter++;
377 }
378 }
379
380 // process ...
381
382 if (diffBitCounter >= _parameters.outputsNumber) // NUMBER_OF_SWITCH_OUTPUTS
383 {
384 uint32_t count = 0;
385 uint64_t group_bit_mask = 0;
386
387 for (auto i = 0; i < T_PIN_NUM; i++)
388 {
389 group_bit_mask |= BitMask::at (i);
390
391 if ((diffActiveBits & BitMask::at (i)) != 0)
392 {
393 if ((++count) >= _parameters.outputsNumber)
394 {
395 count = 0;
396
397 _result &= value;
398 _result |= (value & group_bit_mask);
399
400 vTaskDelay (_parameters.switchPeriod);
401 }
402 }
403 }
404 }
405 }
406
407 _result = value;
408 }
409 }
