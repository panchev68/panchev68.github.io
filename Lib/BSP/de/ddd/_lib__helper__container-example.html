Lib_Helper_Container

Example usage of Lib::Helper::Container

Demonstrates basic usage of the Container class for managing pointers to objects.

Example usage:

#include "Lib/Helper/Container.hpp"
class MyObject {
public:
void doSomething() {}
};
void processCallback(MyObject * obj, void * arg) {
if (obj) obj->doSomething();
}
int main() {
Lib::Helper::Container<MyObject> container;
MyObject obj1, obj2;
container.add(&obj1);
container.add(&obj2);
container.processItems(processCallback, nullptr);
container.remove(&obj1);
container.clear();
return 0;
}
