107 /** Inputs class.
108 * Handles SPI-based digital input operations.
109 */
110 class Inputs : public PortBase {
111 private:
112 PinOutput nssPin; /*!< NSS pin output. */
113 Lib::HAL::SpiPort spiPort; /*!< SPI port abstraction. */
114 const MapField * mappingFields; /*!< Mapping fields for input pins. */
115 std::vector<uint8_t> buffer; /*!< Buffer for SPI data. */
116 const size_t bufferSize; /*!< Buffer size in bytes. */
117 const size_t pinCount; /*!< Number of input pins. */
118 uint64_t value; /*!< Current input value. */
119 uint32_t changesCounter; /*!< Counter for value changes. */
120
121 public:
122 /** Constructor of the Inputs class.
123 * @param config Input configuration.
124 */
125 explicit Inputs(const Config::Inputs & config)
126 : nssPin(config.nssPin),
127 spiPort(config.spiPort),
128 mappingFields(config.mappingFields),
129 buffer((config.pinCount + 7) / 8),
130 bufferSize((config.pinCount + 7) / 8),
131 pinCount(config.pinCount),
132 value(~0ULL),
133 changesCounter(0) {
134 assert(pinCount > 0u);
135 assert(pinCount <= MAX_PINS);
136 assert(mappingFields != nullptr);
137 }
138
139 /** Destructor of the Inputs class. */
140 ~Inputs() override = default;
141
142 /** Setup the input port. */
143 void setup() override {
144 nssPin.write(1);
145 spiPort.init();
146 }
147
148 /** Get the current input value.
149 * @return Current input value.
150 */
151 uint64_t getValue() const { return value; }
152
153 /** Get the changes counter.
154 * @return Number of value changes.
155 */
156 uint32_t getChangesCounter() const { return changesCounter; }
157
158 /** Pre-update callback. */
159 void preUpdateCallback() override {
160 auto handle = spiPort.getHandle();
161 if (!handle)
162 return;
163 HAL_SPI_DeInit(handle);
164 handle->Init.BaudRatePrescaler = SPI_PRESCALER;
165 handle->Init.CLKPolarity = SPI_POLARITY_LOW;
166 handle->Init.CLKPhase = SPI_PHASE_1EDGE;
167 spiPort.init();
168 }
169
170 /** Update the input value. */
171 void update() override {
172 nssPin.write(0);
173 if (!spiPort.receive(buffer.data(), bufferSize, INPUT_SPI_TIMEOUT)) {
174 nssPin.write(1);
175 return;
176 }
177 nssPin.write(1);
178 RTOS::Task::yield();
179
180 uint64_t newValue = 0;
181 for (size_t i = 0; i < pinCount; ++i) {
182 const auto & map = mappingFields[i];
183 const size_t idx = static_cast<size_t>(map.port);
184 if (idx >= bufferSize) {
185 continue; // safety: skip invalid mapping
186 }
187 if ((buffer[idx] & map.mask) != 0) {
188 newValue |= (1ULL << i);
189 }
190 }
191 if (value != newValue) {
192 value = newValue;
193 ++changesCounter;
194 if (onValueChanged)
195 onValueChanged(value);
196 }
197 }
198
199 /** Get SPI handle.
200 * @return SPI_HandleTypeDef pointer.
201 */
202 SPI_HandleTypeDef * getSpiHandle() const override { return spiPort.getHandle(); }
203
204 std::function<void(const uint64_t &)> onValueChanged; /*!< Callback for value changes. */
205 };

207 /** Outputs class.
208 * Handles SPI-based digital output operations.
209 */
210 class Outputs : public PortBase {
211 private:
212 struct MC33879 {
213 struct RxPort {
214 uint8_t dummy; /*!< Dummy byte. */
215 uint8_t status; /*!< Status byte. */
216 };
217
218 struct TxPort {
219 uint8_t control; /*!< Control byte. */
220 uint8_t value; /*!< Value byte. */
221 };
222 };
223
224 static_assert(sizeof(MC33879::TxPort) == 2, "TxPort layout must be 2 bytes");
225 static_assert(sizeof(MC33879::RxPort) == 2, "RxPort layout must be 2 bytes");
226
227 const size_t pinCount; /*!< Number of output pins. */
228 const size_t bufferSize; /*!< Buffer size in bytes. */
229 PinOutput nssPin; /*!< NSS pin output. */
230 PinOutput oePin; /*!< Output enable pin output. */
231 Lib::HAL::SpiPort spiPort; /*!< SPI port abstraction. */
232 const MapField * mappingFields; /*!< Mapping fields for output pins. */
233 std::vector<uint8_t> rxBuffer; /*!< RX buffer for SPI data. */
234 std::vector<uint8_t> txBuffer; /*!< TX buffer for SPI data. */
235 RTOS::Timeout timeout; /*!< Timeout for update operations. */
236 uint64_t value; /*!< Current output value. */
237 uint64_t status; /*!< Current status value. */
238 uint32_t changesCounter; /*!< Counter for value changes. */
239 uint8_t diagCounter; /*!< Diagnostic counter. */
240 bool isValueChanged; /*!< Flag indicating value change. */
241 bool isSpiError; /*!< Flag indicating SPI error. */
242
243 /** Get number of ports available in the TX/RX buffer.
244 * @return Number of ports derived from the buffer size.
245 */
246 size_t portCount() const noexcept { return (bufferSize - 1) / sizeof(MC33879::TxPort); }
247
248 public:
249 /** Constructor of the Outputs class.
250 * @param config Output configuration.
251 */
252 explicit Outputs(const Config::Outputs & config)
253 : pinCount(config.pinCount),
254 bufferSize((((config.pinCount + 7) / 8) * 2) + 1),
255 nssPin(config.nssPin),
256 oePin(config.oePin),
257 spiPort(config.spiPort),
258 mappingFields(config.mappingFields),
259 rxBuffer((((config.pinCount + 7) / 8) * 2) + 1),
260 txBuffer((((config.pinCount + 7) / 8) * 2) + 1),
261 timeout(),
262 value(0),
263 status(0),
264 changesCounter(0),
265 diagCounter(1),
266 isValueChanged(false),
267 isSpiError(false) {
268 assert(pinCount > 0u);
269 assert(pinCount <= MAX_PINS);
270 assert(mappingFields != nullptr);
271 }
272
273 /** Destructor of the Outputs class. */
274 ~Outputs() override = default;
275
276 /** Setup the output port. */
277 void setup() override {
278 nssPin.write(1);
279 spiPort.init();
280 std::fill(txBuffer.begin(), txBuffer.end(), 0);
281 auto txPorts = reinterpret_cast<MC33879::TxPort *>(txBuffer.data() + 1);
282 const size_t n = portCount();
283 for (size_t i = 0; i < n; ++i) {
284 txPorts[i].control = 0xFF;
285 }
286 }
287
288 /** Set the output value.
289 * @param newValue New output value.
290 */
291 void setValue(uint64_t newValue) {
292 if (newValue != value) {
293 value = newValue;
294 isValueChanged = true;
295 }
296 }
297
298 /** Get the current output value.
299 * @return Current output value.
300 */
301 uint64_t getValue() const { return value; }
302
303 /** Get the current status value.
304 * @return Current status value.
305 */
306 uint64_t getStatus() const { return status; }
307
308 /** Get the changes counter.
309 * @return Number of value changes.
310 */
311 uint32_t getChangesCounter() const { return changesCounter; }
312
313 /** Check if there is an SPI error.
314 * @return `true` if there is an SPI error, `false` otherwise.
315 */
316 bool hasSpiError() const { return isSpiError; }
317
318 /** Get SPI handle.
319 * @return SPI_HandleTypeDef pointer.
320 */
321 SPI_HandleTypeDef * getSpiHandle() const override { return spiPort.getHandle(); }
322
323 /** Pre-update callback. */
324 void preUpdateCallback() override {
325 auto handle = spiPort.getHandle();
326 if (!handle)
327 return;
328 spiPort.deinit();
329 handle->Init.BaudRatePrescaler = SPI_PRESCALER;
330 handle->Init.CLKPolarity = SPI_POLARITY_LOW;
331 handle->Init.CLKPhase = SPI_PHASE_2EDGE;
332 spiPort.init();
333 }
334
335 /** Update the output value. */
336 void update() override {
337 bool needTransfer = false;
338 if (isValueChanged) {
339 isValueChanged = false;
340 updateTxBuffer();
341 timeout.set(UPDATE_TIMEOUT_MS);
342 ++changesCounter;
343 needTransfer = true;
344 }
345 if (!needTransfer && timeout.isExpired()) {
346 timeout.set(UPDATE_TIMEOUT_MS);
347 needTransfer = true;
348 }
349 if (!needTransfer)
350 return;
351 performSpiTransfer();
352 processRxBuffer();
353 }
354
355 std::function<void(const uint64_t &)> onValueChanged; /*!< Callback for value changes. */
356 std::function<void(const uint64_t &)> onStatusChanged; /*!< Callback for status changes. */
357 std::function<void()> onSpiError; /*!< Callback for SPI errors. */
358
359 private:
360 /** Update TX buffer based on the current output value. */
361 void updateTxBuffer() {
362 auto txPorts = reinterpret_cast<MC33879::TxPort *>(txBuffer.data() + 1);
363 const size_t ports = portCount();
364 for (size_t i = 0; i < pinCount; ++i) {
365 const auto & map = mappingFields[i];
366 const size_t p = static_cast<size_t>(map.port);
367 if (p >= ports) {
368 continue; // safety: skip invalid mapping
369 }
370 if ((value & (1ULL << i)) != 0) {
371 txPorts[p].value |= map.mask;
372 } else {
373 txPorts[p].value &= static_cast<uint8_t>(~map.mask);
374 }
375 }
376 }
377
378 /** Perform SPI transfer using the TX and RX buffers.
379 * The function updates diagnostic counter and manages OE pin based on response.
380 */
381 void performSpiTransfer() {
382 txBuffer[0] = diagCounter;
383 nssPin.write(0);
384 const bool spiSuccess = spiPort.sendReceive(txBuffer.data(), rxBuffer.data(), bufferSize, OUTPUT_SPI_TIMEOUT);
385 nssPin.write(1);
386 RTOS::Task::yield();
387 if (!spiSuccess)
388 return;
389 if (diagCounter == 0)
390 oePin.write(1);
391 if (++diagCounter == 0)
392 diagCounter = 1;
393 }
394
395 /** Process RX buffer and update status; signal SPI error if detected. */
396 void processRxBuffer() {
397 if ((rxBuffer[bufferSize - 1] != txBuffer[0]) && diagCounter > 0 && !isSpiError) {
398 isSpiError = true;
399 if (onSpiError)
400 onSpiError();
401 }
402 uint64_t newStatus = 0;
403 auto rxPorts = reinterpret_cast<MC33879::RxPort *>(rxBuffer.data() + 1);
404 const size_t ports = portCount();
405 for (size_t i = 0; i < pinCount; ++i) {
406 const auto & map = mappingFields[i];
407 const size_t p = static_cast<size_t>(map.port);
408 if (p >= ports) {
409 continue; // safety: skip invalid mapping
410 }
411 if ((rxPorts[p].status & map.mask) != 0) {
412 newStatus |= (1ULL << i);
413 }
414 }
415 if (status != newStatus) {
416 status = newStatus;
417 if (onStatusChanged)
418 onStatusChanged(status);
419 }
420 }
421 };
422
423 private:
424 std::shared_ptr<Inputs> inputsPtr; /*!< Shared pointer to Inputs instance. */
425 std::shared_ptr<Outputs> outputsPtr; /*!< Shared pointer to Outputs instance. */
426 RTOS::Task task; /*!< RTOS task for periodic updates. */
427
428 public:
429 Inputs & inputs; /*!< Reference to Inputs instance. */
430 Outputs & outputs; /*!< Reference to Outputs instance. */
431
432 /** Constructor of the DigitalPeripheralBase class.
433 * @param config Digital peripheral configuration.
434 */
435 explicit DigitalPeripheralBase(const Config & config)
436 : inputsPtr(std::make_shared<Inputs>(config.inputs)),
437 outputsPtr(std::make_shared<Outputs>(config.outputs)),
438 inputs(*inputsPtr),
439 outputs(*outputsPtr) {}
440
441 /** Destructor of the DigitalPeripheralBase class. */
442 virtual ~DigitalPeripheralBase() = default;
443
444 /** Setup the digital peripheral. */
445 virtual void setup() {
446 inputs.setup();
447 outputs.setup();
448 const bool useSequentialMode = (inputs.getSpiHandle() == outputs.getSpiHandle());
449 if (useSequentialMode) {
450 if (!task.start([this] { taskFn<UpdateMode::SEQUENTIAL>(); }))
451 return;
452 } else {
453 if (!task.start([this] { taskFn<UpdateMode::PARALLEL>(); }))
454 return;
455 }
456 }
457
458 protected:
459 /** Handle pre-update event. */
460 virtual void handlePreUpdateEvent() {}
461
462 /** Handle post-update event. */
463 virtual void handlePostUpdateEvent() {}
464
465 private:
466 /** Task function template.
467 * @tparam T_MODE UpdateMode template parameter selecting sequential or parallel execution.
468 */
469 template <UpdateMode T_MODE>
470 void taskFn() {
471 if (T_MODE == UpdateMode::SEQUENTIAL) {
472 while (true) {
473 handlePreUpdateEvent();
474 inputs.preUpdateCallback();
475 inputs.update();
476 outputs.preUpdateCallback();
477 outputs.update();
478 handlePostUpdateEvent();
479 RTOS::Task::delay(TASK_LOOP_DELAY);
480 }
481 } else {
482 inputs.preUpdateCallback();
483 outputs.preUpdateCallback();
484 while (true) {
485 handlePreUpdateEvent();
486 inputs.update();
487 outputs.update();
488 handlePostUpdateEvent();
489 RTOS::Task::delay(TASK_LOOP_DELAY);
490 }
491 }
492 }
493 };
