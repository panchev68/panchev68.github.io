17/** Kalman filter for generic numeric types.
18 * The Kalman class implements a simple Kalman filter for smoothing and estimating sensor values.
19 * It supports configuration of process noise, sensor noise, and initial error, and provides
20 * methods for filtering, resetting, and parameter adjustment. Internal calculations use double
21 * precision for improved accuracy.
22 * @tparam T Numeric type for input and output values.
23 */
24template <typename T = double> class Kalman {
25public:
26 /** Configuration parameters for Kalman filter.
27 * The Config struct holds process noise, sensor noise, estimated error, and initial value.
28 * It allows fine-tuning of filter behavior for different sensor and signal characteristics.
29 */
30 struct Config {
31 double processNoise = 0.00001; /*!< Process noise covariance. */
32 double sensorNoise = 0.004; /*!< Sensor noise covariance. */
33 double estimatedError = 1.0; /*!< Initial estimation error. */
34 T value = 0; /*!< Initial value for filter. */
35 };
36
37 /** Constructor of Kalman class.
38 * @param config Configuration parameters for the filter.
39 */
40 explicit Kalman(const Config & config)
41 : processNoise(config.processNoise),
42 sensorNoise(config.sensorNoise),
43 estimatedError(config.estimatedError),
44 filteredValue(static_cast<double>(config.value)),
45 initialValue(static_cast<double>(config.value)) {}
46
47 /** Destructor of Kalman class. */
48 virtual ~Kalman() = default;
49
50 /** Reset the filter to its initial state.
51 * @post Filtered value and error are reset to initial configuration.
52 */
53 virtual void reset() {
54 filteredValue = initialValue;
55 estimatedError = 0.0;
56 }
57
58 /** Set filter parameters.
59 * @param config New configuration parameters.
60 * @post Filter parameters and initial value are updated.
61 */
62 virtual void setParam(const Config & config) {
63 processNoise = config.processNoise;
64 sensorNoise = config.sensorNoise;
65 estimatedError = config.estimatedError;
66 initialValue = static_cast<double>(config.value);
67 }
68
69 /** Filter a new value and return the estimated value.
70 * @param value New sensor or input value to filter.
71 * @return Filtered value estimate.
72 * @pre Value must be convertible to double.
73 * @post Internal state is updated with new filtered value.
74 */
75 virtual T operator()(T value) {
76 estimatedError += processNoise;
77 const double gain = estimatedError / (estimatedError + sensorNoise);
78 filteredValue += gain * (static_cast<double>(value) - filteredValue);
79 estimatedError *= (1.0 - gain);
80 return static_cast<T>(filteredValue);
81 }
82
83private:
84 double processNoise; /*!< Process noise covariance. */
85 double sensorNoise; /*!< Sensor noise covariance. */
86 double estimatedError; /*!< Estimation error covariance. */
87 double filteredValue; /*!< Current filtered value. */
88 double initialValue; /*!< Initial value for reset. */
89};
