1/** Timeout utility for FreeRTOS tick-based timing.
2 * This file defines the `Timeout` class, which provides a convenient interface for managing timeouts
3 * using FreeRTOS tick counters. It supports setting, extending, resetting, and querying timeouts in
4 * ticks, milliseconds, and seconds. The class is designed for embedded and RTOS-based applications,
5 * ensuring deterministic and type-safe timeout handling. It relies on the `TickCounter` utility for
6 * conversions and tick operations.
7 * @file Timeout.hpp
8 */

22/** Timeout class for managing tick-based timeouts in FreeRTOS.
23 * Provides methods for setting, extending, resetting, and querying timeouts in ticks, milliseconds,
24 * and seconds. Supports disabled and immediate timeouts, and offers conversion and status utilities.
25 */
26class Timeout {
27public:
28 using TickType = TickCounter::TickType;
29
30 static constexpr TickType DISABLED = portMAX_DELAY; /*!< Disabled timeout value. */
31 static constexpr TickType IMMEDIATE = 0; /*!< Immediate timeout value. */
32
33 /** Constructor of Timeout class. */
34 Timeout() : startTicks(0), duration(DISABLED), isSet(false) {}
35
36 /** Constructor of Timeout class.
37 * @param ticks Timeout duration in ticks.
38 */
39 explicit Timeout(TickType ticks) : startTicks(0), duration(ticks), isSet(ticks != DISABLED) {}
40
41 /** Create Timeout from milliseconds.
42 * @param ms Timeout duration in milliseconds.
43 * @return Timeout object with specified duration.
44 */
45 static Timeout fromMilliseconds(uint32_t ms) { return Timeout(TickCounter::msToTicks(ms)); }
46
47 /** Create Timeout from seconds.
48 * @param sec Timeout duration in seconds.
49 * @return Timeout object with specified duration.
50 */
51 static Timeout fromSeconds(uint32_t sec) { return Timeout(TickCounter::secondsToTicks(sec)); }
52
53 /** Set timeout duration in ticks.
54 * @param ticks Timeout duration in ticks.
55 */
56 void set(TickType ticks) {
57 assert(ticks <= portMAX_DELAY);
58 if (ticks == DISABLED) {
59 disable();
60 return;
61 }
62 startTicks = TickCounter::get();
63 duration = ticks;
64 isSet = true;
65 }
66
67 /** Set timeout duration in milliseconds.
68 * @param ms Timeout duration in milliseconds.
69 */
70 void setMilliseconds(uint32_t ms) { set(TickCounter::msToTicks(ms)); }
71
72 /** Set timeout duration in seconds.
73 * @param sec Timeout duration in seconds.
74 */
75 void setSeconds(uint32_t sec) { set(TickCounter::secondsToTicks(sec)); }
76
77 /** Check if timeout has expired.
78 * @return `true` if timeout has expired, `false` otherwise.
79 */
80 bool isExpired() const {
81 if (isDisabled() || !isSet) return false;
82 return TickCounter::hasTimedOutTicks(startTicks, duration);
83 }
84
85 /** Reset the timeout to start from current tick. */
86 void reset() {
87 if (!isDisabled()) {
88 startTicks = TickCounter::get();
89 isSet = true;
90 }
91 }
92
93 /** Get remaining time in ticks.
94 * @return Remaining ticks until timeout expires.
95 */
96 TickType getRemaining() const {
97 if (isDisabled() || !isSet) return DISABLED;
98 return TickCounter::getRemainingTicks(startTicks, duration);
99 }
100
101 /** Get remaining time in milliseconds.
102 * @return Remaining milliseconds until timeout expires.
103 */
104 uint32_t getRemainingMs() const {
105 TickType rem = getRemaining();
106 return (rem == DISABLED) ? UINT32_MAX : TickCounter::ticksToMs(rem);
107 }
108
109 /** Get elapsed time in ticks since timeout started.
110 * @return Elapsed ticks since timeout started.
111 */
112 TickType getElapsed() const {
113 if (isDisabled() || !isSet) return 0;
114 return TickCounter::getElapsedTicksSince(startTicks);
115 }
116
117 /** Get elapsed time in milliseconds since timeout started.
118 * @return Elapsed milliseconds since timeout started.
119 */
120 uint32_t getElapsedMs() const { return TickCounter::ticksToMs(getElapsed()); }
121
122 /** Get progress percentage of timeout.
123 * @return Progress percentage (0-100).
124 */
125 uint32_t getProgressPercent() const {
126 if (isDisabled() || !isSet || duration == 0) return 0;
127 TickType elapsed = getElapsed();
128 if (elapsed >= duration) return 100;
129 return static_cast<uint32_t>((elapsed * 100) / duration);
130 }

132 /** Assignment operator of Timeout class.
133 * @param ticks Timeout duration in ticks.
134 * @return Reference to this object.

136 Timeout & operator=(TickType ticks) {
137 set(ticks);

139 }

141 /** Equality operator of Timeout class.
142 * @param expired Expired state to compare.
143 * @return `true` if expired state matches, `false` otherwise.
144 */
145 bool operator==(bool expired) const { return isExpired() == expired; }
146
147 /** Inequality operator of Timeout class.
148 * @param expired Expired state to compare.
149 * @return `true` if expired state does not match, `false` otherwise.
150 */
151 bool operator!=(bool expired) const { return isExpired() != expired; }
152
153 /** Addition assignment operator of Timeout class.
154 * @param ticks Number of ticks to add to duration.
155 * @return Reference to this object.
156 */
157 Timeout & operator+=(TickType ticks) {
158 if (!isDisabled() && ticks > 0) {
159 TickType newDuration = duration + ticks;
160 assert(newDuration >= duration);
161 duration = std::min(newDuration, static_cast<TickType>(portMAX_DELAY - 1));
162 }
163 return *this;
164 }
165
166 /** Subtraction assignment operator of Timeout class.
167 * @param ticks Number of ticks to subtract from duration.
168 * @return Reference to this object.
169 */
170 Timeout & operator-=(TickType ticks) {
171 if (!isDisabled() && ticks > 0 && duration > ticks) duration -= ticks;
172 return *this;
173 }
174
175 /** Conversion operator to TickType. */
176 operator TickType() const { return duration; }
177
178 /** Explicit conversion operator to bool.
179 * @return `true` if timeout is active, `false` otherwise.
180 */
181 explicit operator bool() const { return isActive(); }
182
183 /** Disable the timeout. */
184 void disable() {
185 duration = DISABLED;
186 isSet = false;
187 }
188
189 /** Check if timeout is disabled.
190 * @return `true` if timeout is disabled, `false` otherwise.
191 */
192 bool isDisabled() const { return duration == DISABLED; }
193
194 /** Check if timeout is active.
195 * @return `true` if timeout is active, `false` otherwise.
196 */
197 bool isActive() const { return isSet && !isDisabled(); }
198
199 /** Get timeout duration in ticks.
200 * @return Timeout duration in ticks.
201 */
202 TickType getDuration() const { return duration; }
203
204 /** Get timeout duration in milliseconds.
205 * @return Timeout duration in milliseconds.
206 */
207 uint32_t getDurationMs() const {
208 return (duration == DISABLED) ? UINT32_MAX : TickCounter::ticksToMs(duration);
209 }
210
211 /** Get start tick value.
212 * @return Tick value when timeout was started.
213 */
214 TickType getStartTicks() const { return startTicks; }
215
216 /** Extend timeout by specified ticks.
217 * @param ticks Number of ticks to extend.
218 */
219 void extend(TickType ticks) {
220 if (!isDisabled() && ticks > 0) *this += ticks;
221 }
222
223 /** Extend timeout by specified milliseconds.
224 * @param ms Number of milliseconds to extend.
225 */
226 void extendMs(uint32_t ms) { extend(TickCounter::msToTicks(ms)); }
227
228 /** Check if timeout will expire within specified ticks.
229 * @param ticks Number of ticks to check.
230 * @return `true` if timeout will expire within specified ticks, `false` otherwise.
231 */
232 bool willExpireWithin(TickType ticks) const {
233 if (isDisabled() || !isSet) return false;
234 TickType rem = getRemaining();
235 return rem <= ticks;
236 }
237
238 /** Check if timeout will expire within specified milliseconds.
239 * @param ms Number of milliseconds to check.
240 * @return `true` if timeout will expire within specified milliseconds, `false` otherwise.
241 */
242 bool willExpireWithinMs(uint32_t ms) const { return willExpireWithin(TickCounter::msToTicks(ms)); }
243
244 /** Create an immediate timeout.
245 * @return Timeout object that is immediately expired.
246 */
247 static Timeout immediate() {
248 Timeout timeout(IMMEDIATE);
249 timeout.startTicks = TickCounter::get() - 1;
250 return timeout;
251 }
252
253 /** Create a disabled timeout.
254 * @return Timeout object that is disabled.
255 */
256 static Timeout disabled() { return Timeout(); }
257
258private:
259 TickType startTicks; /*!< Tick value when timeout was started. */
260 TickType duration; /*!< Timeout duration in ticks. */
261 bool isSet; /*!< Indicates if timeout is set. */
262};
