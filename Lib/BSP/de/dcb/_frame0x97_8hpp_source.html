19 /** Frame 0x97: Remote AT Command Response.
20 *
21 * Parses and represents a Digi XBee API frame for remote AT command responses. This type provides direct access
22 * to frame ID, destination addresses, AT command, status, and optional parameters, supporting robust embedded
23 * communication.
24 */
25 class Frame0x97 {
26 public:
27 uint8_t frameId; /*!< Frame ID for correlation with request. */
28 uint64_t dstAddr64; /*!< 64-bit destination address. */
29 uint16_t dstAddr16; /*!< 16-bit destination address (typically 0xFFFE). */
30 uint16_t atCmd; /*!< Two ASCII characters identifying the AT command. */
31 uint8_t status; /*!< Command status code. */
32 std::vector<uint8_t> parameters; /*!< Command data (optional). */
33
34 /** Constructor of Frame0x97 class.
35 * @param frame Frame::Data containing the raw response frame.
36 */
37 explicit Frame0x97(const Frame::Data & frame) {
38 using Convert = Frame::Helper::Convert;
39
40 __packed struct Header {
41 uint8_t frameId;
42 uint64_t dstAddr64;
43 uint16_t dstAddr16;
44 uint16_t atCmd;
45 uint8_t status;
46 };
47
48 const auto * payload = frame.vector.data();
49 size_t len = frame.vector.size();
50 const auto * header = reinterpret_cast<const Header *>(payload);
51 frameId = Convert::to(header->frameId);
52 dstAddr64 = Convert::to(header->dstAddr64);
53 dstAddr16 = Convert::to(header->dstAddr16);
54 atCmd = Convert::to(header->atCmd);
55 status = Convert::to(header->status);
56 payload += sizeof(Header);
57 len -= sizeof(Header);
58 parameters.assign(payload, payload + len);
59 }
60 };
