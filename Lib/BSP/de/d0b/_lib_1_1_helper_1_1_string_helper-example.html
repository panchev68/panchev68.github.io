Lib::Helper::StringHelper

Example usage of StringHelper utilities.

Demonstrates converting string case, trimming whitespace, and searching for a string in a null-terminated array.

Example usage:
#include "Lib/Helper/StringHelper.hpp"
#include <iostream>
int main() {
std::string original = " Example String ";
std::string upper = Lib::Helper::StringHelper::toUpperCase(original);
std::string lower = Lib::Helper::StringHelper::toLowerCase(original);
std::string trimmed = Lib::Helper::StringHelper::trim(original);
const char * fruits[] = { "Apple", "Banana", "Cherry", nullptr };
int32_t index = Lib::Helper::StringHelper::findInArray("Banana", fruits);
std::cout << "Original: '" << original << "'\n";
std::cout << "Uppercase: '" << upper << "'\n";
std::cout << "Lowercase: '" << lower << "'\n";
std::cout << "Trimmed: '" << trimmed << "'\n";
std::cout << "Index of 'Banana': " << index << std::endl;
return 0;
}
