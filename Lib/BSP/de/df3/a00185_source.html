226 bool start ()
227 {
228 if (queueButtonPressed.create () == false)
229 {
230 assert (0);
231 return false;
232 }
233 if (FreeRTOS::Task::run ("TouchScreen", [this] { taskFn (); }, 128, 0) == nullptr)
234 {
235 assert (0);
236 return false;
237 }
238 return true;
239 }

259 void taskFn ()
260 {
261 auto extiFunction = [this] (const Lib::HAL::GPIO::EXTIx * exti) {
262 queueButtonPressed.overwriteFromISR (true);
263 };
264
265 // Reads the internal a/d convertor
266 auto readAdc = [this] (uint8_t cmd) {
267 auto writeByte = [this] (uint8_t data) -> void {
268 for (auto i = 0; i < 8; i++)
269 {
270 si.write ((data & 0x80) != 0);
271 __NOP ();
272 data <<= 1;
273 sck.write (0);
274 __NOP ();
275 sck.write (1);
276 __NOP ();
277 }
278 };
279
280 uint32_t num = 0;
281 si.write (0);
282 nss.write (0);
283 writeByte (cmd);
284
285 sck.write (0);
286 __NOP ();
287 sck.write (1);
288 __NOP ();
289
290 for (auto i = 0; i < 16; i++)
291 {
292 num <<= 1;
293 sck.write (0);
294 __NOP ();
295 sck.write (1);
296 __NOP ();
297 if (so.read () == 1) num++;
298 }
299 num >>= 4;
300 nss.write (1);
301
302 return static_cast<uint16_t> (num);
303 };
304
305 nirq.bind (extiFunction);
306 sck.write (1);
307 nss.write (1);
308
309 // Task loop...
310
311 while (1)
312 {
313 isPressed = false;
314
315 // wait for touch screen to be pressed ...
316 bool pinState = false;
317 while (pinState == false) queueButtonPressed.receive (pinState, portMAX_DELAY);
318
319 FreeRTOS::Task::delay (0);
320
321 intergratorX.reset ();
322 intergratorY.reset ();
323 position.X = -1;
324 position.Y = -1;
325
326 timeoutPinDebounce.set (parameters.debounceDuration);
327
328 while (timeoutPinDebounce.check () == false)
329 {
330 if (nirq.read () == true) timeoutPinDebounce.set (parameters.debounceDuration);
331 vTaskDelay (1);
332 }
333
334 while (1)
335 {
336 if (nirq.read () == true) break;
337
338 // Interpolates coordinates...
339
340 rawPosition.X = intergratorX.process (readAdc (CMD_READ_X));
341 rawPosition.Y = intergratorY.process (readAdc (CMD_READ_Y));
342
343 if (intergratorX.isFull () == false)
344 {
345 vTaskDelay (1);
346 continue;
347 }
348
349 // Reads and averages values ​​from the driver..
350
351 rawPosition.X = std::max<uint32_t> (parameters.xMin, rawPosition.X);
352 rawPosition.Y = std::max<uint32_t> (parameters.yMin, rawPosition.Y);
353
354 rawPosition.X = std::min<uint32_t> (rawPosition.X, parameters.xMax);
355 rawPosition.Y = std::min<uint32_t> (rawPosition.Y, parameters.yMax);
356
357 rawPosition.X -= parameters.xMin;
358 rawPosition.Y -= parameters.yMin;
359
360 position.X = static_cast<int32_t> (rawPosition.X / scaleFaconstructorX);
361 position.Y = static_cast<int32_t> (rawPosition.Y / scaleFaconstructorY);
362
363 // The touch screen is pressed and ready to read data.
364
365 isPressed = true;
366
367 switch (rotation)
368 {
369 case 0:
370 // position = position;
371 break;
372 case 1:
373 position.X = parameters.dispWidth - position.X;
374 position.Y = position.Y;
375 break;
376 case 2: // flip X
377 position.X = parameters.dispWidth - position.X;
378 position.Y = parameters.dispHeight - position.Y;
379 break;
380 case 3: // flip Y
381 position.X = position.X;
382 position.Y = parameters.dispHeight - position.Y;
383 break;
384 case 4: // flip X,Y
385 position.X = parameters.dispWidth - position.X;
386 position.Y = parameters.dispHeight - position.Y;
387 break;
388 default:
389 assert (0);
390 break;
391 }
392 vTaskDelay (40);
393 }
394 }
395 }
