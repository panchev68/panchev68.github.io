574 spiHandle->Init.CLKPolarity = SPI_POLARITY_LOW; // sets the clock polarity

608 void update () override
609 {
610 bool hasUpdate (false);
611
612 // checks if there is a value change and processes the output data accordingly
613 if (_isValueChanged == true)
614 {
615 _isValueChanged = false;
616 _bitset = _value;
617
618 auto txBuffer = _txBuffer + 1;
619 for (auto i = 0; i < _numberOfPins; i++)
620 {
621 auto & map = _mapFields [i];
622 auto & value = txBuffer [map.port + offsetof (MC33879::TxPort, value)];
623
624 if (_bitset.test (i)) value |= map.mask;
625 else value &= ~map.mask;
626 }
627 _timeout.set (UPDATE_TIMEOUT_DURATION);
628 _changesCounter++;
629 hasUpdate = true;
630 }
631
632 // updates the timeout timer and checks if it has expired
633 if (_timeout.check ())
634 {
635 _timeout.set (UPDATE_TIMEOUT_DURATION);
636 hasUpdate = true;
637 }
638
639 if (hasUpdate == true)
640 {
641 // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
642 _txBuffer [0] = _diagCounter;
643
644 // sends and receives data to/from the SPI port
645
646 _nss.write (0); // spi select
647
648 if (_spi.sendReceive (_txBuffer, _rxBuffer, _bufferSize, 5) == false)
649 {
650 assert (0);
651 }
652
653 _nss.write (1); // spi deselect
654 FreeRTOS::Task::yield ();
655 if (_diagCounter == 0) _oe.write (1); // outputs enable in first transaction
656 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
657
658 // processing of incoming data
659
660 // 1. Detects an SPI error
661
662 if ((_rxBuffer [_bufferSize - 1] != _txBuffer [0]) && (_diagCounter > 0))
663 {
664 if (_isSpiError == false)
665 {
666 _isSpiError = true;
667 if (onSpiError != nullptr) onSpiError ();
668 }
669 }
670 // 2. Processing the status of MC3879
671
672 for (auto i = 0; i < _numberOfPins; i++)
673 {
674 auto & map = _mapFields [i];
675 _bitset.set (i, ((_rxBuffer [map.port + offsetof (MC33879::RxPort, status)] & map.mask)));
676 }
677
678 // 3. Updates the status if it has changed
679 auto status (_bitset.to_ullong ());
680 if (_status != status)
681 {
682 _status = status;
683 if (onStatusChanged != nullptr) onStatusChanged (status);
684 }
685 }

1027 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::SEQUENTAL> ()
1028 {
1029 // Performs sequential updates of inputs and outputs. This is necessary when the spi port is common to both
1030 // interfaces. and reinitialization is required when servicing each device.
1031
1032 while (1)
1033 {
1034 auto wakeupTime (FreeRTOS::Task::getTickCount ());
1035
1036 handlePreUpdateEvent (); // Pre update event
1037 inputs.preUpdateCallback (); // Inputs pre update callback
1038 inputs.update (); // Inputs update
1039 outputs.preUpdateCallback (); // Outputs pre update callback
1040 outputs.update (); // Outputs update
1041 handlePostUpdateEvent (); // Post update event
1042
1043 if (_onTickEvent) _onTickEvent (); // tick event
1044 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
1045
1046 FreeRTOS::Task::delay (1); // Task delay 1ms
1047 }
1048 }

1066 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::PARALLEL> ()
1067 {
1068 // Performs parallel updates of inputs and outputs. This is possible when the spi port is different for both
1069 // interfaces. The initialization of the spi ports is done at the beginning.
1070
1071 inputs.preUpdateCallback (); // Inputs pre update callback
1072 outputs.preUpdateCallback (); // Outputs pre update callback
1073 while (1)
1074 {
1075 auto wakeupTime (FreeRTOS::Task::getTickCount ()); // Gets the current time
1076 handlePreUpdateEvent (); // Pre update event
1077
1078 inputs.update (); // Inputs update
1079 outputs.update (); // Outputs update
1080
1081 handlePostUpdateEvent (); // Post update event
1082 if (_onTickEvent) _onTickEvent (); // tick event
1083 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
1084 FreeRTOS::Task::delay (1); // Task delay 1ms
1085 }

1095 inline void DigitalPeripheralBase::setup ()

1098 outputs.setup (); // Outputs setup
1099

1101
1102 if (inputs.getSpiHandle () == outputs.getSpiHandle ()) // If the spi port is common to both interfaces

1104 // run the sequental task

1106 }
1107 else
1108 { // run the parallel task
