214 // reads data from sleep port...

216 _nss.write (0);
217
218 // FreeRTOS::Karnel::enterCritical ();
219 if (_spi.receive (_buffer, _bufferSize, 20) == false)

221 assert (0);
222 }
223 // FreeRTOS::Karnel::exitCritical ();

228
229 for (auto i = 0; i < _numberOfPins; i++)
230 {
231 auto & map = _mapFields [i];
232 _bitset.set (i, ((_buffer [map.port] & map.mask)));
233 }
234 if (_value != _bitset.to_ullong ())
235 {
236 _value = _bitset.to_ullong ();
237 if (onValueChanged != nullptr) onValueChanged (_value);

239 }

264 class DigitalPeripheralBase::Outputs : PortBase
265 {
266 public:
267
268 constexpr static inline uint32_t UPDATE_TIMEOUT_DURATION = 50;
269
272 std::function<void (const uint64_t &)> onValueChanged;
273
276 std::function<void (const uint64_t &)> onStatusChanged;
277
280 std::function<void ()> onSpiError;
281
283 Outputs (const Config & config)
284 : _numberOfPins (config.outputs.numberOfPins),
285 _nss (config.outputs.nss),
286 _oe (config.outputs.oe),
287 _spi (config.outputs.spi),
288 _mapFields (config.outputs.mapFields),
289 _bufferSize ((((config.outputs.numberOfPins + 7) / 8) * 2) + 1)
290
291 {
292 _rxBuffer = (new uint8_t [_bufferSize]);
293 _txBuffer = (new uint8_t [_bufferSize]);
294 }
295
297 ~Outputs ()
298 {
299 delete _rxBuffer;
300 delete _txBuffer;
301 }
302
305 void setup () override
306 {
307 _nss.write (1); // NSS = 1
308 _spi.init (); // openns the spi port
309 memset (_txBuffer, 0, _bufferSize); // clears the tx buffer
310
311 // sets th MC33879::TxPort::control to 0xff for all ports ...
312
313 auto port = reinterpret_cast<MC33879::TxPort *> (_txBuffer + 1);
314 for (auto i = 0; i < _bufferSize / sizeof (MC33879::TxPort); i++) port [i].control = 0xff;
315 }
316
319 void setValue (uint64_t value)
320 {
321 _isValueChanged = (value != _value);
322 if (_isValueChanged) _value = value;
323 }
324
327 SPI_HandleTypeDef * getSpiHandle () const override
328 {
329 return _spi.getHandle ();
330 }
331
333 void preUpdateCallback () override
334 {
335 assert (_spi.getHandle ());
336
337 _spi.deinit ();
338 auto spiHandle = _spi.getHandle ();
339 spiHandle->Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_128;
340 spiHandle->Init.CLKPolarity = SPI_POLARITY_LOW;
341 spiHandle->Init.CLKPhase = SPI_PHASE_2EDGE;
342 _spi.init (); // todo: dont use auto `HAL_SPI
343 }
344
347 void update () override
348
349 {
350 bool hasUpdate (false);
351
352 // processing the output data if there is a value change ...
353
354 if (_isValueChanged == true)
355 {
356 _isValueChanged = false;
357 _bitset = _value;
358
359 auto txBuffer = _txBuffer + 1;
360 for (auto i = 0; i < _numberOfPins; i++)
361 {
362 auto & map = _mapFields [i];
363
364 auto & value = txBuffer [map.port + offsetof (MC33879::TxPort, value)];
365 // auto & control = txBuffer [map.port + offsetof (MC33879::TxPort, control)];
366
367 if (_bitset.test (i))
368 {
369 value |= map.mask;
370 // control |= map.mask;
371 }
372 else
373 {
374 value &= ~map.mask;
375 }
376 }
377 _timeout.set (UPDATE_TIMEOUT_DURATION);
378 _changesCounter++;
379 hasUpdate = true;
380 }
381
382 // checks whether the timer that serves the status update has expired...
383
384 if (_timeout.check ())
385 {
386 _timeout.set (UPDATE_TIMEOUT_DURATION);
387 hasUpdate = true;
388 }
389
390 if (hasUpdate == true)
391 {
392 // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
393 _txBuffer [0] = _diagCounter;
394
395 // send / receive data to/from SPI port ...
396
397 _nss.write (0); // spi select
398
399 // FreeRTOS::Karnel::enterCritical ();
400 if (_spi.sendReceive (_txBuffer, _rxBuffer, _bufferSize, 5) == false)
401 {
402 assert (0);
403 }
404 // FreeRTOS::Karnel::exitCritical ();
405 _nss.write (1); // spi deselect
406 FreeRTOS::Task::yield ();
407 if (_diagCounter == 0) _oe.write (1); // outputs enable in first transaction
408 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
409
410 // processing of incoming data...
411
412 // 1. Detects an SPI error...
413
414 if ((_rxBuffer [_bufferSize - 1] != _txBuffer [0]) && (_diagCounter > 0))
415 {
416 if (_isSpiError == false)
417 {
418 _isSpiError = true;
419 if (onSpiError != nullptr) onSpiError ();
420 }
421 }
422 // 2. Processing the status of MC3879...
423
424 for (auto i = 0; i < _numberOfPins; i++)
425 {
426 auto & map = _mapFields [i];
427 _bitset.set (i, ((_rxBuffer [map.port + offsetof (MC33879::RxPort, status)] & map.mask)));
428 }
429
430 auto status (_bitset.to_ullong ());
431 if (_status != status)
432 {
433 _status = status;
434 if (onStatusChanged != nullptr) onStatusChanged (status);
435 }
436 }
437 }

601 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::SEQUENTAL> ()
602 {
603 // Performs sequential updates of inputs and outputs. This is necessary when the spi port is common to both
604 // interfaces. and reinitialization is required when servicing each device.
605
606 while (1)
607 {
608 auto wakeupTime (FreeRTOS::Task::getTickCount ());
609
610 handlePreUpdateEvent ();
611
612 inputs.preUpdateCallback ();
613 inputs.update ();
614
615 outputs.preUpdateCallback ();
616 outputs.update ();
617
618 handlePostUpdateEvent ();
619
620 if (_onTickEvent) _onTickEvent ();
621 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
622
623 FreeRTOS::Task::delay (1);
624 }
625 }
626
627 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::PARALLEL> ()
628 {
629 // Performs parallel updates of inputs and outputs. This is possible when the spi port is different for both
630 // interfaces. The initialization of the spi ports is done at the beginning.
631
632 inputs.preUpdateCallback ();
633 outputs.preUpdateCallback ();
634 while (1)
635 {
636 auto wakeupTime (FreeRTOS::Task::getTickCount ());
637 handlePreUpdateEvent ();
638
639 inputs.update ();
640 outputs.update ();
641
642 handlePostUpdateEvent ();
643 if (_onTickEvent) _onTickEvent ();
644 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
645 FreeRTOS::Task::delay (1);
646 }
647 }

649 inline void DigitalPeripheralBase::setup ()
650 {
651 inputs.setup ();
652 outputs.setup ();
653
654 // Selecting an operating mode...
655
656 if (inputs.getSpiHandle () == outputs.getSpiHandle ())
657 {
658 if (FreeRTOS::Task::run ("I/O Peripheral sequental", [this] { taskFn<UpdateMode::SEQUENTAL> (); }, 128, DEFAULT_TASK_PRIORITY) == nullptr) assert (0 && "Error creating task");
659 }
660 else
661 {
662 if (FreeRTOS::Task::run ("I/O Peripheral parallel", [this] { taskFn<UpdateMode::PARALLEL> (); }, 128, DEFAULT_TASK_PRIORITY) == nullptr) assert (0 && "Error creating task");
663 }
