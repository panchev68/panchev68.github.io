505 void setup () override {

508 memset (_txBuffer, 0, _bufferSize); // clears the tx buffer
509
510 // sets th MC33879::TxPort::control to 0xff for all ports

548 auto spiHandle = _spi.getHandle (); // gets the SPI handle
549 spiHandle->Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_128; // sets the baud rate prescaler
550 spiHandle->Init.CLKPolarity = SPI_POLARITY_LOW; // sets the clock polarity
551 spiHandle->Init.CLKPhase = SPI_PHASE_2EDGE; // sets the clock phase
552 _spi.init (); // initializes the SPI port

584 void update () override {
585 bool hasUpdate (false);
586
587 // checks if there is a value change and processes the output data accordingly
588 if (_isValueChanged == true) {
589 _isValueChanged = false;
590 _bitset = _value;
591
592 auto txBuffer = _txBuffer + 1;
593 for (auto i = 0; i < _numberOfPins; i++) {
594 auto & map = _mapFields [i];
595 auto & value = txBuffer [map.port + offsetof (MC33879::TxPort, value)];
596
597 if (_bitset.test (i)) value |= map.mask;
598 else value &= ~map.mask;
599 }
600 _timeout.set (UPDATE_TIMEOUT_DURATION);
601 _changesCounter++;
602 hasUpdate = true;
603 }
604
605 // updates the timeout timer and checks if it has expired
606 if (_timeout.check ()) {
607 _timeout.set (UPDATE_TIMEOUT_DURATION);
608 hasUpdate = true;
609 }
610
611 if (hasUpdate == true) {
612 // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
613 _txBuffer [0] = _diagCounter;
614
615 // sends and receives data to/from the SPI port
616
617 _nss.write (0); // spi select
618
619 if (_spi.sendReceive (_txBuffer, _rxBuffer, _bufferSize, 5) == false) {
620 assert (0);
621 }
622
623 _nss.write (1); // spi deselect
624 FreeRTOS::Task::yield ();
625 if (_diagCounter == 0) _oe.write (1); // outputs enable in first transaction
626 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
627
628 // processing of incoming data
629
630 // 1. Detects an SPI error
631
632 if ((_rxBuffer [_bufferSize - 1] != _txBuffer [0]) && (_diagCounter > 0)) {
633 if (_isSpiError == false) {
634 _isSpiError = true;
635 if (onSpiError != nullptr) onSpiError ();
636 }
637 }
638 // 2. Processing the status of MC3879
639
640 for (auto i = 0; i < _numberOfPins; i++) {
641 auto & map = _mapFields [i];
642 _bitset.set (i, (_rxBuffer [map.port + offsetof (MC33879::RxPort, status)] & map.mask));
643 }
644
645 // 3. Updates the status if it has changed
646 auto status (_bitset.to_ullong ());
647 if (_status != status) {
648 _status = status;
649 if (onStatusChanged != nullptr) onStatusChanged (status);
650 }
651 }

976 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::SEQUENTAL> () {
977 // Performs sequential updates of inputs and outputs. This is necessary when the spi port is common to both
978 // interfaces. and reinitialization is required when servicing each device.
979
980 while (1) {
981 auto wakeupTime (FreeRTOS::Task::getTickCount ());
982
983 handlePreUpdateEvent (); // Pre update event
984 inputs.preUpdateCallback (); // Inputs pre update callback
985 inputs.update (); // Inputs update
986 outputs.preUpdateCallback (); // Outputs pre update callback
987 outputs.update (); // Outputs update
988 handlePostUpdateEvent (); // Post update event
989
990 if (_onTickEvent) _onTickEvent (); // tick event
991 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
992
993 FreeRTOS::Task::delay (1); // Task delay 1ms
994 }
995 }

1013 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::PARALLEL> () {
1014 // Performs parallel updates of inputs and outputs. This is possible when the spi port is different for both
1015 // interfaces. The initialization of the spi ports is done at the beginning.
1016
1017 inputs.preUpdateCallback (); // Inputs pre update callback
1018 outputs.preUpdateCallback (); // Outputs pre update callback
1019 while (1) {
1020 auto wakeupTime (FreeRTOS::Task::getTickCount ()); // Gets the current time
1021 handlePreUpdateEvent (); // Pre update event
1022
1023 inputs.update (); // Inputs update
1024 outputs.update (); // Outputs update
1025
1026 handlePostUpdateEvent (); // Post update event
1027 if (_onTickEvent) _onTickEvent (); // tick event
1028 // FreeRTOS::Task::delayUntil (wakeupTime, 1);
1029 FreeRTOS::Task::delay (1); // Task delay 1ms
1030 }
1031 }

1040 inline void DigitalPeripheralBase::setup () {
1041 inputs.setup (); // Inputs setup
1042 outputs.setup (); // Outputs setup
1043
1044 // Selecting an operating mode
1045
1046 if (inputs.getSpiHandle () == outputs.getSpiHandle ()) // If the spi port is common to both interfaces
1047 {
1048 // run the sequental task
1049 if (FreeRTOS::Task::run ("I/O Peripheral sequental", [this] { taskFn<UpdateMode::SEQUENTAL> (); }, 128, DEFAULT_TASK_PRIORITY) == nullptr) assert (0 && "Error creating task");
1050 } else { // run the parallel task
1051 if (FreeRTOS::Task::run ("I/O Peripheral parallel", [this] { taskFn<UpdateMode::PARALLEL> (); }, 128, DEFAULT_TASK_PRIORITY) == nullptr) assert (0 && "Error creating task");
1052 }
