Lib::USB::SerialPortRepeaterBase

Example usage of SerialPortRepeaterBase for USB serial repeater functionality.

Demonstrates how to configure, instantiate, and use SerialPortRepeaterBase to enable USB serial communication with connection detection and event callbacks.

Example usage:
#include "Lib/USB/SerialPortRepeaterBase.hpp"
#include <Lib/HAL/GPIO.hpp>
int main() {
// Configure pins for USB detection and disable control
PinHandle usbDetectPin = GPIO::pin(GPIO::Port::A, 9);
PinHandle usbDisablePin = GPIO::pin(GPIO::Port::A, 10);
// Create configuration for SerialPortRepeaterBase
Lib::USB::SerialPortRepeaterBase::Config config;
config.usbDetectPin = usbDetectPin;
config.usbDisablePin = usbDisablePin;
config.baudRate = 115200;
// Instantiate SerialPortRepeaterBase
Lib::USB::SerialPortRepeaterBase serialRepeater(config);
// Register connection callback
serialRepeater.onConnected = [](bool connected) {
// Handle USB connection state change
if (connected) {
// USB connected
} else {
// USB disconnected
}
};
// Open the USB serial repeater port
if (serialRepeater.open()) {
// Port opened successfully
}
// Main loop
while (true) {
// Application logic
}
return 0;
}
