2 * Provides timer-based incremental encoder and button event handling for STM32 MCUs.
3 * This file defines the EncoderTimer class, which interfaces with a hardware timer and GPIO pin to
4 * track incremental encoder position and button events. It uses RTOS queues for event synchronization
5 * and supports detection of button push, pull, hold, and encoder rotation events. The implementation

20 /** Timer-based incremental encoder with button event support.
21 * Provides event-driven handling for encoder rotation and button actions using hardware timer and GPIO.
22 */
23 class EncoderTimer {
24 public:

33 /** Encoder and button event codes.
34 * Enumerates possible events detected by the encoder and button.
35 */
36 enum class EventCode : int8_t {
37 BUTTON_PUSH, /**< Button pressed. */
38 BUTTON_PULL, /**< Button released. */
39 BUTTON_HOLD, /**< Button held. */
40 ENCODER_CW, /**< Encoder rotated clockwise. */
41 ENCODER_CCW, /**< Encoder rotated counterclockwise. */
42 NONE = -1 /**< No event. */
43 };
44
45 /** Constructor of EncoderTimer class.
46 * @param config Encoder timer configuration.
47 */
48 explicit EncoderTimer(const Config & config)
49 : timer(config.timHandle), button(config.button) {}
50
51 /** Destructor of EncoderTimer class. */
52 virtual ~EncoderTimer() = default;
53
54 /** Initializes encoder timer and event queues.
55 * @return `true` if initialization succeeds, `false` otherwise.
56 */
57 virtual bool init() {
58 auto handleButtonPressEvent = [this](const Lib::HAL::GPIO::EXTIBase * exti) {
59 bool pin = !exti->read();
60 queButton.overwriteFromISR(&pin);
61 };
62 auto handleTimerValueChangedEvent = [this](uint32_t value) {
63 if (!quePosition.overwriteFromISR(static_cast<int32_t>(value))) assert(0);
64 };
65 if (!timer.open() || !quePosition.create() || !queButton.create() ||
66 !queueSet.create(2, {quePosition, queButton})) {
67 assert(0);
68 return false;
69 }
70 timer.valueChangedHook = handleTimerValueChangedEvent;
71 button.bind(handleButtonPressEvent);
72 rawPosition = 0;
73 return true;
74 }
75
76 /** Reads encoder and button events.
77 * @return Current event code.
78 */
79 virtual EventCode read() {
80 bool pinState = false;
81 int32_t pos = 0;
82 auto selectedQueue = queueSet.select(0);
83 if (selectedQueue == quePosition) {
84 if (quePosition.receive(pos, 0)) {
85 if (pos < rawPosition) {
86 rawPosition = pos;
87 return EventCode::ENCODER_CW;
88 }
89 if (pos > rawPosition) {
90 rawPosition = pos;
91 return EventCode::ENCODER_CCW;
92 }
93 }
94 } else if (selectedQueue == queButton) {
95 if (queButton.receive(pinState, 0)) {
96 if (pinState) {
97 if (eventCode != EventCode::BUTTON_PUSH) {
98 eventCode = EventCode::BUTTON_PUSH;
99 timeoutButtonHold.set(BUTTON_HOLD_TIME);
100 }
101 } else {
102 eventCode = EventCode::BUTTON_PULL;
103 }
104 return eventCode;
105 }
106 }
107 if (eventCode == EventCode::BUTTON_PUSH && timeoutButtonHold.isExpired()) {
108 eventCode = EventCode::BUTTON_HOLD;
109 return EventCode::BUTTON_HOLD;
110 }
111 return EventCode::NONE;
112 }
113
114 private:
115 static constexpr uint32_t BUTTON_HOLD_TIME = 2000; /**< Button hold time in ms. */
116 using Timer = Lib::HAL::TIM::Timer<Lib::HAL::TIM::Mode::ENCODER>;
117 Timer timer; /**< Encoder timer instance. */
118 Lib::HAL::GPIO::EXTIBase button; /**< Button interrupt handler. */
119 int32_t position{0}; /**< Absolute encoder position. */
120 int32_t rawPosition{0}; /**< Last raw encoder position. */
121 EventCode eventCode{EventCode::NONE}; /**< Current button event code. */
122 RTOS::Queue<int32_t> quePosition; /**< Queue for encoder position events. */
123 RTOS::Queue<bool> queButton; /**< Queue for button state events. */
124 RTOS::QueueSet queueSet; /**< Queue set for event synchronization. */
125 RTOS::Timeout timeoutButtonHold; /**< Timeout for button hold detection. */
126 };
