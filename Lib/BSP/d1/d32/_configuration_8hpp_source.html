16/** Configuration class for OTP memory management.
17 * Provides an interface for reading, writing, and validating configuration data stored in OTP memory.
18 */
19class Configuration {
20 static constexpr uint32_t VALID_MASK = 0x10000000; /*!< Mask for validity check. */
21
22public:
23 using Bitset = std::bitset<31>; /*!< Bitset type for configuration data. */
24
25 /** Constructor of Configuration class.
26 * @param offset Offset from the beginning of OTP memory.
27 * @pre `offset` must be less than the size of the OTP memory region.
28 */
29 explicit Configuration(uint32_t offset)
30 : flashAddress(Lib::HAL::OTP::beginAddr() + offset) {
31 assert(offset < (Lib::HAL::OTP::endAddr() - Lib::HAL::OTP::beginAddr()));
32 }
33
34 /** Destructor of Configuration class. */
35 virtual ~Configuration() = default;
36
37 /** Reads the configuration bitset from OTP memory.
38 * @return Bitset representing the configuration data.
39 */
40 Bitset read() const { return Bitset(readValue(flashAddress)); }
41
42 /** Writes the configuration bitset to OTP memory.
43 * @param value Bitset to write.
44 * @return `true` if write was successful, `false` otherwise.
45 */
46 bool write(const Bitset & value) {
47 auto & otp = Lib::HAL::OTP::getInstance();
48 auto writeToOtp = [&otp](uint32_t addr, uint32_t val) -> bool {
49 return otp.write(addr, val & ~VALID_MASK);
50 };
51
52 if (!otp.unlock()) assert(false);
53 bool result = writeToOtp(flashAddress, value.to_ulong());
54 if (!result) result = writeToOtp(flashAddress, value.to_ulong());
55 if (!otp.lock()) assert(false);
56 return result;
57 }
58
59 /** Checks if the configuration is empty.
60 * @return `true` if configuration is empty, `false` otherwise.
61 */
62 virtual bool isEmpty() const { return readValue(flashAddress) == EMPTY_VALUE; }
63
64 /** Checks if the configuration is valid.
65 * @return `true` if configuration is valid, `false` otherwise.
66 */
67 virtual bool isValid() const { return (readValue(flashAddress) & VALID_MASK) == 0; }
68
69private:
70 static constexpr uint32_t EMPTY_VALUE = 0xFFFFFFFF; /*!< Value representing empty configuration. */
71
72 const uint32_t flashAddress; /*!< Absolute address in OTP memory. */
73
74 /** Reads a 32-bit value from OTP memory.
75 * @param flashAddr Address to read from.
76 * @return 32-bit value read from OTP.
77 */
78 uint32_t readValue(uint32_t flashAddr) const {
79 return Lib::HAL::OTP::getInstance().read<uint32_t>(flashAddr);
80 }
