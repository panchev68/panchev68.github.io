59 /** Converts a MAC address array to a string.
60 * @param addressArray Address array.
61 * @return String representation.
62 */
63 static std::string toString(const std::array<uint8_t, 6> & addressArray) {
64 using Hex = Lib::Helper::Convert::HexHelper;
65 auto toHex = [](uint8_t value) -> std::string {
66 return {Hex::toAscii(value >> 4), Hex::toAscii(value)};
67 };
68 std::string result;
69 result.reserve(17);
70 for (size_t i = 0; i < addressArray.size(); ++i) {
71 if (i > 0) result += ':';
72 result += toHex(addressArray[i]);
73 }
74 return result;
75 }

77 /** Parses a MAC address from a string.
78 * @param macString String to parse.
79 * @param[out] addressArray Output address array.
80 * @return `true` if parsing was successful, `false` otherwise.
81 * @pre macString must be a valid MAC address string in hexadecimal format separated by ':'.
82 */
83 static bool fromString(const std::string & macString, std::array<uint8_t, 6> & addressArray) {
84 std::array<uint8_t, 6> tempArray{};
85 size_t pos = 0;
86 for (size_t i = 0; i < tempArray.size(); ++i) {
87 size_t found = (i < tempArray.size() - 1) ? macString.find(':', pos) : macString.size();
88 if ((found == std::string::npos && i < tempArray.size() - 1) || found == pos) return false;
89 int value = 0;
90 try {
91 value = std::stoi(macString.substr(pos, found - pos), nullptr, 16);
92 } catch (...) { return false; }
93 if (value < 0 || value > 255) return false;
94 tempArray[i] = static_cast<uint8_t>(value);
95 pos = found + 1;
96 }
97 addressArray = tempArray;
98 return true;
99 }

101 /** Validates a MAC address string.
102 * @param macString String to validate.
103 * @return `true` if the string is a valid MAC address, `false` otherwise.
104 */
105 static bool checkValidString(const std::string & macString) {
106 size_t pos = 0;
107 for (size_t i = 0; i < 6; ++i) {
108 size_t found = (i < 5) ? macString.find(':', pos) : macString.size();
109 if ((found == std::string::npos && i < 5) || found == pos) return false;
110 int value = 0;
111 try {
112 value = std::stoi(macString.substr(pos, found - pos), nullptr, 16);
113 } catch (...) { return false; }
114 if (value < 0 || value > 255) return false;
115 pos = found + 1;
116 }
117 return pos >= macString.size() + 1;
118 }

121 /** Generates a MAC address based on device serial number.
122 * STMicroelectronics SRL Manufacturer { 0x00, 0x80, 0xe1, xx, xx, xx }
123 * @return Array containing the MAC address.
124 */
125 static std::array<uint8_t, 6> generate() {
126 std::array<uint8_t, 6> result{0x00, 0x80, 0xe1, 0x00, 0x00, 0x00};
127 uint32_t serialValue = Lib::System::SerialNumber() ^ 0x00123456U;
128 for (size_t i = 0; i < 3; ++i) result[i + 3] = static_cast<uint8_t>(serialValue >> (i * 8));
129 return result;
130 }

133 };
