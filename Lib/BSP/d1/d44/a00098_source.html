94 virtual bool start(bool pauseAfterStart) {
95 assert(state == State::NO_INIT);
96 auto & serviceManager = Service::Manager::getInstance();
97 serviceManager.add(&channelScanner);
98 serviceManager.add(&parameters);
99 serviceManager.add(&networkDiscovery);
100 serviceManager.add(&spectrumAnalyzer);
101 RTOS::Semaphore semaphoreModemStatusHwReset;
102 semaphoreModemStatusHwReset.create();
103 state = module.init() ? State::INIT : State::ERROR;
104 modemStatus.open();
105 modemStatus.onHardwareReset = [this, &semaphoreModemStatusHwReset] { semaphoreModemStatusHwReset.give(); };
106 module.responseEvents.onReceivePacket16bit = [this](const Lib::Digi::Core::Frame0x81 & frame0x81) {
107 handleReceivedPacked16BitsEvent(frame0x81);
108 };
109 module.responseEvents.onTransmitStatus = [this](const Lib::Digi::Core::Frame0x89 & frame0x89) {
110 handleTransmitStatusUpdate(frame0x89);
111 };
112 module.reset();
113 isModulePresent = semaphoreModemStatusHwReset.take(2000);
114 if (!isModulePresent) assert(false);
115 modemStatus.onHardwareReset = nullptr;
116 semaphoreModemStatusHwReset.destroy();
117 if (!isModulePresent) return false;
118 vTaskDelay(100);
119 queueRxMessage.create();
120 queueTransmitDeliveryStatus.create();
121 parameters.read();
122 if (task.start([this] { taskFunc(); }) == nullptr) {
123 assert(false);
124 return false;
125 }
126 state = State::RUNNING;
127 if (pauseAfterStart) pause();
128 return true;
129 }

217 void handleReceivedPacked16BitsEvent(const Lib::Digi::Core::Frame0x81 & frame0x81) {
218 if (state != State::RUNNING) return;
219 connection.info.signalStrength = [](uint32_t value) -> uint8_t {
220 constexpr uint32_t MAX_RSSI_VAL = 90;
221 if (value > MAX_RSSI_VAL) value = MAX_RSSI_VAL;
222 return ((MAX_RSSI_VAL - value) * 100) / MAX_RSSI_VAL;
223 }(frame0x81.rssi);
224 diagnostic.receivedDataMessages++;
225 Core::Message msg(frame0x81.rfData);
226 if (!queueRxMessage.overwrite(msg)) assert(false);
227 }

232 void handleTransmitStatusUpdate(const Lib::Digi::Core::Frame0x89 & frame0x89) {

236 if (++messageFrameId == 0) messageFrameId = 1;
237 if (!queueTransmitDeliveryStatus.overwrite(frame0x89.deliveryStatus))
238 assert(false);
239 }
