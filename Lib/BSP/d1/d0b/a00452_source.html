211 static DayOfWeek getDayOfWeek(const Date & date) {
212 int y = date.year, m = date.month, d = date.day;
213 if (m < 3) {
214 m += 12;
215 --y;
216 }
217 int k = y % 100, j = y / 100;
218 int h = (d + 13 * (m + 1) / 5 + k + k / 4 + j / 4 + 5 * j) % 7;
219 int dow = ((h + 5) % 7) + 1;
220 return static_cast<DayOfWeek>(dow == 7 ? 7 : dow);
221 }

351 static uint32_t pack(const DateTime & dt) {
352 Packet p;
353 p.seconds = dt.time.seconds & 0x3F;
354 p.minutes = dt.time.minutes & 0x3F;
355 p.hours = dt.time.hours & 0x1F;
356 p.day = dt.date.day & 0x1F;
357 p.month = dt.date.month & 0x0F;
358 p.year = (dt.date.year >= Date::FIRST_YEAR) ? (dt.date.year - Date::FIRST_YEAR) & 0x3F : 0;
359 return (p.seconds) | (p.minutes << 6) | (p.hours << 12) | (p.day << 17) | (p.month << 22) | (p.year << 26);
360 }

366 static DateTime unpack(uint32_t value) {
367 Packet p;
368 p.seconds = value & 0x3F;
369 p.minutes = (value >> 6) & 0x3F;
370 p.hours = (value >> 12) & 0x1F;
371 p.day = (value >> 17) & 0x1F;
372 p.month = (value >> 22) & 0x0F;
373 p.year = (value >> 26) & 0x3F;
374 return DateTime(Date(p.day, p.month, Date::FIRST_YEAR + p.year), Time(p.hours, p.minutes, p.seconds));
