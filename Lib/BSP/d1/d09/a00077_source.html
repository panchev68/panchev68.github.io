129 void taskFunction() {
130 /* Write a byte to the SPI bus. */
131 auto writeSpiByte = [this](uint8_t data) {
132 for (int bit = 0; bit < 8; ++bit) {
133 si.write((data & 0x80) != 0);
134 __NOP();
135 data <<= 1;
136 sck.write(0);
137 __NOP();
138 sck.write(1);
139 __NOP();
140 }
141 };
142 /* Read ADC value over SPI. */
143 auto readAdc = [this, &writeSpiByte](uint8_t cmd) -> uint16_t {
144 uint32_t adc = 0;
145 si.write(0);
146 nss.write(0);
147 writeSpiByte(cmd);
148 sck.write(0);
149 __NOP();
150 sck.write(1);
151 __NOP();
152 for (int bit = 0; bit < 16; ++bit) {
153 adc = (adc << 1) | (so.read() ? 1 : 0);
154 sck.write(0);
155 __NOP();
156 sck.write(1);
157 __NOP();
158 }
159 adc >>= 4;
160 nss.write(1);
161 return static_cast<uint16_t>(adc);
162 };
163
164 auto buttonPressHandler = [this](const Lib::HAL::GPIO::EXTIBase *) { queueButtonPressed.overwriteFromISR(true); };
165
166 nirq.bind(buttonPressHandler);
167 sck.write(1);
168 nss.write(1);
169
170 while (true) {
171 isPressed = false;
172 bool pinPressed = false;
173 while (!pinPressed) queueButtonPressed.receive(pinPressed, portMAX_DELAY);
174 RTOS::Task::delay(0);
175 integratorX.resetWindow();
176 integratorY.resetWindow();
177 position.x = -1;
178 position.y = -1;
179 timeoutPinDebounce.set(parameters.debounceMs);
180 while (!timeoutPinDebounce.isExpired()) {
181 if (nirq.read()) timeoutPinDebounce.set(parameters.debounceMs);
182 vTaskDelay(1);
183 }
184 while (true) {
185 if (nirq.read()) break;
186 rawPosition.x = integratorX.processValue(readAdc(CMD_READ_X));
187 rawPosition.y = integratorY.processValue(readAdc(CMD_READ_Y));
188 if (!integratorX.isWindowFull()) {
189 vTaskDelay(1);
190 continue;
191 }
192 rawPosition.x = rawPosition.x < parameters.xMin ? parameters.xMin : rawPosition.x;
193 rawPosition.y = rawPosition.y < parameters.yMin ? parameters.yMin : rawPosition.y;
194 rawPosition.x = rawPosition.x > parameters.xMax ? parameters.xMax : rawPosition.x;
195 rawPosition.y = rawPosition.y > parameters.yMax ? parameters.yMax : rawPosition.y;
196 rawPosition.x -= parameters.xMin;
197 rawPosition.y -= parameters.yMin;
198 position.x = static_cast<int32_t>(rawPosition.x / scaleFactorX);
199 position.y = static_cast<int32_t>(rawPosition.y / scaleFactorY);
200 isPressed = true;
201 switch (rotation) {
202 case 0:
203 break;
204 case 1:
205 position.x = parameters.width - position.x;
206 break;
207 case 2:
208 position.x = parameters.width - position.x;
209 position.y = parameters.height - position.y;
210 break;
211 case 3:
212 position.y = parameters.height - position.y;
213 break;
214 case 4:
215 position.x = parameters.width - position.x;
216 position.y = parameters.height - position.y;
217 break;
218 default:
219 assert(0);
220 break;
221 }
222 vTaskDelay(40);
223 }
224 }
225 }
