64 * @param offsetSeconds The offset in seconds.
65 * @return The UTC minutes or -1 on error.
66 */
67 int convertLocalToUtcMinutes(int year, int month, int day, int hour, int minute, time_t offsetSeconds) const {
68 std::tm timeInfo{};
69 timeInfo.tm_year = year - 1900;
70 timeInfo.tm_mon = month - 1;
71 timeInfo.tm_mday = day;
72 timeInfo.tm_hour = hour;
73 timeInfo.tm_min = minute;
74 time_t localTime = mktime(&timeInfo);
75 return (localTime == -1) ? -1 : static_cast<int>((localTime - offsetSeconds) / 60);
76 }
