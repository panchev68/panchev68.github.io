65 /** Resets LAN9252 chip.
66 * @return `true` if reset successful, `false` otherwise.
67 */
68 bool resetChip() {
69 constexpr uint32_t RESET_LOW_MS = 50;
70 constexpr uint32_t RESET_TIMEOUT_MS = 1000;
71 constexpr uint32_t READY_DELAY_MS = 100;
72 RTOS::Timeout timeout;
73 pinReset.write(0);
74 vTaskDelay(RESET_LOW_MS);
75 pinReset.write(1);
76 timeout.set(RESET_TIMEOUT_MS);
77 while (!timeout.isExpired()) {
78 if (verifyChipId()) {
79 chipState = State::READY;
80 vTaskDelay(READY_DELAY_MS);
81 return true;
82 }
83 }
84 chipState = State::ERROR;
85 return false;
86 }

104 Helper::U32 result;

165 /** Reads CSR register.
166 * @param address Register address.
167 * @param buffer Destination buffer.
168 * @param length Number of bytes to read.
169 */
170 void readCSR(uint16_t address, void * buffer, uint8_t length) {
171 Helper::U32 param{0};
172 param.v[0] = loByte(address);
173 param.v[1] = hiByte(address);
174 param.v[2] = length;
175 param.v[3] = Registers::ESC_READ_BYTE;
176 writeU32(Registers::ESC_CSR_CMD_REG, param.val);
177 while (readU32(Registers::ESC_CSR_CMD_REG) & Registers::ESC_CSR_BUSY) {}
178 param.val = readU32(Registers::ESC_CSR_DATA_REG);
179 std::memcpy(buffer, param.v, length);
180 }

182 /** Writes CSR register.
183 * @param address Register address.
184 * @param buffer Source buffer.
185 * @param length Number of bytes to write.
186 */
187 void writeCSR(uint16_t address, const uint8_t * buffer, uint8_t length) {
188 Helper::U32 param{0};
189 std::memcpy(param.v, buffer, length);
190 writeU32(Registers::ESC_CSR_DATA_REG, param.val);
191 param.v[0] = loByte(address);
192 param.v[1] = hiByte(address);
193 param.v[2] = length;
194 param.v[3] = Registers::ESC_WRITE_BYTE;
195 writeU32(Registers::ESC_CSR_CMD_REG, param.val);
196 while (readU32(Registers::ESC_CSR_CMD_REG) & Registers::ESC_CSR_BUSY) {}
197 }

199 /** Reads from PRAM.
200 * @param address Register address.
201 * @param buffer Destination buffer.
202 * @param length Number of bytes to read.
203 */
204 void readPRAM(uint16_t address, uint8_t * buffer, uint16_t length) {
205 Helper::U64 param{0};
206 writeU32(Registers::PRAM_READ_CMD_REG, Registers::PRAM_RW_ABORT_MASK);
207 while (readU32(Registers::PRAM_READ_CMD_REG) & Registers::PRAM_RW_BUSY_8B) {}
208 param.w[0] = address;
209 param.w[1] = length;
210 param.w[2] = 0;
211 param.w[3] = 0x8000;
212 writeBuffer(Registers::PRAM_READ_ADDR_LEN_REG, reinterpret_cast<uint8_t *>(&param.val), 8);
213 while (!(readU32(Registers::PRAM_READ_CMD_REG) & Registers::IS_PRAM_SPACE_AVBL_MASK)) {}
214 param.val = readU32(Registers::PRAM_READ_FIFO_REG);
215 uint8_t bytePos = address & 0x03;
216 uint8_t partLen = std::min<uint8_t>(4 - bytePos, length);
217 std::memcpy(buffer, &param.v[bytePos], partLen);
218 length -= partLen;
219 uint16_t i = partLen;
220 if (length) {
221 spi.select();
222 spi.write(Registers::CMD_FAST_READ);
223 sendAddr(Registers::PRAM_READ_FIFO_REG);
224 spi.write(Registers::CMD_FAST_READ_DUMMY);
225 while (length) {
226 param.val = readBurstMode();
227 partLen = std::min<uint16_t>(4, length);
228 std::memcpy(buffer + i, &param, partLen);
229 i += partLen;
230 length -= partLen;
231 }
232 spi.deselect();
233 }
234 }

236 /** Writes to PRAM.
237 * @param buffer Source buffer.
238 * @param address Register address.
239 * @param length Number of bytes to write.
240 */
241 void writePRAM(const uint8_t * buffer, uint16_t address, uint16_t length) {
242 Helper::U64 param{0};
243 writeU32(Registers::PRAM_WRITE_CMD_REG, Registers::PRAM_RW_ABORT_MASK);
244 while (readU32(Registers::PRAM_WRITE_CMD_REG) & Registers::PRAM_RW_BUSY_8B) {}
245 param.w[0] = address;
246 param.w[1] = length;
247 param.w[2] = 0;
248 param.w[3] = 0x8000;
249 writeBuffer(Registers::PRAM_WRITE_ADDR_LEN_REG, reinterpret_cast<uint8_t *>(&param.val), 8);
250 while (!(readU32(Registers::PRAM_WRITE_CMD_REG) & Registers::IS_PRAM_SPACE_AVBL_MASK)) {}
251 uint8_t available = readU32(Registers::PRAM_WRITE_CMD_REG) & Registers::PRAM_SPACE_AVBL_COUNT_MASK;
252 uint8_t bytePos = address & 0x03;
253 uint8_t partLen = std::min<uint8_t>(4 - bytePos, length);
254 param.val = 0;
255 std::memcpy(&param.v[bytePos], buffer, partLen);
256 writeU32(Registers::PRAM_WRITE_FIFO_REG, param.val);
257 available--;
258 length -= partLen;
259 uint16_t i = partLen;
260 if (length) {
261 spi.select();
262 spi.write(Registers::CMD_SERIAL_WRITE);
263 sendAddr(Registers::PRAM_WRITE_FIFO_REG);
264 while (length) {
265 partLen = std::min<uint16_t>(4, length);
266 param.val = 0;
267 std::memcpy(&param, buffer + i, partLen);
268 writeBurstMode(param.val);
269 i += partLen;
270 length -= partLen;
271 available--;
272 }
273 spi.deselect();
274 }
275 }

288 */
289 uint32_t readBurstMode() {
290 Helper::U32 retvar;
291 std::array<uint8_t, 4> readBytes{0};
292 spi.readWrite(nullptr, 0, readBytes.data(), readBytes.size());
293 retvar.bytes.lb = readBytes[0];
294 retvar.bytes.hb = readBytes[1];
295 retvar.bytes.ub = readBytes[2];
296 retvar.bytes.mb = readBytes[3];
