72 Error parse(const uint8_t * data, size_t len, Command & command, uint8_t & option) {
73 if (!data || len == 0) return Error::EMPTY_MESSAGE;
74
75 do {
76 if (len < sizeof(MessageHeader)) return Error::EMPTY_MESSAGE;
77 const auto & header = *reinterpret_cast<const MessageHeader *>(data);
78 if (header.delimiter != START_DELIMITER) return Error::INVALID_START_DELIMITER;
79
80 handleIncomingMessageHeader(header);
81 command = static_cast<Command>(header.command);
82 option = header.option;
83
84 size_t payloadLen = header.length;
85 if (len < sizeof(MessageHeader) + payloadLen) return Error::INVALID_PAYLOAD_LENGTH;
86
87 const uint8_t * payloadData = data + sizeof(MessageHeader);
88 size_t nodesNum = 0;
89
90 switch (command) {
91 case Command::ECHO:
92 echoCallback();
93 break;
94 case Command::INPUTS:
95 if (payloadLen % sizeof(uint64_t) != 0) return Error::INVALID_PAYLOAD_LENGTH;
96 nodesNum = payloadLen / sizeof(uint64_t);
97 auto payload = reinterpret_cast<const uint64_t *>(payloadData);
98 updateNodeActivityCallback(nodesNum);
99 for (size_t i = 0; i < nodesNum; ++i) inputsUpdateCallback(static_cast<uint8_t>(i), payload[i]);
100 break;
101 case Command::OUTPUTS:
102 if (payloadLen % sizeof(uint64_t) != 0) return Error::INVALID_PAYLOAD_LENGTH;
103 nodesNum = payloadLen / sizeof(uint64_t);
104 updateNodeActivityCallback(nodesNum);
105 for (size_t i = 0; i < nodesNum; ++i) {
106 uint64_t value;
107 std::memcpy(&value, payloadData + i * sizeof(uint64_t), sizeof(uint64_t));
108 outputsUpdateCallback(static_cast<uint8_t>(i), value);
109 }
110 break;
111 default:
112 return Error::COMMAND_NOT_SUPPORT;
113 }
114
115 data += sizeof(MessageHeader) + payloadLen;
116 len -= sizeof(MessageHeader) + payloadLen;
117 } while (len > 0);
118
119 return Error::OK;
120 }

133 std::vector<uint8_t> build(Command cmd, uint8_t option, uint16_t id, const std::vector<uint8_t> * payload = nullptr) {
134 auto payloadSize = payload ? payload->size() : 0;
135 MessageHeader header = {START_DELIMITER, id, (lastSentTimeStamp = HAL_GetTick()), static_cast<uint8_t>(cmd), option, static_cast<uint8_t>(payloadSize)};
136 std::vector<uint8_t> vector(reinterpret_cast<uint8_t *>(&header), reinterpret_cast<uint8_t *>(&header) + sizeof(header));
137 if (payloadSize > 0) vector.insert(vector.end(), payload->begin(), payload->end());
138 return vector;
