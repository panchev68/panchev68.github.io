90 void transmitBlock(const void * buffer, uint32_t x, uint32_t y, size_t width, size_t height) override {
91 auto pixelData = static_cast<const uint16_t *>(buffer);
92 auto setWindow = [this, &x, &y, &width, &height]() -> size_t {
93 auto setPosReg = [this](Registers reg, uint16_t start, uint16_t end) {
94 writeReg(reg);
95 uint8_t regData[4] = {HI_BYTE(start), LO_BYTE(start), HI_BYTE(end), LO_BYTE(end)};
96 writeData(regData, 4);
97 };
98 y += 20;
99 setPosReg(Registers::CASET, x, x + width - 1);
100 setPosReg(Registers::PASET, y, y + height - 1);
101 writeReg(Registers::RAMWR);
102 return width * height;
103 };
104 setTransmitActive(1);
105 size_t pixelCount = setWindow();
106 assert(pixelCount <= 0xffff /* Pixel count must fit in 16 bits */);
107 auto spiHandle = spi.getHandle();
108 spiHandle->Init.DataSize = SPI_DATASIZE_16BIT;
109 if (HAL_SPI_Init(spiHandle) != HAL_OK) assert(0 /* SPI init failed */);
110 if (!spi.send(const_cast<uint16_t *>(pixelData), pixelCount, 0)) assert(0 /* SPI send failed */);
111 }

130 void writeReg(uint8_t value) {
131 auto spiHandle = spi.getHandle();
132 dc.write(0);
133 if (spiHandle->Init.DataSize != SPI_DATASIZE_8BIT) {
134 spiHandle->Init.DataSize = SPI_DATASIZE_8BIT;
135 if (HAL_SPI_Init(spiHandle) != HAL_OK) assert(0 /* SPI init failed */);
136 }
137 if (HAL_SPI_Transmit(spiHandle, &value, 1, 10) != HAL_OK) assert(0 /* SPI transmit failed */);
138 dc.write(1);
139 }

145 void writeData(const uint8_t * data, size_t length) {
146 assert(data /* Data pointer must not be null */);
147 if (!length) return;
148 auto spiHandle = spi.getHandle();
149 if (spiHandle->Init.DataSize != SPI_DATASIZE_8BIT) {
150 spiHandle->Init.DataSize = SPI_DATASIZE_8BIT;
151 if (HAL_SPI_Init(spiHandle) != HAL_OK) assert(0 /* SPI init failed */);
152 }
153 if (HAL_SPI_Transmit(spiHandle, const_cast<uint8_t *>(data), length, 10) != HAL_OK) assert(0 /* SPI transmit failed */);
154 }
