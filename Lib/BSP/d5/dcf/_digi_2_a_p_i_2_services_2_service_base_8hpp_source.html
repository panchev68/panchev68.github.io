15 /** Base class for Digi API services.
16 *
17 * The ServiceBase class provides a common interface and lifecycle management for all service classes in the
18 * Digi API. It supports open/close state management, update cycles, and integration with service managers.
19 * The class is designed for embedded Digi modules, enabling robust service management, error handling, and
20 * maintainable, type-safe service development.
21 */
22 class ServiceBase {
23 friend class Manager;
24
25 protected:
26 bool isOpen{false}; /*!< Indicates if the service is open. */
27 bool hasClosed{false}; /*!< Indicates if the service has been closed. */
28 bool isFirstUpdate{false}; /*!< Indicates if this is the first update call. */
29
30 /** Constructor of ServiceBase class. */
31 ServiceBase() = default;
32
33 /** Destructor of ServiceBase class. */
34 virtual ~ServiceBase() = default;
35
36 /** Opens the service.
37 * @return `true` if the service was opened successfully, `false` otherwise.
38 */
39 virtual bool open() {
40 hasClosed = false;
41 isFirstUpdate = true;
42 isOpen = true;
43 return true;
44 }
45
46 /** Closes the service. */
47 virtual void close() { hasClosed = true; }
48
49 /** Checks if the service is open.
50 * @return `true` if the service is open, `false` otherwise.
51 */
52 bool isServiceOpen() const { return isOpen; }
53
54 /** Initializes the service. Called once after opening. */
55 virtual void beginUpdate() {}
56
57 /** Updates the service. Called periodically. */
58 virtual void update() {}
59
60 /** Called after the service is closed. */
61 virtual void endUpdate() {}
62
63 private:
64 /** Updates the service state and invokes callbacks. */
65 void updateHandler() {
66 if (!isOpen) return;
67 if (isFirstUpdate) {
68 isFirstUpdate = false;
69 beginUpdate();
70 }
71 update();
72 if (hasClosed) {
73 hasClosed = false;
74 isOpen = false;
75 endUpdate();
76 }
77 }
78 };
