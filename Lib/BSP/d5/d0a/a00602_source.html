132 // Initialize the display
133
134 const uint8_t * initSequence = initData;
135 for (;;) {
136 auto * row = reinterpret_cast<const Command *> (initSequence); // Cast the initSequence to Command
137 if (row->len == 0) break; // If the row length is 0, break
138 writeCommand (row->cmd); // Write the command
139 initSequence = &row->data; // Set the initSequence to the data
140 for (auto i = 0; i < row->len - 1; i++) writeData (initSequence [i]); //
141 initSequence += row->len - 1; // Increment the initSequence
142 }
143 vTaskDelay (120); // Delay for 120ms
144 writeCommand (Registers::DISPON); // Write the display on command
145 vTaskDelay (150); // Delay for 150ms
146 setBacklight (100); // Set the backlight to 100 %
147 postInitCallback (); // Call the postInitCallback
148 return true; // Return true

165 void transmitBlock (const void * src, uint32_t x, uint32_t y, size_t width, size_t height) override {
166 auto data = reinterpret_cast<const uint16_t *> (src);
167 auto sendBuffer = [this, &data, &x, &y, &width, &height] () -> size_t {
168 auto writeCommandExt = [this] (uint8_t reg, uint32_t begin, uint32_t end) {
169 std::array<uint8_t, 4> data{HI_BYTE (begin), LO_BYTE (begin), HI_BYTE (end), LO_BYTE (end)};
170 writeCommand (reg);
171 if (HAL_SPI_Transmit (_spi.getHandle (), data.data (), data.size (), 100) != HAL_OK) assert (0);
172 };
173
174 auto spiHandle = _spi.getHandle ();
175 spiHandle->Init.DataSize = SPI_DATASIZE_8BIT;
176
177 if (HAL_SPI_Init (spiHandle) != HAL_OK) assert (0);
178
179 // Sets the position
180 writeCommandExt (Registers::CASET, x, x + width - 1);
181 writeCommandExt (Registers::PASET, y, y + height - 1);
182 writeCommand (Registers::RAMWR);
183
184 spiHandle->Init.DataSize = SPI_DATASIZE_16BIT;
185 if (HAL_SPI_Init (spiHandle) != HAL_OK) assert (0);
186 return width * height;
187 };
