23 /** Protocol class for Digi XBee API frames.
24 *
25 * Handles parsing, validation, and processing of Digi XBee API frames, including frame construction,
26 * checksum verification, and event dispatching for Digi XBee modules.
27 */
28 class Protocol {
29 public:
30 static constexpr uint8_t START_DELIMITER = 0x7E; /*!< Start delimiter for Digi API frames. */
31 static constexpr ApiMode API_MODE = ApiMode::API1; /*!< API mode used. */
32
33 /** Parses the incoming Digi API frame.
34 * @param apiFrame Incoming Digi API frame.
35 * @param[out] frameData Output frame data.
36 * @return `true` if parsing is successful, `false` otherwise.
37 */
38 static bool parse(const ApiFrameBuffer & apiFrame, Frame::Data & frameData) {
39 if (!isValid(apiFrame)) return false;
40 const auto * dst = apiFrame.getData();
41 const auto * header = reinterpret_cast<const Frame::Header *>(dst);
42 size_t payloadLen = static_cast<size_t>((header->lengthMSB << 8) | header->lengthLSB);
43 dst += sizeof(Frame::Header);
44 frameData.type = *(dst++);
45 --payloadLen;
46 frameData.vector.clear();
47 frameData.vector.insert(frameData.vector.end(), dst, dst + payloadLen);
48 return true;
49 }
50
51 /** Processes a received Digi API frame and dispatches events.
52 * @param apiFrame Reference to Digi API frame (incoming message).
53 * @return `true` if the message was processed successfully, `false` otherwise.
54 */
55 static bool processMessage(const ApiFrameBuffer & apiFrame) {
56 static Frame::Data frameData;
57 auto & events = ResponseEvents::getInstance();
58 if (!parse(apiFrame, frameData)) {
59 assert(false);
60 return false;
61 }
62 switch (frameData.type) {
63 case Frame::Type::RECEIVE_PACKET_64BIT:
64 if (events.onReceivePacket64bit) events.onReceivePacket64bit(Frame0x80{frameData});
65 break;
66 case Frame::Type::RECEIVE_PACKET_16BIT:
67 if (events.onReceivePacket16bit) events.onReceivePacket16bit(Frame0x81{frameData});
68 break;
69 case Frame::Type::LOCAL_AT_COMMAND_RESPONSE:
70 if (events.onLocalAtCommand) events.onLocalAtCommand(Frame0x88{frameData});
71 break;
72 case Frame::Type::TRANSMIT_STATUS:
73 if (events.onTransmitStatus) events.onTransmitStatus(Frame0x89{frameData});
74 break;
75 case Frame::Type::MODEM_STATUS:
76 if (events.onModemStatusChanged) events.onModemStatusChanged(Frame0x8a{frameData});
77 break;
78 case Frame::Type::EXTENDED_TRANSMIT_STATUS:
79 if (events.onExtendedTransmitStatusChanged) events.onExtendedTransmitStatusChanged(Frame0x8b{frameData});
80 break;
81 case Frame::Type::RECEIVE_PACKET:
82 if (events.onPacketReceived) events.onPacketReceived(Frame0x90{frameData});
83 break;
84 case Frame::Type::REMOTE_AT_COMMAND_RESPONSE:
85 if (events.onRemoteATCommand) events.onRemoteATCommand(Frame0x97{frameData});
86 break;
87 default: break;
88 }
89 return true;
90 }

92 /** Checks whether the Digi API frame is valid (non-escaped message).
93 * @param apiFrame Reference to Digi API frame.
94 * @return `true` if the message is valid, `false` otherwise.
95 */
96 static bool isValid(const ApiFrameBuffer & apiFrame) {
97 size_t msgSize = apiFrame.getSize();
98 if (msgSize < sizeof(Frame::Header) + 1) return false;
99 const auto & header = *reinterpret_cast<const Frame::Header *>(apiFrame.getData());
100 if (header.startDelimiter != START_DELIMITER) return false;
101 size_t headerLen = static_cast<size_t>((header.lengthMSB << 8) | header.lengthLSB);
102 size_t crcLen = sizeof(Frame::Header) + headerLen;
103 if (crcLen > msgSize - 1) return false;
104 return Checksum(apiFrame) == apiFrame.getData()[crcLen];
105 }
106
107 /** Checks whether the message is valid (non-escaped message).
108 * @param dst Pointer to the raw message buffer.
109 * @return `true` if the message is valid, `false` otherwise.
110 */
111 static bool isValid(const uint8_t * dst) {
112 assert(dst);
113 const auto & header = *reinterpret_cast<const Frame::Header *>(dst);
114 if (header.startDelimiter != START_DELIMITER) return false;
115 size_t headerLen = static_cast<size_t>((header.lengthMSB << 8) | header.lengthLSB);
116 size_t crcLen = sizeof(Frame::Header) + headerLen;
117 return Checksum(dst) == dst[crcLen];
118 }

120 /** Builds a Digi API frame from frame data.
121 * @param frameData Reference to the frame data.
122 * @return Reference to the built ApiFrameBuffer object.
123 */
124 const static ApiFrameBuffer & build(const Frame::Data & frameData) {
125 static ApiFrameBuffer msg;
126 assert(API_MODE == ApiMode::API1);
127 size_t frameSize = frameData.vector.size() + 1;
128 msg.clear();
129 msg.append(START_DELIMITER);
130 msg.append(static_cast<uint8_t>(frameSize >> 8));
131 msg.append(static_cast<uint8_t>(frameSize & 0xff));
132 msg.append(frameData.type);
133 msg.append(frameData.vector);
134 msg.append(Checksum(msg));
135 return msg;
136 }

143 Protocol & operator=(Protocol &&) = delete;
144 ~Protocol() = default;
145 };
