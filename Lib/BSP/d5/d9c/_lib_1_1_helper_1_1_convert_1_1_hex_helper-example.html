Lib::Helper::Convert::HexHelper

Example usage of HexHelper.

Demonstrates conversion between hexadecimal, ASCII, and BCD formats using HexHelper static methods.

Example usage:

#include "Lib/Helper/Convert/HexHelper.hpp"
#include <iostream>
int main() {
using Lib::Helper::Convert::HexHelper;
uint8_t hexValue = 0xB;
char asciiChar = HexHelper::toAscii(hexValue);
std::cout << "Hex 0xB to ASCII: " << asciiChar << std::endl;
char asciiInput = 'F';
uint8_t value = HexHelper::fromAscii(asciiInput);
std::cout << "ASCII 'F' to value: " << static_cast<int>(value) << std::endl;
uint16_t number = 1234;
uint32_t bcd = HexHelper::toBcd(number);
std::cout << "1234 to BCD: 0x" << std::hex << bcd << std::endl;
uint8_t smallNumber = 42;
uint8_t bcd8 = HexHelper::toBcd8(smallNumber);
std::cout << "42 to BCD8: 0x" << std::hex << static_cast<int>(bcd8) << std::endl;
return 0;
}
