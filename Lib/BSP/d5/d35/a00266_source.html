112 void taskFn() {
113 Socket::Error error;
114 size_t length;
115 DataHandle<> incomingData;
116 DataHandle<> outgoingData;
117
118 setConnectionState(ConnectionState::INIT);
119 initCallback();
120
121 while (true) {
122 FreeRTOS::delay(1);
123
124 while (!readLinkState()) {
125 if (getConnectionState() == ConnectionState::CONNECTED) {
126 setConnectionState(ConnectionState::DISCONNECTED);
127 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
128 }
129 FreeRTOS::delay(100);
130 }
131
132 if (getConnectionState() != ConnectionState::CONNECTED) {
133 if ((error = sockets[0].open(Socket::Protocol::TCP, remote.port, 0)) == Socket::Error::OK) {
134 if ((error = sockets[0].connect(remote)) == Socket::Error::OK) {
135 setConnectionState(ConnectionState::CONNECTED);
136 } else {
137 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
138 FreeRTOS::delay(100);
139 }
140 } else assert(0);
141 } else {
142 if ((length = sockets[0].getReceivedSize()) > 0) {
143 updateRxLedCallback(true);
144 diagnostic.rxMessageCounter++;
145 incomingData(new uint8_t[length], length);
146 if ((error = sockets[0].receive(incomingData, length)) == Socket::Error::OK) {
147 handleIncomingMessage(incomingData);
148 } else {
149 if (error != Socket::Error::ERR_DATALEN) {
150 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
151 }
152 }
153 delete incomingData.data();
154 }
155
156 if (handleOutgoingMessage(outgoingData)) {
157 if ((error = sockets[0].send(outgoingData, length)) == Socket::Error::OK) {
158 updateTxLedCallback(true);
159 diagnostic.txMessageCounter++;
160 } else {
161 if ((error = sockets[0].close()) != Socket::Error::OK) assert(0);
162 setConnectionState(ConnectionState::DISCONNECTED);
163 }
164 }
165 }
166
167 taskYIELD();
168 applicationLoopCallback();
169 }
170 }
