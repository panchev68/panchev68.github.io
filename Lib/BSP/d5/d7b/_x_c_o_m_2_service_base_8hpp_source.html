24 /** Convert value to vector.
25 * Converts a value of type `T` to a vector of bytes.
26 * @tparam T Type of the value.
27 * @tparam SIZE Size of the value in bytes.
28 * @param value Value to convert.
29 * @return Vector of bytes representing the value.

31 template <typename T, size_t SIZE = sizeof(T)>
32 static std::vector<uint8_t> toVec(const T & value) {
33 std::vector<uint8_t> vec;
34 vec.reserve(SIZE);
35 auto ptr = reinterpret_cast<const uint8_t *>(&value);
36 for (size_t i = 0; i < SIZE; ++i) vec.push_back(ptr[i]);
37 return vec;
38 }

53 /** Extract value from vector.
54 * Extracts a value of type `T` from a vector of bytes at the specified offset.
55 * @tparam T Type of the value.
56 * @param vec Source vector.
57 * @param offset Offset in the vector.
58 * @return Extracted value.
59 */
60 template <typename T>
61 static T fromVec(const std::vector<uint8_t> & vec, uint32_t offset = 0) {
62 T val = 0;
63 size_t size = std::min(vec.size() - offset, sizeof(T));
64 auto ptr = reinterpret_cast<uint8_t *>(&val);
65 for (size_t i = 0; i < size; ++i) ptr[i] = vec.at(i + offset);
66 return val;
