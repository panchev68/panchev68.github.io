Lib::Helper::Convert::Percent

Example usage of Percent class for percentage to dBM conversion.

Demonstrates converting percentage values to dBM and vice versa using static methods.

Example usage:

#include "Lib/Helper/Convert/Percent.hpp"
#include <iostream>
int main() {
uint8_t percentage = 75;
int8_t dbm = Lib::Helper::Convert::Percent::toDBm(percentage);
std::cout << "Percentage: " << static_cast<int>(percentage) << "% -> dBM: " << static_cast<int>(dbm) << std::endl;
int8_t dbmValue = -80;
uint8_t percent = Lib::Helper::Convert::Percent::fromDBm(dbmValue);
std::cout << "dBM: " << static_cast<int>(dbmValue) << " -> Percentage: " << static_cast<int>(percent) << "%" << std::endl;
return 0;
}
