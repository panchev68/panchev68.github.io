1/** EventGroup class and event bits utilities for RTOS event synchronization.
2 * This file provides the `EventGroup` class and related types for managing event groups in FreeRTOS.
3 * It supports bitwise event signaling, waiting, and synchronization with configurable bit width
4 * depending on the FreeRTOS tick type. The class encapsulates creation, destruction, and manipulation
5 * of event bits, including ISR-safe operations and timeout-based waiting. It is designed for use
6 * in embedded systems with FreeRTOS and provides a type-safe, C++-style interface.
7 * @file EventGroup.hpp
8 */

282 /** Waits for specified bits with a timeout in seconds.
283 * @param bits Bits to wait for.
284 * @param timeoutSec Timeout in seconds.
285 * @param clearMode Whether to clear bits after wait.
286 * @param waitMode Whether to wait for any or all bits.
287 * @return The value of the event bits after waiting.
288 */
289 EventBits waitForBitsSeconds(const EventBits & bits, uint32_t timeoutSec, ClearMode clearMode = ClearMode::NO_CLEAR, WaitMode waitMode = WaitMode::ANY) const {
290 static constexpr uint64_t MS_PER_SECOND = 1000ULL;
291 static constexpr uint64_t MAX_TIMEOUT_MS = std::numeric_limits<uint32_t>::max();
292
293 const uint64_t timeoutMs = static_cast<uint64_t>(timeoutSec) * MS_PER_SECOND;
294 const uint32_t clampedTimeout = (timeoutMs > MAX_TIMEOUT_MS) ? static_cast<uint32_t>(MAX_TIMEOUT_MS) : static_cast<uint32_t>(timeoutMs);
295 return waitForBitsMs(bits, clampedTimeout, clearMode, waitMode);
296 }

512 * void WaiterTask(void *param) {
513 * (void)param;
514 * printf("Waiter: Waiting for bit %u...\n", BIT_INDEX);

529 * vTaskDelete(nullptr);

532 * int main() {
533 * xTaskCreate(WaiterTask, TASK1_NAME, 256, nullptr, 2, nullptr);
534 * xTaskCreate(SetterTask, TASK2_NAME, 256, nullptr, 2, nullptr);

537 * return 0;
538 * }
