44/** EventGroup class for FreeRTOS event group management.
45 * Provides a modern, type-safe C++17 wrapper for FreeRTOS event groups, supporting bitwise signaling,
46 * waiting, and synchronization with compile-time validation and utility methods.

228 /** Waits for specified event bits.
229 * @param bits Bits to wait for.
230 * @param clearMode Whether to clear bits after waiting.
231 * @param waitMode Wait for any or all bits.
232 * @param waitTime Timeout in ticks.
233 * @return Event bits after waiting.

235 EventBits waitForBits(
236 const EventBits & bits, ClearMode clearMode = ClearMode::CLEAR_NO_CLEAR, WaitMode waitMode = WaitMode::WAIT_ANY,
237 TickType waitTime = DEFAULT_WAIT) const {
238 if (!isCreated()) return EventBits(0);
239 if (!bits.any()) return getBits();
240 return fromBitMask(xEventGroupWaitBits(

242 (waitMode == WaitMode::WAIT_ALL) ? pdTRUE : pdFALSE, waitTime));

259 /** Waits for specified event bits with timeout in seconds.
260 * @param bits Bits to wait for.
261 * @param timeoutSec Timeout in seconds.
262 * @param clearMode Whether to clear bits after waiting.
263 * @param waitMode Wait for any or all bits.
264 * @return Event bits after waiting.
265 */
266 EventBits waitForBitsSeconds(
267 const EventBits & bits, uint32_t timeoutSec, ClearMode clearMode = ClearMode::CLEAR_NO_CLEAR,
268 WaitMode waitMode = WaitMode::WAIT_ANY) const {
269 const uint64_t timeoutMs = static_cast<uint64_t>(timeoutSec) * 1000ULL;
270 const uint32_t clampedTimeout =
271 (timeoutMs > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max()))
272 ? std::numeric_limits<uint32_t>::max()
273 : static_cast<uint32_t>(timeoutMs);
274 return waitForBitsMs(bits, clampedTimeout, clearMode, waitMode);
275 }
