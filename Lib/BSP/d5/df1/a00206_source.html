131 void taskFn ()
132 {
133 // Check pressed
134 auto checkPressed = [this] () {
135 bool pinState = (_nirq.read () == 0);
136
137 if (_debountTime == 0) return pinState;
138
139 if (pinState == true)
140 {
141 if (_lastPressState == false)
142 {
143 _lastPressState = true;
144 _timeoutDebount.set (_debountTime);
145 }
146 else return _timeoutDebount.check ();
147 }
148 else _lastPressState = false;
149 return false;
150 };
151
152 // Read the internal a/d convertor
153
154 auto readAdc = [this] (uint8_t cmd) {
155 auto writeByte = [this] (uint8_t data) -> void {
156 for (auto i = 0; i < 8; i++)
157 {
158 _si.write ((data & 0x80) != 0);
159 __NOP ();
160 data <<= 1;
161 _sck.write (0);
162 __NOP ();
163 _sck.write (1);
164 __NOP ();
165 }
166 };
167
168 uint32_t num = 0;
169 _si.write (0);
170 _nss.write (0);
171 writeByte (cmd);
172
173 _sck.write (0);
174 __NOP ();
175 _sck.write (1);
176 __NOP ();
177
178 for (auto i = 0; i < 16; i++)
179 {
180 num <<= 1;
181 _sck.write (0);
182 __NOP ();
183 _sck.write (1);
184 __NOP ();
185 if (_so.read () == 1) num++;
186 }
187 num >>= 4;
188 _nss.write (1);
189
190 return static_cast<uint16_t> (num);
191 };
192
193 _sck.write (1);
194 _nss.write (1);
195
196 bool isPressed = false;
197
198 // Task loop
199
200 for (;;)
201 {
202 if (checkPressed ())
203 {
204 if (isPressed == false) // 1st time afrer press
205 {
206 isPressed = true;
207 _intergratorX.reset ();
208 _intergratorY.reset ();
209 position.X = -1;
210 position.Y = -1;
211 }
212
213 // Interpolate coordinates
214
215 rawPosition.X = _intergratorX.process (readAdc (CMD_READ_X));
216 rawPosition.Y = _intergratorY.process (readAdc (CMD_READ_Y));
217
218 if (_intergratorX.isFull () == false)
219 {
220 vTaskDelay (1);
221 continue;
222 }
223
224 // Read and averages the values from the XPT2046
225
226 _rawPosition.X = std::max<uint32_t> (_parameters.xMin, rawPosition.X);
227 _rawPosition.Y = std::max<uint32_t> (_parameters.yMin, rawPosition.Y);
228
229 _rawPosition.X = std::min<uint32_t> (_rawPosition.X, _parameters.xMax);
230 _rawPosition.Y = std::min<uint32_t> (_rawPosition.Y, _parameters.yMax);
231
232 _rawPosition.X -= _parameters.xMin;
233 _rawPosition.Y -= _parameters.yMin;
234
235 _position.X = static_cast<int32_t> (_rawPosition.X / _devX);
236 _position.Y = static_cast<int32_t> (_rawPosition.Y / _devY);
237
238 // Touch is pressed ad ready read data
239
240 _isPressed = true;
241
242 switch (_rotation)
243 {
244 case 0:
245 position = _position;
246 break;
247 case 1:
248 position.X = _parameters.dispWidth - _position.X;
249 position.Y = _position.Y;
250 break;
251 case 2: // flip X
252 position.X = _parameters.dispWidth - _position.X;
253 position.Y = _parameters.dispHeight - _position.Y;
254 break;
255 case 3: // flip Y
256 position.X = _position.X;
257 position.Y = _parameters.dispHeight - _position.Y;
258 break;
259 case 4: // flip X,Y
260 position.X = _parameters.dispWidth - _position.X;
261 position.Y = _parameters.dispHeight - _position.Y;
262 break;
263 default:
264 assert (0);
265 break;
266 }
267 }
268 else
269 {
270 isPressed = false;
271 _isPressed = false;
272 }
273 vTaskDelay (20);
274 }
275 }
