Lib::System::ProduceDate

Example usage of ProduceDate for OTP production date management.

Demonstrates how to read and write a production date to OTP memory using the ProduceDate class.

Example usage:

#include <Lib/System/ProduceDate.hpp>
#include <Lib/Helper/DateTime.hpp>
int main() {
Lib::System::ProduceDate produceDate(0x10); // Offset for OTP address
Lib::Helper::Date dateToWrite{2024, 6, 1};
if (produceDate.isEmpty()) {
if (produceDate.write(dateToWrite)) {
Lib::Helper::Date readDate;
if (produceDate.read(readDate)) {
// Successfully read back the written date
}
}
}
return 0;
}
