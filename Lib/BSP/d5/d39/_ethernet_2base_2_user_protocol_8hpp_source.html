27 /** User protocol Message Header.
28 * Represents the header of a user protocol message, including delimiter, ID, timestamp, command, option,
29 * and payload length.
30 */
31 __packed struct MessageHeader {
32 uint8_t delimiter; /*!< Start delimiter of frame, must be `0x7e`. */
33 uint16_t id; /*!< Frame ID, incremented for each message. */
34 uint32_t timestamp; /*!< Timestamp, typically systick value. */
35 uint8_t command; /*!< Command identifier. */
36 uint8_t option; /*!< Message option. */
37 uint8_t length; /*!< Payload length in bytes. */
38 };

43 enum class Command : uint8_t {

60 /** Parses the incoming message.
61 * @param data Pointer to incoming message.
62 * @param size Size of incoming message.
63 * @param[out] command Reference to store the parsed command.
64 * @param[out] option Reference to store the parsed option.
65 * @return Error code indicating the result of parsing.
66 * @pre data must not be nullptr; size must be greater than 0.
67 * @pre Message must start with the correct delimiter.
68 * @pre Payload length must be valid for the command.
69 * @post command and option are set if parsing is successful.
70 * @see [Message Format](https://github.com/panchev68-Library/EWARM/blob/2.02/EWARM/Ethernet/doc/EthernetCommunication.md#message-format)
71 */
72 Error parse(const uint8_t * data, size_t size, Command & command, uint8_t & option) {
73 if (!data || size == 0) return Error::EMPTY_MESSAGE;
74
75 size_t offset = 0;
76 while (size - offset >= sizeof(MessageHeader)) {
77 const auto & header = *reinterpret_cast<const MessageHeader *>(data + offset);
78 if (header.delimiter != START_DELIMITER) return Error::INVALID_START_DELIMITER;
79
80 handleIncomingMessageHeader(header);
81 command = static_cast<Command>(header.command);
82 option = header.option;
83
84 size_t payloadLength = header.length;
85 if (size - offset < sizeof(MessageHeader) + payloadLength) return Error::INVALID_PAYLOAD_LENGTH;
86
87 const uint8_t * payloadData = data + offset + sizeof(MessageHeader);
88 size_t nodesCount = 0;
89
90 switch (command) {
91 case Command::ECHO:
92 echoCallback();
93 break;
94 case Command::INPUTS:
95 if (payloadLength % sizeof(uint64_t) != 0) return Error::INVALID_PAYLOAD_LENGTH;
96 nodesCount = payloadLength / sizeof(uint64_t);
97 updateNodeActivityCallback(nodesCount);
98 for (size_t i = 0; i < nodesCount; ++i) {
99 uint64_t value;
100 std::memcpy(&value, payloadData + i * sizeof(uint64_t), sizeof(uint64_t));
101 inputsUpdateCallback(static_cast<uint8_t>(i), value);
102 }
103 break;
104 case Command::OUTPUTS:
105 if (payloadLength % sizeof(uint64_t) != 0) return Error::INVALID_PAYLOAD_LENGTH;
106 nodesCount = payloadLength / sizeof(uint64_t);
107 updateNodeActivityCallback(nodesCount);
108 for (size_t i = 0; i < nodesCount; ++i) {
109 uint64_t value;
110 std::memcpy(&value, payloadData + i * sizeof(uint64_t), sizeof(uint64_t));
111 handleOututsUpdate(static_cast<uint8_t>(i), value);
112 }
113 break;
114 default:
115 return Error::COMMAND_NOT_SUPPORT;
116 }
117
118 offset += sizeof(MessageHeader) + payloadLength;
119 }
120
121 if (offset != size) return Error::INVALID_PAYLOAD_LENGTH;
122 return Error::OK;
123 }

125 /** Builds the outgoing message.
126 * @param command Command to send.
127 * @param option Message option.
128 * @param id Frame ID.
129 * @param payload Pointer to payload data (optional).
130 * @return Vector containing the constructed message bytes.
131 * @pre command must be a valid Command.
132 * @pre option must be a valid option value.
133 * @pre id must be a valid frame ID.
134 * @post Returns a vector containing the message header and payload.
135 */
136 std::vector<uint8_t> build(Command command, uint8_t option, uint16_t id, const std::vector<uint8_t> * payload = nullptr) {
137 size_t payloadSize = payload ? payload->size() : 0;
138 MessageHeader header = {START_DELIMITER, id, (lastSentTimeStamp = HAL_GetTick()), static_cast<uint8_t>(command), option, static_cast<uint8_t>(payloadSize)};
139 std::vector<uint8_t> result(reinterpret_cast<uint8_t *>(&header), reinterpret_cast<uint8_t *>(&header) + sizeof(header));
140 if (payloadSize > 0) result.insert(result.end(), payload->begin(), payload->end());
141 return result;
142 }
