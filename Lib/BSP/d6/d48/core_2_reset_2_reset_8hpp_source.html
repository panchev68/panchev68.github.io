17/** Reset class for STM32 HAL reset management.
18 * Provides methods to query and trigger system resets, encapsulating STM32 HAL reset flags and operations.
19 */
20class Reset {
21public:
22 /** Enumeration of possible reset states.
23 * Represents the various reset sources detected by STM32 hardware.
24 */
25 enum class State : uint8_t {
26 UNKNOWN = 0, /*!< Unknown reset source. */
27 LOW_POWER_RESET, /*!< Low power reset. */
28 WINDOW_WATCHDOG_RESET, /*!< Window watchdog reset. */
29 INDEPENDENT_WATCHDOG_RESET, /*!< Independent watchdog reset. */
30 SOFTWARE_RESET, /*!< Software reset. */
31 POWER_ON_POWER_DOWN_RESET, /*!< Power-on or power-down reset. */
32 EXTERNAL_RESET_PIN_RESET, /*!< External reset pin reset. */
33 BROWNOUT_RESET /*!< Brownout reset. */
34 };

63 /** Reads the current reset state from STM32 HAL flags.
64 * @return Detected reset state.
65 */
66 static State readState() {
67 if (__HAL_RCC_GET_FLAG(RCC_FLAG_LPWRRST)) return State::LOW_POWER_RESET;
68 if (__HAL_RCC_GET_FLAG(RCC_FLAG_WWDGRST)) return State::WINDOW_WATCHDOG_RESET;
69 if (__HAL_RCC_GET_FLAG(RCC_FLAG_IWDGRST)) return State::INDEPENDENT_WATCHDOG_RESET;
70 if (__HAL_RCC_GET_FLAG(RCC_FLAG_SFTRST)) return State::SOFTWARE_RESET;
71 if (__HAL_RCC_GET_FLAG(RCC_FLAG_PORRST)) return State::POWER_ON_POWER_DOWN_RESET;
72 if (__HAL_RCC_GET_FLAG(RCC_FLAG_PINRST)) return State::EXTERNAL_RESET_PIN_RESET;
73 if (__HAL_RCC_GET_FLAG(RCC_FLAG_BORRST)) return State::BROWNOUT_RESET;
74 return State::UNKNOWN;
75 }
76
77 /** Constructor of the Reset class. */
