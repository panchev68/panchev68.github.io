19/** Utility class for microsecond delays using DWT cycle counter.
20 * This class provides static methods to perform busy-wait delays in microseconds
21 * and nanoseconds. It is non-instantiable and uses the ARM DWT for cycle-accurate timing.
22 * The delay is calibrated based on the current HCLK frequency.
23 */
24class Delay {
25public:
26 /** Performs a busy-wait delay for the specified number of microseconds.
27 * @param delay The delay duration in microseconds (must be greater than 0).
28 */
29 static void microseconds(uint32_t delay) {
30 assert(delay > 0 && "Delay must be greater than 0 microseconds");
31
32 const uint32_t hclkFreq = HAL_RCC_GetHCLKFreq();
33 const uint64_t targetTicks = (static_cast<uint64_t>(delay) * hclkFreq) / 1000000ULL;
34 performDelay(targetTicks);
35 }

37 /** Performs a busy-wait delay for the specified number of nanoseconds.

39 */
40 static void nanoseconds(uint32_t delay) {

48private:
49 /** Performs the actual busy-wait delay using DWT cycle counter.
50 * @param targetTicks The number of ticks to wait.
51 */
52 static void performDelay(uint64_t targetTicks) {
53 /* Enable DWT cycle counter */
54 CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
55 DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
56 DWT->CYCCNT = 0U;
57 const uint32_t startTicks = DWT->CYCCNT;
58
59 /* Busy-wait until target ticks are reached */
60 while (static_cast<uint64_t>(DWT->CYCCNT - startTicks) < targetTicks) {
61 }
62 }
