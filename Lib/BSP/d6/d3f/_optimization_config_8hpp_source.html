93/** Should-inline function helper.
94 * Determines if a function of given size should be inlined.
95 * @tparam SIZE Function size in bytes.
96 * @return `true` if the function should be inlined, `false` otherwise.

98template <size_t SIZE> constexpr bool should_inline_function() {

102/** Zero-overhead type check.
103 * Determines if a type is zero-overhead (trivial or pointer-sized).
104 * @tparam T Type to check.
105 * @return `true` if the type is zero-overhead, `false` otherwise.
106 */
107template <typename T> constexpr bool is_zero_overhead() {
108 return sizeof(T) == sizeof(void *) || std::is_trivial<T>::value;
109}

111/** Optimal pool size calculation.
112 * Calculates the optimal pool size for a given element size and count, aligned to cache line size.
113 * @tparam ELEMENT_SIZE Size of each element in bytes.
114 * @tparam COUNT Number of elements.
115 * @return Optimal pool size in bytes.
116 */
117template <size_t ELEMENT_SIZE, size_t COUNT> constexpr size_t optimal_pool_size() {
118 constexpr size_t alignedSize = (ELEMENT_SIZE + CACHE_LINE_SIZE - 1) & ~(CACHE_LINE_SIZE - 1);
119 return alignedSize * COUNT;
120}

122/** Optimal stack size calculation.
123 * Calculates the optimal stack size based on the current optimization level.
124 * @tparam BASE_SIZE Base stack size in bytes.
125 * @return Optimal stack size in bytes.
126 */
127template <size_t BASE_SIZE> constexpr size_t optimal_stack_size() {
128 switch (CURRENT_LEVEL) {
129 case Level::SPACE: return BASE_SIZE;
130 case Level::BALANCED: return BASE_SIZE + (BASE_SIZE / 4);
131 case Level::SPEED: return BASE_SIZE + (BASE_SIZE / 2);
132 case Level::REALTIME: return BASE_SIZE * 2;
133 default: return BASE_SIZE;
134 }
135}

137/** Optimal buffer count calculation.
138 * Calculates the optimal buffer count for a given type and requested count.
139 * @tparam T Buffer element type.
140 * @tparam REQUESTED_COUNT Requested buffer count.
141 * @return Optimal buffer count.
142 */
143template <typename T, size_t REQUESTED_COUNT> constexpr size_t optimal_buffer_count() {
144 constexpr size_t elementSize = sizeof(T);
145 if constexpr (elementSize <= 8) return REQUESTED_COUNT * 2;
146 else if constexpr (elementSize <= 64) return REQUESTED_COUNT;
147 else return (REQUESTED_COUNT + 1) / 2;
148}
