69/** Calculate optimal pool size with alignment.
70 * @tparam N_ELEMENT_SIZE Size of each element.
71 * @tparam N_COUNT Number of elements.
72 * @return Optimal pool size.
73 */
74template <size_t N_ELEMENT_SIZE, size_t N_COUNT>
75constexpr size_t optimalPoolSize() {
76 constexpr size_t alignedSize = (N_ELEMENT_SIZE + CACHE_LINE_SIZE - 1) & ~(CACHE_LINE_SIZE - 1);
77 return alignedSize * N_COUNT;
78}

80/** Calculate optimal stack size based on level.
81 * @tparam N_BASE_SIZE Base stack size.
82 * @return Optimal stack size.
83 */
84template <size_t N_BASE_SIZE>
85constexpr size_t optimalStackSize() {
86 switch (CURRENT_LEVEL) {
87 case Level::SPACE: return N_BASE_SIZE;
88 case Level::BALANCED: return N_BASE_SIZE + (N_BASE_SIZE / 4);
89 case Level::SPEED: return N_BASE_SIZE + (N_BASE_SIZE / 2);
90 case Level::REALTIME: return N_BASE_SIZE * 2;
91 default: return N_BASE_SIZE;
92 }
93}

95/** Calculate optimal buffer count based on type size.
96 * @tparam T_TYPE Buffer element type.
97 * @tparam N_REQUESTED_COUNT Requested count.
98 * @return Optimal buffer count.
99 */
100template <typename T_TYPE, size_t N_REQUESTED_COUNT>
101constexpr size_t optimalBufferCount() {
102 constexpr size_t elementSize = sizeof(T_TYPE);
103 if constexpr (elementSize <= 8) return N_REQUESTED_COUNT * 2;
104 else if constexpr (elementSize <= 64) return N_REQUESTED_COUNT;
105 else return (N_REQUESTED_COUNT + 1) / 2;
106}
