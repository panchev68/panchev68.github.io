27/** Optimization level for RTOS.
28 * Defines the optimization strategy for space, speed, or real-time requirements.
29 */
30enum class Level : uint8_t {
31 SPACE = 0, /*!< Optimize for minimal memory usage. */
32 BALANCED = 1, /*!< Balanced optimization. */
33 SPEED = 2, /*!< Optimize for speed. */
34 REALTIME = 3 /*!< Optimize for real-time constraints. */
35};
36
37/*!< Current optimization level. */
38static constexpr Level CURRENT_LEVEL = Level::SPEED;
39/*!< Enable aggressive inlining if optimization level is SPEED or higher. */
40static constexpr bool AGGRESSIVE_INLINING = (CURRENT_LEVEL >= Level::SPEED);
41/*!< Enable loop unrolling if optimization level is SPEED or higher. */
42static constexpr bool ENABLE_LOOP_UNROLLING = (CURRENT_LEVEL >= Level::SPEED);
43/*!< Enable branch prediction if optimization level is BALANCED or higher. */
44static constexpr bool ENABLE_BRANCH_PREDICTION = (CURRENT_LEVEL >= Level::BALANCED);
45/*!< Enable prefetching if optimization level is SPEED or higher. */
46static constexpr bool ENABLE_PREFETCHING = (CURRENT_LEVEL >= Level::SPEED);
47/*!< Threshold in bytes for function inlining. */
48static constexpr size_t INLINE_THRESHOLD_BYTES = 64;
49/*!< Threshold for small function inlining. */
50static constexpr size_t SMALL_FUNCTION_THRESHOLD = 16;
51/*!< Cache line size in bytes. */
52static constexpr size_t CACHE_LINE_SIZE = 32;

66/** Check if function should be inlined.
67 * @tparam SIZE Size of the function in bytes.
68 * @return `true` if function should be inlined.
69 */
70template <size_t SIZE> constexpr bool should_inline_function() {
71 return SIZE <= INLINE_THRESHOLD_BYTES && AGGRESSIVE_INLINING;

74/** Check if type is zero-overhead.
75 * @tparam T Type to check.
76 * @return `true` if type is trivial or pointer-sized.
77 */
78template <typename T> constexpr bool is_zero_overhead() {
79 return sizeof(T) == sizeof(void *) || std::is_trivial<T>::value;
80}

82/** Calculate optimal pool size.
83 * @tparam ELEMENT_SIZE Size of each element.
84 * @tparam COUNT Number of elements.
85 * @return Optimal pool size in bytes.
86 */
87template <size_t ELEMENT_SIZE, size_t COUNT> constexpr size_t optimal_pool_size() {
88 constexpr size_t alignedSize = (ELEMENT_SIZE + CACHE_LINE_SIZE - 1) & ~(CACHE_LINE_SIZE - 1);
89 return alignedSize * COUNT;
90}

92/** Calculate optimal stack size.
93 * @tparam BASE_SIZE Base stack size.
94 * @return Optimal stack size based on optimization level.
95 */
96template <size_t BASE_SIZE> constexpr size_t optimal_stack_size() {
97 switch (CURRENT_LEVEL) {
98 case Level::SPACE: return BASE_SIZE;
99 case Level::BALANCED: return BASE_SIZE + (BASE_SIZE / 4);
100 case Level::SPEED: return BASE_SIZE + (BASE_SIZE / 2);
101 case Level::REALTIME: return BASE_SIZE * 2;
102 default: return BASE_SIZE;
103 }
104}

106/** Calculate optimal buffer count.
107 * @tparam T Buffer element type.
108 * @tparam REQUESTED_COUNT Requested number of elements.
109 * @return Optimal buffer count.
110 */
111template <typename T, size_t REQUESTED_COUNT> constexpr size_t optimal_buffer_count() {
112 constexpr size_t elementSize = sizeof(T);
113 if constexpr (elementSize <= 8) return REQUESTED_COUNT * 2;
114 else if constexpr (elementSize <= 64) return REQUESTED_COUNT;
115 else return (REQUESTED_COUNT + 1) / 2;
116}
