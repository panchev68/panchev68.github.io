30/** TickCounter utility for FreeRTOS tick/time conversions and elapsed/timeout calculations.
31 * Provides static methods for converting between ticks and milliseconds/seconds, calculating
32 * elapsed time, and checking timeouts. Ensures type safety and overflow protection.
33 */
34class TickCounter {
35public:
36 using TickType = TickType_t;
37
38 /** Get the current tick count.
39 * @return Current tick count.
40 */
41 static TickType get() { return xTaskGetTickCount(); }
42
43 /** Get the current tick count from ISR context.
44 * @return Current tick count.
45 */
46 static TickType getFromISR() { return xTaskGetTickCountFromISR(); }
47
48 /** Convert milliseconds to ticks.
49 * @param ms Milliseconds to convert.
50 * @return Equivalent ticks.
51 * @pre ms must be less than or equal to max representable ticks.
52 */
53 static TickType msToTicks(uint32_t ms) {
54 if (ms == 0) return 0;
55 const uint32_t TICK_PERIOD_MS = getTickPeriodMs();
56 return static_cast<TickType>((ms + TICK_PERIOD_MS - 1) / TICK_PERIOD_MS);
57 }
58
59 /** Convert ticks to milliseconds.
60 * @param ticks Ticks to convert.
61 * @return Equivalent milliseconds.
62 */
63 static uint32_t ticksToMs(TickType ticks) { return static_cast<uint32_t>(ticks) * getTickPeriodMs(); }
64
65 /** Convert seconds to ticks.
66 * @param seconds Seconds to convert.
67 * @return Equivalent ticks.
68 */
69 static TickType secondsToTicks(uint32_t seconds) { return msToTicks(seconds * 1000U); }
70
71 /** Convert ticks to seconds.
72 * @param ticks Ticks to convert.
73 * @return Equivalent seconds.
74 */
75 static uint32_t ticksToSeconds(TickType ticks) { return ticksToMs(ticks) / 1000U; }
76
77 /** Get the tick period in milliseconds.
78 * @return Tick period in milliseconds.
79 */
80 static constexpr uint32_t getTickPeriodMs() { return static_cast<uint32_t>(1000U / configTICK_RATE_HZ); }
81
82 /** Get the tick rate in Hertz.
83 * @return Tick rate in Hertz.
84 */
85 static constexpr uint32_t getTickRateHz() { return static_cast<uint32_t>(configTICK_RATE_HZ); }
86
87 /** Get the maximum delay in ticks.
88 * @return Maximum delay in ticks.
89 */
90 static constexpr TickType getMaxDelay() { return portMAX_DELAY; }
91
92 /** Calculate elapsed ticks between two tick counts (handles overflow).
93 * @param startTicks Start tick count.
94 * @param endTicks End tick count.
95 * @return Elapsed ticks.
96 */
97 static TickType getElapsedTicks(TickType startTicks, TickType endTicks) {
98 return (endTicks >= startTicks) ? (endTicks - startTicks)
99 : (std::numeric_limits<TickType>::max() - startTicks + endTicks + 1);
100 }
101
102 /** Calculate elapsed milliseconds between two tick counts.
103 * @param startTicks Start tick count.
104 * @param endTicks End tick count.
105 * @return Elapsed milliseconds.
106 */
107 static uint32_t getElapsedMs(TickType startTicks, TickType endTicks) {
108 return ticksToMs(getElapsedTicks(startTicks, endTicks));
109 }
110
111 /** Calculate elapsed milliseconds since a start tick count.
112 * @param startTicks Start tick count.
113 * @return Elapsed milliseconds.
114 */
115 static uint32_t getElapsedMsSince(TickType startTicks) { return getElapsedMs(startTicks, get()); }
116
117 /** Calculate elapsed ticks since a start tick count.
118 * @param startTicks Start tick count.
119 * @return Elapsed ticks.
120 */
121 static TickType getElapsedTicksSince(TickType startTicks) { return getElapsedTicks(startTicks, get()); }
122
123 /** Check if a timeout (in milliseconds) has occurred since start tick.
124 * @param startTicks Start tick count.
125 * @param timeoutMs Timeout in milliseconds.
126 * @return `true` if timed out, `false` otherwise.
127 */
128 static bool hasTimedOut(TickType startTicks, uint32_t timeoutMs) {
129 return getElapsedMsSince(startTicks) >= timeoutMs;
130 }
131
132 /** Check if a timeout (in ticks) has occurred since start tick.
133 * @param startTicks Start tick count.
134 * @param timeoutTicks Timeout in ticks.
135 * @return `true` if timed out, `false` otherwise.
136 */
137 static bool hasTimedOutTicks(TickType startTicks, TickType timeoutTicks) {
138 return getElapsedTicksSince(startTicks) >= timeoutTicks;
139 }
140
141 /** Get remaining time in milliseconds before timeout.
142 * @param startTicks Start tick count.
143 * @param timeoutMs Timeout in milliseconds.
144 * @return Remaining milliseconds before timeout.
145 */
146 static uint32_t getRemainingTimeMs(TickType startTicks, uint32_t timeoutMs) {
147 uint32_t elapsed = getElapsedMsSince(startTicks);
148 return (elapsed >= timeoutMs) ? 0U : (timeoutMs - elapsed);
149 }
150
151 /** Get remaining ticks before timeout.
152 * @param startTicks Start tick count.
153 * @param timeoutTicks Timeout in ticks.
154 * @return Remaining ticks before timeout.
155 */
156 static TickType getRemainingTicks(TickType startTicks, TickType timeoutTicks) {
157 TickType elapsed = getElapsedTicksSince(startTicks);
158 return (elapsed >= timeoutTicks) ? 0 : (timeoutTicks - elapsed);
159 }
160
161private:
162 /** Constructor of the TickCounter class. */
163 TickCounter() = delete;
164
165 /** Copy constructor of the TickCounter class. */
166 TickCounter(const TickCounter &) = delete;
167
168 /** Assignment operator of the TickCounter class.
169 * @param other Other TickCounter instance.
170 * @return Reference to this instance.
171 */
172 TickCounter & operator=(const TickCounter &) = delete;
173
174 /** Destructor of the TickCounter class. */
175 ~TickCounter() = delete;
176};
