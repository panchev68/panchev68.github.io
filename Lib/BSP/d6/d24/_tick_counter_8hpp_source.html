29/** Tick counter utility class.
30 * Provides static methods for tick counting, time conversion, and elapsed/remaining time
31 * calculations using FreeRTOS tick counters.
32 */
33class TickCounter {
34public:
35 using TickType = TickType_t; /*!< Alias for FreeRTOS tick type. */
36
37 /** Get current tick count.
38 * @return Current tick count.
39 */
40 static TickType get() { return xTaskGetTickCount(); }
41
42 /** Get current tick count from ISR context.
43 * @return Current tick count (from ISR).
44 */
45 static TickType getFromISR() { return xTaskGetTickCountFromISR(); }
46
47 /** Convert milliseconds to ticks.
48 * @param ms Milliseconds to convert.
49 * @return Number of ticks corresponding to the given milliseconds.
50 */
51 static TickType msToTicks(uint32_t ms) {
52 if (ms == 0) return 0;
53 const uint32_t TICK_PERIOD_MS = getTickPeriodMs();
54 // Използвай pdMS_TO_TICKS ако е дефиниран, иначе ръчно изчисление
55#ifdef pdMS_TO_TICKS
56 return static_cast<TickType>(pdMS_TO_TICKS(ms));
57#else
58 return static_cast<TickType>((ms + TICK_PERIOD_MS - 1) / TICK_PERIOD_MS);
59#endif
60 }
61
62 /** Convert ticks to milliseconds.
63 * @param ticks Number of ticks.
64 * @return Milliseconds corresponding to the given ticks.
65 */
66 static uint32_t ticksToMs(TickType ticks) {
67 const uint32_t TICK_PERIOD_MS = getTickPeriodMs();
68 return static_cast<uint32_t>(ticks) * TICK_PERIOD_MS;
69 }
70
71 /** Convert seconds to ticks.
72 * @param seconds Seconds to convert.
73 * @return Number of ticks corresponding to the given seconds.
74 */
75 static TickType secondsToTicks(uint32_t seconds) {
76 if (seconds == 0) return 0;
77 constexpr uint32_t MS_PER_SEC = 1000U;
78 // Проверка за препълване
79 if (seconds > (std::numeric_limits<uint32_t>::max() / MS_PER_SEC))
80 return getMaxDelay();
81 return msToTicks(seconds * MS_PER_SEC);
82 }
83
84 /** Convert ticks to seconds.
85 * @param ticks Number of ticks.
86 * @return Seconds corresponding to the given ticks.
87 */
88 static uint32_t ticksToSeconds(TickType ticks) {
89 constexpr uint32_t MS_PER_SEC = 1000U;
90 return ticksToMs(ticks) / MS_PER_SEC;
91 }
92
93 /** Get tick period in milliseconds.
94 * @return Tick period in milliseconds.
95 */
96 static constexpr uint32_t getTickPeriodMs() {
97 static_assert(configTICK_RATE_HZ != 0, "configTICK_RATE_HZ must not be zero");
98 return static_cast<uint32_t>(1000U / configTICK_RATE_HZ);
99 }
100
101 /** Get tick rate in Hz.
102 * @return Tick rate in Hz.
103 */
104 static constexpr uint32_t getTickRateHz() { return static_cast<uint32_t>(configTICK_RATE_HZ); }
105
106 /** Get maximum delay in ticks.
107 * @return Maximum delay in ticks.
108 */
109 static constexpr TickType getMaxDelay() { return portMAX_DELAY; }
110
111 /** Get elapsed ticks between two tick counts.
112 * @param startTicks Start tick count.
113 * @param endTicks End tick count.
114 * @return Elapsed ticks between start and end, handling overflow.
115 */
116 static TickType getElapsedTicks(TickType startTicks, TickType endTicks) {
117 // FreeRTOS tick counter overflow-safe разлика
118 return (endTicks >= startTicks)
119 ? (endTicks - startTicks)
120 : (static_cast<TickType>(std::numeric_limits<TickType>::max() - startTicks + endTicks + 1));
121 }
122
123 /** Get elapsed milliseconds between two tick counts.
124 * @param startTicks Start tick count.
125 * @param endTicks End tick count.
126 * @return Elapsed milliseconds between start and end.
127 */
128 static uint32_t getElapsedMs(TickType startTicks, TickType endTicks) {
129 return ticksToMs(getElapsedTicks(startTicks, endTicks));
130 }
131
132 /** Get elapsed milliseconds since a given tick count.
133 * @param startTicks Start tick count.
134 * @return Elapsed milliseconds since start.
135 */
136 static uint32_t getElapsedMsSince(TickType startTicks) {
137 return getElapsedMs(startTicks, get());
138 }
139
140 /** Get elapsed ticks since a given tick count.
141 * @param startTicks Start tick count.
142 * @return Elapsed ticks since start.
143 */
144 static TickType getElapsedTicksSince(TickType startTicks) {
145 return getElapsedTicks(startTicks, get());
146 }
147
148 /** Check if a timeout (in milliseconds) has occurred since a given tick count.
149 * @param startTicks Start tick count.
150 * @param timeoutMs Timeout in milliseconds.
151 * @return `true` if timeout has occurred, `false` otherwise.
152 */
153 static bool hasTimedOut(TickType startTicks, uint32_t timeoutMs) {
154 if (timeoutMs == 0) return true;
155 return getElapsedMsSince(startTicks) >= timeoutMs;
156 }
157
158 /** Check if a timeout (in ticks) has occurred since a given tick count.
159 * @param startTicks Start tick count.
160 * @param timeoutTicks Timeout in ticks.
161 * @return `true` if timeout has occurred, `false` otherwise.
162 */
163 static bool hasTimedOutTicks(TickType startTicks, TickType timeoutTicks) {
164 if (timeoutTicks == 0) return true;
165 return getElapsedTicksSince(startTicks) >= timeoutTicks;
166 }
167
168 /** Get remaining time in milliseconds before timeout.
169 * @param startTicks Start tick count.
170 * @param timeoutMs Timeout in milliseconds.
171 * @return Remaining milliseconds before timeout, or 0 if already timed out.
172 */
173 static uint32_t getRemainingTimeMs(TickType startTicks, uint32_t timeoutMs) {
174 if (timeoutMs == 0) return 0U;
175 const uint32_t elapsed = getElapsedMsSince(startTicks);
176 return (elapsed >= timeoutMs) ? 0U : (timeoutMs - elapsed);
177 }
178
179 /** Get remaining ticks before timeout.
180 * @param startTicks Start tick count.
181 * @param timeoutTicks Timeout in ticks.
182 * @return Remaining ticks before timeout, or 0 if already timed out.
183 */
184 static TickType getRemainingTicks(TickType startTicks, TickType timeoutTicks) {
185 if (timeoutTicks == 0) return 0;
186 const TickType elapsed = getElapsedTicksSince(startTicks);
187 return (elapsed >= timeoutTicks) ? 0 : (timeoutTicks - elapsed);
188 }

201 ~TickCounter() = delete;
