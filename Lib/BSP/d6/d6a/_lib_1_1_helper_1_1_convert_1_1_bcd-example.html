Lib::Helper::Convert::Bcd

Example usage of Bcd conversion utilities.

Demonstrates converting BCD values to ASCII strings, ASCII characters, and hexadecimal integers.

Example usage:

#include "Lib/Helper/Convert/Bcd.hpp"
#include <iostream>
int main() {
uint16_t bcdValue = 0x1234;
std::string asciiStr = Lib::Helper::Convert::Bcd::toString(bcdValue);
char asciiChar = Lib::Helper::Convert::Bcd::toAsc(0x05);
uint16_t hexValue = Lib::Helper::Convert::Bcd::toHex(bcdValue);
std::cout << "BCD value: 0x" << std::hex << bcdValue << std::dec << std::endl;
std::cout << "ASCII string: " << asciiStr << std::endl;
std::cout << "ASCII char: " << asciiChar << std::endl;
std::cout << "Hex value: " << hexValue << std::endl;
return 0;
}
