118 virtual void transmitBlock (const void * src, uint32_t x, uint32_t y, size_t w, size_t h) override
119 {

121 auto setWindow = [this, &x, &y, &w, &h] () -> size_t {
122 auto setPosReg = [this] (Registers reg, uint16_t start, uint16_t end) -> void {
123 _impl.writeReg (reg);
124 uint8_t data [4] = {HI_BYTE (start), LO_BYTE (start), HI_BYTE (end), LO_BYTE (end)};
125 _impl.writeData (data, sizeof (data));
126 };
127 y += 20; // todo: for 1.69" Color 280x240 Round Rectangle IPS TFT LCD Display Panel ST7789
128 setPosReg (Registers::CASET, x, x + w - 1); // 0x2A
129 setPosReg (Registers::PASET, y, y + h - 1); // 0x2B
130 _impl.writeReg (Registers::RAMWR);
