static uint32_t counter = 0;
static FreeRTOS::Timeout timeout;
if (_isStarted)

if (timeout.check ())

timeout.set (_updatePeriod);
uint64_t value;
switch (_mode)

counter++;
counter %= 24;

case 1:
value = outputsTest1 [counter];
counter++;
counter %= outputsTest1.size ();
break;
case 2:
value = outputsTest2 [counter];
counter++;
counter %= outputsTest2.size ();
break;
}
DAL::peripheral.outputs = value;
}
}
