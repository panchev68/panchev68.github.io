29/** System class encapsulates system configuration and state.
30 * Manages reset state, production and sale dates, configuration, serial number, and version
31 * information for embedded applications. Ensures type safety, input validation, and reliable
32 * error handling.
33 */
34class System {
35public:
36 using LastResetState = Lib::HAL::LastResetState;
37
38 LastResetState lastResetState; /*!< State of the last system reset. */
39 ProduceDate produceDate; /*!< Production date of the system. */
40 ProduceDate saleDate; /*!< Sale date of the system. */
41 Configuration configuration; /*!< System configuration settings. */
42 SerialNumber serialNumber; /*!< Serial number of the system. */
43 FirmwareVersion firmwareVersion; /*!< Firmware version of the system. */
44 HardwareVersion hardwareVersion; /*!< Hardware version of the system. */
45
46 /** Constructor of the System class.
47 * @param fwVersionString Firmware version string.
48 * @param fwSubversion Firmware subversion.
49 * @param hwVersionString Hardware version string.
50 * @param hwSubversion Hardware subversion.
51 */
52 System(const std::string & fwVersionString, uint32_t fwSubversion,
53 const std::string & hwVersionString, uint32_t hwSubversion)
54 : lastResetState(),
55 produceDate(offsetof(DataMap, produceDate)),
56 saleDate(offsetof(DataMap, saleDate)),
57 configuration(offsetof(DataMap, configuration)),
58 firmwareVersion(fwVersionString, fwSubversion),
59 hardwareVersion(hwVersionString, hwSubversion) {}
60
61 /** Destructor of the System class. */
62 ~System() = default;
63
64 /** Restarts the system by triggering a system reset. */
65 void restart() { NVIC_SystemReset(); }
66
67private:
68 /** DataMap structure for product dates and configuration.
69 * Maps production date, sale date, and configuration settings to memory offsets for OTP.

71 struct DataMap {
72 uint32_t produceDate; /*!< Production date. */
73 uint32_t saleDate; /*!< Sale date. */
74 uint32_t configuration; /*!< Configuration settings. */
75 };
