26 /** System class encapsulates system configuration and state.
27 * Manages reset state, production/sale dates, configuration, serial number, and version info for
28 * embedded applications. Ensures type safety, input validation, and reliable error handling.
29 */
30 class System {
31 public:
32 using LastResetState = Lib::HAL::LastResetState;
33
34 LastResetState lastResetState; /*!< State of the last system reset. */
35 ProduceDate produceDate; /*!< Production date of the system. */
36 ProduceDate saleDate; /*!< Sale date of the system. */
37 Configuration configuration; /*!< System configuration settings. */
38 SerialNumber serialNumber; /*!< Serial number of the system. */
39 FirmwareVersion firmwareVersion; /*!< Firmware version of the system. */
40 HardwareVersion hardwareVersion; /*!< Hardware version of the system. */
41
42 /** Constructor of System class. Initializes firmware and hardware versions.
43 * @param fwVersionString Firmware version string.
44 * @param fwSubversion Firmware subversion.
45 * @param hwVersionString Hardware version string.
46 * @param hwSubversion Hardware subversion.
47 */
48 System(const std::string & fwVersionString, uint32_t fwSubversion,
49 const std::string & hwVersionString, uint32_t hwSubversion)
50 : lastResetState(),
51 produceDate(offsetof(DataMap, produceDate)),
52 saleDate(offsetof(DataMap, saleDate)),
53 configuration(offsetof(DataMap, configuration)),
54 firmwareVersion(fwVersionString, fwSubversion),
55 hardwareVersion(hwVersionString, hwSubversion) {}
56
57 /** Destructor of System class. */
58 ~System() = default;
59
60 /** Restarts the system by triggering a system reset. */
61 void restart() { NVIC_SystemReset(); }
62
63 private:
64 /** DataMap structure for product dates and configuration.
65 * Maps production date, sale date, and configuration settings to memory offsets for OTP.
66 */
67 struct DataMap {
68 uint32_t produceDate; /*!< Production date. */
69 uint32_t saleDate; /*!< Sale date. */
70 uint32_t configuration; /*!< Configuration settings. */
71 };
72 };
