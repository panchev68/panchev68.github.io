23 /** Converts a string to a long integer.
24 * @param str Input string to convert.
25 * @return Converted long integer value.
26 */
27 constexpr long atol(const char * str) {
28 long result{0};
29 long sign{(*str == '-') ? -1 : 1};
30 if (sign == -1) ++str;
31 while (*str && std::isdigit(*str)) result = result * 10 + (*str++ - '0');
32 return sign * result;
33 }

35 /** Converts a string to a double-precision floating-point value.
36 * @param str Input string to convert.
37 * @param[out] end Pointer to the character after the last parsed character.
38 * @return Converted double value.
39 */
40 double strtod(const char * str, char ** end) {
41 constexpr int MAX_EXPONENT = 511;
42 constexpr std::array<double, 9> POWERS_OF_10{{10., 100., 1.0e4, 1.0e8, 1.0e16, 1.0e32, 1.0e64, 1.0e128, 1.0e256}};
43 const char * s{str};
44 while (std::isspace(*s)) ++s;
45 bool isNegative{*s == '-'};
46 if (isNegative || *s == '+') ++s;
47 int mantissa{0}, decimal{-1}, fracExp{0}, exp{0};
48 double frac{0.0}, dExp{1.0};
49 const char * expPtr;
50 for (; std::isdigit(*s) || (*s == '.' && decimal < 0); ++mantissa)
51 if (*s++ == '.') decimal = mantissa;
52 expPtr = s -= mantissa;
53 if (decimal < 0) decimal = mantissa;
54 else {
55 --mantissa;
56 fracExp = decimal - mantissa;
57 }
58 if (mantissa > 18) {
59 fracExp += mantissa - 18;
60 mantissa = 18;
61 }
62 if (mantissa > 0) {
63 long long f1{0}, f2{0};
64 for (; mantissa > 9; --mantissa) f1 = 10 * f1 + (*s++ - '0');
65 for (; mantissa > 0; --mantissa) f2 = 10 * f2 + (*s++ - '0');
66 frac = (1.0e9 * f1) + f2;
67 }
68 if (*expPtr == 'E' || *expPtr == 'e') {
69 s = ++expPtr;
70 bool expNegative{*s == '-'};
71 if (expNegative || *s == '+') ++s;
72 while (std::isdigit(*s)) exp = exp * 10 + (*s++ - '0');
73 exp = expNegative ? fracExp - exp : fracExp + exp;
74 }
75 bool expNegative{exp < 0};
76 if (expNegative) exp = -exp;
77 if (exp > MAX_EXPONENT) {
78 exp = MAX_EXPONENT;
79 errno = ERANGE;
80 }
81 auto p = POWERS_OF_10.begin();
82 for (int e = exp; e; e >>= 1, ++p)
83 if (e & 1) dExp *= *p;
84 frac = expNegative ? frac / dExp : frac * dExp;
85 if (end) *end = const_cast<char *>(s);
86 return isNegative ? -frac : frac;
87 }

89 /** Converts an integer to a string.
90 * @param value Integer value to convert.
91 * @param buffer Output string buffer.
92 * @param size Length of the buffer.
93 * @param base Numerical base for conversion (e.g., 10 for decimal, 16 for hexadecimal).
94 * @return Pointer to the resulting string, or `nullptr` if the buffer is too small.
95 */
96 inline char * itoa(int value, char * buffer, size_t size, int base) {
97 if (size < 2 || base < 2 || base > 16) return nullptr;
98 char * ptr{buffer + size - 1};
99 *ptr = '\0';
100 bool isNegative{value < 0};
101 unsigned int absValue{static_cast<unsigned int>(isNegative ? -value : value)};
102 do {
103 *--ptr = "0123456789abcdef"[absValue % base];
104 absValue /= base;
105 } while (absValue && ptr > buffer);
106 if (isNegative && ptr > buffer) *--ptr = '-';
107 return ptr;
108 }
