96 * @pre buffer must be valid and large enough to hold the result; size > 1; base between 2 and 16.

98inline char * itoa(int value, char * buffer, size_t size, int base) {

112/** Truncates a value to the specified range.
113 * Returns the value limited to the range [`minValue`, `maxValue`].
114 * @param value Input value to be limited.
115 * @param minValue Minimum allowed value.
116 * @param maxValue Maximum allowed value.
117 * @return Limited value within the specified range.
118 * @pre minValue must be less than or equal to maxValue.
119 * @post Returns `minValue` if `value` is less than `minValue`, `maxValue` if greater than `maxValue`, otherwise `value`.
120 */
121template <typename T> inline T trip(const T value, const T minValue, const T maxValue) {
122 assert(minValue <= maxValue);
123 if (value < minValue) return minValue;
124 if (value > maxValue) return maxValue;
125 return value;
126}
