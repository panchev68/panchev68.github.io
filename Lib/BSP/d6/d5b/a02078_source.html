225 template <EventBit T_BIT> bool waitForEvent(uint32_t timeout) {
226 if (timeout == 0)
227 return true;
228 const auto bits = eventGroup.waitForBits(T_BIT, true, false, timeout).to_ulong();
229 assert(bits & T_BIT);
230 /* Return true if the specified event bit was set. */
231 return bits & T_BIT;
232 }

237 void eventHandler(CallbackId callbackId) {
238 switch (callbackId) {
239 case CallbackId::RX_COMPLETE:
240 /* Set RX completed event bit from ISR. */
241 if (eventGroup.setBitsFromISR(RX_COMPLETED) == pdFAIL)
242 assert(false);
243 handleRxCompleteEvent();
244 break;
245 case CallbackId::TX_COMPLETE:
246 /* Set TX completed event bit from ISR. */
247 if (eventGroup.setBitsFromISR(TX_COMPLETED) == pdFAIL)
248 assert(false);
249 handleTxCompleteEvent();
250 break;
251 case CallbackId::TX_RX_COMPLETE:
252 /* Set TX/RX completed event bit from ISR. */
253 if (eventGroup.setBitsFromISR(TX_RX_COMPLETED) == pdFAIL)
254 assert(false);
255 handleTxRxCompleteEvent();
256 break;
257 case CallbackId::ABORT:
258 /* Set abort completed event bit from ISR. */
259 if (eventGroup.setBitsFromISR(ABORT_COMPLETED) == pdFAIL)
260 assert(false);
261 handleAbortCompleteEvent();
262 break;
263 case CallbackId::ERROR:
264 /* Set error event bit from ISR and handle error. */
265 if (eventGroup.setBitsFromISR(ERROR) == pdFAIL)
266 assert(false);
267 handleErrorEvent(static_cast<ErrorCode>(HAL_SPI_GetError(handle.getHandle())));
268 break;
269 default:
270 assert(false);
271 break;
272 }
