162 /** Starts the task with the given function.
163 * @param func Function to execute in the task.
164 * @return Task handle.
165 */
166 Handle start(std::function<void()> func) {
167 if (!func || data.handle) return data.handle;
168 auto callable = std::make_unique<std::function<void()>>(std::move(func));
169 auto taskEntry = [](void * param) {
170 std::unique_ptr<std::function<void()>> callable(static_cast<std::function<void()> *>(param));
171 if (callable) (*callable)();
172 vTaskDelete(nullptr);
173 };
174 BaseType_t result = xTaskCreate(
175 taskEntry,
176 data.name.empty() ? "Task" : data.name.c_str(),
177 data.stackSize / sizeof(StackType_t),
178 callable.get(),
179 data.priority,
180 &data.handle);
181 if (result == pdPASS) callable.release();
182 else data.handle = nullptr;
183 return data.handle;
184 }

391 * @param timeout Timeout in ticks.
392 * @return Pair of result and notification bits.

394 static std::pair<bool, NotificationBits> waitForNotificationBits(
395 NotificationBits bitsToWaitFor,
396 bool clearOnExit = true,
397 bool waitForAllBits = false,
398 TickType_t timeout = portMAX_DELAY) {
399 uint32_t notificationValue = 0;
400 const uint32_t mask = static_cast<uint32_t>(bitsToWaitFor.to_ulong());
401 const uint32_t exitClearMask = clearOnExit ? mask : 0;
402 bool result = xTaskNotifyWait(0, exitClearMask, &notificationValue, timeout) == pdPASS;
403 if (result) {
404 const uint32_t matchedBits = notificationValue & mask;
405 result = waitForAllBits ? matchedBits == mask : matchedBits != 0;
406 }
407 return {result, NotificationBits(notificationValue)};
