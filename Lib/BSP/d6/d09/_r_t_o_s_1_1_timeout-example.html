RTOS::Timeout

Example usage of Timeout class.

Demonstrates how to create, set, extend, and check timeouts using the RTOS::Timeout class.

Example usage:
#include "Timeout.hpp"
#include <cstdio>
int main() {
RTOS::Timeout timeout = RTOS::Timeout::fromMilliseconds(500);
timeout.setMilliseconds(1000);
timeout.extendMs(500);
if (timeout.isExpired()) {
// Timeout expired
printf("Timeout expired!\n");
} else {
// Timeout still active
printf("Timeout active, remaining: %u ms\n", timeout.getRemainingMs());
}
return 0;
}
