Lib::Helper::JaggedArray

Example usage of JaggedArray for managing jagged arrays of integers.

Demonstrates how to create a jagged array, add rows and elements, and access values.

Example usage:

#include "Lib/Helper/JaggedArray.hpp"
#include <iostream>
int main() {
using namespace Lib::Helper;
JaggedArray<int> jaggedArray;
// Create rows
auto * row1 = new JaggedArrayRow<int>();
auto * row2 = new JaggedArrayRow<int>();
// Add elements to rows
int a = 10, b = 20, c = 30, d = 40;
row1->addItem(&a);
row1->addItem(&b);
row2->addItem(&c);
row2->addItem(&d);
// Add rows to jagged array
jaggedArray.addItem(row1);
jaggedArray.addItem(row2);
// Access and print elements
for (size_t i = 0; i < jaggedArray.getSize(); ++i) {
auto & row = jaggedArray[i];
for (size_t j = 0; j < row.getSize(); ++j) {
std::cout << row[j] << " ";
}
std::cout << std::endl;
}
// Clean up
delete row1;
delete row2;
return 0;
}
