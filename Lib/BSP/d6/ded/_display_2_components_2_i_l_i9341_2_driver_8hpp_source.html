69 16, Registers::GMCTRN1, 0x00, 0x00, 0x10, 0x03, 0x0f, 0x05, 0x2c, 0xa2, 0x3f, 0x05, 0x0e, 0x0c, 0x37, 0x3c, 0x0F,

73 constexpr uint8_t BACKLIGHT_OFF = 0;
74 constexpr uint8_t BACKLIGHT_ON = 100;
75 constexpr uint32_t DELAY_SHORT = 100;
76 constexpr uint32_t DELAY_LONG = 120;
77 constexpr uint32_t DELAY_FINAL = 150;
78 csPin.write(1);
79 spiPort.init();
80 setBacklight(BACKLIGHT_OFF);
81 vTaskDelay(DELAY_SHORT);
82 resetDevice();
83 csPin.write(0);
84 for (const uint8_t * seq = INIT_SEQUENCE; *seq; seq += seq[0] + 1) {
85 sendCommand(seq[1]);
86 for (uint8_t i = 0; i < seq[0] - 1; ++i) sendData(seq[2 + i]);
87 }
88 vTaskDelay(DELAY_LONG);
89 sendCommand(Registers::DISPON);
90 vTaskDelay(DELAY_FINAL);
91 setBacklight(BACKLIGHT_ON);
92 return true;
93 }

95 /** Transmits a rectangular block of pixel data to the display.
96 * @param buffer Pointer to pixel data buffer.
97 * @param x X-coordinate of block start.
98 * @param y Y-coordinate of block start.
99 * @param width Width of the block.
100 * @param height Height of the block.
101 */
102 void transmitBlock(const void * buffer, uint32_t x, uint32_t y, size_t width, size_t height) override {
103 assert(buffer);
104 assert(width);
105 assert(height);
106 assert(width * height <= 0xffff);
107 const auto * pixelData = static_cast<const uint16_t *>(buffer);
108 auto spi = spiPort.getHandle();
109 if (spi->Init.DataSize != SPI_DATASIZE_8BIT) {
110 spi->Init.DataSize = SPI_DATASIZE_8BIT;
111 HAL_SPI_Init(spi);
112 }
113 setAddressWindow(x, y, width, height);
114 sendCommand(Registers::RAMWR);
115 if (spi->Init.DataSize != SPI_DATASIZE_16BIT) {
116 spi->Init.DataSize = SPI_DATASIZE_16BIT;
117 HAL_SPI_Init(spi);
118 }
119 setTransferActive(1);
120 spiPort.sendBlock(pixelData, width * height);
121 }

136 /** Sends a block of pixel data.
137 * @param buffer Pointer to pixel data.
138 * @param size Number of pixels.
139 * @return `true` if transmission started, `false` otherwise.

141 bool sendBlock(const uint16_t * buffer, size_t size) {
142 assert(buffer);
143 assert(size);

145 }

147 /** Sends a single byte via SPI.
148 * @param value Byte value to send.
149 * @return `true` if sent, `false` otherwise.
150 */
151 bool sendByte(uint8_t value) {
152 bool result = (HAL_SPI_Transmit(getHandle(), &value, sizeof(uint8_t), 10) == HAL_OK);
153 assert(result);
154 return result;
155 }

167
168 /** Sends data buffer using DMA via SPI.
169 * @param buffer Pointer to data buffer.
170 * @param bufferSize Size of buffer in bytes.
171 * @return `true` if DMA transmission started, `false` otherwise.
172 * @note Assertion fails if DMA transmission could not be started.
173 */
174 bool sendData(const void * buffer, size_t bufferSize) override {
175 bool result = HAL_SPI_Transmit_DMA(getHandle(), static_cast<uint8_t *>(const_cast<void *>(buffer)), bufferSize) == HAL_OK;
176 assert(result);
177 return result;
178 }

186 /** Sets the address window for pixel data transfer.
187 * @param x X-coordinate of window start.
188 * @param y Y-coordinate of window start.
189 * @param width Width of the window.
190 * @param height Height of the window.
191 */
192 void setAddressWindow(uint32_t x, uint32_t y, size_t width, size_t height) {
193 setPosition(Registers::CASET, x, x + width - 1);
194 setPosition(Registers::PASET, y, y + height - 1);
195 }
196
197 /** Sends position command and data.
198 * @param reg Register address.
199 * @param start Start value.
200 * @param end End value.
201 */
202 void setPosition(uint8_t reg, uint32_t start, uint32_t end) {
203 uint8_t d[4] = {HI_BYTE(start), LO_BYTE(start), HI_BYTE(end), LO_BYTE(end)};
204 sendCommand(reg);
205 for (uint8_t b : d) spiPort.sendByte(b);
206 }

208 /** Sets display rotation mode.
209 * @param rotate `true` for rotated orientation, `false` for default.
210 */
211 void setRotation(bool rotate) {
212 sendCommand(Registers::MADCTL);
213 sendData(rotate ? 0b10111000 : 0b01111000);
214 }
