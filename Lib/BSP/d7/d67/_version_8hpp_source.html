32 /** Converts time and date strings to DateTime.
33 * @param timeString Time string (e.g., __TIME__).
34 * @param dateString Date string (e.g., __DATE__).
35 * @return DateTime object representing the parsed time and date.
36 */
37 static Lib::Helper::DateTime stringToDateTime(const char * timeString, const char * dateString) {
38 assert(timeString);
39 assert(dateString);
40 auto getDigit = [](const char * str, size_t offset) -> uint8_t {
41 assert(str);
42 return static_cast<uint8_t>(str[offset] - '0');
43 };
44 Lib::Helper::DateTime dateTime;
45 dateTime.time.hours = 10 * getDigit(timeString, 0) + getDigit(timeString, 1);
46 dateTime.time.minutes = 10 * getDigit(timeString, 3) + getDigit(timeString, 4);
47 dateTime.time.seconds = 10 * getDigit(timeString, 6) + getDigit(timeString, 7);
48 dateTime.date.year = 1000 * getDigit(dateString, 7) + 100 * getDigit(dateString, 8) +
49 10 * getDigit(dateString, 9) + getDigit(dateString, 10);
50 dateTime.date.month = parseMonth(dateString);
51 dateTime.date.day = 10 * (dateString[4] == ' ' ? 0 : getDigit(dateString, 4)) + getDigit(dateString, 5);
52 return dateTime;
53 }

55 /** Converts version string to uint32_t.
56 * @param versionString Version string in the format "X.Y".
57 * @return 32-bit unsigned integer representing the version.
58 */
59 static uint32_t stringToU32(const std::string & versionString) {
60 std::array<uint8_t, 2> versionParts{0, 0};
61 size_t begin = 0;
62 for (size_t i = 0; i < versionParts.size(); ++i) {
63 size_t found = versionString.find('.', begin);
64 if (found == std::string::npos) found = versionString.size();
65 versionParts[i] = static_cast<uint8_t>(std::stoul(versionString.substr(begin, found - begin)));
66 begin = found + 1;
67 if (found == versionString.size()) break;
68 }
69 return (static_cast<uint32_t>(versionParts[0]) << 8) | static_cast<uint32_t>(versionParts[1]);
70 }

103 static uint8_t parseMonth(const char * dateString) {
104 switch (dateString[2]) {
105 case 'b': return 2;
106 case 'y': return 5;
107 case 'l': return 7;
108 case 'g': return 8;
109 case 'p': return 9;
110 case 't': return 10;
111 case 'v': return 11;
112 case 'c': return 12;
113 case 'n': return dateString[1] == 'a' ? 1 : 6;
114 default: return dateString[1] == 'a' ? 3 : 4;
115 }
