235 /* Write data to the EEPROM 8-bit address */
236 inline bool DriverBase::write8 (uint32_t address, const void * buffer, size_t bufferSize) {
237 if (_isReady == false) {
238 assert (0);
239 return false;
240 }
241
242 if ((address + bufferSize) > _chipSize) {
243 assert (0 && "Address is out of range");
244 return false;
245 }
246
247 uint8_t * ptr = reinterpret_cast<uint8_t *> (const_cast<void *> (buffer));
248 while (bufferSize > 0) {
249 auto pageOffset = address % _pageSize;
250 size_t wlen = std::min<size_t> (bufferSize, (_pageSize - pageOffset));
251
252 if (_i2c->send (_deviceAddress, I2C_MEMADD_SIZE_8BIT, address, ptr, wlen, I2C_TIMEOUT) == false) {
253 assert (0);
254 return false;
255 }
256 vTaskDelay (5);
257
258 address += wlen;
259 ptr += wlen;
260 bufferSize -= wlen;
261 }
262 return true;
263 }

266 inline bool DriverBase::write16 (uint32_t address, const void * buffer, size_t bufferSize) {
267 // Check if the device is ready

279 while (bufferSize > 0) {

281 size_t wlen = std::min<size_t> (bufferSize, (_pageSize - pageOffset));
282 if (_i2c->send (_deviceAddress, I2C_MEMADD_SIZE_16BIT, address, ptr, wlen, I2C_TIMEOUT) == false) {
283 assert (0);
284 return false;

288 ptr += wlen;
