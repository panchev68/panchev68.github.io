1/** PID controller implementation for control systems.
2 * This file provides a robust and flexible PID controller class for discrete-time control systems. The implementation
3 * supports advanced features such as output clamping, integral windup protection, derivative kick prevention, and
4 * bumpless transfer between automatic and manual modes. The controller can be configured for direct or reverse action
5 * and allows runtime adjustment of gains, output limits, and sample time. The design is suitable for embedded and
6 * real-time applications, ensuring safe parameter validation and bounded control outputs. Type aliases for single and
7 * double precision are provided for convenience.
8 * @file Pid.hpp
9 */

19 /** PID controller for control systems.
20 * Implements a discrete-time PID controller with anti-windup, output clamping, and derivative kick prevention.
21 * Supports both automatic and manual modes with bumpless transfer. The controller is suitable for embedded and
22 * real-time applications, providing safe parameter validation and bounded control outputs.
23 * @tparam T Numeric type for calculations (typically float or double).
24 */
25 template <typename T = float>
26 class Pid final {
27 public:
28 /** PID controller parameters structure.
29 * Contains all tuning parameters for the PID controller, including gains, sample time, and output limits.
30 */
31 struct Parameters {
32 T kp; /*!< Proportional gain. */
33 T ki; /*!< Integral gain. */
34 T kd; /*!< Derivative gain. */
35 T sampleTime; /*!< Sample time in seconds. */
36 T outputMin; /*!< Minimum output value. */
37 T outputMax; /*!< Maximum output value. */
38
39 /** Default constructor of Parameters class. Initializes with safe default values. */
40 constexpr Parameters()
41 : kp(1.0), ki(0.0), kd(0.0), sampleTime(0.1), outputMin(-100.0), outputMax(100.0) {}
42
43 /** Constructor of Parameters class.
44 * @param proportional Proportional gain.
45 * @param integral Integral gain.
46 * @param derivative Derivative gain.
47 * @param sample Sample time in seconds.
48 * @param minOutput Minimum output value.
49 * @param maxOutput Maximum output value.
50 */
51 constexpr Parameters(T proportional, T integral, T derivative, T sample, T minOutput, T maxOutput)
52 : kp(proportional),
53 ki(integral),
54 kd(derivative),
55 sampleTime(sample),
56 outputMin(minOutput),
57 outputMax(maxOutput) {
58 if (!(sampleTime > T{0})) assert(false);
59 if (!(outputMin <= outputMax)) assert(false);
60 }
61 };
62
63 /** PID controller state structure.
64 * Contains internal state variables for the controller, including previous input, accumulated integral term,
65 * last output, and a flag for first execution.
66 */
67 struct State {
68 T lastInput; /*!< Previous input value for derivative calculation. */
69 T integralSum; /*!< Accumulated integral term. */
70 T lastOutput; /*!< Previous output value for manual mode. */
71 bool firstRun; /*!< Flag indicating first execution. */
72
73 /** Default constructor of State class. */
74 constexpr State() : lastInput(T{0}), integralSum(T{0}), lastOutput(T{0}), firstRun(true) {}
75 };
76
77 private:
78 Parameters params; /*!< PID controller parameters. */
79 State state; /*!< PID controller state. */
80 bool autoMode; /*!< Automatic mode flag. */
81 bool reverseDirection; /*!< Reverse control direction flag. */
82
83 public:
84 /** Default constructor of Pid class. */
85 constexpr Pid() : autoMode(false), reverseDirection(false) {}
86
87 /** Constructor of Pid class.
88 * @param parameters PID controller parameters.
89 */
90 explicit constexpr Pid(const Parameters & parameters)
91 : params(parameters), autoMode(false), reverseDirection(false) {}
92
93 /** Constructor of Pid class.
94 * @param kp Proportional gain.
95 * @param ki Integral gain.
96 * @param kd Derivative gain.
97 * @param sampleTime Sample time in seconds.
98 */
99 constexpr Pid(T kp, T ki, T kd, T sampleTime)
100 : params(kp, ki, kd, sampleTime, -std::numeric_limits<T>::max(), std::numeric_limits<T>::max()),
101 autoMode(false),
102 reverseDirection(false) {}
103
104 /** Compute PID output.
105 * @param setpoint Desired value.
106 * @param input Current process variable.
107 * @return PID controller output.
108 */
109 T compute(T setpoint, T input) {
110 if (!autoMode) return state.lastOutput;
111
112 T error = reverseDirection ? (input - setpoint) : (setpoint - input);
113
114 if (state.firstRun) {
115 state.lastInput = input;
116 state.firstRun = false;
117 }
118
119 T proportional = params.kp * error;
120 state.integralSum += params.ki * error * params.sampleTime;
121 state.integralSum =
122 std::clamp(state.integralSum, params.outputMin - proportional, params.outputMax - proportional);
123
124 T derivative = -params.kd * (input - state.lastInput) / params.sampleTime;
125 T output = std::clamp(proportional + state.integralSum + derivative, params.outputMin, params.outputMax);
126
127 state.lastInput = input;
128 state.lastOutput = output;
129
130 return output;
131 }
132
133 /** Set PID parameters.
134 * @param parameters New PID parameters.
135 */
136 void setParameters(const Parameters & parameters) {
137 params = parameters;
138 if (!(params.sampleTime > T{0})) assert(false);
139 if (!(params.outputMin <= params.outputMax)) assert(false);
140 }
141
142 /** Set PID gains.
143 * @param kp Proportional gain.
144 * @param ki Integral gain.
145 * @param kd Derivative gain.
146 */
147 void setGains(T kp, T ki, T kd) {
148 params.kp = kp;
149 params.ki = ki;
150 params.kd = kd;
151 }
152
153 /** Set output limits.
154 * @param minOutput Minimum output value.
155 * @param maxOutput Maximum output value.
156 */
157 void setOutputLimits(T minOutput, T maxOutput) {
158 if (!(minOutput <= maxOutput)) assert(false);
159 params.outputMin = minOutput;
160 params.outputMax = maxOutput;
161
162 if (autoMode) {
163 T proportional = params.kp * state.lastInput;
164 state.integralSum = std::clamp(state.integralSum, minOutput - proportional, maxOutput - proportional);
165 state.lastOutput = std::clamp(state.lastOutput, minOutput, maxOutput);
166 }
167 }
168
169 /** Set sample time.
170 * @param sampleTime New sample time in seconds.
171 */
172 void setSampleTime(T sampleTime) {
173 if (!(sampleTime > T{0})) assert(false);
174
175 T ratio = sampleTime / params.sampleTime;
176 params.ki *= ratio;
177 params.kd /= ratio;
178 params.sampleTime = sampleTime;
179 }
180
181 /** Set controller mode.
182 * @param automatic `true` for automatic mode, `false` for manual mode.
183 */
184 void setMode(bool automatic) {
185 if (automatic && !autoMode) {
186 state.integralSum = std::clamp(state.lastOutput, params.outputMin, params.outputMax);
187 state.firstRun = true;
188 }
189 autoMode = automatic;
190 }
191
192 /** Set control direction.
193 * @param reverse `true` for reverse action, `false` for direct action.
194 */
195 void setDirection(bool reverse) {
196 if (reverse != reverseDirection) {
197 params.kp = -params.kp;
198 params.ki = -params.ki;
199 params.kd = -params.kd;
200 reverseDirection = reverse;
201 }
202 }
203
204 /** Reset controller state. Clears integral sum and resets internal state. */
205 void reset() { state = State{}; }
206
207 /** Set manual output value.
208 * @param output Manual output value.
209 */
210 void setManualOutput(T output) { state.lastOutput = std::clamp(output, params.outputMin, params.outputMax); }
211
212 /** Get current PID parameters.
213 * @return Current parameters.
214 */
215 const constexpr Parameters & getParameters() const { return params; }
216
217 /** Get current controller state.
218 * @return Current state.
219 */
220 const constexpr State & getState() const { return state; }
221
222 /** Check if controller is in automatic mode.
223 * @return `true` if in automatic mode, `false` if in manual mode.
224 */
225 constexpr bool isAutomatic() const { return autoMode; }
226
227 /** Check if controller has reverse direction.
228 * @return `true` if reverse direction, `false` if direct direction.
229 */
230 constexpr bool isReverse() const { return reverseDirection; }
231
232 /** Get last computed output.
233 * @return Last output value.
234 */
235 constexpr T getOutput() const { return state.lastOutput; }
236
237 /** Get current integral term.
238 * @return Current integral sum.
239 */
240 constexpr T getIntegralTerm() const { return state.integralSum; }
241 };
