20/** MemoryManager class for FreeRTOS heap allocation and diagnostics.
21 * Provides static APIs for allocation, deallocation, diagnostics, and heap usage.

26 /** Status codes for memory operations. */
27 enum class Status : uint8_t {
28 SUCCESS = 0, /*!< Operation successful. */
29 INVALID_SIZE = 1, /*!< Size parameter is zero. */
30 OUT_OF_MEMORY = 2, /*!< Allocation failed due to insufficient memory. */
31 NULL_POINTER = 3, /*!< Null pointer provided for deallocation. */
32 SIZE_TOO_LARGE = 4 /*!< Requested size exceeds total heap. */
33 };

111 /** Checks if size is valid for allocation.
112 * @param size Number of bytes.
113 * @return `true` if size is valid, `false` otherwise.
114 */
115 static constexpr bool isValidSize(size_t size) noexcept {
116 // Ensure non-zero, fits with header, and does not exceed configured heap minus header.
117 if (size == 0) return false;
118 if (size > (kMaxSize - kHeaderSize)) return false;
119#if defined(configTOTAL_HEAP_SIZE)
120 const size_t totalHeap = static_cast<size_t>(configTOTAL_HEAP_SIZE);
121 const size_t maxAlloc = (totalHeap > kHeaderSize) ? (totalHeap - kHeaderSize) : 0u;
122 return size <= maxAlloc;
123#else
124 return true;
125#endif
126 }

128 /** Gets heap usage as percent.
129 * @return Heap usage percent (0-100).
130 */
131 static uint8_t getHeapUsagePercent() noexcept {
132#if defined(configTOTAL_HEAP_SIZE)
133 const size_t totalHeap = static_cast<size_t>(configTOTAL_HEAP_SIZE);
134 if (totalHeap == 0) return 0;
135 const size_t usedHeap = (totalHeap > getFreeHeapSize()) ? (totalHeap - getFreeHeapSize()) : 0u;
136 const uint64_t percent = (static_cast<uint64_t>(usedHeap) * 100ull) / static_cast<uint64_t>(totalHeap);
137 return static_cast<uint8_t>(percent > 100ull ? 100u : percent);
138#else
139 return 0;
140#endif
141 }

161 /** Centralized allocation implementation.
162 * Returns pointer to user area or nullptr and sets status if provided.
163 */
164 static void * allocateImpl(size_t size, Status * status) noexcept {
165 if (!isValidSize(size)) {
166 if (status) {
167 *status = (size == 0) ? Status::INVALID_SIZE : Status::SIZE_TOO_LARGE;
168 }
169#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
170 diagnostic.failedAllocations++;
171#endif
172 return nullptr;
173 }
174
175 // Prevent overflow when adding header
176 if (size > (kMaxSize - kHeaderSize)) {
177 if (status) *status = Status::SIZE_TOO_LARGE;
178#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
179 diagnostic.failedAllocations++;
180#endif
181 return nullptr;
182 }
183
184 const size_t totalSize = size + kHeaderSize;
185 void * raw = pvPortMalloc(totalSize);
186 if (!raw) {
187 if (status) *status = Status::OUT_OF_MEMORY;
188 handleAllocationFailure();
189#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
190 diagnostic.failedAllocations++;
191#endif
192 return nullptr;
193 }
194
195 auto * hdr = static_cast<AllocationHeader *>(raw);
196 hdr->size = size;
197 hdr->magic = ALLOC_HEADER_MAGIC;
198
199 if (status) *status = Status::SUCCESS;
200
201#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
202 recordAlloc(size);
203#endif
204
205 return static_cast<void *>(static_cast<uint8_t *>(raw) + kHeaderSize);
206 }
207
208 /** Centralized deallocation implementation.
209 * Frees memory and sets status if provided.
210 */
211 static void deallocateImpl(void * pointer, Status * status) noexcept {
212 if (!pointer) {
213 if (status) *status = Status::NULL_POINTER;
214 return;
215 }
216
217 auto * raw = static_cast<uint8_t *>(pointer) - kHeaderSize;
218 auto * hdr = reinterpret_cast<AllocationHeader *>(raw);
219
220 if (hdr->magic == ALLOC_HEADER_MAGIC) {
221 const size_t allocSize = hdr->size;
222#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
223 recordDealloc(allocSize);
224#endif
225 vPortFree(raw);
226 if (status) *status = Status::SUCCESS;
227 } else {
228 // Best-effort free for pointers not allocated by this manager.
229#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
230 // Update diagnostics conservatively.
231 if (diagnostic.active > 0) diagnostic.active--;
232 diagnostic.deallocated++;
233#endif
234 vPortFree(raw);
235 if (status) *status = Status::SUCCESS;
236 }
237 }
238
239#if defined(MEMORY_MANAGER_DIAG) || defined(__INTELLISENSE__) || defined(__DOXYGEN__)
240 static void recordAlloc(size_t size) noexcept {
241 diagnostic.allocated++;
242 diagnostic.active++;
243 // Safe add: avoid overflow
244 if (diagnostic.currentUsage <= (kMaxSize - size)) {
245 diagnostic.currentUsage += size;
246 if (diagnostic.currentUsage > diagnostic.peakUsage) diagnostic.peakUsage = diagnostic.currentUsage;
247 } else {
248 diagnostic.currentUsage = kMaxSize;
249 diagnostic.peakUsage = diagnostic.currentUsage;
250 }
251 }
252
253 static void recordDealloc(size_t size) noexcept {
254 if (diagnostic.currentUsage >= size) diagnostic.currentUsage -= size;
255 else diagnostic.currentUsage = 0;
256 diagnostic.deallocated++;
257 if (diagnostic.active > 0) diagnostic.active--;
258 }
259#endif
260
261 /** Handles allocation failure (asserts and throws if exceptions enabled).
262 * @warning Throws std::bad_alloc if exceptions enabled.
263 */
264 static void handleAllocationFailure() noexcept {
265 assert(false);
266#if defined(__cpp_exceptions) && (__cpp_exceptions == 1)
267 throw std::bad_alloc();
268#endif
269 }
