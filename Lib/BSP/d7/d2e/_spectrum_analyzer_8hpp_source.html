68 /** Updates the service and processes spectrum data. */
69 void update() override {
70 static std::vector<uint8_t> response;
71 if (!hasWaitForResponse) {
72 if (timeout.isExpired()) {
73 if (!localAtCommand->send({"ED"})) {
74 assert(false);
75 return;
76 }
77 timeout.set(1000);
78 hasWaitForResponse = true;
79 }
80 } else {
81 if (!timeout.isExpired()) {
82 if (localAtCommand->receive(response, 0)) {
83 for (auto & value : response)
84 value = Lib::Helper::Convert::dBM::toDigiPercent(value);
85 for (size_t i = 0; i < response.size() && i < currentArray.size(); ++i) {
86 uint8_t value = static_cast<uint8_t>(response[i] + currentArray[i] / 2);
87 currentArray[i] = value;
88 if (averageArray[i] > value) averageArray[i] -= 1;
89 else if (averageArray[i] < value) averageArray[i] += 2;
90 if (value > maxArray[i]) maxArray[i] = value;
91 }
92 isNewData = true;
93 hasWaitForResponse = false;
94 timeout.set(250);
95 }
96 } else {
97 hasWaitForResponse = false;
98 timeout.set(500);
99 }
100 }
101 }
