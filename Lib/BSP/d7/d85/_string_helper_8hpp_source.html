21 /** Utility functions for string operations.
22 *
23 * The StringHelper class provides static methods for converting string case, trimming whitespace,
24 * and searching for strings in null-terminated arrays. All methods validate inputs and use standard

29 /** Converts a string to uppercase.
30 * @param input Input string to convert.
31 * @return New string with all characters in uppercase.
32 */
33 static std::string toUpperCase(const std::string & input) {
34 assert(!input.empty());
35 std::string result(input);
36 std::transform(result.begin(), result.end(), result.begin(),
37 [](unsigned char c) { return static_cast<char>(std::toupper(c)); });
38 return result;
39 }

41 /** Converts a string to lowercase.
42 * @param input Input string to convert.
43 * @return New string with all characters in lowercase.
44 */
45 static std::string toLowerCase(const std::string & input) {
46 assert(!input.empty());
47 std::string result(input);
48 std::transform(result.begin(), result.end(), result.begin(),
49 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
50 return result;
51 }
52
53 /** Trims whitespace from both ends of a string.
54 * @param input Input string to trim.
55 * @return New string with leading and trailing whitespace removed.
56 */
57 static std::string trim(const std::string & input) {
58 assert(!input.empty());
59 auto first = std::find_if_not(input.begin(), input.end(),
60 [](unsigned char c) { return std::isspace(c); });
61 if (first == input.end()) return std::string();
62 auto last = std::find_if_not(input.rbegin(), input.rend(),
63 [](unsigned char c) { return std::isspace(c); })
64 .base();
65 return std::string(first, last);
66 }
67
68 /** Finds the index of a string in a null-terminated array of strings.
69 * @param target String to search for.
70 * @param array Null-terminated array of strings.
71 * @return Index of the target string if found, `-1` if not found.
72 */
73 static int32_t findInArray(const char * target, const char * const * array) {
74 assert(target);
75 assert(array);
76 int32_t index = 0;
77 while (*array) {
78 if (std::strcmp(target, *array) == 0) return index;
79 ++array;
80 ++index;
81 }
82 return -1;
83 }
84 };
