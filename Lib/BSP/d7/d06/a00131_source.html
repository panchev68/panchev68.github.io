62 virtual bool update () override
63 {
64 _nss.write (0);
65
66 // send data via spi without waitForCompleted (timeout duration = 0) ...
67 if (_spi.receive (_rxBuffer.data (), _rxBuffer.size (), 0) == false)

69 assert (0);

78 virtual bool waitForCompleted (uint32_t timeout) override
79 {
80 auto res = _spi.waitForRxCompleted (timeout);
81
82 _nss.write (1); // de-select spi interface
83 if (res == false)
84 {
85 assert (0);
86 return false;
87 }
88
89 //> Set inputs in array
90
91 uint64_t value = 0;
92
93 for (auto i = 0; i < T_PIN_NUM; i++)
94 {
95 //> Set inputs
96
97 auto & map = _map [i];
98 auto input = _array.at (i).update ((_rxBuffer [map.port] & map.mask));
99 if (input == true) value |= BitMask::at (i);
100 }
101
102 //> Detect changes amd emit value changed
103
104 if (value != _lastValue)
105 {
106 _lastValue = value;
107 _changesCounter++;
108
109 //> Handle value changed event
110 handleValueChangedEvent (value);
111 }
112 return true;
113 }
