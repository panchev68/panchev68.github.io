95inline void DigitalPeripheralBase::Inputs::update ()
96{
97 // reads data from sleep port...
98
99 _nss.write (0);
100
101 if (_spi.receive (_buffer, _bufferSize, 5) == false)
102 {
103 assert (0);
104 }
105 _nss.write (1);
106
107 // processing of received data...
108
109 for (auto i = 0; i < _numberOfPins; i++)
110 {
111 auto & map = _mapFields [i];
112 _bitset.set (i, ((_buffer [map.port] & map.mask)));
113 }
114 if (_value != _bitset.to_ullong ())
115 {
116 _value = _bitset.to_ullong ();
117 if (onValueChanged != nullptr) onValueChanged (_value);
118 _changesCounter++;
119 }
120}
