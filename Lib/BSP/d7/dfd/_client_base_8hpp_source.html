95 void beginUpdate() override {

101 /** Task loop callback. */
102 void update() override {
103 auto & messageIn = taskData.messageIn;
104 auto & messageOut = taskData.messageOut;
105 auto & timeoutConnection = taskData.timeoutConnection;
106 auto & timeoutRetries = taskData.timeoutRetries;
107 if (queueRxMessage.receive(messageIn, 0)) {
108 if (!handleIncomingMessage(messageIn)) assert(!"Failed to handle incoming message");
109 else {
110 if (connection.state != Connection::State::CONNECTED) {
111 connection.state = Connection::State::CONNECTED;
112 connection.info.transmitPeriod.beginMeasurement();
113 } else {
114 connection.info.transmitPeriod.endMeasurement();
115 connection.info.transmitPeriod.beginMeasurement();
116 }
117 timeoutRetries.set(ApplicationBase::MAX_RESPONSE_TIME);
118 timeoutConnection.set(ApplicationBase::CONNECTION_TIMEOUT);
119 handleOutgoingMessage(messageOut);
120 if (!sendData(messageOut)) assert(!"Failed to send outgoing message");
121 }
122 } else if (connection.state == Connection::State::CONNECTED) {
123 if (timeoutRetries.isExpired()) {
124 timeoutRetries.set(portMAX_DELAY);
125 connection.info.retriesCounter++;
126 }
127 if (timeoutConnection.isExpired()) {
128 timeoutConnection.set(portMAX_DELAY);
129 connection.state = Connection::State::DISCONNECTED;
130 connection.info.signalStrength = 0;
131 connection.info.transmitPeriod.reset();
132 connection.info.reconnectionCounter++;
133 }
134 }
135 }
