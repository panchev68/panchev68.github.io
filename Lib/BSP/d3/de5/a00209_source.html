136 bool start ()
137 {
138 if (queueButtonPressed.create () == false)
139 {
140 assert (0);
141 return false;
142 }
143 if (FreeRTOS::Task::run ("TouchScreen", [this] { taskFn (); }, 128, 0) == nullptr)
144 {
145 assert (0);
146 return false;
147 }
148 return true;
149 }
150
152 void taskFn ()
153 {
154 auto extiFunction = [this] (const Lib::HAL::GPIO::ExtInterrupt * exti) {
155 queueButtonPressed.overwriteFromISR (true);
156 };
157
158 // Reads the internal a/d convertor
159 auto readAdc = [this] (uint8_t cmd) {
160 auto writeByte = [this] (uint8_t data) -> void {
161 for (auto i = 0; i < 8; i++)
162 {
163 si.write ((data & 0x80) != 0);
164 __NOP ();
165 data <<= 1;
166 sck.write (0);
167 __NOP ();
168 sck.write (1);
169 __NOP ();
170 }
171 };
172
173 uint32_t num = 0;
174 si.write (0);
175 nss.write (0);
176 writeByte (cmd);
177
178 sck.write (0);
179 __NOP ();
180 sck.write (1);
181 __NOP ();
182
183 for (auto i = 0; i < 16; i++)
184 {
185 num <<= 1;
186 sck.write (0);
187 __NOP ();
188 sck.write (1);
189 __NOP ();
190 if (so.read () == 1) num++;
191 }
192 num >>= 4;
193 nss.write (1);
194
195 return static_cast<uint16_t> (num);
196 };
197
198 nirq.bind (extiFunction);
199 sck.write (1);
200 nss.write (1);
201
202 // Task loop...
203
204 while (1)
205 {
206 isPressed = false;
207
208 // wait for touch screen to be pressed ...
209 bool pinState = false;
210 while (pinState == false) queueButtonPressed.receive (pinState, portMAX_DELAY);
211
212 FreeRTOS::Task::delay (0);
213
214 intergratorX.reset ();
215 intergratorY.reset ();
216 position.X = -1;
217 position.Y = -1;
218
219 timeoutPinDebounce.set (parameters.debounceDuration);
220
221 while (timeoutPinDebounce.check () == false)
222 {
223 if (nirq.read () == true) timeoutPinDebounce.set (parameters.debounceDuration);
224 vTaskDelay (1);
225 }
226
227 while (1)
228 {
229 if (nirq.read () == true) break;
230
231 // Interpolates coordinates...
232
233 rawPosition.X = intergratorX.process (readAdc (CMD_READ_X));
234 rawPosition.Y = intergratorY.process (readAdc (CMD_READ_Y));
235
236 if (intergratorX.isFull () == false)
237 {
238 vTaskDelay (1);
239 continue;
240 }
241
242 // Reads and averages values ​​from the driver..
243
244 rawPosition.X = std::max<uint32_t> (parameters.xMin, rawPosition.X);
245 rawPosition.Y = std::max<uint32_t> (parameters.yMin, rawPosition.Y);
246
247 rawPosition.X = std::min<uint32_t> (rawPosition.X, parameters.xMax);
248 rawPosition.Y = std::min<uint32_t> (rawPosition.Y, parameters.yMax);
249
250 rawPosition.X -= parameters.xMin;
251 rawPosition.Y -= parameters.yMin;
252
253 position.X = static_cast<int32_t> (rawPosition.X / scaleFaconstructorX);
254 position.Y = static_cast<int32_t> (rawPosition.Y / scaleFaconstructorY);
255
256 // The touch screen is pressed and ready to read data.
257
258 isPressed = true;
259
260 switch (rotation)
261 {
262 case 0:
263 // position = position;
264 break;
265 case 1:
266 position.X = parameters.dispWidth - position.X;
267 position.Y = position.Y;
268 break;
269 case 2: // flip X
270 position.X = parameters.dispWidth - position.X;
271 position.Y = parameters.dispHeight - position.Y;
272 break;
273 case 3: // flip Y
274 position.X = position.X;
275 position.Y = parameters.dispHeight - position.Y;
276 break;
277 case 4: // flip X,Y
278 position.X = parameters.dispWidth - position.X;
279 position.Y = parameters.dispHeight - position.Y;
280 break;
281 default:
282 assert (0);
283 break;
284 }
285 vTaskDelay (40);
286 }
287 }
288 }
