Lib::BSP::EEPROM::I2C::M24C64W

Example usage of the M24C64W EEPROM driver.

This example demonstrates how to initialize, open, write, read, and close the M24C64W EEPROM device using the provided driver interface.

Example usage:
#include "M24C64W.hpp"
int main() {
Lib::BSP::EEPROM::I2C::M24C64W::Config config{...};
Lib::BSP::EEPROM::I2C::M24C64W eeprom(config);
if (!eeprom.open()) return -1;
uint8_t writeData[8] = {1,2,3,4,5,6,7,8};
if (!eeprom.write(0x0000, writeData, sizeof(writeData))) return -2;
uint8_t readData[8] = {};
if (!eeprom.read(0x0000, readData, sizeof(readData))) return -3;
eeprom.close();
return 0;
}
