178 static bool parseIpString(const std::string & str, std::array<uint8_t, 4> * address) {
179 if (str.empty() || str.size() > 15 || str.find(':') != std::string::npos) return false;
180 std::array<uint8_t, 4> temp{};
181 size_t pos = 0;
182 for (size_t i = 0; i < 4; ++i) {
183 if (pos >= str.size()) return false;
184 size_t found = (i < 3) ? str.find('.', pos) : std::string::npos;
185 if (i < 3 && (found == std::string::npos || found == pos)) return false;
186 uint32_t value = 0;
187 size_t end = (i < 3) ? found : str.size();
188 for (size_t j = pos; j < end; ++j) {
189 char c = str[j];
190 if (c < '0' || c > '9') return false;
191 value = value * 10 + static_cast<uint32_t>(c - '0');
192 if (value > 255) return false;
193 }
194 if (pos == end) return false;
195 temp[i] = static_cast<uint8_t>(value);
196 pos = (i < 3) ? found + 1 : str.size();
197 }
198 if (str.find('.', pos) != std::string::npos) return false;
199 if (address) *address = temp;
200 return true;
201 }
