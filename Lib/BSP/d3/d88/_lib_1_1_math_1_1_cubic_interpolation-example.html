Lib::Math::CubicInterpolation

Example: Cubic spline interpolation with standard output

This example demonstrates how to use the CubicInterpolationF class to perform cubic spline interpolation on a set of data points and print the interpolated values and derivatives to the standard output.

Example usage:
#include <Lib/Math/Interpolation/CubicInterpolation.hpp>
#include <cstdio>
int main() {
using Lib::Math::CubicInterpolationF;
using Lib::Math::CubicInterpolationF::Point;
// Define interpolation points
constexpr std::array<Point, 5> points = {{
{0.0f, 0.0f},
{1.0f, 2.0f},
{2.0f, 0.0f},
{3.0f, 2.0f},
{4.0f, 0.0f}
}};
// Create cubic interpolation object with natural boundary
CubicInterpolationF spline(points);
// Interpolate and print results
for (float x = 0.0f; x <= 4.0f; x += 0.5f) {
float y = spline.interpolate(x);
float dy = spline.getDerivative(x);
std::printf("x = %.2f, y = %.4f, dy/dx = %.4f\n", x, y, dy);
}
return 0;
}
