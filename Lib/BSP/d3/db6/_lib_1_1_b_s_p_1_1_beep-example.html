Lib::BSP::Beep

Example usage of the Beep class.

Demonstrates how to configure and start a beep sequence using the Beep class.

Example usage:

#include "Beep.hpp"
TIM_HandleTypeDef htim1;
int main() {
Lib::BSP::Beep::Config beepConfig{&htim1, TIM_CHANNEL_1};
Lib::BSP::Beep beep(beepConfig);
beep.start(50, 3, 20); // 3 beeps, each 50 ticks, 20 ticks pause
while (1) {
// RTOS tick service should call beep.handleTickEvent() periodically
}
return 0;
}
