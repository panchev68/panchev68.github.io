147 inline bool DriverBase::write16 (uint32_t eepAddr, const void * src, size_t len)
148 {
149 if (_impl.isReady == false)
150 {
151 assert (0);
152 return false;
153 }
154
155 if ((eepAddr + len) > _impl.chipSize)
156 {
157 assert (0 && "Address is out of range");
158 return false;
159 }
160
161 uint8_t * ptr = reinterpret_cast<uint8_t *> (const_cast<void *> (src));
162
163 while (len > 0)
164 {
165 auto offs_page = eepAddr % _impl.pageSize;
166 size_t wlen = std::min<size_t> (len, (_impl.pageSize - offs_page));
167
168 if (_impl.i2c.send (_impl.devAddr, HAL::I2C::Master::MemAddrSize::ADDR_16BITS, eepAddr, ptr, wlen) == false)
169 {
170 assert (0);
171 return false;
172 }
173 vTaskDelay (5);
174
175 eepAddr += wlen;
176 ptr += wlen;
177 len -= wlen;
178 }
179 return true;
180 }
