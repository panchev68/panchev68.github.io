19/** Base class for I2C EEPROM drivers.
20 * Provides common functionality for interacting with I2C EEPROM devices.
21 */
22class DriverBase {
23public:
24 /** Configuration structure for DriverBase. */
25 struct Config {
26 I2CPort * port; /*!< Pointer to the I2C port interface. */

30 static constexpr uint32_t I2C_TIMEOUT = 100; /*!< Timeout for I2C operations (ms). */
31
32 /** Constructor of the DriverBase class.
33 * @param config Configuration structure.
34 * @param pageSize Size of the EEPROM page.
35 * @param chipSize Total size of the EEPROM chip.
36 */
37 explicit DriverBase(const Config & config, uint8_t pageSize, size_t chipSize)
38 : i2cPort(config.port),
39 deviceAddr(static_cast<uint8_t>((config.addr & 0x0F) | 0xA0)),
40 pageSize(pageSize),
41 chipSize(chipSize),
42 isReady(false) {
43 assert(i2cPort);
44 assert(pageSize > 0);
45 assert(chipSize > 0);
46 }
47
48 /** Destructor of the DriverBase class. */
49 virtual ~DriverBase() = default;
50
51 /** Opens the I2C connection.
52 * @return `true` if successful, `false` otherwise.
53 */
54 virtual bool open() {
55 if (!i2cPort->init()) return false;
56 isReady = i2cPort->isReady(deviceAddr);
57 assert(isReady);
58 return isReady;
59 }
60
61 /** Closes the I2C connection.
62 * @return `true` if successful, `false` otherwise.
63 */
64 virtual bool close() {
65 isReady = false;
66 i2cPort->deinit();
67 return true;
68 }
69
70 /** Reads data from the EEPROM.
71 * @param addr Memory address to read from.
72 * @param buffer Pointer to the buffer to store read data.
73 * @param bufferSize Size of the buffer.
74 * @return `true` if successful, `false` otherwise.
75 */
76 virtual bool read(uint32_t addr, void * buffer, size_t bufferSize) = 0;
77

79 * @param addr Memory address to write to.
80 * @param buffer Pointer to the buffer containing data to write.
81 * @param bufferSize Size of the buffer.
82 * @return `true` if successful, `false` otherwise.
83 */
84 virtual bool write(uint32_t addr, const void * buffer, size_t bufferSize) = 0;
85
86 /** Gets the total size of the EEPROM chip.
87 * @return Size of the EEPROM chip.
88 */
89 virtual size_t getChipSize() const { return chipSize; }
90
91protected:
92 /** Reads data from the EEPROM using 8-bit memory addressing.
93 * @param addr Memory address to read from.
94 * @param buffer Pointer to the buffer to store read data.
95 * @param size Size of the buffer.
96 * @return `true` if successful, `false` otherwise.
97 */
98 virtual bool read8(uint32_t addr, void * buffer, size_t size) {
99 if (!isReady) {
100 assert(false);
101 return false;
102 }
103 return i2cPort->receive(deviceAddr, I2C_MEMADD_SIZE_8BIT, addr, buffer, size, I2C_TIMEOUT);
104 }
105
106 /** Reads data from the EEPROM using 16-bit memory addressing.
107 * @param addr Memory address to read from.
108 * @param buffer Pointer to the buffer to store read data.
109 * @param size Size of the buffer.
110 * @return `true` if successful, `false` otherwise.
111 */
112 virtual bool read16(uint32_t addr, void * buffer, size_t size) {
113 if (!isReady) {
114 assert(false);
115 return false;
116 }
117 return i2cPort->receive(deviceAddr, I2C_MEMADD_SIZE_16BIT, addr, buffer, size, I2C_TIMEOUT);
118 }
119
120 /** Writes data to the EEPROM using 8-bit memory addressing.
121 * @param addr Memory address to write to.
122 * @param buffer Pointer to the buffer containing data to write.
123 * @param size Size of the buffer.
124 * @return `true` if successful, `false` otherwise.
125 */
126 bool write8(uint32_t addr, const void * buffer, size_t size) {
127 return write(addr, buffer, size, I2C_MEMADD_SIZE_8BIT);
128 }
129
130 /** Writes data to the EEPROM using 16-bit memory addressing.
131 * @param addr Memory address to write to.
132 * @param buffer Pointer to the buffer containing data to write.
133 * @param size Size of the buffer.
134 * @return `true` if successful, `false` otherwise.
135 */
136 bool write16(uint32_t addr, const void * buffer, size_t size) {
137 return write(addr, buffer, size, I2C_MEMADD_SIZE_16BIT);
138 }
139
140private:
141 /** Internal helper for writing data to the EEPROM.
142 * @param addr Memory address to write to.
143 * @param buffer Pointer to the buffer containing data to write.
144 * @param size Size of the buffer.
145 * @param memAddSize Memory address size (8-bit or 16-bit).
146 * @return `true` if successful, `false` otherwise.
147 */
148 bool write(uint32_t addr, const void * buffer, size_t size, uint16_t memAddSize) {
149 if (!isReady || (addr + size) > chipSize) {
150 assert(false);
151 return false;
152 }
153 auto * dataPtr = static_cast<uint8_t *>(const_cast<void *>(buffer));
154 size_t remaining = size;
155 uint32_t currentAddr = addr;
156 while (remaining > 0) {
157 size_t pageOffset = currentAddr % pageSize;
158 size_t writeLen = std::min(remaining, pageSize - pageOffset);
159 if (!i2cPort->send(deviceAddr, memAddSize, currentAddr, dataPtr, writeLen, I2C_TIMEOUT)) {
160 assert(false);
161 return false;
162 }
163 vTaskDelay(5);
164 currentAddr += writeLen;
165 dataPtr += writeLen;
166 remaining -= writeLen;
167 }
168 return true;
169 }
170
171 I2CPort * i2cPort; /*!< Pointer to the I2C port interface. */
172 uint8_t deviceAddr; /*!< I2C device address. */
173 const uint8_t pageSize; /*!< Size of the EEPROM page. */
174 const size_t chipSize; /*!< Total size of the EEPROM chip. */
175 bool isReady; /*!< Indicates if the device is ready. */
176};
