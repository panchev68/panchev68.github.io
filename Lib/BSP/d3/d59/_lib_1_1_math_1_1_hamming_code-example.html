Lib::Math::HammingCode

Example usage of HammingCode for encoding and decoding with error correction.

This example demonstrates how to use the HammingCode class to encode a 16-bit value, introduce a single-bit error, and then decode and correct the error using the Hamming(21,16) code.

Example usage:
#include "Lib/Math/HammingCode.hpp"
#include <iostream>
#include <iomanip>
int main() {
Lib::Math::HammingCode hamming;
uint16_t original = 0xBEEF; // Example 16-bit value
uint32_t encoded = hamming.encode(original);
std::cout << "Original: 0x" << std::hex << std::setw(4) << std::setfill('0') << original << std::endl;
std::cout << "Encoded : 0x" << std::hex << std::setw(6) << std::setfill('0') << encoded << std::endl;
// Introduce a single-bit error at bit 5 (1-based)
uint32_t corrupted = encoded ^ (1U << (5 - 1));
std::cout << "Corrupted: 0x" << std::hex << std::setw(6) << std::setfill('0') << corrupted << std::endl;
uint16_t decoded = hamming.decode(corrupted);
std::cout << "Decoded : 0x" << std::hex << std::setw(4) << std::setfill('0') << decoded << std::endl;
std::cout << "Correction " << (decoded == original ? "successful" : "failed") << std::endl;
return 0;
}
