60 Error parse (const uint8_t * data, size_t len, Command & command, uint8_t & option)
61 {
62 if ((data == nullptr) || (len == 0))
63 {
64 assert (0 && "Invalid parameters");
65 return Error::EMPTY_MESSAGE;
66 }
67
68 do
69 {
70 // Message Header process...
71
72 auto & header = reinterpret_cast<const MessageHeader &> (*data);
73
74 // Chech for a start delimiter..
75
76 if (header.delimiter != START_DELIMITER)
77 {
78 assert (0);
79 return Error::INVALID_START_DELIMITER;
80 }
81
82 handleIncomingMessageHeader (header); // Handle incoming message header
83
84 command = static_cast<Command> (header.command);
85 option = header.option;
86
87 size_t nodesNum; // number of nodes
88 size_t payloadLen = header.lenght; // payload lenght
89
90 data += sizeof (MessageHeader);
91 len -= sizeof (MessageHeader);
92
93 // Command prosess...
94
95 switch (command)
96 {
97 // Echo command (client/server function)
98 case Command::ECHO:
99 echoCallback ();
100 break;
101
102 // Inputs update
103 case Command::INPUTS:
104
105 if ((payloadLen % sizeof (uint64_t)) == 0)
106 {
107 nodesNum = payloadLen / sizeof (uint64_t);
108 _payload = reinterpret_cast<const uint64_t *> (data);
109 }
110 else
111 {
112 assert (0);
113 return Error::INVALID_PAYLOAD_LENGHT;
114 }
115
116 // update node activity
117 updateNodeActivityCallback (nodesNum);
118
119 // outputs update
120
121 for (auto i = 0; i < nodesNum; i++) inputsUpdateCallback (i, _payload [i]);
122 break;
123
124 // Outputs update
125 case Command::OUTPUTS:
126
127 uint64_t value;
128
129 if ((payloadLen % sizeof (uint64_t)) == 0)
130 {
131 nodesNum = payloadLen / sizeof (uint64_t);
132
133 // update node activity...
134 updateNodeActivityCallback (nodesNum);
135
136 // update outputs of local and extended board...
137 for (auto i = 0; i < nodesNum; i++)
138 {
139 memcpy (&value, reinterpret_cast<const uint64_t *> (data) + i, sizeof (uint64_t));
140 outputsUpdateCallback (i, value);
141 }
142 }
143 else
144 {
145 assert (0);
146 return Error::INVALID_PAYLOAD_LENGHT;
147 }
148 break;
149
150 // Unknown command
151 default:
152
153 return Error::COMMAND_NOT_SUPPORT;
154 }
155
156 len -= payloadLen;
157 data += payloadLen;
158 } while (len > 0);
159
160 return Error::OK;
161 }

168 std::vector<uint8_t> build (Command cmd, uint8_t option, uint16_t id, const std::vector<uint8_t> * payload = nullptr)
169 {
170 std::vector<uint8_t> vector;
171 auto payloadSize = (payload != nullptr) ? payload->size () : 0;
172
173 // build the message header
174 MessageHeader header = {0x7e, id, (_lastSentTimeStamp = HAL_GetTick ()), static_cast<uint8_t> (cmd), option, static_cast<uint8_t> (payloadSize)};
175
176 // copy header
177 vector.assign (reinterpret_cast<uint8_t *> (&header), reinterpret_cast<uint8_t *> (&header) + sizeof (header));
178
179 // copy payload
180 if (payloadSize > 0) vector.insert (vector.end (), payload->begin (), payload->end ());
181 return vector;
182 }
