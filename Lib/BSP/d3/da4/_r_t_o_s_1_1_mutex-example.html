RTOS::Mutex

Advanced example: Using RTOS::Mutex for protecting a shared resource in concurrent tasks.

This example demonstrates the use of RTOS::Mutex and MutexGuard to synchronize access to a shared resource (sharedBuffer) between two concurrent tasks in a FreeRTOS-based application. The example shows how to create a recursive mutex, use RAII-based locking, and safely update a buffer from multiple tasks. It also demonstrates how to check lock status and handle lock acquisition failures.

Example usage:
#include "Lib/RTOS/Mutex.hpp"
#include <cstdio>
#include <cstring>
RTOS::Mutex sharedBufferMutex(RTOS::Mutex::MutexType::RECURSIVE);
char sharedBuffer[32] = {0};
void writerTask(const char *text) {
if (!sharedBufferMutex.isValid()) sharedBufferMutex.create();
for (int i = 0; i < 3; ++i) {
MutexGuard guard(sharedBufferMutex, 50);
if (guard) {
std::snprintf(sharedBuffer, sizeof(sharedBuffer), "Write %d: %s", i, text);
std::printf("Writer: %s\n", sharedBuffer);
} else {
std::printf("Writer failed to acquire lock\n");
}
}
}
void readerTask() {
if (!sharedBufferMutex.isValid()) sharedBufferMutex.create();
for (int i = 0; i < 3; ++i) {
MutexGuard guard(sharedBufferMutex, 50);
if (guard.isLocked()) {
std::printf("Reader: %s\n", sharedBuffer);
} else {
std::printf("Reader failed to acquire lock\n");
}
}
}
int main() {
sharedBufferMutex.create();
writerTask("Hello");
readerTask();
writerTask("World");
readerTask();
sharedBufferMutex.destroy();
return 0;
}
