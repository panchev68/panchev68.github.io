Lib::Helper::Convert::Ascii

Example usage of Ascii class for ASCII character conversion.

Demonstrates printable ASCII check, digit and hex digit detection, and conversion between ASCII and numeric values.

Example usage:

#include "Lib/Helper/Convert/Ascii.hpp"
#include <iostream>
int main() {
char ch = 'A';
std::cout << "Is '" << ch << "' printable ASCII? " << (Lib::Helper::Convert::Ascii::isPrintableAscii(ch) ? "Yes" : "No") << std::endl;
std::cout << "Is '" << ch << "' ASCII? " << (Lib::Helper::Convert::Ascii::isAscii(ch) ? "Yes" : "No") << std::endl;
std::cout << "Is '" << ch << "' digit? " << (Lib::Helper::Convert::Ascii::isDigit(ch) ? "Yes" : "No") << std::endl;
std::cout << "Is '" << ch << "' hex digit? " << (Lib::Helper::Convert::Ascii::isHexDigit(ch) ? "Yes" : "No") << std::endl;
std::cout << "'" << ch << "' to hex: " << static_cast<int>(Lib::Helper::Convert::Ascii::toHex(ch)) << std::endl;
uint8_t value = 10;
std::cout << "Value " << static_cast<int>(value) << " to hex char: " << Lib::Helper::Convert::Ascii::fromHex(value) << std::endl;
return 0;
}
