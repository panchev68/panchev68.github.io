Lib::BSP::DigitalPeripheralBase

Example usage of DigitalPeripheralBase.

This example demonstrates how to configure and use the DigitalPeripheralBase class for SPI-based digital input and output management in an embedded system.

Example usage:
#include <Lib/BSP/DigitalPeripheral/DigitalPeripheralBase.hpp>
using namespace Lib::BSP;
// Define mapping fields for inputs and outputs
constexpr DigitalPeripheralBase::MapField inputMapping[] = {
{0, 0x01}, {0, 0x02}, {1, 0x01}
};
constexpr DigitalPeripheralBase::MapField outputMapping[] = {
{0, 0x01}, {0, 0x02}, {1, 0x01}
};
// SPI and GPIO handles (provided by HAL)
extern SPI_HandleTypeDef hspi1;
extern PinHandle nssInputPin;
extern PinHandle nssOutputPin;
extern PinHandle oeOutputPin;
int main() {
DigitalPeripheralBase::Config config = {
{3, &hspi1, nssInputPin, inputMapping},
{3, &hspi1, nssOutputPin, oeOutputPin, outputMapping}
};
DigitalPeripheralBase peripheral(config);
peripheral.inputs.onValueChanged = [](const uint64_t & value) {
// Handle input value change
};
peripheral.outputs.onStatusChanged = [](const uint64_t & status) {
// Handle output status change
};
peripheral.outputs.onSpiError = []() {
// Handle SPI error
};
peripheral.setup();
while (true) {
// Application main loop
RTOS::Task::delay(100);
}
return 0;
}
