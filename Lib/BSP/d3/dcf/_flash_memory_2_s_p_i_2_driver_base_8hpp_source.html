1/** Base class for SPI memory drivers.
2 * Provides initialization, open/close, and abstract read/write/erase methods for SPI memory devices.
3 * @file DriverBase.hpp
4 */
5#pragma once
6
7#include "Lib/HAL/GPIO.hpp"
8#include "Lib/HAL/SPI.hpp"
9
10namespace Lib::BSP::FMEM::SPI {
11
12 /** Base class for SPI memory drivers.
13 * Defines interface and common members for SPI memory device drivers.
14 */
15 class DriverBase {
16 public:
17 /** SPI driver configuration.
18 * Holds SPI handle and NSS (chip select) pin.
19 */
20 struct Config {
21 Lib::HAL::SpiPort::Handle spiHandle; /**< SPI handle. */
22 PinHandle nssPin; /**< NSS (chip select) pin. */
23 };
24
25 /** SPI memory information.
26 * Contains chip ID, manufacturer, capacity, page/sector sizes, and counts.
27 */
28 struct Info {
29 uint32_t chipId{0}; /**< Chip ID. */
30 uint32_t manufacturerId{0}; /**< Manufacturer ID. */
31 uint32_t capacityBytes{0}; /**< Memory capacity (bytes). */
32 uint32_t pageCount{0}; /**< Number of pages. */
33 size_t pageSize{0}; /**< Page size (bytes). */
34 size_t sectorSize{0}; /**< Sector size (bytes). */
35 size_t sectorCount{0}; /**< Number of sectors. */
36 size_t pagesPerSector{0}; /**< Pages per sector. */
37 };
38
39 /** Driver status codes.
40 * Indicates driver state.
41 */
42 enum class Status {
43 NO_INIT, /**< Not initialized. */
44 READY, /**< Ready. */
45 POWER_OFF, /**< Powered off. */
46 ERRORS /**< Error state. */
47 };
48
49 /** Constructor of DriverBase.
50 * @param config SPI driver configuration.
51 */
52 explicit DriverBase(const Config & config)
53 : spiPort{config.spiHandle}, nss{config.nssPin}, info{}, status{Status::NO_INIT} {}
54
55 /** Destructor of DriverBase. */
56 virtual ~DriverBase() = default;
57
58 /** Opens the SPI driver.
59 * @return `true` if opened successfully, `false` otherwise.
60 */
61 virtual bool open() {
62 nss.write(1);
63 spiPort.init();
64 return true;
65 }
66
67 /** Closes the SPI driver.
68 * @return `true` if closed successfully, `false` otherwise.
69 */
70 virtual bool close() {
71 spiPort.deinit();
72 nss.write(1);
73 return true;
74 }
75
76 /** Reads data from SPI memory.
77 * @param address Source address.
78 * @param destination Destination buffer.
79 * @param size Number of bytes to read.
80 * @return `true` if successful, `false` otherwise.
81 */
82 virtual bool read(uint32_t address, void * destination, size_t size) = 0;
83
84 /** Writes data to SPI memory.
85 * @param address Destination address.
86 * @param source Source buffer.
87 * @param size Number of bytes to write.
88 * @return `true` if successful, `false` otherwise.
89 */
90 virtual bool write(uint32_t address, const void * source, size_t size) = 0;
91
92 /** Erases the entire SPI memory chip.
93 * @return `true` if successful, `false` otherwise.
94 */
95 virtual bool eraseChip() = 0;
96
97 /** Erases a sector of SPI memory.
98 * @param address Sector address.
99 * @return `true` if successful, `false` otherwise.
100 */
101 virtual bool eraseSector(uint32_t address) = 0;
102
103 /** Gets SPI memory information.
104 * @return Reference to Info structure.
105 */
106 const inline Info & getInfo() const { return info; }
107
108 protected:
109 Lib::HAL::SpiPort spiPort; /**< SPI port. */
110 PinOutput nss; /**< NSS (chip select) pin. */
111 Info info; /**< SPI memory information. */
112 Status status; /**< Driver status. */
113 };
114
115} // namespace Lib::BSP::FMEM::SPI
