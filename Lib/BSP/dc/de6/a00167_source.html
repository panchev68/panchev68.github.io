218 // Raw chip info struct
219 __packed class RawChipInfo
220 {
221 public:
222
223 uint8_t chipManufacturer = 0;
224 uint8_t memoryInterfaceType = 0;
225 uint8_t memoryDensity = 0;
226 };
227
228 static W25Q128JVSIQ::Info info;
229 RawChipInfo raw_info;
230
231 if (info.memoryCapacity != 0) return info; // already read
232
233 select ();
234
235 if (sendCommand (Register::READ_JEDEC_ID) == false) goto exit;
236 if (receive (reinterpret_cast<uint8_t *> (&raw_info), sizeof (RawChipInfo)) == false) goto exit;
237
238 info.chipManufacturer = raw_info.chipManufacturer;
239
240 if (raw_info.memoryDensity == 0x18)
241 {
242 info.memoryCapacity = 128 * (1024 * 1024 / 8); // in bytes
243 info.numberOfPages = 65536; // W25Q128JV area is organized into 65536 programmable pages of 256 bytes each.
244 info.pageSize = 256;
245 info.sectorSize = 4096;
246 info.sectorCount = info.memoryCapacity / info.sectorSize; // 4096
247 info.pagePersector = info.sectorSize / 256; // 16
248 }
249 // else if (raw_chip_info.MemoryDensity == 0x19)
250 // {
251 // info.MemoryCapacity = 256 MB;
252 // }
253 else
254 {
255 assert (0 && "Unsupported chip");
256 info.memoryCapacity = 0;
257 }
258 exit:
259 deselect ();
260 return info;

263 // ------------ Waiting for writing to complete ---------------
264
268 bool waitingForWritingToComplete (uint32_t timeout = MAX_WRITE_TIME)
269 {
270 uint8_t statusReg;
271 auto endTicks = HAL_GetTick () + timeout;
272
273 if (readStatusRegister (statusReg) == false)
274 {
275 assert (0);
276 return false;
277 }
278 while (HAL_GetTick () < endTicks)
279 {
280 if (readStatusRegister (statusReg) == false)
281 {
282 assert (0);
283 return false;
284 }
285 if ((statusReg & 0x01) == 0) return true; // Bit 0 - Erase/Write in progress Busy status only
286 }
287 return false;
288 }

486
492 bool read (uint32_t address, uint8_t * buffer, size_t len)
493 {
494 assert (buffer);
495 assert (len);
496 assert ((address + len) < _impl.getChipInfo ().memoryCapacity);
497 bool result = false;
498
499 if (_impl.isPresent == false) return false;
500
501 while (len > 0)
502 {
503 auto rlen = std::min (len, BLOCK_READ_SIZE);
504 _impl.select ();
505 if (_impl.sendCommandA24 (Register::READ_DATA, address) == false)
506 {
507 assert (0);
508 goto exit;
509 }
510 if (_impl.receive (buffer, rlen, 200) == false)
511 {
512 assert (0);
513 goto exit;
514 }
515
516 address += rlen;
517 buffer += rlen;
518 len -= rlen;
519 _impl.deselect ();
520 }
521 result = true;
522 exit:
523 _impl.deselect ();
524 return result;
525 }

532 bool readsector (uint8_t * buffer, uint32_t sector, uint32_t count)
533 {
534 bool result = false;
535 assert (buffer);
536 assert (count);
537

541 {

543 if (read (address, buffer, _impl.getChipInfo ().sectorSize) == false)
544 {

549 buffer += _impl.getChipInfo ().sectorSize;

551 result = true;
552 exit:

556 // ---------------------- Write functions -------------------------
557
563 bool write (uint32_t address, const uint8_t * buffer, size_t len)
564 {
565 bool result = false;
566 assert (buffer);
567 assert ((address + len) < _impl.getChipInfo ().memoryCapacity);
568
569 uint32_t page_start = address & 0xffff00UL;
570 uint32_t page_offs = address & 0xffUL;
571 uint8_t * data_ptr = (uint8_t *) buffer;
572
573 if (_impl.isPresent == false) return false;
574
575 while (len > 0)
576 {
577 auto write_addr = page_start + page_offs;
578 auto block_size = std::min<size_t> (len, (page_offs));
579
580 if (_impl.writeEnable () == false)
581 {
582 assert (0);
583 goto exit;
584 }
585 _impl.select ();
586
587 if (_impl.sendCommandA24 (Register::PAGE_PROGRAM, write_addr) == false)
588 {
589 assert (0);
590 goto exit;
591 }
592 if (_impl.send (data_ptr, block_size) == false)
593 {
594 assert (0);
595 goto exit;
596 }
597 _impl.deselect ();
598
599 if (_impl.waitingForWritingToComplete () == false)
600 {
601 assert (0);
602 goto exit;
603 }
604
605 page_start += 0x100; // next page
606 data_ptr += block_size;
607 write_addr += block_size;
608 page_offs = 0;
609 len -= block_size;
610 }
611 result = true;
612
613 exit:
614 if (_impl.writeDisable () == false)
615 {
616 assert (0);
617 result = false;
618 }
619
620 return result;
621 }

628 bool writeSector (const uint8_t * buffer, uint32_t sector, uint32_t count, bool hasSectorErase = true)
629 {
630 auto & chip_info = _impl.getChipInfo ();
631
632 auto data_ptr = const_cast<uint8_t *> (buffer);
633
634 assert (buffer);
635 assert (sector < chip_info.sectorCount);
636
637 if (_impl.isPresent == false) return false;
638
639 for (auto i = 0; i < count; i++)
640 {
641 auto address = (sector + i) * chip_info.sectorSize; // start sector address
642
643 if (hasSectorErase)
644 {
645 if (eraseSector (address) == false)
646 {
647 assert (0);
648 return false;
649 }
650 }
651
652 for (auto i = 0; i < chip_info.pagePersector; i++)
653 {
654 if (writePage (data_ptr, address) == false)
655 {
656 assert (0);
657 return false;
658 }
659
660 address += chip_info.pageSize;
661 data_ptr += chip_info.pageSize;
662 }
663 }
664 return true;
665 }
