307 const W25Q128JVSIQ::Info & getChipInfo ()
308 {
309 __packed struct RawChipInfo
310 {
311 uint8_t chipManufacturer = 0; // Chip Manufacturer
312 uint8_t memoryInterfaceType = 0; // Memory Interface Type
313 uint8_t memoryDensity = 0; // Memory Density
314 };
315
316 static W25Q128JVSIQ::Info info; // Flash Memory Info
317 RawChipInfo rawInfo; // Raw Chip Info
318
319 // If the memory capacity is already set, return the info
320 if (info.memoryCapacity != 0) return info;
321
322 // Read the JEDEC ID from the chip
323
324 // Select the device
325 select ();
326 // Send the READ_JEDEC_ID command
327 if (sendCommand (Register::READ_JEDEC_ID) == false) goto exit;
328 // Receive the raw chip info
329 if (receive (reinterpret_cast<uint8_t *> (&rawInfo), sizeof (RawChipInfo)) == false) goto exit;
330
331 // Populate the chip info
332 info.chipManufacturer = rawInfo.chipManufacturer;
333 // Check the memory density
334 if (rawInfo.memoryDensity == 0x18)
335 {
336 // Set the memory capacity to 128Mb
337 info.memoryCapacity = 128 * (1024 * 1024 / 8);
338 info.numberOfPages = 65536;
339 info.pageSize = 256;
340 info.sectorSize = 4096;
341 info.sectorCount = info.memoryCapacity / info.sectorSize;
342 info.pagePersector = info.sectorSize / 256;
343 }
344 else
345 {
346 // error: Unsupported chip
347
348 assert (0 && "Unsupported chip");
349 info.memoryCapacity = 0;
350 }
351 exit:
352 // Deselect the device
353 deselect ();
354 // Return the chip info
355 return info;
356 }

778 bool write (uint32_t address, const uint8_t * buffer, size_t len)
779 {
780 bool result = false;
781 assert (buffer);
782 assert ((address + len) < _impl.getChipInfo ().memoryCapacity);
783
784 uint32_t page_start = address & 0xffff00UL;
785 uint32_t page_offs = address & 0xffUL;
786 uint8_t * data_ptr = (uint8_t *) buffer;
787
788 if (_impl.isPresent == false) return false;
789
790 while (len > 0)
791 {
792 auto write_addr = page_start + page_offs;
793 auto block_size = std::min<size_t> (len, (page_offs));
794
795 if (_impl.writeEnable () == false)
796 {
797 assert (0);
798 goto exit;
799 }
800 _impl.select ();
801
802 if (_impl.sendCommandA24 (Register::PAGE_PROGRAM, write_addr) == false)
803 {
804 assert (0);
805 goto exit;
806 }
807 if (_impl.send (data_ptr, block_size) == false)
808 {
809 assert (0);
810 goto exit;
811 }
812 _impl.deselect ();
813
814 if (_impl.waitingForWritingToComplete () == false)
815 {
816 assert (0);
817 goto exit;
818 }
819
820 page_start += 0x100;
821 data_ptr += block_size;
822 write_addr += block_size;
823 page_offs = 0;
824 len -= block_size;
825 }
826 result = true;
827
828 exit:
829 if (_impl.writeDisable () == false)
830 {
831 assert (0);
832 result = false;
833 }
