4 * The CanServiceBase class enables robust CAN bus communication, diagnostics tracking, and extensibility for custom

19/** Base class for CAN bus services.
20 * Provides start/stop, send/receive, diagnostics, and event handling for CAN messages.
21 */
22class CanServiceBase : public CanPortBase {
23public:
24 /** Diagnostic info for CAN bus communication.
25 * Holds counters and error codes for CAN bus communication.
26 */
27 struct Diagnostic {
28 uint32_t receivedMessages{0}; /*!< Number of messages received. */
29 uint32_t sentMessages{0}; /*!< Number of messages sent. */
30 uint32_t txMailboxAbortCounter{0}; /*!< Transmit mailbox aborts. */
31 uint32_t rxFifoFullCounter{0}; /*!< Receive FIFO full events. */
32 uint32_t errorCounter{0}; /*!< Error count. */
33 uint32_t errorCode{0}; /*!< Last error code. */
34 };
35
36 /** Constructor of CanServiceBase class.
37 * @param handle CAN peripheral handle.
38 */
39 explicit CanServiceBase(CAN_HandleTypeDef * handle) : CanPortBase(handle) {}
40
41 /** Virtual destructor of CanServiceBase class. */
42 virtual ~CanServiceBase() = default;
43
44 /** Starts the service with the given identifier.
45 * Initializes filter, opens port, and starts the task.
46 * @param serviceId Service identifier.
47 * @return `true` if started successfully, `false` otherwise.
48 */
49 virtual bool start(uint8_t serviceId) {
50 id = serviceId;
51 CanFilter filter{serviceId};
52 filter.setId(0);
53 filter.setMask(0);
54 filter.setBank(0);
55 if (!queueRxMessage.create() || !CanPortBase::setFilter(filter) || !CanPortBase::open()) {
56 assert(!"CanServiceBase initialization failed");
57 return false;
58 }
59 return task.start([this] { taskFn(); });
60 }
61
62 /** Stops the service.
63 * Stops the task and closes the port.
64 * @return `true` if stopped successfully, `false` otherwise.
65 */
66 virtual bool stop() {
67 task.stop();
68 return CanPortBase::close();
69 }
70
71 /** Receives a message from the queue.
72 * @param[out] message Reference to received message.
73 * @param timeout Timeout in milliseconds.
74 * @return `true` if received, `false` otherwise.
75 */
76 virtual bool receive(CanRxMessage & message, uint32_t timeout) { return queueRxMessage.receive(message, timeout); }
77
78 /** Sends a message with the given identifier.
79 * @param messageId Message identifier.
80 * @param timeout Timeout in milliseconds. Default is SEND_TIMEOUT.
81 * @return `true` if sent, `false` otherwise.
82 */
83 virtual bool send(uint8_t messageId, uint32_t timeout = SEND_TIMEOUT) {
84 return send(messageId, nullptr, 0, timeout);
85 }
86
87 /** Sends a message with data.
88 * @param messageId Message identifier.
89 * @param data Pointer to data buffer.
90 * @param length Data length in bytes.
91 * @param timeout Timeout in milliseconds. Default is SEND_TIMEOUT.
92 * @return `true` if sent, `false` otherwise.
93 */
94 virtual bool send(uint8_t messageId, const void * data, size_t length, uint32_t timeout = SEND_TIMEOUT) override {
95 if (CanPortBase::send(messageId, data, length, timeout)) {
96 ++diagnostic.sentMessages;
97 return true;
98 }
99 ++diagnostic.errorCounter;
100 return false;
101 }
102
103 /** Sends a 64-bit value.
104 * @param messageId Message identifier.
105 * @param value 64-bit value to send.
106 * @param timeout Timeout in milliseconds. Default is SEND_TIMEOUT.
107 * @return `true` if sent, `false` otherwise.
108 */
109 virtual bool send(uint8_t messageId, uint64_t value, uint32_t timeout = SEND_TIMEOUT) {
110 return send(messageId, &value, sizeof(uint64_t), timeout);
111 }
112
113 /** Gets diagnostic information.
114 * @return Reference to Diagnostic struct.
115 */
116 const Diagnostic & getDiagnostic() const { return diagnostic; }
117
118 /** Handles a received message event.
119 * Increments received counter and sends message to queue.
120 * @param message Reference to received CAN message.
121 */
122 virtual void handleMessageReceivedEvent(const CanRxMessage & message) override {
123 ++diagnostic.receivedMessages;
124 queueRxMessage.send(message, 1);
125 }
126
127 /** Handles an error event.
128 * Increments error counter and updates error code.
129 * @param errorCode Error code value.
130 */
131 virtual void handleErrorEvent(uint32_t errorCode) override {
132 ++diagnostic.errorCounter;
133 diagnostic.errorCode = errorCode;
134 }
135
136 /** Task function (pure virtual).
137 * @note Must be implemented by derived classes to provide service task behavior.
138 */
139 virtual void taskFn() = 0;
140
141protected:
142 static constexpr uint32_t SEND_TIMEOUT{50}; /*!< Send timeout (ms). */
143 static constexpr uint32_t RECEIVE_TIMEOUT{50}; /*!< Receive timeout (ms). */
144 uint8_t id{0}; /*!< Service identifier. */
145 RTOS::Task task; /*!< FreeRTOS task. */
146 Diagnostic diagnostic; /*!< Diagnostic info. */
147 RTOS::Queue<CanRxMessage, 8> queueRxMessage; /*!< Rx message queue. */
148};
