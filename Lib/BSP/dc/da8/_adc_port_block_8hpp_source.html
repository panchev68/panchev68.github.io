19/*! Specialization of the Port class for BLOCK mode.
20 * Implements ADC port operations for BLOCK driver mode.
21 */
22template <> class Port<DriverMode::BLOCK> : public AdcPortBase {
23public:
24 /*! Deleted default constructor of the Port class. */
25 Port() = delete;
26
27 /*! Constructor of the Port class.
28 * @param adcHandle Pointer to the ADC handle.
29 * @param channelConfig Reference to the ADC channel configuration.
30 */
31 explicit Port(ADC_HandleTypeDef * adcHandle, const AdcChannel::Config & channelConfig)
32 : AdcPortBase(adcHandle, channelConfig) {}
33
34 /*! Destructor of the Port class. */
35 ~Port() override = default;
36
37 /*! Setup the ADC port in BLOCK mode. */
38 void setup() override {
39 if (HAL_ADC_Start(getHandle()) != HAL_OK) assert(false);
40 }
41
42 /*! Release the ADC port in BLOCK mode. */
43 void release() override {
44 if (HAL_ADC_Stop(getHandle()) != HAL_OK) assert(false);
45 }
46
47 /*! Start the ADC port in BLOCK mode.
48 * @return `true` if start was successful, `false` otherwise.
49 */
50 bool start() override {
51 if (!AdcChannel::setup(getHandle())) {
52 assert(false);
53 return false;
54 }
55 if (HAL_ADC_Start(getHandle()) != HAL_OK) {
56 assert(false);
57 return false;
58 }
59 return true;
60 }
61
62 /*! Stop the ADC port in BLOCK mode.
63 * @return `true` if stop was successful, `false` otherwise.
64 */
65 bool stop() override {
66 if (HAL_ADC_Stop(getHandle()) != HAL_OK) {
67 assert(false);
68 return false;
69 }
70 return true;
71 }
72
73 /*! Read ADC value in BLOCK mode.
74 * @return The ADC value.
75 */
76 uint16_t read() const override { return HAL_ADC_GetValue(getHandle()); }
77};
