25template <typename T_DEVICE, typename T_HANDLE>
26class DeviceManagerBase {

34 /** Adds a device to the manager.
35 * @param device Pointer to the device.
36 * @return `true` if the device was added, `false` otherwise.
37 */
38 virtual bool addDevice(T_DEVICE * device) {
39 if (!device) return false;
40 T_HANDLE * handle = device->getHandle();
41 if (!handle) return false;
42 return deviceMap.emplace(handle, device).second;
43 }
44
45 /** Removes a device from the manager.
46 * @param device Pointer to the device.
47 * @return `true` if the device was removed, `false` otherwise.
48 */
49 virtual bool removeDevice(T_DEVICE * device) {
50 if (!device) return false;
51 T_HANDLE * handle = device->getHandle();
52 if (!handle) return false;
53 auto it = deviceMap.find(handle);
54 if (it == deviceMap.end() || it->second != device) return false;
55 deviceMap.erase(it);
56 return true;
57 }
58
59 /** Finds a device by its handle.
60 * @param handle Pointer to the handle.
61 * @return Pointer to the device if found, `nullptr` otherwise.
62 */
63 virtual T_DEVICE * findDevice(T_HANDLE * handle) const {
64 auto it = deviceMap.find(handle);
65 return it != deviceMap.end() ? it->second : nullptr;
66 }
67
68private:
69 std::unordered_map<T_HANDLE *, T_DEVICE *> deviceMap; /*!< Map of handles to devices. */
70};
