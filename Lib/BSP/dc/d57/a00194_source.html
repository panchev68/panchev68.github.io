229 bool start () {
230 if (queueButtonPressed.create () == false) {
231 assert (0);
232 return false;
233 }
234 if (FreeRTOS::Task::run ("TouchScreen", [this] { taskFn (); }, 128, 0) == nullptr) {
235 assert (0);
236 return false;
237 }
238 return true;
239 }
240
259 void taskFn () {
260 auto extiFunction = [this] (const Lib::HAL::GPIO::EXTIx * exti) {
261 queueButtonPressed.overwriteFromISR (true);
262 };
263
264 // Reads the internal a/d convertor
265 auto readAdc = [this] (uint8_t cmd) {
266 auto writeByte = [this] (uint8_t data) -> void {
267 for (auto i = 0; i < 8; i++) {
268 si.write ((data & 0x80) != 0);
269 __NOP ();
270 data <<= 1;
271 sck.write (0);
272 __NOP ();
273 sck.write (1);
274 __NOP ();
275 }
276 };
277
278 uint32_t num = 0;
279 si.write (0);
280 nss.write (0);
281 writeByte (cmd);
282
283 sck.write (0);
284 __NOP ();
285 sck.write (1);
286 __NOP ();
287
288 for (auto i = 0; i < 16; i++) {
289 num <<= 1;
290 sck.write (0);
291 __NOP ();
292 sck.write (1);
293 __NOP ();
294 if (so.read () == 1) num++;
295 }
296 num >>= 4;
297 nss.write (1);
298
299 return static_cast<uint16_t> (num);
300 };
301
302 nirq.bind (extiFunction);
303 sck.write (1);
304 nss.write (1);
305
306 // Task loop...
307
308 while (1) {
309 isPressed = false;
310
311 // wait for touch screen to be pressed ...
312 bool pinState = false;
313 while (pinState == false) queueButtonPressed.receive (pinState, portMAX_DELAY);
314
315 FreeRTOS::Task::delay (0);
316
317 intergratorX.reset ();
318 intergratorY.reset ();
319 position.X = -1;
320 position.Y = -1;
321
322 timeoutPinDebounce.set (parameters.debounceDuration);
323
324 while (timeoutPinDebounce.check () == false) {
325 if (nirq.read () == true) timeoutPinDebounce.set (parameters.debounceDuration);
326 vTaskDelay (1);
327 }
328
329 while (1) {
330 if (nirq.read () == true) break;
331
332 // Interpolates coordinates...
333
334 rawPosition.X = intergratorX.process (readAdc (CMD_READ_X));
335 rawPosition.Y = intergratorY.process (readAdc (CMD_READ_Y));
336
337 if (intergratorX.isFull () == false) {
338 vTaskDelay (1);
339 continue;
340 }
341
342 // Reads and averages values ​​from the driver..
343
344 rawPosition.X = std::max<uint32_t> (parameters.xMin, rawPosition.X);
345 rawPosition.Y = std::max<uint32_t> (parameters.yMin, rawPosition.Y);
346
347 rawPosition.X = std::min<uint32_t> (rawPosition.X, parameters.xMax);
348 rawPosition.Y = std::min<uint32_t> (rawPosition.Y, parameters.yMax);
349
350 rawPosition.X -= parameters.xMin;
351 rawPosition.Y -= parameters.yMin;
352
353 position.X = static_cast<int32_t> (rawPosition.X / scaleFaconstructorX);
354 position.Y = static_cast<int32_t> (rawPosition.Y / scaleFaconstructorY);
355
356 // The touch screen is pressed and ready to read data.
357
358 isPressed = true;
359
360 switch (rotation) {
361 case 0:
362 // position = position;
363 break;
364 case 1:
365 position.X = parameters.dispWidth - position.X;
366 position.Y = position.Y;
367 break;
368 case 2: // flip X
369 position.X = parameters.dispWidth - position.X;
370 position.Y = parameters.dispHeight - position.Y;
371 break;
372 case 3: // flip Y
373 position.X = position.X;
374 position.Y = parameters.dispHeight - position.Y;
375 break;
376 case 4: // flip X,Y
377 position.X = parameters.dispWidth - position.X;
378 position.Y = parameters.dispHeight - position.Y;
379 break;
380 default:
381 assert (0);
382 break;
383 }
384 vTaskDelay (40);
385 }
386 }
387 }
