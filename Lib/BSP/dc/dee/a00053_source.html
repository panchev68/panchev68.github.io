104 bool read(uint32_t address, uint8_t * destination, size_t size) {
105 assert(destination && size && (address + size) < getChipInfo().memoryCapacity);
106 if (!isPresent()) return false;
107 while (size > 0) {
108 auto readLength = (size < BLOCK_READ_SIZE) ? size : BLOCK_READ_SIZE;
109 select();
110 bool isSuccess = sendCommandA24(Register::READ_DATA, address) &&
111 receive(destination, readLength, SPI_TIMEOUT);
112 deselect();
113 if (!isSuccess) return false;
114 address += readLength;
115 destination += readLength;
116 size -= readLength;
117 }
118 return true;
119 }

145 bool write(uint32_t address, const uint8_t * source, size_t size) {
146 assert(source && (address + size) < getChipInfo().memoryCapacity);
147 if (!isPresent()) return false;
148 const auto & chipInfo = getChipInfo();
149 size_t pageSize = chipInfo.pageSize ? chipInfo.pageSize : 256;
150 while (size > 0) {
151 size_t pageOffset = address % pageSize;
152 size_t writeLength = pageSize - pageOffset;
153 if (writeLength > size) writeLength = size;
154 if (!writeEnable() || !sendCommandA24(Register::PAGE_PROGRAM, address) ||
155 !send(source, writeLength) || !waitForWriteComplete()) return false;
156 address += writeLength;
157 source += writeLength;
158 size -= writeLength;
159 }
160 return writeDisable();
161 }

170 bool writeSector(const uint8_t * source, uint32_t sectorIndex, uint32_t sectorCount, bool eraseBeforeWrite = true) {
171 const auto & chipInfo = getChipInfo();
172 assert(source && sectorIndex < chipInfo.sectorCount);
173 if (!isPresent()) return false;
174 auto dataPtr = source;
175 for (uint32_t i = 0; i < sectorCount; ++i) {
176 auto address = (sectorIndex + i) * chipInfo.sectorSize;
177 if (eraseBeforeWrite && !eraseSector(address)) return false;
178 for (size_t j = 0; j < chipInfo.pagePerSector; ++j) {
179 if (!writePage(dataPtr, address)) return false;
180 address += chipInfo.pageSize;
181 dataPtr += chipInfo.pageSize;
182 }
183 }
184 return true;
185 }

316 const Info & getChipInfo() {
317 static Info info;
318 if (info.memoryCapacity) return info;
319
320 __packed struct RawInfo {
321 uint8_t manufacturer;
322 uint8_t interfaceType;
323 uint8_t density;
324 } rawInfo = {};
325
326 select();
327 bool isSuccess = sendCommand(Register::READ_JEDEC_ID) &&
328 receive(reinterpret_cast<uint8_t *>(&rawInfo), sizeof(rawInfo));
329 deselect();
330 if (!isSuccess) return info;
331 info.chipManufacturer = rawInfo.manufacturer;
332 if (rawInfo.density == 0x18) {
333 constexpr uint32_t MEMORY_CAPACITY = 128 * (1024 * 1024 / 8);
334 constexpr size_t SECTOR_SIZE = 4096;
335 constexpr size_t PAGE_SIZE = 256;
336 info = {
337 rawInfo.manufacturer,
338 MEMORY_CAPACITY,
339 PAGE_SIZE,
340 SECTOR_SIZE,
341 MEMORY_CAPACITY / SECTOR_SIZE,
342 SECTOR_SIZE / PAGE_SIZE};
343 }
344 return info;
345 }

351 inline bool waitForWriteComplete(uint32_t timeoutMs = MAX_WRITE_TIME) {
352 uint8_t statusReg = 0;
353 const auto endTicks = HAL_GetTick() + timeoutMs;
354 do {
355 if (!readStatusRegister(statusReg)) return false;
356 if ((statusReg & 0x01) == 0) return true;
357 } while (HAL_GetTick() < endTicks);
358 return false;
359 }

384 if (address >= getChipInfo().memoryCapacity) return false;

386 static_cast<uint8_t>(address >> 8), static_cast<uint8_t>(address & 0xff)};
