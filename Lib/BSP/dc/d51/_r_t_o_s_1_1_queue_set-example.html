RTOS::QueueSet

Example usage of QueueSet with FreeRTOS queue sets.

This example demonstrates how to create a QueueSet, add queues to it, and select a ready member.

Example usage:

#include "Lib/RTOS/QueueSet.hpp"
#include "FreeRTOS.h"
#include "queue.h"
void queueSetExample() {
constexpr UBaseType_t queueLength = 5;
constexpr UBaseType_t itemSize = sizeof(int);
RTOS::QueueSet queueSet = RTOS::createQueueSet(2);
QueueHandle_t queue1 = xQueueCreate(queueLength, itemSize);
QueueHandle_t queue2 = xQueueCreate(queueLength, itemSize);
queueSet.addToSet(queue1);
queueSet.addToSet(queue2);
int value = 42;
xQueueSend(queue1, &value, 0);
RTOS::QueueSet::Member readyMember = queueSet.selectFromSetNow();
if (readyMember == queue1) {
int received = 0;
xQueueReceive(queue1, &received, 0);
// Use received value
}
queueSet.removeFromSet(queue1);
queueSet.removeFromSet(queue2);
vQueueDelete(queue1);
vQueueDelete(queue2);
}
