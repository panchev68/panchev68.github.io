108 inline bool Interface::readWrite (const void * src, size_t src_len, void * dst, size_t dst_len)
109 {
110 bool res = false;
111 auto size = src_len + dst_len;
112
113 uint8_t * wbuf = new uint8_t [size];
114 uint8_t * rbuf = new uint8_t [size];
115
116 memset (wbuf, 0xff, size);
117
118 if ((src != nullptr) && (src_len > 0)) memcpy (wbuf, src, src_len);
119
120 res = spiPort.sendReceive (wbuf, rbuf, size, 100);
121
122 if (res == true)
123 {
124 if ((dst != nullptr) && (dst_len > 0)) memcpy (dst, (rbuf + src_len), dst_len);
125 }
126 else assert (0);
127 taskYIELD ();
128 delete wbuf;
129 delete rbuf;
130 return res;
131 }
