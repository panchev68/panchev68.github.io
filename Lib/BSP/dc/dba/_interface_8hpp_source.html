21/** SPI interface for LAN9252 EtherCAT Slave Controller.
22 * Encapsulates SPI and NSS pin operations with mutex protection.
23 */
24class Interface {
25public:
26 /** SPI configuration for LAN9252 interface.
27 * Holds SPI handle and NSS pin.

29 struct Config {

34 static constexpr uint32_t SPI_TIMEOUT = 100; /*!< SPI timeout in ms. */
35
36 /** Constructor of Interface class.
37 * @param config SPI and NSS pin configuration.
38 */
39 explicit Interface(const Config & config) : spi(config.spiHandle), pinNss(config.nssPin) {}
40
41 /** Initializes SPI and mutex. */
42 void open() {
43 spi.init();
44 if (!mutexBlock.create()) assert(0);
45 }
46
47 /** Deinitializes SPI and destroys mutex. */
48 void close() {
49 spi.deinit();
50 mutexBlock.destroy();
51 }
52
53 /** Selects device (NSS low) and locks mutex. */
54 void select() {
55 mutexBlock.take(portMAX_DELAY);
56 pinNss.write(0);
57 }
58
59 /** Deselects device (NSS high) and unlocks mutex. */
60 void deselect() {
61 pinNss.write(1);
62 mutexBlock.give();
63 }
64
65 /** SPI full-duplex transfer.
66 * @param source Source buffer.
67 * @param sourceLength Source length.
68 * @param destination Destination buffer.
69 * @param destinationLength Destination length.
70 * @return `true` if successful, `false` otherwise.
71 * @pre source and destination must be valid pointers if sourceLength/destinationLength > 0.
72 * @post Data is transferred; destination contains received bytes.
73 */
74 bool readWrite(const void * source, size_t sourceLength, void * destination, size_t destinationLength) {
75 size_t totalLength = sourceLength + destinationLength;
76 if (totalLength > MAX_BUFFER_SIZE) return false;
77 std::fill(tempTxBuffer.begin(), tempTxBuffer.begin() + totalLength, 0xFF);
78 if (source && sourceLength) std::copy(static_cast<const uint8_t *>(source), static_cast<const uint8_t *>(source) + sourceLength, tempTxBuffer.begin());
79 bool ok = spi.sendReceive(tempTxBuffer.data(), tempRxBuffer.data(), totalLength, SPI_TIMEOUT);
80 if (ok && destination && destinationLength) std::copy(tempRxBuffer.begin() + sourceLength, tempRxBuffer.begin() + totalLength, static_cast<uint8_t *>(destination));
81 return ok;
82 }
83
84 /** Writes data to device.
85 * @param source Source data pointer.
86 * @param length Data length.
87 * @return `true` if successful, `false` otherwise.
88 * @pre source must be valid pointer and length > 0.
89 * @post Data is written to device.
90 */
91 bool write(const void * source, size_t length) {
92 assert(source && length);
93 bool ok = spi.send(source, length, SPI_TIMEOUT);
94 return ok;
95 }
96
97 /** Writes single byte to device.
98 * @param value Byte to write.
99 * @return `true` if successful, `false` otherwise.
100 */
101 bool write(uint8_t value) { return write(&value, sizeof(value)); }
102
103 /** Reads data from device.
104 * @param destination Destination buffer.
105 * @param length Data length.
106 * @return `true` if successful, `false` otherwise.
107 * @pre destination must be valid pointer and length > 0.
108 * @post Data is read from device into destination.
109 */
110 bool read(void * destination, size_t length) {
111 assert(destination && length);
112 bool ok = spi.receive(destination, length, SPI_TIMEOUT);
113 return ok;
114 }
115
116 /** Reads single byte from device.
117 * @param value Reference to store byte.
118 * @return `true` if successful, `false` otherwise.
119 */
120 bool read(uint8_t & value) { return read(&value, sizeof(uint8_t)); }
121
122private:
123 static constexpr size_t MAX_BUFFER_SIZE = 256; /*!< Maximum buffer size for SPI transfers. */
124 Lib::HAL::SpiPort spi; /*!< SPI port. */
125 PinOutput pinNss; /*!< NSS pin. */
126 RTOS::Mutex mutexBlock; /*!< Mutex for resource access. */
127 std::array<uint8_t, MAX_BUFFER_SIZE> tempTxBuffer; /*!< Temporary TX buffer. */
128 std::array<uint8_t, MAX_BUFFER_SIZE> tempRxBuffer; /*!< Temporary RX buffer. */
129};
