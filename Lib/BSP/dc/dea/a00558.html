#include "BSP.hpp"
#include "i2c.h"
namespace BSP
{
Backlight backlight; // TFT Baclkight driver
Display tft; // TFT Driver
TouchScreenDrv touchScreen; // TouchScreen Driver
DigitalPeripheral digitalPeripheral; // Digital Peripheral Driver
Leds leds ({{PIN (LED_RDY)}, {PIN (LED_ERR)}, 200}); // Leds
Voltmeter voltmeter; // Voltmeter
Beeper beeper ({&htim1, TIM_CHANNEL_1}); // Beeper
Lib::BSP::EEPROM::I2C::M24C64W eeprom ({&hi2c1, 0xa0}); // I2C eeprom
void init ()
{
auto & driverManager = Lib::BSP::DriverManager<256>::getInstance ();
PinOutput (PIN (FMEM_NSS)).write (1); // Disable Flash Memory
digitalPeripheral.init (); // Initialize the Digital I/O Driver
voltmeter.init (); // Voltmeter init
tft.init (); // Initialize tft Driver
driverManager.bind (beeper); // register beeper driver to divers list
driverManager.bind (leds); // register beeper driver to divers list
driverManager.bind (backlight); // register backlight driver to device list
driverManager.init (); // initialize all drivers
#ifndef DISABLE_BEEP_AFTER_RESET
beeper.play (100);
#endif
}
} // namespace BSP
