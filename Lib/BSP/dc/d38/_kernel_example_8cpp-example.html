Example usage of the Kernel class for task management and timing.

int main() {
// Delay the current task for 1 second
RTOS::Kernel::delayMs(1000);
// Get current tick count
auto ticks = RTOS::Kernel::getTickCount();
// Get memory information
auto memoryInfo = RTOS::Kernel::getMemoryInfo();
// Use critical section
RTOS::Kernel::withCriticalSection([]() {
// Critical code here
});
return 0;
}
