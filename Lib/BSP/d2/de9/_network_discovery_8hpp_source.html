85 /** Starts network discovery and scans the wireless network.
86 * @return Reference to the vector of discovered nodes.
87 */
88 const std::vector<Node> & start() {
89 data.clear();
90 std::vector<uint8_t> result;
91 RTOS::Timeout timeout;
92 if (atCommand->send({"ND"})) {
93 timeout.set(discoveryTimeout);
94 while (!timeout.isExpired()) {
95 if (atCommand->receive(result, 1)) data.emplace_back(result.data());
96 }
97 } else assert(false);
98 return data;
99 }

113 /** Sends an AT command and receives the response.
114 * @param command AT command string.
115 * @param rxTimeout Receive timeout in milliseconds.
116 * @return Response as a vector of bytes.
117 */
118 std::vector<uint8_t> sendAtCommand(const std::string & command, uint32_t rxTimeout) {
119 std::vector<uint8_t> result;
120 if (!atCommand->send(command)) {
121 assert(false);
122 return result;
123 }
124 if (!atCommand->receive(result, rxTimeout)) assert(false);
125 return result;
126 }
