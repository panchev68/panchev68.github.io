42 /** Encode a 16-bit value using Hamming(21,16) code.
43 * @param value 16-bit value to encode.
44 * @return Encoded 21-bit Hamming code word as a 32-bit value.
45 */
46 uint32_t encode(uint16_t value) {
47 uint32_t encoded = 0;
48 for (size_t i = 0; i < DATA_BITS; ++i)
49 if (value & (1U << i)) encoded |= 1U << (dataPositions[i] - 1);
50 for (size_t i = 0; i < PARITY_BITS; ++i) {
51 const auto parityPos = parityPositions[i];
52 uint32_t parity = 0;
53 for (int j = 1; j <= CODE_BITS; ++j)
54 if ((j & parityPos) && (encoded & (1U << (j - 1)))) parity ^= 1;
55 if (parity) encoded |= 1U << (parityPos - 1);
56 }
57 encodedWord = std::bitset<CODE_BITS>(encoded);
58 return encoded;
59 }

61 /** Decode a 21-bit Hamming encoded value and correct a single-bit error if present.
62 * @param value 21-bit Hamming encoded value.
63 * @return Decoded 16-bit original data.
64 */
65 uint16_t decode(uint32_t value) const {
66 uint32_t errorPos = 0;
67 for (size_t i = 0; i < PARITY_BITS; ++i) {
68 const auto parityPos = parityPositions[i];
69 uint32_t parity = 0;
70 for (int j = 1; j <= CODE_BITS; ++j)
71 if ((j & parityPos) && (value & (1U << (j - 1)))) parity ^= 1;
72 if (parity) errorPos |= parityPos;
73 }
74 if (errorPos > 0 && errorPos <= CODE_BITS) value ^= (1U << (errorPos - 1));
75 uint16_t decoded = 0;
76 for (size_t i = 0; i < DATA_BITS; ++i)
77 if (value & (1U << (dataPositions[i] - 1))) decoded |= 1U << i;
78 return decoded;
79 }
