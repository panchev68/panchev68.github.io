22/** Linear interpolation class for embedded applications.
23 * Provides efficient linear interpolation between two points or within a lookup table.
24 * Supports both runtime and compile-time interpolation with type safety and overflow protection.
25 * Suitable for embedded and real-time systems requiring fast, reliable interpolation between
26 * variable data points, including lookup tables and direct value interpolation.
27 * @tparam T_VALUE Numeric type for values (typically float, double, or fixed-point types).
28 * @tparam T_INPUT Numeric type for input coordinates (typically same as T_VALUE).

31public:

35 /** Data point structure for interpolation table.
36 *
37 * Represents a single data point with input and output values for interpolation.
38 * Provides constructors for default and explicit initialization.
39 */
40 struct Point {
41 T_INPUT x; /*!< Input coordinate value. */
42 T_VALUE y; /*!< Output value corresponding to x. */
43
44 /** Default constructor of Point struct. */
45 constexpr Point() : x{}, y{} {}

47 /** Constructor of Point struct.
48 * @param inputX Input coordinate value.

109 /** Perform linear interpolation for a given input value.
110 * @param x Input value to interpolate.
111 * @return Interpolated output value.
112 */
113 T_VALUE interpolate(T_INPUT x) const {
114 assert(pointCount >= 2);
115 if (!isSorted) const_cast<LinearInterpolation *>(this)->sortPoints();
116
117 /* Handle boundary conditions */
118 if (x <= points[0].x) return points[0].y;
119 if (x >= points[pointCount - 1].x) return points[pointCount - 1].y;
120
121 /* Find the appropriate interval */
122 size_t index = findInterval(x);
123
124 /* Perform linear interpolation */
125 return interpolateBetween(points[index], points[index + 1], x);
126 }
127
128 /** Perform linear interpolation between two points.
129 * @param point1 First interpolation point.
130 * @param point2 Second interpolation point.
131 * @param x Input value to interpolate.
132 * @return Interpolated output value.
133 */
134 static constexpr T_VALUE interpolateBetween(const Point & point1, const Point & point2, T_INPUT x) {
135 assert(point1.x != point2.x);
136 const auto deltaX = point2.x - point1.x;
137 const auto deltaY = point2.y - point1.y;
138 const auto factor = (x - point1.x) / deltaX;
139 return point1.y + static_cast<T_VALUE>(factor * deltaY);
140 }
141
142 /** Perform simple linear interpolation between two values.
143 * @param x1 First input coordinate.
144 * @param y1 First output value.
145 * @param x2 Second input coordinate.
146 * @param y2 Second output value.
147 * @param x Input value to interpolate.
148 * @return Interpolated output value.
149 */
150 static constexpr T_VALUE interpolateSimple(T_INPUT x1, T_VALUE y1, T_INPUT x2, T_VALUE y2, T_INPUT x) {
151 return interpolateBetween(Point(x1, y1), Point(x2, y2), x);
152 }

154 /** Get a point from the interpolation table.
155 * @param index Index of the point to retrieve.
156 * @return Reference to the point at the specified index.
157 */
158 const Point & getPoint(size_t index) const {
159 assert(index < pointCount);
160 return points[index];
161 }

163 /** Sort points by x coordinate for efficient interpolation. */
164 void sortPoints() {
165 if (isSorted || pointCount <= 1) return;
166 /* Simple bubble sort - adequate for small embedded tables */
167 for (size_t i = 0; i < pointCount - 1; ++i)
168 for (size_t j = 0; j < pointCount - i - 1; ++j)
169 if (points[j].x > points[j + 1].x) {
170 const Point temp = points[j];
171 points[j] = points[j + 1];
172 points[j + 1] = temp;
173 }
174 isSorted = true;
175 }
176
177private:
178 /** Find the interval containing the input value.
179 * @param x Input value to locate.
180 * @return Index of the lower bound point.
181 */
182 size_t findInterval(T_INPUT x) const {
183 assert(pointCount >= 2);
184 assert(isSorted);
185 /* Binary search for efficiency with larger tables */
186 size_t left = 0, right = pointCount - 1;
187 while (right - left > 1) {
188 const size_t middle = left + (right - left) / 2;
189 if (points[middle].x <= x) left = middle;
190 else right = middle;
191 }
192 return left;
193 }
