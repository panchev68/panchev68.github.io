27/** USB serial port interface.
28 *
29 * Provides methods for USB serial communication, including open/close, send, reset,
30 * baud rate management, and event callbacks. Encapsulates UART port management, USB reset pin
31 * control, RTOS task handling, and message buffering for efficient and reliable USB serial data
32 * transfer.
33 */
34class SerialPort {
35public:
36 /** SerialPort configuration.
37 *
38 * Holds UART port handles and reset pin for SerialPort.
39 */
40 struct Config {
41 HAL::UARTBase & rxPort; /*!< UART port for receiving. */
42 HAL::UARTBase & txPort; /*!< UART port for transmitting. */
43 PinHandle pinNRST; /*!< Pin handle for USB reset. */
44 };

46 /** Constructor of the SerialPort class.
47 * @param config SerialPort configuration.
48 */
49 explicit SerialPort(const Config & config)
50 : rxPort(config.rxPort),
51 txPort(config.txPort),
52 pinNRST(config.pinNRST),
53 taskRx("USB_RX", 256),
54 taskTx("USB_TX", 256) {}

56 /** Virtual destructor of the SerialPort class. */
57 virtual ~SerialPort() = default;

59 /** Open the serial port.
60 * @return `true` if opened successfully, `false` otherwise.
61 */
62 virtual bool open() {
63 if (!messageBuffer.create(USB_MSG_BUF_SIZE)) return false;
64 rxPort.init();
65 txPort.init();
66 if (!taskRx.start([this] { receiveTask(); })) return false;
67 if (!taskTx.start([this] { transmitTask(); })) return false;
68 setResetPin(false);
69 vTaskDelay(50);
70 setResetPin(true);
71 return true;
72 }
73
74 /** Close the serial port. */
75 virtual void close() {
76 taskRx.stop();
77 taskTx.stop();
78 rxPort.deinit();
79 txPort.deinit();
80 messageBuffer.destroy();
81 setResetPin(false);
82 }
83
84 /** Set the reset pin to a value.
85 * @param value Pin value (`true` = high, `false` = low).
86 */
87 void setResetPin(bool value) { pinNRST.write(value); }
88
89 /** Send data through the serial port.
90 * @param data Data buffer pointer.
91 * @param length Data length.
92 * @return `true` if sent successfully, `false` otherwise.
93 */
94 bool send(const uint8_t * data, size_t length) {
95 if (!data) return false;
96 if (length >= USB_MSG_BUF_SIZE) return false;
97 return messageBuffer.sendFromISR(data, length) == length;
98 }
99
100 /** Set baud rate for the serial port.
101 * @param baudRate Baud rate.
102 */
103 void setBaudRate(uint32_t baudRate) {
104 rxPort.setBaudRate(baudRate);
105 txPort.setBaudRate(baudRate);
106 }

108 /** Callback after data is received.
109 * @param data Received data pointer.
110 * @param length Received data size.
111 */
112 virtual void postReceivedCallback(const uint8_t * data, size_t length) {}
113
114 /** Callback after transmit completes. */
115 virtual void postTransmitCallback() {}
116
117private:
118 RTOS::Task taskRx; /*!< RX task. */
119 RTOS::Task taskTx; /*!< TX task. */
120 HAL::UARTBase & rxPort; /*!< RX UART port. */
121 HAL::UARTBase & txPort; /*!< TX UART port. */
122 PinOutput pinNRST; /*!< USB reset pin. */
123 RTOS::MessageBuffer messageBuffer; /*!< Message buffer. */
124
125 /** RX task loop. */
126 void receiveTask() {
127 std::vector<uint8_t> buffer;
128 uint8_t value = 0;
129 for (;;) {
130 if (rxPort.receive(&value, sizeof(uint8_t), 1) == sizeof(uint8_t)) {
131 buffer.push_back(value);
132 } else if (!buffer.empty()) {
133 postReceivedCallback(buffer.data(), buffer.size());
134 buffer.clear();
135 }
136 }
137 }
138
139 /** TX task loop. */
140 void transmitTask() {
141 std::array<uint8_t, USB_MSG_BUF_SIZE> buffer{};
142 for (;;) {
143 auto length = messageBuffer.receive(buffer.data(), buffer.size(), portMAX_DELAY);
144 if (!txPort.send(buffer.data(), length, 100)) {
145 assert(false);
146 }
147 postTransmitCallback();
148 }
149 }
