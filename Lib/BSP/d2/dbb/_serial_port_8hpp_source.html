28 /** USB serial port interface.
29 *
30 * Provides methods for USB serial communication, including open/close, send, reset,
31 * baud rate management, and event callbacks. Encapsulates UART port management, USB reset pin
32 * control, RTOS task handling, and message buffering for efficient and reliable USB serial data
33 * transfer.
34 */
35 class SerialPort {
36 public:
37 /** SerialPort configuration.
38 *
39 * Holds UART port handles and reset pin for SerialPort.
40 */
41 struct Config {
42 HAL::UARTBase & rxPort; /*!< UART port for receiving. */
43 HAL::UARTBase & txPort; /*!< UART port for transmitting. */
44 PinHandle pinNRST; /*!< Pin handle for USB reset. */
45 };
46
47 /** Constructor of SerialPort class.
48 * @param config SerialPort configuration.
49 */
50 explicit SerialPort(const Config & config)
51 : rxPort(config.rxPort),
52 txPort(config.txPort),
53 pinNRST(config.pinNRST),
54 taskRx("USB_RX", 256),
55 taskTx("USB_TX", 256) {}
56
57 /** Virtual destructor of SerialPort class. */
58 virtual ~SerialPort() = default;
59
60 /** Open the serial port.
61 * @return `true` if opened successfully, `false` otherwise.
62 */
63 virtual bool open() {
64 if (!messageBuffer.create(USB_MSG_BUF_SIZE)) return false;
65 rxPort.init();
66 txPort.init();
67 if (!taskRx.start([this] { receiveTask(); })) return false;
68 if (!taskTx.start([this] { transmitTask(); })) return false;
69 resetPin(false);
70 vTaskDelay(50);
71 resetPin(true);
72 return true;
73 }
74
75 /** Close the serial port. */
76 virtual void close() {
77 taskRx.stop();
78 taskTx.stop();
79 rxPort.deinit();
80 txPort.deinit();
81 messageBuffer.destroy();
82 resetPin(false);
83 }
84
85 /** Reset the pin to a value.
86 * @param value Pin value (`true` = high, `false` = low).
87 */
88 void resetPin(bool value) { pinNRST.write(value); }
89
90 /** Send data through the serial port.
91 * @param data Data buffer pointer.
92 * @param length Data length.
93 * @return `true` if sent successfully, `false` otherwise.
94 */
95 bool send(const uint8_t * data, size_t length) {
96 if (!data) return false;
97 if (length >= USB_MSG_BUF_SIZE) return false;
98 return messageBuffer.sendFromISR(data, length) == length;
99 }
100
101 /** Set baud rate for the serial port.
102 * @param baud Baud rate.
103 */
104 void setBaudRate(uint32_t baud) {
105 rxPort.setBaudRate(baud);
106 txPort.setBaudRate(baud);
107 }
108
109 /** Callback after data is received.
110 * @param data Received data pointer.
111 * @param length Received data size.
112 */
113 virtual void postReceivedCallback(const uint8_t * data, size_t length) {}
114
115 /** Callback after transmit completes. */
116 virtual void postTransmitCallback() {}
117
118 private:
119 RTOS::Task taskRx; /*!< RX task. */
120 RTOS::Task taskTx; /*!< TX task. */
121 HAL::UARTBase & rxPort; /*!< RX UART port. */
122 HAL::UARTBase & txPort; /*!< TX UART port. */
123 PinOutput pinNRST; /*!< USB reset pin. */
124 RTOS::MessageBuffer messageBuffer; /*!< Message buffer. */
125
126 /** RX task loop. */
127 void receiveTask() {
128 std::vector<uint8_t> buffer;
129 uint8_t data = 0;
130 for (;;) {
131 if (rxPort.receive(&data, sizeof(uint8_t), 1) == sizeof(uint8_t)) {
132 buffer.push_back(data);
133 } else if (!buffer.empty()) {
134 postReceivedCallback(buffer.data(), buffer.size());
135 buffer.clear();
136 }
137 }
138 }
139
140 /** TX task loop. */
141 void transmitTask() {
142 std::array<uint8_t, USB_MSG_BUF_SIZE> buff{};
143 for (;;) {
144 auto length = messageBuffer.receive(buff.data(), buff.size(), portMAX_DELAY);
145 if (!txPort.send(buff.data(), length, 100)) {
146 assert(false); /* Failed to send data */
147 }
148 postTransmitCallback();
149 }
150 }
151 };
