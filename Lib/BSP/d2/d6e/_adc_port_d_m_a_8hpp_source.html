23/** ADC DMA Port specialization.
24 * Specialization of the `Lib::HAL::Port` template for `DriverMode::DMA`. Provides DMA-based ADC
25 * port functionality, implementing the required interface for ADC operations using DMA.
26 */
27template <> class Lib::HAL::Port<Lib::HAL::DriverMode::DMA> : public AdcTransparentPortBase {
28public:
29 /** Constructor of Port class.
30 * @param adcHandle Pointer to the ADC handle.
31 * @param channelConfig Reference to the ADC channel configuration.
32 */
33 explicit Port(ADC_HandleTypeDef * adcHandle, const AdcChannel::Config & channelConfig)
34 : AdcTransparentPortBase(adcHandle, channelConfig), result{0} {}
35
36 /** Destructor of Port class. */
37 ~Port() = default;
38
39 /** Sets up the ADC port. */
40 void setup() override { AdcTransparentPortBase::setup(); }
41
42 /** Releases the ADC port. */
43 void release() override { AdcTransparentPortBase::release(); }
44
45 /** Starts ADC conversion using DMA.
46 * @return `true` if the operation was successful, `false` otherwise.
47 */
48 bool start() override { return HAL_ADC_Start_DMA(getHandle(), result.data(), result.size()) == HAL_OK; }
49
50 /** Stops ADC conversion using DMA.
51 * @return `true` if the operation was successful, `false` otherwise.
52 */
53 bool stop() override { return HAL_ADC_Stop_DMA(getHandle()) == HAL_OK; }
54
55 /** Reads ADC value with timeout.
56 * @param[out] value Reference to store the read value.
57 * @param timeout Timeout for the read operation.
58 * @return `true` if the read was successful, `false` otherwise.
59 */
60 bool read(uint16_t & value, uint32_t timeout) override { return AdcTransparentPortBase::read(value, timeout); }
61
62 /** Reads the latest ADC value.
63 * @return The latest ADC value.
64 */
65 uint16_t read() const override { return static_cast<uint16_t>(result[0]); }
66
67private:
68 std::array<uint32_t, 1> result; /*!< DMA result buffer. */
69
70 /** Handles ADC interrupt events.
71 * @param callbackId The callback identifier.
72 */
73 void handleInterruptEvent(HAL_ADC_CallbackIDTypeDef callbackId) override {}
74};
