145inline void DigitalPeripheralBase::Outputs::update ()
146{
147 bool hasUpdate (false);
148
149 // processing the output data if there is a value change ...
150
151 if (_isValueChanged == true)
152 {
153 _isValueChanged = false;
154 _bitset = _value;
155
156 auto txBuffer = _txBuffer + 1;
157 for (auto i = 0; i < _numberOfPins; i++)
158 {
159 auto & map = _mapFields [i];
160
161 auto & value = txBuffer [map.port + offsetof (MC33879::TxPort, value)];
162 auto & control = txBuffer [map.port + offsetof (MC33879::TxPort, control)];
163
164 if (_bitset.test (i))
165 {
166 value |= map.mask;
167 // control |= map.mask;
168 }
169 else
170 {
171 value &= ~map.mask;
172 }
173 }
174 _timeout.set (UPDATE_TIMEOUT_DURATION);
175 _changesCounter++;
176 hasUpdate = true;
177 }
178
179 // checks whether the timer that serves the status update has expired...
180
181 if (_timeout.check ())
182 {
183 _timeout.set (UPDATE_TIMEOUT_DURATION);
184 hasUpdate = true;
185 }
186
187 if (hasUpdate == true)
188 {
189 // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
190 _txBuffer [0] = _diagCounter;
191
192 // send / receive data to/from SPI port ...
193
194 _nss.write (0); // spi select
195 if (_spi.sendReceive (_txBuffer, _rxBuffer, _bufferSize, 5) == false)
196 {
197 assert (0);
198 }
199 _nss.write (1); // spi deselect
200 if (_diagCounter == 0) _oe.write (1); // outputs enable in first transaction
201 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
202
203 // processing of incoming data...
204
205 // 1. Detects an SPI error...
206
207 if ((_rxBuffer [_bufferSize - 1] != _txBuffer [0]) && (_diagCounter > 0))
208 {
209 if (_isSpiError == false)
210 {
211 _isSpiError = true;
212 if (onSpiError != nullptr) onSpiError ();
213 }
214 }
215 // 2. Processing the status of MC3879...
216
217 for (auto i = 0; i < _numberOfPins; i++)
218 {
219 auto & map = _mapFields [i];
220 _bitset.set (i, ((_rxBuffer [map.port + offsetof (MC33879::RxPort, status)] & map.mask)));
221 }
222
223 auto status (_bitset.to_ullong ());
224 if (_status != status)
225 {
226 _status = status;
227 if (onStatusChanged != nullptr) onStatusChanged (status);
228 }
229 }
230}
