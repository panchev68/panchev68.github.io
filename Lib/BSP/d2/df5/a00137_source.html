198 HAL_SPI_DeInit (spiHandle);

208 { // reads data from sleep port...
209
210 _nss.write (0);
211
212 if (_spi.receive (_buffer, _bufferSize, 20) == false)
213 {
214 assert (0);
215 }
216 _nss.write (1);
217
218 // processing of received data...
219
220 for (auto i = 0; i < _numberOfPins; i++)
221 {
222 auto & map = _mapFields [i];
223 _bitset.set (i, ((_buffer [map.port] & map.mask)));
224 }
225 if (_value != _bitset.to_ullong ())
226 {
227 _value = _bitset.to_ullong ();
228 if (onValueChanged != nullptr) onValueChanged (_value);
229 _changesCounter++;
230 }
231 }

256 class DigitalPeripheralBase::Outputs : PortBase
257 {
258 public:
259
260 constexpr static inline uint32_t UPDATE_TIMEOUT_DURATION = 50;
261
264 std::function<void (const uint64_t &)> onValueChanged;
265
268 std::function<void (const uint64_t &)> onStatusChanged;
269
272 std::function<void ()> onSpiError;
273
275 Outputs (const Config & config)
276 : _numberOfPins (config.outputs.numberOfPins),
277 _nss (config.outputs.nss),
278 _oe (config.outputs.oe),
279 _spi (config.outputs.spi),
280 _mapFields (config.outputs.mapFields),
281 _bufferSize ((((config.outputs.numberOfPins + 7) / 8) * 2) + 1)
282
283 {
284 _rxBuffer = (new uint8_t [_bufferSize]);
285 _txBuffer = (new uint8_t [_bufferSize]);
286 }
287
289 ~Outputs ()
290 {
291 delete _rxBuffer;
292 delete _txBuffer;
293 }
294
297 void setup () override
298 {
299 _nss.write (1); // NSS = 1
300 _spi.open (); // openns the spi port
301 memset (_txBuffer, 0, _bufferSize); // clears the tx buffer
302
303 // sets th MC33879::TxPort::control to 0xff for all ports ...
304
305 auto port = reinterpret_cast<MC33879::TxPort *> (_txBuffer + 1);
306 for (auto i = 0; i < _bufferSize / sizeof (MC33879::TxPort); i++) port [i].control = 0xff;
307 }
308
311 void setValue (uint64_t value)
312 {
313 _isValueChanged = (value != _value);
314 if (_isValueChanged) _value = value;
315 }
316
319 SPI_HandleTypeDef * getSpiHandle () const override
320 {
321 return _spi;
322 }
323
325 void preUpdateCallback () override
326 {
327 assert (_spi.getHandle ());
328
329 _spi.deinit ();
330 auto spiHandle = _spi.getHandle ();
331 spiHandle->Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_128;
332 spiHandle->Init.CLKPolarity = SPI_POLARITY_LOW;
333 spiHandle->Init.CLKPhase = SPI_PHASE_2EDGE;
334 _spi.init (); // todo: dont use auto `HAL_SPI
335 }
336
339 void update () override
340
341 {
342 bool hasUpdate (false);
343
344 // processing the output data if there is a value change ...
345
346 if (_isValueChanged == true)
347 {
348 _isValueChanged = false;
349 _bitset = _value;
350
351 auto txBuffer = _txBuffer + 1;
352 for (auto i = 0; i < _numberOfPins; i++)
353 {
354 auto & map = _mapFields [i];
355
356 auto & value = txBuffer [map.port + offsetof (MC33879::TxPort, value)];
357 // auto & control = txBuffer [map.port + offsetof (MC33879::TxPort, control)];
358
359 if (_bitset.test (i))
360 {
361 value |= map.mask;
362 // control |= map.mask;
363 }
364 else
365 {
366 value &= ~map.mask;
367 }
368 }
369 _timeout.set (UPDATE_TIMEOUT_DURATION);
370 _changesCounter++;
371 hasUpdate = true;
372 }
373
374 // checks whether the timer that serves the status update has expired...
375
376 if (_timeout.check ())
377 {
378 _timeout.set (UPDATE_TIMEOUT_DURATION);
379 hasUpdate = true;
380 }
381
382 if (hasUpdate == true)
383 {
384 // note: This is due to the fact that SPI on the MC33879 does not work without an active OE
385 _txBuffer [0] = _diagCounter;
386
387 // send / receive data to/from SPI port ...
388
389 _nss.write (0); // spi select
390 if (_spi.sendReceive (_txBuffer, _rxBuffer, _bufferSize, 5) == false)
391 {
392 assert (0);
393 }
394 _nss.write (1); // spi deselect
395 if (_diagCounter == 0) _oe.write (1); // outputs enable in first transaction
396 if ((++_diagCounter) == 0) _diagCounter = 1; // increment _diagCounter
397
398 // processing of incoming data...
399
400 // 1. Detects an SPI error...
401
402 if ((_rxBuffer [_bufferSize - 1] != _txBuffer [0]) && (_diagCounter > 0))
403 {
404 if (_isSpiError == false)
405 {
406 _isSpiError = true;
407 if (onSpiError != nullptr) onSpiError ();
408 }
409 }
410 // 2. Processing the status of MC3879...
411
412 for (auto i = 0; i < _numberOfPins; i++)
413 {
414 auto & map = _mapFields [i];
415 _bitset.set (i, ((_rxBuffer [map.port + offsetof (MC33879::RxPort, status)] & map.mask)));
416 }
417
418 auto status (_bitset.to_ullong ());
419 if (_status != status)
420 {
421 _status = status;
422 if (onStatusChanged != nullptr) onStatusChanged (status);
423 }
424 }
425 }

587 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::SEQUENTAL> ()
588 {
589 // Performs sequential updates of inputs and outputs. This is necessary when the spi port is common to both
590 // interfaces. and reinitialization is required when servicing each device.
591
592 while (1)
593 {
594 handlePreUpdateEvent ();
595
596 FreeRTOS::Karnel::enterCritical ();
597 inputs.preUpdateCallback ();
598 inputs.update ();
599 outputs.preUpdateCallback ();
600 outputs.update ();
601 FreeRTOS::Karnel::exitCritical ();
602
603 handlePostUpdateEvent ();
604
605 if (_onTickEvent) _onTickEvent ();
606 FreeRTOS::Task::delay (1);
607 }
608 }
609
610 template <> inline void DigitalPeripheralBase::taskFn<DigitalPeripheralBase::UpdateMode::PARALLEL> ()
611 {
612 // Performs parallel updates of inputs and outputs. This is possible when the spi port is different for both
613 // interfaces. The initialization of the spi ports is done at the beginning.
614
615 inputs.preUpdateCallback ();
616 outputs.preUpdateCallback ();
617 while (1)
618 {
619 handlePreUpdateEvent ();
620
621 FreeRTOS::Karnel::enterCritical ();
622 inputs.update ();
623 outputs.update ();
624 FreeRTOS::Karnel::exitCritical ();
625
626 handlePostUpdateEvent ();
627 if (_onTickEvent) _onTickEvent ();
628 FreeRTOS::Task::delay (1);
629 }
630 }

632 inline void DigitalPeripheralBase::setup ()
633 {
634 inputs.setup ();
635 outputs.setup ();
636
637 // Selecting an operating mode...
638
639 if (inputs.getSpiHandle () == outputs.getSpiHandle ())
640 {
641 if (FreeRTOS::Task::run ("I/O Peripheral sequental", [this] { taskFn<UpdateMode::SEQUENTAL> (); }, 128, 0) == nullptr) assert (0 && "Error creating task");
642 }
643 else
644 {
645 if (FreeRTOS::Task::run ("I/O Peripheral parallel", [this] { taskFn<UpdateMode::PARALLEL> (); }, 128, 0) == nullptr) assert (0 && "Error creating task");
646 }
