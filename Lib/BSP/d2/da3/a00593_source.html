35 static Lib::Helper::DateTime stringToDateTime(const char * time, const char * date) {
36 auto getDigit = [](const char * str, uint32_t offset) -> uint8_t {
37 assert(str);
38 return static_cast<uint8_t>(str[offset] - '0');
39 };
40 Lib::Helper::DateTime dt;
41 dt.time.hours = 10 * getDigit(time, 0) + getDigit(time, 1);
42 dt.time.minutes = 10 * getDigit(time, 3) + getDigit(time, 4);
43 dt.time.seconds = 10 * getDigit(time, 6) + getDigit(time, 7);
44 dt.date.year = 1000 * getDigit(date, 7) + 100 * getDigit(date, 8) +
45 10 * getDigit(date, 9) + getDigit(date, 10);
46 dt.date.month = parseMonth(date);
47 dt.date.day = 10 * (date[4] == ' ' ? 0 : getDigit(date, 4)) + getDigit(date, 5);

55 static uint32_t stringToU32(const std::string & str) {
56 std::array<uint8_t, 2> array{0, 0};
57 size_t begin = 0;
58 for (size_t i = 0; i < array.size(); i++) {
59 size_t found = str.find('.', begin);
60 if (found == std::string::npos) found = str.size();
61 array[i] = static_cast<uint8_t>(std::stoul(str.substr(begin, found - begin)));
62 begin = found + 1;
63 if (found == str.size()) break;
64 }
65 return (static_cast<uint32_t>(array[0]) << 8) | static_cast<uint32_t>(array[1]);
66 }
