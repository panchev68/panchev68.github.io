MoonPhase

Example demonstrating moon phase calculation for a specific date. This example shows how to use the MoonPhase class to determine the lunar phase for a given date.

#include "Lib/Helper/MoonPhase.hpp"
#include "Lib/Helper/DateTime.hpp"
#include <iostream>
int main() {
Lib::Helper::DateTime::Date date{2024, 6, 1};
Lib::Helper::MoonPhase moonPhase(date);
std::cout << "Moon phase index: " << moonPhase.index() << std::endl;
std::cout << "Moon phase name: " << moonPhase.name() << std::endl;
return 0;
}
