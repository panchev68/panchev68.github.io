439 bool initialize(const Config & config) {
440 if (handle) return false;
441 if (!config.callback) {
442 assert(0);
443 return false;
444 }
445 auto timerCallback = [](Handle timerHandle) {
446 if (!timerHandle) return;
447 auto * instance = reinterpret_cast<Timer *>(pvTimerGetTimerID(timerHandle));
448 if (instance && instance->callback) instance->callback();
449 };
450 handle = xTimerCreate(
451 config.name.empty() ? "Timer" : config.name.c_str(),
452 config.period,
453 (config.mode == Mode::AUTO_RELOAD) ? pdTRUE : pdFALSE,
454 reinterpret_cast<void *>(this),
455 timerCallback);
456 if (!handle) {
457 assert(0);
458 return false;
459 }
460 mode = config.mode;
461 return true;
462 }
