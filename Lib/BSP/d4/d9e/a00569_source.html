209 inline bool DriverBase::read8 (uint32_t eepAddr, void * dst, size_t len)
210 {
211 if (_isReady == false)
212 {
213 assert (0);
214 return false;
215 }
216 uint8_t addr_h = (_devAddr & 0xf0) | ((eepAddr >> 7) & 0x0e);
217 uint8_t addr_l = eepAddr & 0xff;
218
219 return _i2c.receive (addr_h, Lib::HAL::I2Cx::MemAddrSize::ADDR_8BITS, addr_l, dst, len);
220 }

232 inline bool DriverBase::write16 (uint32_t eepAddr, const void * src, size_t len)
233 {
234 if (_isReady == false)
235 {
236 assert (0);
237 return false;
238 }
239
240 if ((eepAddr + len) > _chipSize)
241 {
242 assert (0 && "Address is out of range");
243 return false;
244 }
245
246 uint8_t * ptr = reinterpret_cast<uint8_t *> (const_cast<void *> (src));
247
248 while (len > 0)
249 {
250 auto offs_page = eepAddr % _pageSize;
251 size_t wlen = std::min<size_t> (len, (_pageSize - offs_page));
252
253 if (_i2c.send (_devAddr, Lib::HAL::I2Cx::MemAddrSize::ADDR_16BITS, eepAddr, ptr, wlen) == false)
254 {
255 assert (0);
256 return false;
257 }
258 vTaskDelay (5);
259
260 eepAddr += wlen;
261 ptr += wlen;
262 len -= wlen;
263 }
264 return true;
265 }
