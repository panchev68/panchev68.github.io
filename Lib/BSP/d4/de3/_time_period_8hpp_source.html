16/** Measures time periods using FreeRTOS TickCounter.
17 * Enables starting and stopping a time measurement, returning the elapsed ticks.

19class TimePeriod {

27 /** Starts the time measurement.
28 * @post startTick == RTOS::TickCounter::get()
29 */
30 void start() {
31 startTick = RTOS::TickCounter::get();
32 }
33
34 /** Stops the time measurement and retrieves the elapsed ticks.
35 * @param[out] ticks Reference to store the elapsed ticks.
36 * @pre startTick != 0
37 * @post startTick == 0, ticks == elapsed ticks
38 * @return `true` if the measurement was active, `false` otherwise.
39 */
40 bool stop(uint32_t & ticks) {
41 if (startTick == 0) return false;
42 ticks = RTOS::TickCounter::get() - startTick;
43 startTick = 0;
44 return true;
45 }
46
47private:
48 uint32_t startTick{0}; /*!< The start time of the measurement in ticks; 0 indicates not started. */
49};
