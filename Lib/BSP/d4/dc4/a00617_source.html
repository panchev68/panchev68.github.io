78 bool start(uint32_t intervalMs = 1) {
79 assert(intervalMs > 0);
80 assert(intervalMs <= 1000);

82 tickInterval = intervalMs;
83 auto taskFunction = [this]() {
84 while (true) {
85 auto startTime = TickCounter::get();
86 {
87 MutexGuard guard(mutex);
88 for (auto * client : clients)

92 auto targetTicks = TickCounter::msToTicks(tickInterval);
93 if (elapsed < targetTicks) Task::delayTicks(targetTicks - elapsed);
94 }
95 };
