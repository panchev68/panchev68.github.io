18/** ProduceDate handles reading and writing production dates in OTP memory.
19 * Enables robust access, validation, and storage of production dates in OTP memory for embedded

22class ProduceDate {
23public:

27 explicit ProduceDate(uint32_t offset)

30 /** Reads production date from OTP memory.
31 * @param[out] result Date read from OTP.
32 * @return `true` if valid date was read, `false` otherwise.
33 */
34 bool read(Date & result) const {
35 constexpr uint32_t EMPTY_VALUE = 0xFFFFFFFF;
36 uint32_t value = Lib::HAL::OTP::getInstance().read<uint32_t>(flashAddress);
37 if (value == EMPTY_VALUE) return false;
38 DateTime dateTime(DateTime::unpack(value));
39 if (!dateTime.isValid()) return false;
40 result = dateTime.date;
41 return true;
42 }
43
44 /** Writes production date to OTP memory.
45 * @param value Date to write.
46 * @return `true` if write succeeded, `false` otherwise.
47 */
48 bool write(const Date & value) {
49 constexpr uint32_t EMPTY_VALUE = 0xFFFFFFFF;
50 if (!isEmpty()) return false;
51 if (!value.isValid()) return false;
52
53 auto & otp = Lib::HAL::OTP::getInstance();
54 if (!otp.unlock()) {
55 assert(false);
56 return false;
57 }
58
59 DateTime dateTime;
60 dateTime.date = value;
61 uint32_t packed = DateTime::pack(dateTime);
62
63 bool writeResult = otp.write(flashAddress, packed);
64 if (!writeResult) writeResult = otp.write(flashAddress, packed);
65
66 assert(writeResult);
67
68 if (!otp.lock()) {
69 assert(false);
70 return false;
71 }
72 return writeResult;
73 }
74
75 /** Checks if OTP memory at address is empty.
76 * @return `true` if memory is empty, `false` otherwise.
77 */
78 bool isEmpty() const {
79 constexpr uint32_t EMPTY_VALUE = 0xFFFFFFFF;
80 return Lib::HAL::OTP::getInstance().read<uint32_t>(flashAddress) == EMPTY_VALUE;
81 }
82
83private:
84 const uint32_t flashAddress; /*!< OTP memory address for production date. */
85};
