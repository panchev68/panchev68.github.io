27/** I2CPort class.
28 * Type-safe, event-driven wrapper for STM32 HAL I2C peripheral with callback registration,
29 * memory and device operations, and error/status handling.
30 */
31class I2CPort : public Handle<I2C_HandleTypeDef> {
32 friend class Lib::HAL::DeviceManagerBase<I2CPort, I2C_HandleTypeDef>;
33
34public:
35 using CallbackID = HAL_I2C_CallbackIDTypeDef; /*!< HAL callback ID type. */
36 using State = HAL_I2C_StateTypeDef; /*!< HAL I2C state type. */
37
38 /** Memory address size for I2C memory operations. */
39 enum class MemAddrSize : uint32_t {
40 ADDR_8BITS = I2C_MEMADD_SIZE_8BIT, /*!< 8-bit memory address. */
41 ADDR_16BITS = I2C_MEMADD_SIZE_16BIT /*!< 16-bit memory address. */
42 };
43 /** Error codes for I2C operations. */
44 enum class ErrorCode : uint32_t {
45 NONE = HAL_I2C_ERROR_NONE, /*!< No error. */
46 BUS_ERROR = HAL_I2C_ERROR_BERR, /*!< Bus error. */
47 ARBITRATION_LOST = HAL_I2C_ERROR_ARLO, /*!< Arbitration lost. */
48 ACK_FAILURE = HAL_I2C_ERROR_AF, /*!< Acknowledge failure. */
49 OVERRUN = HAL_I2C_ERROR_OVR, /*!< Overrun/Underrun. */
50 DMA_ERROR = HAL_I2C_ERROR_DMA, /*!< DMA transfer error. */
51 TIMEOUT = HAL_I2C_ERROR_TIMEOUT, /*!< Timeout error. */
52 SIZE_ERROR = HAL_I2C_ERROR_SIZE, /*!< Size management error. */
53 DMA_PARAM_ERROR = HAL_I2C_ERROR_DMA_PARAM, /*!< DMA parameter error. */
54 WRONG_START_CONDITION = HAL_I2C_WRONG_START /*!< Wrong start condition. */
55 };
56 /** Event bits for I2C operations. */
57 enum class EventBit : uint16_t {
58 TX_COMPLETED = 0x01, /*!< Transmission completed. */
59 RX_COMPLETED = 0x02, /*!< Reception completed. */
60 MEM_RX_COMPLETED = 0x04, /*!< Memory reception completed. */
61 MEM_TX_COMPLETED = 0x08, /*!< Memory transmission completed. */
62 ABORT_COMPLETED = 0x10, /*!< Abort completed. */
63 ERROR = 0x0100 /*!< Error occurred. */
64 };
65
66 /** Constructor of the I2CPort class.
67 * @param handle Pointer to the I2C_HandleTypeDef structure.
68 */
69 explicit I2CPort(I2C_HandleTypeDef * handle) : Handle<I2C_HandleTypeDef>(handle) { deviceManager.addDevice(this); }
70
71 /** Destructor of the I2CPort class. */
72 ~I2CPort() { deviceManager.removeDevice(this); }

74 /** Initializes the I2C peripheral and registers callbacks.
75 * @return `true` if initialization was successful, `false` otherwise.
76 */
77 bool init() {
78 if ((!eventGroup.isCreated() && !eventGroup.create()) || HAL_I2C_Init(getHandle()) != HAL_OK) {
79 assert(false);
80 return false;
81 }
82 static constexpr std::array<std::pair<CallbackID, void (*)(I2C_HandleTypeDef *)>, 6> callbacks{{{CallbackID::HAL_I2C_MASTER_RX_COMPLETE_CB_ID, handleInterrupt<CallbackID::HAL_I2C_MASTER_RX_COMPLETE_CB_ID>},
83 {CallbackID::HAL_I2C_MASTER_TX_COMPLETE_CB_ID, handleInterrupt<CallbackID::HAL_I2C_MASTER_TX_COMPLETE_CB_ID>},
84 {CallbackID::HAL_I2C_MEM_RX_COMPLETE_CB_ID, handleInterrupt<CallbackID::HAL_I2C_MEM_RX_COMPLETE_CB_ID>},
85 {CallbackID::HAL_I2C_MEM_TX_COMPLETE_CB_ID, handleInterrupt<CallbackID::HAL_I2C_MEM_TX_COMPLETE_CB_ID>},
86 {CallbackID::HAL_I2C_ABORT_CB_ID, handleInterrupt<CallbackID::HAL_I2C_ABORT_CB_ID>},
87 {CallbackID::HAL_I2C_ERROR_CB_ID, handleInterrupt<CallbackID::HAL_I2C_ERROR_CB_ID>}}};
88 for (const auto & cb : callbacks) {
89 if (HAL_I2C_RegisterCallback(getHandle(), cb.first, cb.second) != HAL_OK) {
90 assert(false);
91 return false;
92 }
93 }
94 return true;
95 }
96
97 /** Deinitializes the I2C peripheral.
98 * @return `true` if deinitialization was successful, `false` otherwise.
99 */
100 bool deinit() { return HAL_I2C_DeInit(getHandle()) == HAL_OK; }
101
102 /** Receives data from an I2C device (master mode, interrupt).
103 * @param deviceAddress I2C device address.
104 * @param destination Pointer to destination buffer.
105 * @param size Number of bytes to receive.
106 * @param timeout Timeout in milliseconds.
107 * @return `true` if receive was successful, `false` otherwise.
108 */
109 bool receive(uint32_t deviceAddress, void * destination, size_t size, uint32_t timeout = 0) {
110 assert(destination && size > 0);
111 return HAL_I2C_Master_Receive_IT(getHandle(), deviceAddress, static_cast<uint8_t *>(destination), size) == HAL_OK &&
112 waitForComplete<EventBit::RX_COMPLETED>(timeout);
113 }
114
115 /** Sends data to an I2C device (master mode, interrupt).
116 * @param deviceAddress I2C device address.
117 * @param source Pointer to source buffer.
118 * @param size Number of bytes to send.
119 * @param timeout Timeout in milliseconds.
120 * @return `true` if send was successful, `false` otherwise.
121 */
122 bool send(uint32_t deviceAddress, const void * source, size_t size, uint32_t timeout = 0) {
123 assert(source && size > 0);
124 return sendFunc(deviceAddress, source, size, timeout) && waitForComplete<EventBit::TX_COMPLETED>(timeout);
125 }
126
127 /** Receives data from an I2C memory device.
128 * @param deviceAddress I2C device address.
129 * @param addressSize Memory address size.
130 * @param memoryAddress Memory address to read from.
131 * @param destination Pointer to destination buffer.
132 * @param size Number of bytes to receive.
133 * @param timeout Timeout in milliseconds.
134 * @return `true` if receive was successful, `false` otherwise.
135 */
136 bool receive(uint32_t deviceAddress, MemAddrSize addressSize, uint32_t memoryAddress, void * destination, size_t size, uint32_t timeout = 0) {
137 return receiveDevFunc(deviceAddress, addressSize, memoryAddress, destination, size, timeout) &&
138 waitForComplete<EventBit::MEM_RX_COMPLETED>(timeout);
139 }
140
141 /** Sends data to an I2C memory device.
142 * @param deviceAddress I2C device address.
143 * @param addressSize Memory address size.
144 * @param memoryAddress Memory address to write to.
145 * @param source Pointer to source buffer.
146 * @param size Number of bytes to send.
147 * @param timeout Timeout in milliseconds.
148 * @return `true` if send was successful, `false` otherwise.
149 */
150 bool send(uint32_t deviceAddress, MemAddrSize addressSize, uint32_t memoryAddress, const void * source, size_t size, uint32_t timeout = 0) {
151 return sendDevFunc(deviceAddress, addressSize, memoryAddress, source, size, timeout) &&
152 waitForComplete<EventBit::MEM_TX_COMPLETED>(timeout);
153 }
154
155 /** Aborts an ongoing I2C transfer.
156 * @param deviceAddress I2C device address.
157 * @param timeout Timeout in milliseconds.
158 * @return `true` if abort was successful, `false` otherwise.
159 */
160 bool abort(uint32_t deviceAddress, uint32_t timeout = 0) {
161 return abortFunc(deviceAddress) && waitForComplete<EventBit::ABORT_COMPLETED>(timeout);
162 }
163
164 /** Checks if an I2C device is ready.
165 * @param deviceAddress I2C device address.
166 * @param timeout Timeout in milliseconds.
167 * @return `true` if device is ready, `false` otherwise.
168 */
169 bool isReady(uint8_t deviceAddress, uint32_t timeout = 100) {
170 return HAL_I2C_IsDeviceReady(getHandle(), deviceAddress, 64, timeout) == HAL_OK;
171 }

173private:
174 RTOS::EventGroup eventGroup; /*!< RTOS event group for signaling I2C events. */

176};
