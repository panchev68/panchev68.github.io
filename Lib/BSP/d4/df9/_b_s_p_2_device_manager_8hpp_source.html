20/** Device manager for Lib::BSP devices.
21 * Manages registration and tick handling of devices. Devices are registered automatically upon
22 * construction and unregistered upon destruction. Provides static setup for initialization and
23 * tick timer management.
24 */
25class DeviceManager {

27 /** Base class for devices managed by DeviceManager.
28 * Registers and unregisters itself with DeviceManager on construction and destruction.
29 */
30 class DeviceBase {

49 /** Setup all devices and start tick timer.
50 * Initializes all registered devices and starts a periodic tick timer.
51 * @return `true` if setup and timer start were successful, `false` otherwise.
52 */
53 static bool setup() {
54 auto & manager = instance();
55 for (auto * device : manager.deviceList) device->setup();
56 RTOS::Timer::Config timerConfig{
57 "Tick",
58 [&manager] {
59 for (auto * device : manager.deviceList) device->handleTickEvent();
60 }};
61 static std::unique_ptr<RTOS::Timer> tickTimer = RTOS::Timer::create(std::move(timerConfig));
62 return tickTimer && tickTimer->start();
63 }
64
65private:

67
68 /** Get singleton instance of DeviceManager.
69 * @return Reference to the singleton DeviceManager.

72 static DeviceManager instance;
73 return instance;
74 }

76 /** Add a device to the manager.
77 * @param device Pointer to the device to add.
78 * @return `true` if the device was added, `false` if already present.
79 */
80 bool addDevice(DeviceBase * device) {
81 if (std::find(deviceList.begin(), deviceList.end(), device) != deviceList.end()) return false;
82 deviceList.push_back(device);
83 return true;
84 }
85
86 /** Remove a device from the manager.
87 * @param device Pointer to the device to remove.
88 * @return `true` if the device was removed, `false` if not found.
89 */
90 bool removeDevice(DeviceBase * device) {
91 auto iter = std::find(deviceList.begin(), deviceList.end(), device);
92 if (iter == deviceList.end()) return false;
93 deviceList.erase(iter);
94 return true;
95 }
