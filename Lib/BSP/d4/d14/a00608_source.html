330 inline bool Base::resetChip ()
331 {
332 FreeRTOS::Timeout timeout;
333
334 _pinReset.write (0);
335 vTaskDelay (50);
336 _pinReset.write (1);
337 timeout.set (1000);
338 while (1)
339 {
340 if (verifyChipID () == true)
341 {
342 _state = State::READY;
343 vTaskDelay (100);
344 return true;
345 }
346 else if (timeout.check ())
347 {
348 _state = State::ERROR;
349 return false;
350 }
351 }

402 inline void Base::readCSR (uint16_t addr, void * dst, size_t len)
403 {
404 UU32 param = {0};
405 UU16 waddr;
406 waddr.val = addr;
407
408 param.v [0] = waddr.bytes.lb;
409 param.v [1] = waddr.bytes.hb;
410 param.v [2] = len;
411 param.v [3] = Registers::ESC_READ_BYTE;
412
413 writeU32 (Registers::ESC_CSR_CMD_REG, param.val);
414
415 do
416 {
417 param.val = readU32 (Registers::ESC_CSR_CMD_REG);
418 } while (param.v [3] & Registers::ESC_CSR_BUSY);
419
420 param.val = readU32 (Registers::ESC_CSR_DATA_REG);
421 memcpy (dst, param.v, len);
422 }

424 inline void Base::writeCSR (uint16_t addr, const uint8_t * src, uint8_t len)
425 {
426 Helper::UU32 param = {0};
427 Helper::UU16 waddr;
428
429 memcpy (param.v, src, len);
430
431 writeU32 (Registers::ESC_CSR_DATA_REG, param.val);
432 waddr.val = addr;
433 param.v [0] = waddr.bytes.lb;
434 param.v [1] = waddr.bytes.hb;
435 param.v [2] = len;
436 param.v [3] = Registers::ESC_WRITE_BYTE;
437
438 writeU32 (0x304, param.val);
439
440 do
441 {
442 param.val = readU32 (0x304);
443 } while (param.v [3] & Registers::ESC_CSR_BUSY);
444 }

446 inline void Base::readPRAM (uint16_t addr, uint8_t * dst, uint16_t len)
447 {
448 Helper::UU64 param = {0};
449 uint8_t i = 0, part_len, byte_pos;
450 uint8_t available;
451
452 param.val = Registers::PRAM_RW_ABORT_MASK;
453
454 writeU32 (Registers::PRAM_READ_CMD_REG, param.val);
455
456 do
457 {
458 param.val = readU32 (Registers::PRAM_READ_CMD_REG);
459 } while ((param.v [3] & Registers::PRAM_RW_BUSY_8B));
460
461 param.w [0] = addr;
462 param.w [1] = len;
463 param.w [2] = 0x0;
464 param.w [3] = 0x8000;
465
466 writeBuffer (Registers::PRAM_READ_ADDR_LEN_REG, (uint8_t *) &param.val, 8);
467
468 do
469 {
470 param.val = readU32 (Registers::PRAM_READ_CMD_REG);
471 } while (!(param.v [0] & Registers::IS_PRAM_SPACE_AVBL_MASK));
472
473 available = param.v [1] & Registers::PRAM_SPACE_AVBL_COUNT_MASK;
474
475 param.val = readU32 (Registers::PRAM_READ_FIFO_REG);
476 available--;
477 byte_pos = (addr & 0x03);
478 part_len = (4 - byte_pos) > len ? len : (4 - byte_pos);
479 memcpy (dst + i, &param.v [byte_pos], part_len);
480 len -= part_len;
481 i += part_len;
482
483 Interface::select ();
484 Interface::write (Registers::CMD_FAST_READ);
485 sendAddr (Registers::PRAM_READ_FIFO_REG);
486 Interface::write (Registers::CMD_FAST_READ_DUMMY);
487
488 while (len)
489 {
490 param.val = readBurstMode ();
491
492 part_len = len > 4 ? 4 : len;
493 memcpy ((dst + i), &param, part_len);
494
495 i += part_len;
496 len -= part_len;
497 available--;
498 }
499
500 Interface::deselect ();
501 }

503 inline void Base::writePRAM (const uint8_t * src, uint16_t addr, uint16_t len)
504 {
505 Helper::UU64 param = {0};
506
507 uint8_t i = 0, part_len, byte_pos, available;
508
509 param.val = Registers::PRAM_RW_ABORT_MASK;
510
511 writeU32 (Registers::PRAM_WRITE_CMD_REG, param.val);
512
513 do
514 {
515 param.val = readU32 (Registers::PRAM_WRITE_CMD_REG);
516 } while ((param.v [3] & Registers::PRAM_RW_BUSY_8B));
517
518 param.w [0] = addr;
519 param.w [1] = len;
520 param.w [2] = 0x0;
521 param.w [3] = 0x8000;
522
523 writeBuffer (Registers::PRAM_WRITE_ADDR_LEN_REG, (uint8_t *) &param.val, 8);
524
525 do
526 {
527 param.val = readU32 (Registers::PRAM_WRITE_CMD_REG);
528 } while (!(param.v [0] & Registers::IS_PRAM_SPACE_AVBL_MASK));
529
530 available = param.v [1] & Registers::PRAM_SPACE_AVBL_COUNT_MASK;
531
532 byte_pos = (addr & 0x03);
533
534 part_len = (4 - byte_pos) > len ? len : (4 - byte_pos);
535
536 param.val = 0;
537 memcpy (&param.v [byte_pos], src + i, part_len);
538
539 writeU32 (Registers::PRAM_WRITE_FIFO_REG, param.val);
540
541 available--;
542 len -= part_len;
543 i += part_len;
544
545 Interface::select ();
546 Interface::write (Registers::CMD_SERIAL_WRITE);
547 sendAddr (Registers::PRAM_WRITE_FIFO_REG);
548
549 while (len)
550 {
551 part_len = len > 4 ? 4 : len;
552 param.val = 0;
553 memcpy (&param, (src + i), part_len);
554
555 writeBurstMode (param.val);
556 i += part_len;
557 len -= part_len;
558 available--;
559 }
560
561 Interface::deselect ();
562 }
