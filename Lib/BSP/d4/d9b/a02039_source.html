119 bool processSensorData(const uint16_t * data, size_t length) {
120 auto calcChecksum = [](const uint8_t * d) -> uint8_t {
121 assert(d /* Data pointer must not be null */);
122 return static_cast<uint8_t>((d[0] + d[1] + d[2] + d[3]) & 0xFF);
123 };
124 auto extractRaw = [](const uint8_t * d) -> int16_t {
125 assert(d /* Data pointer must not be null */);
126 return static_cast<int16_t>((static_cast<uint16_t>(d[0]) << 8) | d[1]);
127 };
128
129 std::array<uint8_t, 5> buffer{0};
130 uint32_t last = data[0];
131 for (size_t i = 3; i < length; ++i) {
132 uint32_t delta = data[i] - last;
133 last = data[i];
134 if (delta > TIMER_THRESHOLD)
135 buffer[(i - 3) / 8] |= Base::BIT_MASK_8MSB.at((i - 3) % 8);
136 }
137
138 diagnostic.frameChecksum = buffer[4];
139 diagnostic.calculatedChecksum = calcChecksum(buffer.data());
140 if (diagnostic.frameChecksum != diagnostic.calculatedChecksum) {
141 diagnostic.invalidFrameChecksum++;
142 return false;
143 }
144
145 int16_t temp = extractRaw(buffer.data() + 2);
146 int16_t hum = extractRaw(buffer.data());
147 temp = (temp & 0x8000) ? static_cast<int16_t>(-(temp & 0x7FFF)) : temp;
148
149 if (temp <= MIN_TEMPERATURE || temp >= MAX_TEMPERATURE) {
150 diagnostic.invalidTemperature++;
151 return false;
152 }
153 if (hum <= MIN_HUMIDITY || hum >= MAX_HUMIDITY) {
154 diagnostic.invalidHumidity++;
155 return false;
156 }
157
158 if (rawTemperature != temp) {
159 rawTemperature = temp;
160 temperature = static_cast<float>(rawTemperature) / 10.0F;
161 isChanged = true;
162 }
163 if (rawHumidity != hum) {
164 rawHumidity = hum;
165 humidity = static_cast<float>(rawHumidity) / 10.0F;
166 isChanged = true;
167 }
168
169 return true;
170 }
