21/** Semaphore class.
22 * Provides a high-level wrapper for FreeRTOS semaphores, supporting binary, mutex, and signal semantics.
23 */
24class Semaphore : public SemaphoreBase {
25public:
26 using Handle = SemaphoreHandle_t; /*!< Underlying FreeRTOS semaphore handle. */
27 using TickType = TickCounter::TickType; /*!< Tick type for wait durations. */
28
29 static constexpr TickType DEFAULT_WAIT = portMAX_DELAY; /*!< Default wait duration (infinite). */
30 static constexpr TickType NO_WAIT = 0; /*!< No wait duration (immediate return). */
31
32 /** State of the semaphore.
33 * Represents the current state of the semaphore: available, taken, or invalid.
34 */
35 enum class State : uint8_t {
36 AVAILABLE = 0, /*!< Semaphore is available. */
37 TAKEN = 1, /*!< Semaphore is taken. */
38 INVALID = 2 /*!< Semaphore is invalid or not created. */
39 };
40
41 /** Default constructor of Semaphore class. */
42 Semaphore() = default;

44 /** Constructor of Semaphore class.
45 * @param isAutoCreate If `true`, the semaphore is created automatically.
46 */
47 explicit Semaphore(bool isAutoCreate) {
48 if (isAutoCreate) SemaphoreBase::create();
49 }

51 /** Move constructor of Semaphore class.

54 Semaphore(Semaphore && otherSemaphore) : SemaphoreBase(std::move(otherSemaphore)) {}

60 Semaphore & operator=(Semaphore && otherSemaphore) {

68 /** Creates a new Semaphore instance.
69 * @param isAutoCreate If `true`, the semaphore is created automatically.
70 * @return Unique pointer to the created Semaphore, or `nullptr` if creation failed.
71 */
72 static std::unique_ptr<Semaphore> make(bool isAutoCreate = true) {
73 auto instance = std::make_unique<Semaphore>(isAutoCreate);
74 return (isAutoCreate && !instance->isCreated()) ? nullptr : std::move(instance);
75 }
76
77 /** Creates a new Semaphore with initial state.
78 * @param isInitiallyGiven If `true`, the semaphore is initially given.
79 * @return Unique pointer to the created Semaphore.

81 static std::unique_ptr<Semaphore> createWithState(bool isInitiallyGiven = false) {
82 auto instance = make(true);
83 if (instance && isInitiallyGiven) instance->give();
84 return instance;
85 }

87 /** Creates a mutex semaphore.

90 static std::unique_ptr<Semaphore> makeMutex() { return createWithState(true); }

92 /** Creates a signal semaphore.

95 static std::unique_ptr<Semaphore> makeSignal() { return createWithState(false); }

97 /** Creates the semaphore.
98 * @return `true` if creation was successful, `false` otherwise.
99 */
100 bool create() override {
101 if (!handle) {
102 handle = xSemaphoreCreateBinary();
103 assert(handle);
104 return handle != nullptr;
105 }
106 return true;
107 }

109 /** Destroys the semaphore. */

118 /** Takes the semaphore with a timeout in milliseconds.

122 bool takeMs(uint32_t waitMs) { return take(TickCounter::msToTicks(waitMs)); }

130 /** Tries to take the semaphore without waiting.

133 bool tryTake() { return take(NO_WAIT); }

169 bool isCreated() const override { return SemaphoreBase::isCreated(); }

171 /** Checks if the semaphore is available.
172 * @return `true` if the semaphore is available, `false` otherwise.
173 */
174 bool isAvailable() const { return handle && uxSemaphoreGetCount(handle) > 0; }
175
176 /** Checks if the semaphore is taken.
177 * @return `true` if the semaphore is taken, `false` otherwise.
178 */
179 bool isTaken() const { return handle && uxSemaphoreGetCount(handle) == 0; }
180
181 /** Gets the current state of the semaphore.
182 * @return State of the semaphore.
183 */
184 State getState() const {
185 if (!handle) return State::INVALID;
186 return uxSemaphoreGetCount(handle) > 0 ? State::AVAILABLE : State::TAKEN;
187 }
188
189 /** Gets the current count of the semaphore.
190 * @return Current count value.
191 */
192 uint32_t getCount() const { return handle ? uxSemaphoreGetCount(handle) : 0; }
193
194 /** Checks if the semaphore is valid.
195 * @return `true` if the semaphore is valid, `false` otherwise.
196 */
197 bool isValid() const { return isCreated(); }
198
199 /** Conversion operator to Handle reference.

202 operator Handle &() override { return SemaphoreBase::operator Handle &(); }
203
204 /** Conversion operator to bool.
205 * @return `true` if the semaphore is created, `false` otherwise.
206 */
207 explicit operator bool() const { return isCreated(); }
