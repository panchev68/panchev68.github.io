46 /** Calculates CRC-16/CCITT-FALSE for a buffer.
47 * @param data Pointer to input data.
48 * @param size Size of input data in bytes.
49 * @return 16-bit CRC value.
50 */
51 static uint16_t calculate(const uint8_t * data, size_t size) {
52 constexpr uint16_t POLYNOMIAL = 0x1021U;
53 constexpr uint16_t INITIAL_VALUE = 0xFFFFU;
54 if (data == nullptr || size == 0U) return INITIAL_VALUE;
55 uint16_t crc = INITIAL_VALUE;
56 for (size_t index = 0U; index < size; ++index) {
57 crc ^= static_cast<uint16_t>(data[index]) << 8U;
58 for (uint8_t bit = 0U; bit < 8U; ++bit)
59 crc = (crc & 0x8000U) ? ((crc << 1U) ^ POLYNOMIAL) : (crc << 1U);
60 }
61 return crc;
62 }
