24/** C++ RAII wrapper for FreeRTOS mutex.
25 * Manages lifecycle, safe creation, destruction, locking, unlocking, and status queries.
26 * Supports both normal and recursive mutex types for STM32/FreeRTOS systems.
27 */
28class Mutex {
29public:
30 static constexpr uint32_t NO_WAIT = 0; /*!< No wait timeout for lock operations. */
31 static constexpr uint32_t WAIT_FOREVER = portMAX_DELAY; /*!< Infinite wait timeout for lock operations. */
32
33 /** Mutex type enumeration.
34 * Specifies whether the mutex is normal or recursive.
35 */
36 enum class MutexType : uint8_t {
37 NORMAL = 0, /*!< Standard mutex type. */
38 RECURSIVE = 1 /*!< Recursive mutex type. */
39 };
40
41 /** Scoped lock guard for Mutex.
42 * Acquires and releases mutex lock automatically using RAII.
43 * Prevents deadlocks and resource leaks by unlocking on destruction.
44 */
45 class LockGuard {
46 public:
47 /** Constructor of the LockGuard class.
48 * @param mutex Mutex to lock.
49 * @param timeout Timeout for lock operation.
50 */
51 explicit LockGuard(Mutex & mutex, uint32_t timeout = WAIT_FOREVER)
52 : mutex(mutex), lockAcquired(mutex.lock(timeout)) {}
53
54 /** Destructor of the LockGuard class. */
55 ~LockGuard() {
56 if (lockAcquired) mutex.unlock();
57 }
58
59 LockGuard(const LockGuard &) = delete;
60 LockGuard & operator=(const LockGuard &) = delete;
61 LockGuard(LockGuard &&) = delete;
62 LockGuard & operator=(LockGuard &&) = delete;
63
64 /** Checks if lock was acquired.
65 * @return `true` if lock acquired, `false` otherwise.
66 */
67 bool isLocked() const { return lockAcquired; }
68
69 /** Conversion to bool for lock status.
70 * @return `true` if lock acquired, `false` otherwise.
71 */
72 explicit operator bool() const { return lockAcquired; }
73
74 private:
75 Mutex & mutex; /*!< Reference to associated mutex. */
76 const bool lockAcquired; /*!< Indicates if lock was acquired. */
77 };
78
79 /** Constructor of the Mutex class.
80 * @param mutexType Type of mutex (normal or recursive).
81 */
82 explicit Mutex(MutexType mutexType = MutexType::NORMAL) : mutexType(mutexType), handle(nullptr) {}
83
84 /** Destructor of the Mutex class. */
85 ~Mutex() { destroy(); }
86
87 /** Creates the mutex.
88 * @return `true` if created, `false` otherwise.
89 * @pre Mutex not already created.
90 * @post Mutex is created and handle is valid if return is `true`.
91 */
92 bool create() {
93 if (handle) return false;
94 if (mutexType == MutexType::RECURSIVE) {
95#if configUSE_RECURSIVE_MUTEXES == 1
96 handle = xSemaphoreCreateRecursiveMutex();
97#else
98 assert(false);
99 handle = nullptr;
100#endif
101 } else {
102#if configUSE_MUTEXES == 1
103 handle = xSemaphoreCreateMutex();
104#else
105 assert(false);
106 handle = nullptr;
107#endif
108 }
109 assert(handle != nullptr);
110 return handle != nullptr;
111 }
112
113 /** Destroys the mutex.
114 * @post Mutex is destroyed and handle is null.
115 */
116 void destroy() {
117 if (handle) {
118 vSemaphoreDelete(handle);
119 handle = nullptr;
120 }
121 }
122
123 Mutex(const Mutex &) = delete;
124 Mutex & operator=(const Mutex &) = delete;
125
126 /** Move constructor of the Mutex class.
127 * @param other Source Mutex to move from.
128 */
129 Mutex(Mutex && other) noexcept : mutexType(other.mutexType), handle(other.handle) { other.handle = nullptr; }
130
131 /** Move assignment operator of the Mutex class.
132 * @param other Source Mutex to move from.
133 * @return Reference to this Mutex.
134 */
135 Mutex & operator=(Mutex && other) noexcept {
136 if (this != &other) {
137 destroy();
138 mutexType = other.mutexType;
139 handle = other.handle;
140 other.handle = nullptr;
141 }
142 return *this;
143 }
144
145 /** Locks the mutex with optional timeout.
146 * @param timeout Timeout in FreeRTOS ticks.
147 * @return `true` if lock acquired, `false` otherwise.
148 * @pre Mutex is created.
149 * @post Mutex is locked if return is `true`.
150 */
151 bool lock(uint32_t timeout = WAIT_FOREVER) {
152 if (!handle) {
153 assert(false);
154 return false;
155 }
156 const TickType_t ticks = (timeout == WAIT_FOREVER) ? portMAX_DELAY : pdMS_TO_TICKS(timeout);
157 if (mutexType == MutexType::RECURSIVE) {
158#if configUSE_RECURSIVE_MUTEXES == 1
159 return xSemaphoreTakeRecursive(handle, ticks) == pdTRUE;
160#else
161 assert(false);
162 return false;
163#endif
164 }
165#if configUSE_MUTEXES == 1
166 return xSemaphoreTake(handle, ticks) == pdTRUE;
167#else
168 assert(false);
169 return false;
170#endif
171 }
172
173 /** Attempts to lock the mutex without waiting.
174 * @return `true` if lock acquired, `false` otherwise.
175 * @pre Mutex is created.
176 * @post Mutex is locked if return is `true`.
177 */
178 bool tryLock() { return lock(NO_WAIT); }
179
180 /** Unlocks the mutex.
181 * @pre Mutex is locked.
182 * @post Mutex is unlocked.
183 */
184 void unlock() {
185 if (!handle) return;
186 if (mutexType == MutexType::RECURSIVE) {
187#if configUSE_RECURSIVE_MUTEXES == 1
188 xSemaphoreGiveRecursive(handle);
189#else
190 assert(false);
191#endif
192 return;
193 }
194#if configUSE_MUTEXES == 1
195 xSemaphoreGive(handle);
196#else
197 assert(false);
198#endif
199 }
200
201 /** Checks if mutex is valid.
202 * @return `true` if valid, `false` otherwise.
203 */
204 bool isValid() const { return handle != nullptr; }
205
206 /** Gets native FreeRTOS mutex handle.
207 * @return Native FreeRTOS handle.
208 */
209 SemaphoreHandle_t getNativeHandle() const { return handle; }
210
211 /** Gets mutex type.
212 * @return Mutex type.
213 */
214 MutexType getType() const { return mutexType; }
215
216private:
217 MutexType mutexType; /*!< Mutex type (normal or recursive). */
218 SemaphoreHandle_t handle; /*!< FreeRTOS semaphore handle. */
219};
