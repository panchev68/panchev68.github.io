RTOS::MemoryManager

Example usage of RTOS::MemoryManager

Demonstrates allocation, deallocation, and diagnostics using the MemoryManager class.

Example usage:
#include "Lib/RTOS/MemoryManager.hpp"
int main() {
using RTOS::MemoryManager;
constexpr size_t ALLOC_SIZE = 128;
void *ptr = MemoryManager::allocate(ALLOC_SIZE);
if (ptr) {
// Use the allocated memory...
MemoryManager::deallocate(ptr);
}
MemoryManager::Status status;
void *ptr2 = MemoryManager::allocate(ALLOC_SIZE, status);
if (status == MemoryManager::Status::SUCCESS) {
// Use the allocated memory...
MemoryManager::deallocate(ptr2, status);
}
#if defined(MEMORY_MANAGER_DIAG)
const auto &diag = MemoryManager::getDiagnostics();
// diag.allocated, diag.deallocated, diag.active, etc.
#endif
return 0;
}
