RTOS::MemoryManager

Example usage of RTOS::MemoryManager for heap allocation and diagnostics.

This example demonstrates how to use the RTOS::MemoryManager class to allocate and deallocate memory from the FreeRTOS heap, check allocation status, and access diagnostic information. It shows both successful and failed allocations, as well as how to use the diagnostics API.

Example usage:
#include "Lib/RTOS/MemoryManager.hpp"
#include <cstdio>
int main() {
using RTOS::MemoryManager;
using RTOS::MemoryManager::Status;
// Allocate 128 bytes
Status status;
void *ptr = MemoryManager::allocate(128, status);
if (status == Status::SUCCESS && ptr) {
std::printf("Allocated 128 bytes at %p\n", ptr);
} else {
std::printf("Allocation failed: %u\n", static_cast<unsigned>(status));
}
// Try to allocate zero bytes (should fail)
void *zeroPtr = MemoryManager::allocate(0, status);
if (!zeroPtr) {
std::printf("Zero-size allocation failed as expected: %u\n", static_cast<unsigned>(status));
}
// Deallocate the first allocation
MemoryManager::deallocate(ptr, status);
if (status == Status::SUCCESS) {
std::printf("Deallocated memory at %p\n", ptr);
}
#if defined(MEMORY_MANAGER_DIAG)
// Print diagnostic information
const auto &diag = MemoryManager::getDiagnostics();
std::printf("Allocations: %u, Deallocations: %u, Active: %u, Failed: %u\n",
diag.allocated, diag.deallocated, diag.active, diag.failedAllocations);
std::printf("Peak usage: %zu, Current usage: %zu\n", diag.peakUsage, diag.currentUsage);
#endif
return 0;
}
