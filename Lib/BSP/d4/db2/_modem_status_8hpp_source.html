3 * Handles modem status responses and provides event hooks for various modem status changes. This class

19 /** Modem Status handler for Digi XBee modules.
20 *
21 * Handles modem status responses and emits events for status changes. Provides event hooks for hardware
22 * reset, network join/leave, coordinator actions, security updates, BLE events, firmware updates, and
23 * error conditions. Ensures type safety and efficient event dispatch for embedded applications.
24 */
25 class ModemStatus {
26 public:
27 std::function<void()> onHardwareReset; /*!< Hardware reset event. */
28 std::function<void()> onWatchdogTimerReset; /*!< Watchdog timer reset event. */
29 std::function<void()> onJoinedNetwork; /*!< Joined network event. */
30 std::function<void()> onLeftNetwork; /*!< Left network event. */
31 std::function<void()> onCoordinatorStarted; /*!< Coordinator started event. */
32 std::function<void()> onNetworkSecurityKeyWasUpdated; /*!< Network security key updated event. */
33 std::function<void()> onNetworkWokeUp; /*!< Network woke up event. */
34 std::function<void()> onNetworkWentToSleep; /*!< Network went to sleep event. */
35 std::function<void()> onVoltageSupplyLimitExceeded; /*!< Voltage supply limit exceeded event. */
36 std::function<void()> onRemoteManagerConnected; /*!< Remote manager connected event. */
37 std::function<void()> onRemoteManagerDisconnected; /*!< Remote manager disconnected event. */
38 std::function<void()> onModemConfigurationChangedWhileJoinInProgress; /*!< Modem config changed during join event. */
39 std::function<void()> onAccessFault; /*!< Access fault event. */
40 std::function<void()> onFatalError; /*!< Fatal error event. */
41 std::function<void()> onSecureSessionSuccessfullyEstablished; /*!< Secure session established event. */
42 std::function<void()> onSecureSessionEnded; /*!< Secure session ended event. */
43 std::function<void()> onSecureEessionAuthenticationFailed; /*!< Secure session authentication failed event. */
44 std::function<void()> onCoordinatorDetectedAPanIdConflictButTookNoAction; /*!< PAN ID conflict detected, no action event. */
45 std::function<void()> onCoordinatorChangedPanIdDueToAConflict; /*!< PAN ID changed due to conflict event. */
46 std::function<void()> onBLEConnect; /*!< BLE connect event. */
47 std::function<void()> onBLEDisconnect; /*!< BLE disconnect event. */
48 std::function<void()> onBandmaskConfigurationFailed; /*!< Bandmask configuration failed event. */
49 std::function<void()> onCellularComponentUpdateStarted; /*!< Cellular component update started event. */
50 std::function<void()> onCellularComponentUpdateFailed; /*!< Cellular component update failed event. */
51 std::function<void()> onCellularComponentUpdateCompleted; /*!< Cellular component update completed event. */
52 std::function<void()> onXbeeFirmwareUpdateStarted; /*!< XBee firmware update started event. */
53 std::function<void()> onXbeeFirmwareUpdateFailed; /*!< XBee firmware update failed event. */
54 std::function<void()> onXbeeFirmwareUpdateApplying; /*!< XBee firmware update applying event. */
55 std::function<void()> onRouterPANIDWasShangedByCoordinatorDueToConflict; /*!< Router PAN ID changed by coordinator event. */
56 std::function<void()> onNetworkWatchdogTimeoutExpired; /*!< Network watchdog timeout expired event. */
57 std::function<void()> onStackError; /*!< Stack error event. */

59 /** Constructor of ModemStatus class.
60 * @param module Pointer to the Digi XBee module.
61 */
62 explicit ModemStatus(Module * module) : module(module) { assert(module); }
63
64 /** Destructor of ModemStatus class. */
65 ~ModemStatus() = default;
66
67 /** Opens the modem status handler.
68 * @pre `module` must be valid and not already opened.
69 */
70 void open() {
71 assert(module->responseEvents.onModemStatusChanged == nullptr);
72 module->responseEvents.onModemStatusChanged = [this](const Frame0x8a & frame) { modemStatusHandler(frame); };
73 }
74
75 /** Closes the modem status handler.
76 * @pre Handler must be opened.
77 */
78 void close() { module->responseEvents.onModemStatusChanged = nullptr; }
79
80 private:
81 Module * module; /*!< Pointer to Digi XBee module. */
82
83 /** Processes the modem status frame and emits the corresponding event.
84 * @param frame Reference to the modem status frame.
85 */
86 void modemStatusHandler(const Frame0x8a & frame) {
87 using ModemStatusEnum = Frame0x8a::ModemStatus;
88 auto emitEvent = [](std::function<void()> & event) { if (event) event(); };
89 switch (frame.modemStatus) {
90 case ModemStatusEnum::HARDWARE_RESET: emitEvent(onHardwareReset); break;
91 case ModemStatusEnum::WATCHDOG_TIMER_RESET:
92 emitEvent(onWatchdogTimerReset);
93 assert(false);
94 break;
95 case ModemStatusEnum::JOINED_NETWORK: emitEvent(onJoinedNetwork); break;
96 case ModemStatusEnum::LEFT_NETWORK: emitEvent(onLeftNetwork); break;
97 case ModemStatusEnum::COORDINATOR_STARTED: emitEvent(onCoordinatorStarted); break;
98 case ModemStatusEnum::NETWORK_SECURITY_KEY_WAS_UPDATED: emitEvent(onNetworkSecurityKeyWasUpdated); break;
99 case ModemStatusEnum::NETWORK_WOKE_UP: emitEvent(onNetworkWokeUp); break;
100 case ModemStatusEnum::NETWORK_WENT_TO_SLEEP: emitEvent(onNetworkWentToSleep); break;
101 case ModemStatusEnum::VOLTAGE_SUPPLY_LIMIT_EXCEEDED: emitEvent(onVoltageSupplyLimitExceeded); break;
102 case ModemStatusEnum::REMOTE_MANAGER_CONNECTED: emitEvent(onRemoteManagerConnected); break;
103 case ModemStatusEnum::REMOTE_MANAGER_DISCONNECTED: emitEvent(onRemoteManagerDisconnected); break;
104 case ModemStatusEnum::MODEM_CONFIGURATION_CHANGED_WHILE_JOIN_IN_PROGRESS: emitEvent(onModemConfigurationChangedWhileJoinInProgress); break;
105 case ModemStatusEnum::ACCESS_FAULT:
106 emitEvent(onAccessFault);
107 assert(false);
108 break;
109 case ModemStatusEnum::FATAL_ERROR:
110 emitEvent(onFatalError);
111 assert(false);
112 break;
113 case ModemStatusEnum::SECURE_SESSION_SUCCESSFULLY_ESTABLISHED: emitEvent(onSecureSessionSuccessfullyEstablished); break;
114 case ModemStatusEnum::SECURE_SESSION_ENDED: emitEvent(onSecureSessionEnded); break;
115 case ModemStatusEnum::SECURE_SESSION_AUTHENTICATION_FAILED: emitEvent(onSecureEessionAuthenticationFailed); break;
116 case ModemStatusEnum::COORDINATOR_DETECTED_A_PAN_ID_CONFLICT_BUT_TOOK_NO_ACTION: emitEvent(onCoordinatorDetectedAPanIdConflictButTookNoAction); break;
117 case ModemStatusEnum::COORDINATOR_CHANGED_PAN_ID_DUE_TO_A_CONFLICT:
118 emitEvent(onCoordinatorChangedPanIdDueToAConflict);
119 assert(false);
120 break;
121 case ModemStatusEnum::BLE_CONNECT: emitEvent(onBLEConnect); break;
122 case ModemStatusEnum::BLE_DISCONNECT: emitEvent(onBLEDisconnect); break;
123 case ModemStatusEnum::BANDMASK_CONFIGURATION_FAILED:
124 emitEvent(onBandmaskConfigurationFailed);
125 assert(false);
126 break;
127 case ModemStatusEnum::CELLULAR_COMPONENT_UPDATE_STARTED: emitEvent(onCellularComponentUpdateStarted); break;
128 case ModemStatusEnum::CELLULAR_COMPONENT_UPDATE_FAILED: emitEvent(onCellularComponentUpdateFailed); break;
129 case ModemStatusEnum::CELLULAR_COMPONENT_UPDATE_COMPLETED: emitEvent(onCellularComponentUpdateCompleted); break;
130 case ModemStatusEnum::XBEE_FIRMWARE_UPDATE_STARTED: emitEvent(onXbeeFirmwareUpdateStarted); break;
131 case ModemStatusEnum::XBEE_FIRMWARE_UPDATE_FAILED:
132 emitEvent(onXbeeFirmwareUpdateFailed);
133 assert(false);
134 break;
135 case ModemStatusEnum::XBEE_FIRMWARE_UPDATE_APPLYING:
136 emitEvent(onXbeeFirmwareUpdateApplying);
137 assert(false);
138 break;
139 case ModemStatusEnum::ROUTER_PAN_ID_WAS_CHANGED_BY_COORDINATOR_DUE_TO_A_CONFLICT: emitEvent(onRouterPANIDWasShangedByCoordinatorDueToConflict); break;
140 case ModemStatusEnum::NETWORK_WATCHDOG_TIMEOUT_EXPIRED:
141 emitEvent(onNetworkWatchdogTimeoutExpired);
142 assert(false);
143 break;
144 default:
145 if (frame.modemStatus < ModemStatusEnum::STACK_ERROR) emitEvent(onStackError);
146 assert(false);
147 break;
148 }
149 }
150 };
