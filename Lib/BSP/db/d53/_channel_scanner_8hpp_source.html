125 /** Task function for channel scanning. */
126 void taskFn() {
127 enum class State {
128 IDLE,
129 BEGIN,
130 OPEN_THE_CHANNEL,
131 SCAN,
132 NEXT_CHANNEL,
133 END
134 };
135 State state{State::BEGIN};
136 size_t index{0};
137 Channel * current{nullptr};
138 auto emitError = [this, &state](Error error) {
139 if (onError) onError(error);
140 state = State::IDLE;
141 };
142 while (true) {
143 switch (state) {
144 case State::IDLE:
145 break;
146 case State::BEGIN:
147 moduleParameters->open();
148 if (!connection->state.isConnected()) {
149 assert(false);
150 emitError(Error::MODULE_IS_NOT_CONNECTED);
151 break;
152 }
153 for (size_t i = 0; i < array.size(); ++i) {
154 array[i] = static_cast<uint8_t>(i);
155 array[i].setConnectionQuality(0);
156 }
157 index = 0;
158 state = State::OPEN_THE_CHANNEL;
159 break;
160 case State::OPEN_THE_CHANNEL:
161 current = &array[index];
162 moduleParameters->setChannel(*current);
163 moduleParameters->writeChanges();
164 connection->quality.setScannerMode();
165 state = State::SCAN;
166 break;
167 case State::SCAN: {
168 Connection::Quality * sender{nullptr};
169 if (queQuality.receive(sender, 0)) current->setConnectionQuality(sender->getPercent());
170 state = State::NEXT_CHANNEL;
171 break;
172 }
173 case State::NEXT_CHANNEL:
174 ++index;
175 if (index < array.size()) {
176 state = State::OPEN_THE_CHANNEL;
177 } else {
178 state = State::END;
179 }
180 break;
181 case State::END:
182 vTaskDelay(SCAN_CHANNEL_PERIOD);
183 state = State::BEGIN;
184 break;
185 }
186 vTaskDelay(1);
187 }
188 }
189
190 /** Update Connection Quality Callback.
191 * @param sender Pointer to Connection::Quality.
192 */

194 if (!queQuality.overwrite(sender)) assert(false);
195 }
196 };
