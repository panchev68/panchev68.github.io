18/** Template class for computing moving average.
19 * This class maintains a circular buffer of values and computes the average of the last
20 * T_WINDOW_SIZE values added. It provides methods to add values, compute the result,
21 * process values, and reset the window.
22 * @tparam T_VALUE The arithmetic type of the values to be integrated.
23 * @tparam T_WINDOW_SIZE The size of the moving window.
24 */
25template <typename T_VALUE, size_t T_WINDOW_SIZE> class Integrator {
26public:
27 static_assert(T_WINDOW_SIZE > 0, "Window size must be greater than 0");
28 static_assert(std::is_arithmetic_v<T_VALUE>, "T_VALUE must be an arithmetic type");
29
30 /** Constructor of Integrator class. */
31 constexpr Integrator() = default;
32
33 /** Adds a value to the integrator window.
34 * @param value The value to add.
35 */
36 void addValue(T_VALUE value) {
37 assert(T_WINDOW_SIZE > 0);
38 values[index] = value;
39 index = (index + 1 < T_WINDOW_SIZE) ? (index + 1) : 0;
40 if (index == 0) full = true;
41 }
42
43 /** Computes the average of the values in the window.
44 * @return The computed average.
45 */
46 T_VALUE computeResult() const {
47 assert(full || index > 0);
48 const size_t count = full ? T_WINDOW_SIZE : index;
49 const auto sum = std::accumulate(values.begin(), values.begin() + count, T_VALUE{0});
50 if (count == 0) return T_VALUE{0};
51 return sum / static_cast<T_VALUE>(count);
52 }
53
54 /** Adds a value and computes the result.
55 * @param value The value to process.
56 * @return The computed average after adding the value.
57 */
58 T_VALUE processValue(T_VALUE value) {
59 addValue(value);
60 return computeResult();
61 }
62
63 /** Function call operator for processing a value.
64 * @param value The value to process.
65 * @return The computed average.
66 */
67 T_VALUE operator()(T_VALUE value) { return processValue(value); }
68
69 /** Resets the integrator window. */
70 void resetWindow() {
71 index = 0;
72 full = false;
73 values.fill(T_VALUE{0});
74 }
75
76 /** Checks if the window is full.
77 * @return true if the window is full, false otherwise.
78 */
79 bool isWindowFull() const { return full; }
80
81private:
82 std::array<T_VALUE, T_WINDOW_SIZE> values{}; /*!< The circular buffer of values. */
83 size_t index{0}; /*!< The current index in the buffer. */
84 bool full{false}; /*!< Flag indicating if the buffer is full. */
85};

87} // namespace Lib::Helper
