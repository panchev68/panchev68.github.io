187 inline bool Interface::readWrite (const void * src, size_t src_len, void * dst, size_t dst_len)
188 {
189 bool res = false;
190 auto size = src_len + dst_len;
191
192 uint8_t * wbuf = new uint8_t [size];
193 uint8_t * rbuf = new uint8_t [size];
194
195 memset (wbuf, 0xff, size);
196
197 if ((src != nullptr) && (src_len > 0)) memcpy (wbuf, src, src_len);
198
199 res = spi.sendReceive (wbuf, rbuf, size, 100);
200
201 if (res == true)
202 {
203 if ((dst != nullptr) && (dst_len > 0)) memcpy (dst, (rbuf + src_len), dst_len);
204 }
205 else assert (0);
206 taskYIELD ();
207 delete wbuf;
208 delete rbuf;
209 return res;
210 }
