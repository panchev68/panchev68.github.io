14/** ADC channel configuration and management class.
15 * The AdcChannel class encapsulates configuration and management of ADC channels using the STM32 HAL library.
16 * It provides a type-safe interface for ADC channel setup, utilizing a nested Config class for parameter storage.
17 */
18class AdcChannel {
19public:
20 using Handle = ADC_HandleTypeDef; /*!< ADC handle type. */
21
22 /** ADC channel configuration structure.
23 * The Config class extends ADC_ChannelConfTypeDef and provides a constructor for explicit initialization
24 * of channel parameters such as channel number, rank, and sampling time.
25 */
26 class Config : public ADC_ChannelConfTypeDef {
27 public:
28 /** Constructor of Config class.
29 * @param channel ADC channel number.
30 * @param rank ADC rank in the sequencer.
31 * @param samplingTime ADC sampling time.
32 */
33 Config(uint32_t channel, uint32_t rank, uint32_t samplingTime)
34 : ADC_ChannelConfTypeDef{} {
35 Channel = channel;
36 Rank = rank;
37 SamplingTime = samplingTime;
38 Offset = 0;
39 }
40 };
41
42 /** Constructor of AdcChannel class.
43 * @param config Reference to a Config object.
44 */
45 explicit AdcChannel(const Config & config);
46
47 /** Configures the ADC channel.
48 * @param handle Pointer to the ADC handle.
49 * @return `true` if the operation was successful, `false` otherwise.
50 */
51 bool setup(Handle * handle);
52
53private:
54 Config config; /*!< ADC channel configuration. */
55};
