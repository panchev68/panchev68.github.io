RTOS::EventGroup

Example usage of RTOS::EventGroup

Demonstrates basic creation, setting, waiting, and clearing of event bits using the EventGroup class.

Example usage:
#include "EventGroup.hpp"
int main() {
using RTOS::EventGroup;
using RTOS::EventBits;
// Create an event group
auto eventGroup = EventGroup::create();
if (!eventGroup) return -1;
// Set bit 0
eventGroup->setBit(0);
// Wait for bit 0 to be set (non-blocking)
bool isSet = eventGroup->waitForBit(0, EventGroup::NO_WAIT);
// Clear bit 0
eventGroup->clearBit(0);
// Set multiple bits (1 and 2)
std::array<EventGroup::BitIndex, 2> indices = {1, 2};
eventGroup->setBits(indices);
// Wait for both bits to be set (blocking)
EventBits result = eventGroup->waitForBits(EventBits().set(1).set(2));
return isSet ? 0 : 1;
}
