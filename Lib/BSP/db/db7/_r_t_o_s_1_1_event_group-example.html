RTOS::EventGroup

Example usage of RTOS::EventGroup for task synchronization.

This example demonstrates how to use the EventGroup class to synchronize two tasks in FreeRTOS. One task waits for a specific event bit, while another task sets the bit after a delay.

Example usage:
#include "Lib/RTOS/EventGroup.hpp"
#include "FreeRTOS.h"
#include "task.h"
using RTOS::EventGroup;
using RTOS::EventBits;
using RTOS::EventGroupHandle_t;
constexpr EventGroup::BitIndex BIT_INDEX = 3;
constexpr char const *TASK1_NAME = "Waiter";
constexpr char const *TASK2_NAME = "Setter";
EventGroup eventGroup(true);
void WaiterTask(void *param) {
(void)param;
printf("Waiter: Waiting for bit %u...\n", BIT_INDEX);
bool bitSet = eventGroup.waitForBit(BIT_INDEX, 10000, EventGroup::ClearMode::CLEAR);
if (bitSet) {
printf("Waiter: Bit %u was set!\n", BIT_INDEX);
} else {
printf("Waiter: Timeout waiting for bit %u.\n", BIT_INDEX);
}
vTaskDelete(nullptr);
}
void SetterTask(void *param) {
(void)param;
vTaskDelay(2000);
printf("Setter: Setting bit %u.\n", BIT_INDEX);
eventGroup.setBit(BIT_INDEX);
vTaskDelete(nullptr);
}
int main() {
xTaskCreate(WaiterTask, TASK1_NAME, 256, nullptr, 2, nullptr);
xTaskCreate(SetterTask, TASK2_NAME, 256, nullptr, 2, nullptr);
vTaskStartScheduler();
for (;;);
return 0;
}
