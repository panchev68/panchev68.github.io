Sunset

Example demonstrating usage of the Sunset class for sunrise, sunset, and moon phase calculations. This example shows how to set the geographic position, date, and retrieve sun event times and moon phase.

#include "Lib/Helper/Sunset.hpp"
#include <iostream>
int main() {
Lib::Helper::Sunset sunset(42.7, 23.3, 2.0); // Sofia, Bulgaria, UTC+2
sunset.setCurrentDate(2024, 6, 21); // Summer solstice
double sunrise = sunset.calculateSunrise();
double sunsetTime = sunset.calculateSunset();
auto sunriseTime = Lib::Helper::Sunset::convertToTime(sunrise);
auto sunsetTimeObj = Lib::Helper::Sunset::convertToTime(sunsetTime);
std::cout << "Sunrise: " << sunriseTime.hour << ":" << sunriseTime.minute << std::endl;
std::cout << "Sunset: " << sunsetTimeObj.hour << ":" << sunsetTimeObj.minute << std::endl;
int moonPhase = sunset.calculateCurrentMoonPhase();
std::cout << "Current moon phase index: " << moonPhase << std::endl;
return 0;
}
