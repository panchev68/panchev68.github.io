22/** CAN port base class.
23 * Provides base interface and logic for CAN port management, including message transmission,
24 * reception, filtering, and event handling.
25 */
26class CanPortBase : protected Lib::HAL::Handle<CAN_HandleTypeDef> {
27public:
28 using CallbackID = HAL_CAN_CallbackIDTypeDef;
29
30 /** Constructor of CanPortBase class.
31 * @param handle Pointer to the CAN handle.
32 */
33 explicit CanPortBase(CAN_HandleTypeDef * handle)

35 canTask(),
36 txCompleteSemaphore(),
37 txCompletedSemaphore(),
38 errorSemaphore(),

40 txMessages(),

42
43 /** Destructor of CanPortBase class. */

46 /** Opens the CAN port.
47 * @return `true` if the port was opened successfully, `false` otherwise.
48 */
49 virtual bool open() {
50 if (!rxMessages.create() || !txMessages.create() || !errorSemaphore.create() ||
51 !txCompleteSemaphore.create() || !txCompletedSemaphore.create()) return false;
52 constexpr uint32_t CAN_NOTIFICATIONS =
53 CAN_IT_TX_MAILBOX_EMPTY | CAN_IT_RX_FIFO0_MSG_PENDING | CAN_IT_RX_FIFO0_FULL | CAN_IT_RX_FIFO0_OVERRUN;
54 return (HAL_CAN_ActivateNotification(getHandle(), CAN_NOTIFICATIONS) == HAL_OK &&
55 HAL_CAN_Start(getHandle()) == HAL_OK &&
56 canTask.start([this] { taskFunction(); }));
57 }
58
59 /** Closes the CAN port.
60 * @return `true` if the port was closed successfully, `false` otherwise.
61 */
62 virtual bool close() {
63 canTask.stop();
64 constexpr uint32_t CAN_NOTIFICATIONS =
65 CAN_IT_TX_MAILBOX_EMPTY | CAN_IT_RX_FIFO0_MSG_PENDING | CAN_IT_RX_FIFO0_FULL | CAN_IT_RX_FIFO0_OVERRUN;
66 bool isSuccess = (HAL_CAN_DeactivateNotification(getHandle(), CAN_NOTIFICATIONS) == HAL_OK &&
67 HAL_CAN_Stop(getHandle()) == HAL_OK);
68 rxMessages.destroy();
69 txCompleteSemaphore.destroy();
70 txCompletedSemaphore.destroy();
71 errorSemaphore.destroy();
72 txMessages.destroy();
73 return isSuccess;
74 }
75
76 /** Gets the CAN handle.
77 * @return Pointer to the CAN handle.
78 */
79 virtual CAN_HandleTypeDef * getHandle() const { return Handle<CAN_HandleTypeDef>::getHandle(); }
80
81 /** Sets the CAN filter.
82 * @param filter CAN filter to set.
83 * @return `true` if the filter was set successfully, `false` otherwise.
84 */
85 virtual bool setFilter(const CanFilter & filter) {
86 this->filter = filter;

89 }
90
91 /** Sends a CAN message.
92 * @param messageId Message identifier.
93 * @param source Pointer to the source data.
94 * @param length Length of the data.
95 * @param timeout Timeout for the operation.
96 * @return `true` if the message was sent successfully, `false` otherwise.
97 */
98 virtual bool send(uint8_t messageId, const void * source, size_t length, uint32_t timeout) {
99 return send(CanTxMessage(messageId, source, length), timeout);
100 }
101
102 /** Sends a CAN message.
103 * @param txMessage CAN transmit message.
104 * @param timeout Timeout for the operation.
105 * @return `true` if the message was sent successfully, `false` otherwise.
106 */
107 virtual bool send(const CanTxMessage & txMessage, uint32_t timeout) {
108 return txMessages.send(txMessage, 1) && txCompletedSemaphore.take(timeout);
109 }
110
111 /** Handles CAN events.
112 * @param callbackId Callback identifier.
113 */
114 virtual void handleEvent(CallbackID callbackId) {
115 if (callbackId == CallbackID::HAL_CAN_ERROR_CB_ID) {
116 errorSemaphore.giveFromISR();
117 } else if (callbackId == CallbackID::HAL_CAN_TX_MAILBOX0_COMPLETE_CB_ID ||
118 callbackId == CallbackID::HAL_CAN_TX_MAILBOX1_COMPLETE_CB_ID ||
119 callbackId == CallbackID::HAL_CAN_TX_MAILBOX2_COMPLETE_CB_ID) {
120 txCompleteSemaphore.giveFromISR();
121 } else if (callbackId == CallbackID::HAL_CAN_RX_FIFO0_MSG_PENDING_CB_ID ||
122 callbackId == CallbackID::HAL_CAN_RX_FIFO1_MSG_PENDING_CB_ID) {
123 const uint32_t fifo = (callbackId == CallbackID::HAL_CAN_RX_FIFO0_MSG_PENDING_CB_ID) ? CAN_RX_FIFO0 : CAN_RX_FIFO1;
124 CanRxMessage rxMessage;
125 if (HAL_CAN_GetRxMessage(getHandle(), fifo, &rxMessage, rxMessage.getData()) == HAL_OK)
126 rxMessages.sendToBackFromISR(rxMessage);
127 }
128 }
129
130 /** Handles received CAN message event.
131 * @param rxMessage Received CAN message.
132 */
133 virtual void handleMessageReceivedEvent(const CanRxMessage & rxMessage) {}
134
135 /** Handles pre-send event. */
136 virtual void handlePreSendEvent() {}
137
138 /** Handles post-send event. */
139 virtual void handlePostSendEvent() {}
140
141protected:
142 /** Handles CAN error event.
143 * @param errorCode Error code.
144 */
145 virtual void handleErrorEvent(uint32_t errorCode) {}
146
147 /** CAN task function. */
148 void taskFunction() {
149 CanRxMessage rxMessage;
150 CanTxMessage txMessage;
151 auto sendMessage = [this](const CanTxMessage & message, uint32_t & mailbox) {
152 return HAL_CAN_AddTxMessage(getHandle(), &message, const_cast<uint8_t *>(message.getData()), &mailbox) == HAL_OK;
153 };
154 for (;;) {
155 if (rxMessages.receive(rxMessage, 10)) handleMessageReceivedEvent(rxMessage);
156 if (txMessages.receive(txMessage, 0)) {
157 handlePreSendEvent();
158 if (sendMessage(txMessage, txMailbox) && txCompleteSemaphore.take(100)) txCompletedSemaphore.give();
159 handlePostSendEvent();
160 }
161 if (errorSemaphore.take(0)) handleErrorEvent(HAL_CAN_GetError(getHandle()));
162 }
163 }
164
165private:
166 CanFilter filter; /*!< CAN filter. */
167 RTOS::Task canTask; /*!< CAN task. */
168 RTOS::Semaphore txCompleteSemaphore; /*!< Semaphore for TX complete. */
169 RTOS::Semaphore txCompletedSemaphore; /*!< Semaphore for TX completed. */
170 RTOS::Semaphore errorSemaphore; /*!< Semaphore for error events. */
171 RTOS::Queue<CanRxMessage, 8> rxMessages; /*!< RX message queue. */
172 RTOS::Queue<CanTxMessage, 8> txMessages; /*!< TX message queue. */
173 uint32_t txMailbox; /*!< TX mailbox index. */
174};
