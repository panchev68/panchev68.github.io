15/** Generic pointer container for object management.
16 * @tparam T Type of the managed objects.
17 */
18template <typename T>
19class Container {
20public:
21 /** Constructor of Container class. */
22 explicit Container() = default;
23
24 /** Destructor of Container class. */
25 virtual ~Container() = default;

27 /** Adds a pointer to the container if not already present.
28 * @param obj Pointer to the object to add.
29 * @return `true` if the object was added, `false` otherwise.
30 */
31 virtual bool add(T * obj) {
32 if (obj == nullptr) return false;
33 if (std::find(items.begin(), items.end(), obj) != items.end()) return false;
34 items.emplace_back(obj);
35 return true;
36 }
37
38 /** Adds multiple pointers from an initializer list.
39 * @param objs Initializer list of pointers to add.
40 * @return `true` if all objects were added, `false` otherwise.
41 */
42 virtual bool add(const std::initializer_list<T *> & objs) {
43 bool allAdded = true;
44 for (auto * obj : objs) allAdded &= add(obj);
45 return allAdded;
46 }
47
48 /** Removes a pointer from the container.
49 * @param obj Pointer to the object to remove.
50 * @return `true` if the object was removed, `false` otherwise.
51 */
52 virtual bool remove(T * obj) {
53 auto it = std::find(items.begin(), items.end(), obj);
54 if (it == items.end()) return false;
55 items.erase(it);
56 return true;
57 }
58
59 /** Clears all pointers from the container. */
60 virtual void clear() { items.clear(); }
61
62 /** Processes all items with a callback function.
63 * @param callback Function pointer to process each item.
64 * @param arg Optional argument passed to the callback.
65 */
66 virtual void processItems(void (*callback)(T *, void *), void * arg = nullptr) {
67 if (!callback) assert(false);
68 for (auto * item : items) callback(item, arg);
69 }
70
71 /** Returns a reference to the underlying vector.
72 * @return Reference to the vector of pointers.
73 */
74 std::vector<T *> & getVector() { return items; }
75
76private:
77 std::vector<T *> items; /*!< Vector of managed object pointers. */
78};
