163 template <size_t BASE_SIZE> constexpr size_t optimal_stack_size() {
164 switch (CURRENT_LEVEL) {
165 case Level::SPACE:
166 return BASE_SIZE;
167 case Level::BALANCED:
168 return BASE_SIZE + (BASE_SIZE / 4);
169 case Level::SPEED:
170 return BASE_SIZE + (BASE_SIZE / 2);
171 case Level::REALTIME:
172 return BASE_SIZE * 2;

174 return BASE_SIZE;
175 }
