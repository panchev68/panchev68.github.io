Lib::System::Version

Example usage of Version class for hardware and firmware version management.

Demonstrates conversion between string and integer formats, and parsing build date/time.

Example usage:
#include "Lib/System/Version.hpp"
int main() {
Lib::System::Version::Hardware hardwareVersion("2.5");
Lib::System::Version::Firmware firmwareVersion("3.7");
uint32_t hwValue = hardwareVersion;
uint32_t fwValue = firmwareVersion;
std::string hwString = static_cast<std::string>(hardwareVersion);
std::string fwString = static_cast<std::string>(firmwareVersion);
// Print version values
printf("Hardware Version: %s (%u)\n", hwString.c_str(), hwValue);
printf("Firmware Version: %s (%u)\n", fwString.c_str(), fwValue);
return 0;
}
