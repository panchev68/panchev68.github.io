18/** Abstraction for the STM32 Independent Watchdog (IWDG) peripheral.
19 * Provides configuration and refresh methods, singleton access, and safe initialization.
20 */
21class Watchdog {
22public:
23 /** Type alias for IWDG handle pointer. */
24 using Handle = IWDG_HandleTypeDef *;
25 static constexpr uint32_t DEFAULT_PRESCALER = IWDG_PRESCALER_8; /**< Default prescaler. */
26 static constexpr uint32_t DEFAULT_RELOAD = 4095; /**< Default reload value (max timeout). */
27 static inline Watchdog * instance = nullptr; /**< Singleton instance pointer. */
28
29 /** Constructor of Watchdog class.
30 * @param handle Pointer to the IWDG handle structure.
31 */
32 explicit Watchdog(Handle handle) : handle(handle) { instance = this; }
33
34 /** Configure and start the IWDG peripheral.
35 * @param prescaler Prescaler value (default: IWDG_PRESCALER_8).
36 * @param reload Reload value (default: 4095, max timeout).
37 * @return `true` if initialization succeeds, `false` otherwise.
38 * @pre `prescaler` and `reload` must be valid for the STM32 IWDG peripheral.
39 */
40 bool setup(uint32_t prescaler = DEFAULT_PRESCALER, uint32_t reload = DEFAULT_RELOAD) {
41 assert(IS_IWDG_PRESCALER(prescaler));
42 assert(IS_IWDG_RELOAD(reload));
43 handle->Instance = IWDG;
44 handle->Init.Prescaler = prescaler;
45 handle->Init.Reload = reload;
46 if (HAL_IWDG_Init(handle) != HAL_OK) return false;
47 return true;
48 }
49
50 /** Refresh (kick) the IWDG to prevent a system reset. */
51 void refresh() { __HAL_IWDG_RELOAD_COUNTER(handle); }
52
53 /** Destructor of Watchdog class. */
54 ~Watchdog() { instance = nullptr; }
55
56 Watchdog(const Watchdog &) = delete;
57 Watchdog & operator=(const Watchdog &) = delete;
58
59private:
60 Handle handle; /*!< Pointer to the IWDG handle structure. */
61};
