20/** Cubic spline interpolation class for embedded applications.
21 * The CubicInterpolation class template provides smooth cubic interpolation between data points using natural
22 * or clamped boundary conditions. It uses a tridiagonal matrix solution for efficient spline coefficient
23 * calculation and supports up to MAX_POINTS data points. All internal state variables are initialized and
24 * updated according to standard cubic spline equations.
25 * @tparam T_VALUE Numeric type for values (float or double).
26 * @tparam T_INPUT Numeric type for input coordinates (typically same as T_VALUE).
27 */
28template <typename T_VALUE, typename T_INPUT = T_VALUE>
29class CubicInterpolation final {
30public:
31 static_assert(std::is_floating_point<T_VALUE>::value, "T_VALUE must be floating point type");
32 static_assert(std::is_floating_point<T_INPUT>::value, "T_INPUT must be floating point type");

62 /** Spline segment coefficients for cubic polynomial.
63 * This structure encapsulates the four coefficients (constant, linear, quadratic, and cubic)
64 * that define a single segment of a cubic spline interpolation. Each coefficient corresponds
65 * to a term in the cubic polynomial:
66 * f(x) = a + b*(x) + c*(x^2) + d*(x^3)
67 * where:
68 * - a: Constant coefficient, representing the offset of the segment.
69 * - b: Linear coefficient, controlling the slope of the segment.
70 * - c: Quadratic coefficient, affecting the curvature of the segment.
71 * - d: Cubic coefficient, determining the rate of change of curvature.
72 * @tparam T_VALUE Type used for the coefficients (e.g., float, double).
73 * @note This struct provides both a default constructor, which initializes all coefficients
74 * to their default values, and a parameterized constructor for explicit initialization.
75 */
76 struct SplineSegment {
77 T_VALUE a; /*!< Constant coefficient. */
78 T_VALUE b; /*!< Linear coefficient. */
79 T_VALUE c; /*!< Quadratic coefficient. */
80 T_VALUE d; /*!< Cubic coefficient. */
81
82 /** Default constructor of SplineSegment struct.*/
83 constexpr SplineSegment() : a{}, b{}, c{}, d{} {}
84
85 /** Constructor of SplineSegment struct.
86 * @param coeffA Constant coefficient.
87 * @param coeffB Linear coefficient.
88 * @param coeffC Quadratic coefficient.
89 * @param coeffD Cubic coefficient.
90 */
91 constexpr SplineSegment(T_VALUE coeffA, T_VALUE coeffB, T_VALUE coeffC, T_VALUE coeffD)
92 : a(coeffA), b(coeffB), c(coeffC), d(coeffD) {}
93 };

189 /** Perform cubic spline interpolation for a given input value.
190 * @param x Input value.
191 * @return Interpolated output value.
192 */
193 T_VALUE interpolate(T_INPUT x) {
194 assert(pointCount >= 2);
195 if (!isSorted) sortPoints();
196 if (!isComputed) computeSplineCoefficients();
197 if (x <= points[0].x) return points[0].y;
198 if (x >= points[pointCount - 1].x) return points[pointCount - 1].y;
199 size_t index = findInterval(x);
200 return evaluateSplineSegment(segments[index], x - points[index].x);
201 }
202
203 /** Get the first derivative at a given input value.
204 * @param x Input value.
205 * @return First derivative value.
206 */
207 T_VALUE getDerivative(T_INPUT x) {
208 assert(pointCount >= 2);
209 if (!isSorted) sortPoints();
210 if (!isComputed) computeSplineCoefficients();
211 if (x <= points[0].x) return segments[0].b;
212 if (x >= points[pointCount - 1].x) {
213 size_t lastIndex = pointCount - 2;
214 T_VALUE dx = points[pointCount - 1].x - points[lastIndex].x;
215 return segments[lastIndex].b + 2 * segments[lastIndex].c * dx + 3 * segments[lastIndex].d * dx * dx;
216 }
217 size_t index = findInterval(x);
218 T_VALUE dx = x - points[index].x;
219 return segments[index].b + 2 * segments[index].c * dx + 3 * segments[index].d * dx * dx;
220 }

231 /** Sort points by x coordinate.*/
232 void sortPoints() {
233 if (isSorted || pointCount <= 1) return;
234 for (size_t i = 0; i < pointCount - 1; ++i)
235 for (size_t j = 0; j < pointCount - i - 1; ++j)
236 if (points[j].x > points[j + 1].x) {
237 auto temp = points[j];
238 points[j] = points[j + 1];
239 points[j + 1] = temp;
240 }
241 isSorted = true;
242 isComputed = false;
243 }
244
245private:
246 /** Compute spline coefficients using tridiagonal matrix solution.*/
247 void computeSplineCoefficients() {
248 if (isComputed || pointCount < 2) return;
249 if (pointCount == 2) {
250 T_VALUE slope = (points[1].y - points[0].y) / (points[1].x - points[0].x);
251 segments[0] = SplineSegment(points[0].y, slope, T_VALUE{}, T_VALUE{});
252 isComputed = true;
253 return;
254 }
255 size_t n = pointCount - 1;
256 std::array<T_VALUE, MAX_POINTS> h, alpha, l, mu, z, c;
257 for (size_t i = 0; i < n; ++i) {
258 h[i] = points[i + 1].x - points[i].x;
259 assert(h[i] > T_VALUE{});
260 }
261 for (size_t i = 1; i < n; ++i)
262 alpha[i] = (3 / h[i]) * (points[i + 1].y - points[i].y) - (3 / h[i - 1]) * (points[i].y - points[i - 1].y);
263 if (boundaryType == BoundaryType::NATURAL) {
264 l[0] = T_VALUE{1};
265 mu[0] = T_VALUE{};
266 z[0] = T_VALUE{};
267 l[n] = T_VALUE{1};
268 z[n] = T_VALUE{};
269 c[n] = T_VALUE{};
270 } else {
271 l[0] = 2 * h[0];
272 mu[0] = T_VALUE{0.5};
273 alpha[0] = (3 / h[0]) * (points[1].y - points[0].y) - 3 * leftDerivative;
274 z[0] = alpha[0] / l[0];
275 l[n] = h[n - 1] * (2 - mu[n - 1]);
276 alpha[n] = 3 * rightDerivative - (3 / h[n - 1]) * (points[n].y - points[n - 1].y);
277 z[n] = (alpha[n] - h[n - 1] * z[n - 1]) / l[n];
278 c[n] = z[n];
279 }
280 for (size_t i = 1; i < n; ++i) {
281 l[i] = 2 * (points[i + 1].x - points[i - 1].x) - h[i - 1] * mu[i - 1];
282 mu[i] = h[i] / l[i];
283 z[i] = (alpha[i] - h[i - 1] * z[i - 1]) / l[i];
284 }
285 for (size_t j = n - 1; j > 0; --j) c[j] = z[j] - mu[j] * c[j + 1];
286 c[0] = boundaryType == BoundaryType::NATURAL ? T_VALUE{} : z[0] - mu[0] * c[1];
287 for (size_t j = 0; j < n; ++j) {
288 T_VALUE a = points[j].y;
289 T_VALUE b = (points[j + 1].y - points[j].y) / h[j] - h[j] * (c[j + 1] + 2 * c[j]) / 3;
290 T_VALUE d = (c[j + 1] - c[j]) / (3 * h[j]);
291 segments[j] = SplineSegment(a, b, c[j], d);
292 }
293 isComputed = true;
294 }
295
296 /** Find the interval containing the input value.
297 * @param x Input value.
298 * @return Index of lower bound point.
299 */
300 size_t findInterval(T_INPUT x) const {
301 assert(pointCount >= 2);
302 assert(isSorted);
303 size_t left = 0, right = pointCount - 1;
304 while (right - left > 1) {
305 size_t middle = left + (right - left) / 2;
306 if (points[middle].x <= x) left = middle;
307 else right = middle;
308 }
309 return left;
310 }

312 /** Evaluate cubic polynomial at given offset.
313 * @param segment Spline segment coefficients.
314 * @param dx Offset from segment start.
315 * @return Polynomial value.
316 */
317 static constexpr T_VALUE evaluateSplineSegment(const SplineSegment & segment, T_VALUE dx) {
318 return segment.a + segment.b * dx + segment.c * dx * dx + segment.d * dx * dx * dx;
319 }
