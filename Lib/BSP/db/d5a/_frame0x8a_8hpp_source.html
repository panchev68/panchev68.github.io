22 /** ModemStatus enum class for frame 0x8A.
23 *
24 * Represents modem status codes returned by Digi XBee modules.
25 */
26 enum class ModemStatus : uint8_t {
27 HARDWARE_RESET = 0x00, /*!< Hardware reset occurred. */
28 WATCHDOG_TIMER_RESET = 0x01, /*!< Watchdog timer reset occurred. */
29 JOINED_NETWORK = 0x02, /*!< Device joined a network. */

33 NETWORK_WOKE_UP = 0x0B, /*!< Network woke up. */
34 NETWORK_WENT_TO_SLEEP = 0x0C, /*!< Network went to sleep. */
35 VOLTAGE_SUPPLY_LIMIT_EXCEEDED = 0x0D, /*!< Voltage supply limit exceeded. */
36 REMOTE_MANAGER_CONNECTED = 0x0E, /*!< Remote manager connected. */
37 REMOTE_MANAGER_DISCONNECTED = 0x0F, /*!< Remote manager disconnected. */
38 MODEM_CONFIGURATION_CHANGED_WHILE_JOIN_IN_PROGRESS = 0x11, /*!< Modem config changed during join. */
39 ACCESS_FAULT = 0x12, /*!< Access fault occurred. */
40 FATAL_ERROR = 0x13, /*!< Fatal error occurred. */
41 BLE_CONNECT = 0x32, /*!< BLE connection established. */
42 BLE_DISCONNECT = 0x33, /*!< BLE disconnected. */
43 BANDMASK_CONFIGURATION_FAILED = 0x34, /*!< Bandmask configuration failed. */
44 CELLULAR_COMPONENT_UPDATE_STARTED = 0x35, /*!< Cellular update started. */
45 CELLULAR_COMPONENT_UPDATE_FAILED = 0x36, /*!< Cellular update failed. */
46 CELLULAR_COMPONENT_UPDATE_COMPLETED = 0x37, /*!< Cellular update completed. */
47 XBEE_FIRMWARE_UPDATE_STARTED = 0x38, /*!< XBee firmware update started. */
48 XBEE_FIRMWARE_UPDATE_FAILED = 0x39, /*!< XBee firmware update failed. */
49 XBEE_FIRMWARE_UPDATE_APPLYING = 0x3A, /*!< XBee firmware update applying. */
50 SECURE_SESSION_SUCCESSFULLY_ESTABLISHED = 0x3B, /*!< Secure session established. */
51 SECURE_SESSION_ENDED = 0x3C, /*!< Secure session ended. */
52 SECURE_SESSION_AUTHENTICATION_FAILED = 0x3D, /*!< Secure session authentication failed. */
53 COORDINATOR_DETECTED_A_PAN_ID_CONFLICT_BUT_TOOK_NO_ACTION = 0x3E, /*!< PAN ID conflict detected, no action. */
54 COORDINATOR_CHANGED_PAN_ID_DUE_TO_A_CONFLICT = 0x3F, /*!< PAN ID changed due to conflict. */
55 ROUTER_PAN_ID_WAS_CHANGED_BY_COORDINATOR_DUE_TO_A_CONFLICT = 0x40, /*!< Router PAN ID changed by coordinator. */
56 NETWORK_WATCHDOG_TIMEOUT_EXPIRED = 0x42, /*!< Network watchdog timeout expired. */
57 STACK_ERROR = 0x80 /*!< Stack error occurred. */
58 };
59
60 ModemStatus modemStatus; /*!< Current modem status. */
61
62 /** Constructor of Frame0x8a class.
63 * @param frame Frame::Data containing the raw response frame.
64 */
65 explicit Frame0x8a(const Frame::Data & frame)
66 : modemStatus(static_cast<ModemStatus>(frame.vector.at(0))) {}
67 };
