16/** Beep control class.
17 * Provides beep pattern generation using a timer peripheral. Supports configurable beep duration,
18 * count, and pause between beeps. Integrates with RTOS tick service for timing.
19 */
20class Beep : public RTOS::TickService::ClientBase {
21public:
22 /** Beep configuration structure.
23 * Holds timer handle and channel for beep output.
24 */
25 struct Config {
26 TIM_HandleTypeDef * timerHandle; /*!< Timer handle for beep output. */
27 uint32_t timerChannel; /*!< Timer channel for beep output. */
28 };
29
30 static constexpr TIM_OC_InitTypeDef DEFAULT_OC_CONFIG = {
31 TIM_OCMODE_TOGGLE, 100,
32 TIM_OCPOLARITY_LOW, TIM_OCNPOLARITY_LOW,
33 TIM_OCFAST_DISABLE, TIM_OCIDLESTATE_RESET,
34 TIM_OCNIDLESTATE_RESET}; /*!< Default output compare configuration. */
35
36 /** Constructor of the Beep class.
37 * @param config Beep configuration structure.
38 */
39 explicit Beep(const Config & config)
40 : config(config),
41 ocConfig(DEFAULT_OC_CONFIG),
42 currentState(State::IDLE),
43 beepDuration(0),
44 pauseDuration(0),
45 beepCount(0) {}
46
47 /** Destructor of the Beep class. */
48 virtual ~Beep() = default;
49
50 /** Starts the beep sequence.
51 * @param duration Duration of each beep in ticks.
52 * @param count Number of beeps to generate.
53 * @param pause Pause duration between beeps in ticks.
54 */
55 void start(uint32_t duration, int32_t count, uint32_t pause) {
56 if (currentState != State::IDLE) return;
57 beepDuration = duration;
58 beepCount = count;
59 pauseDuration = pause;
60 currentState = State::START;
61 }
62
63private:
64 /** Beep state machine.
65 * Represents the current state of the beep sequence.
66 */
67 enum class State : uint8_t {
68 IDLE, /*!< Idle state. */
69 START, /*!< Start beep state. */
70 WAIT_FOR_COMPLETED, /*!< Waiting for beep completion. */
71 WAIT_FOR_PAUSE_COMPLETED /*!< Waiting for pause completion. */
72 };
73
74 State currentState; /*!< Current state of the beep sequence. */
75 uint32_t beepDuration; /*!< Duration of each beep in ticks. */
76 uint32_t pauseDuration; /*!< Pause duration between beeps in ticks. */
77 int32_t beepCount; /*!< Remaining beep count. */
78 RTOS::Timeout timeout; /*!< Timeout for beep and pause timing. */
79 const Config config; /*!< Beep configuration. */
80 const TIM_OC_InitTypeDef ocConfig; /*!< Output compare configuration. */
81
82 /** Handles RTOS tick events for beep timing. */
83 void handleTickEvent() override {
84 switch (currentState) {
85 case State::IDLE: break;
86 case State::START: {
87 auto * timer = config.timerHandle;
88 auto channel = config.timerChannel;
89 if (HAL_TIM_OC_ConfigChannel(timer, &ocConfig, channel) != HAL_OK) assert(0);
90 if (HAL_TIMEx_OCN_Start(timer, channel) != HAL_OK) assert(0);
91 if (HAL_TIM_OC_Start(timer, channel) != HAL_OK) assert(0);
92 timeout.set(beepDuration);
93 currentState = State::WAIT_FOR_COMPLETED;
94 break;
95 }
96 case State::WAIT_FOR_COMPLETED:
97 if (!timeout.isExpired()) break;
98 HAL_TIM_OC_Stop(config.timerHandle, config.timerChannel);
99 HAL_TIMEx_OCN_Stop(config.timerHandle, config.timerChannel);
100 if (--beepCount <= 0) {
101 timeout.disable();
102 currentState = State::IDLE;
103 } else {
104 timeout.set(pauseDuration);
105 currentState = State::WAIT_FOR_PAUSE_COMPLETED;
106 }
107 break;
108 case State::WAIT_FOR_PAUSE_COMPLETED:
109 if (timeout.isExpired()) currentState = State::START;
110 break;
111 }
112 }
113};
114
115} // namespace Lib::BSP
