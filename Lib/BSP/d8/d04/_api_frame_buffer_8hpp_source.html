19 /** XBee API1 Frame buffer for serialized data.
20 *
21 * Stores and manages serialized data for XBee API1 frames, supporting assignment, appending, and clearing.
22 * The buffer is bounded and provides direct access to the underlying data array for efficient embedded use.
23 */
24 class ApiFrameBuffer {
25 public:
26 static constexpr size_t BUFFER_SIZE = 255; /*!< Maximum buffer size. */

28 /** Constructor of ApiFrameBuffer class. */
29 ApiFrameBuffer() = default;

42 const uint8_t * getData() const { return array.data(); }

44 /** Assign data from a vector.
45 * @param buffer Source vector.
46 * @return `true` if assignment succeeded, `false` otherwise.
47 */
48 bool assign(const std::vector<uint8_t> & buffer) {
49 if (buffer.size() > BUFFER_SIZE) return false;
50 std::memcpy(array.data(), buffer.data(), buffer.size());
51 size = buffer.size();
52 return true;
53 }
54
55 /** Assign data from a buffer.
56 * @param srcBuffer Source buffer.
57 * @param length Length of data.
58 * @return `true` if assignment succeeded, `false` otherwise.
59 */
60 bool assign(const uint8_t * srcBuffer, size_t length) {
61 if (length > BUFFER_SIZE) return false;
62 std::memcpy(array.data(), srcBuffer, length);
63 size = length;
64 return true;
65 }
66
67 /** Append data from a vector.
68 * @param buffer Source vector.
69 * @return `true` if append succeeded, `false` otherwise.
70 */
71 bool append(const std::vector<uint8_t> & buffer) {
72 return append(buffer.data(), buffer.size());
73 }
74
75 /** Append a single byte.
76 * @param value Byte to append.
77 * @return `true` if append succeeded, `false` otherwise.
78 */
79 bool append(uint8_t value) {
80 if (size >= BUFFER_SIZE) return false;
81 array[size++] = value;
82 return true;
83 }
84
85 /** Append a block of data.
86 * @param srcBuffer Source buffer.
87 * @param length Length of data.
88 * @return `true` if append succeeded, `false` otherwise.
89 */
90 bool append(const void * srcBuffer, size_t length) {
91 if ((size + length) > BUFFER_SIZE) return false;
92 std::memcpy(&array[size], srcBuffer, length);
93 size += length;
94 return true;
95 }
96
97 /** Get the size of stored data.
98 * @return Number of bytes stored.
99 */
100 size_t getSize() const { return size; }
101
102 /** Clear the buffer. */
103 void clear() {
104 size = 0;
105 std::memset(array.data(), 0, BUFFER_SIZE);
106 }
107
108 private:
109 size_t size{0}; /*!< Number of stored bytes in the buffer. */
110 std::array<uint8_t, BUFFER_SIZE> array; /*!< Data buffer for XBee API1 frame. */
111 };
