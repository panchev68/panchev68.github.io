19/** Utility class for Binary-Coded Decimal (BCD) conversions.
20 *
21 * The Bcd class provides static methods for converting BCD values to ASCII strings and characters. It supports
22 * conversion of 16-bit BCD values to their ASCII string representations and conversion of individual BCD digits
23 * to ASCII characters. The class is designed for use in embedded systems and applications requiring efficient

26class Bcd {
27public:
28 /** Converts a 16-bit BCD value to its ASCII string representation.
29 * @param bcd 16-bit BCD value to convert.
30 * @return ASCII string representation of the BCD input.
31 */
32 static std::string toString(uint16_t bcd) {
33 std::string str;
34 do {
35 auto digit = bcd & 0x0F;
36 str.push_back(static_cast<char>(digit + '0'));
37 bcd >>= 4;
38 } while (bcd > 0);
39 std::reverse(str.begin(), str.end());
40 return str;
41 }
42
43 /** Converts a BCD digit to its ASCII character representation.
44 * @param bcd BCD value (lower 4 bits considered).
45 * @return ASCII character corresponding to the BCD digit ('0' to '9').
46 * @note Asserts if the lower nibble of the input is not a valid BCD digit (0-9).
47 */
48 static constexpr char toAsc(uint8_t bcd) {
49 auto digit = bcd & 0x0F;
50 assert(digit <= 0x09);
51 return static_cast<char>(digit + '0');
52 }
53
54 /** Converts a 16-bit BCD value to its hexadecimal integer representation.
55 * @param bcd 16-bit BCD value to convert.
56 * @return Hexadecimal integer representation of the BCD input.
57 * @note Asserts if any nibble is not a valid BCD digit (0-9).
58 */
59 static constexpr uint16_t toHex(uint16_t bcd) {
60 uint16_t result = 0;
61 uint16_t factor = 1;
62 for (size_t i = 0; i < 4; ++i) {
63 auto digit = (bcd >> (i * 4)) & 0x0F;
64 assert(digit <= 0x09);
65 result += digit * factor;
66 factor *= 10;
67 }
68 return result;
69 }
