1/** Counting semaphore wrapper for FreeRTOS synchronization primitives.
2 * This file provides the SemaphoreCounting class, which implements a counting semaphore abstraction for
3 * real-time STM32 embedded applications using FreeRTOS. The class enables creation and management of
4 * counting semaphores with configurable maximum and initial counts, supporting resource pooling and
5 * multi-token synchronization. It offers methods to query the current count, maximum capacity, initial
6 * value, and state of the semaphore, facilitating robust resource management and signaling between tasks.
7 * The class enforces input validation, safe handle management, and exposes utility methods for checking
8 * semaphore fullness and emptiness. It inherits from SemaphoreBase, leveraging RAII and move semantics
9 * for safe resource cleanup and transfer.
10 * @file SemaphoreCounting.hpp
11 */

22/** Counting semaphore for resource pooling and multi-token synchronization.
23 * This class provides a counting semaphore abstraction with configurable maximum and initial counts,
24 * enabling robust resource management and signaling between tasks in RTOS environments.
25 */
26class SemaphoreCounting : public SemaphoreBase {
27public:
28 /** Constructor of the SemaphoreCounting class.
29 * @param maxCount Maximum count value for the semaphore.
30 * @param initialCount Initial count value for the semaphore.
31 */
32 SemaphoreCounting(UBaseType_t maxCount, UBaseType_t initialCount)
33 : SemaphoreBase(), maxCount(maxCount), initialCount(initialCount) {
34 assert(maxCount > 0);
35 assert(initialCount <= maxCount);
36 }
37
38 /** Destructor of the SemaphoreCounting class. */
39 ~SemaphoreCounting() = default;
40
41 /** Create the counting semaphore with specified maximum and initial counts.
42 * @return `true` if semaphore was successfully created, `false` otherwise.
43 */
44 bool create() override {
45 assert(handle == nullptr);
46 if (handle) return false;
47 handle = xSemaphoreCreateCounting(maxCount, initialCount);
48 return handle != nullptr;
49 }
50
51 /** Get the current count value of the semaphore.
52 * @return Current semaphore count value, or 0 if semaphore is not created.
53 */
54 UBaseType_t getCount() const {
55 assert(handle != nullptr);
56 return handle ? uxSemaphoreGetCount(handle) : 0;
57 }
58
59 /** Get the maximum count value of the semaphore.
60 * @return Maximum count value configured for the semaphore.
61 */
62 UBaseType_t getMaxCount() const { return maxCount; }
63
64 /** Get the initial count value of the semaphore.
65 * @return Initial count value configured for the semaphore.
66 */
67 UBaseType_t getInitialCount() const { return initialCount; }
68
69 /** Check if the semaphore is at its maximum count.
70 * @return `true` if semaphore count is at maximum, `false` otherwise.
71 */
72 bool isAtMaxCount() const { return getCount() >= maxCount; }
73
74 /** Check if the semaphore is empty (count is zero).
75 * @return `true` if semaphore count is zero, `false` otherwise.
76 */
77 bool isEmpty() const { return getCount() == 0; }
78
79 /** Get the number of available tokens in the semaphore.
80 * @return Number of available tokens (current count value).
81 */
82 UBaseType_t getAvailableCount() const { return getCount(); }
83
84 /** Get the remaining capacity before reaching maximum count.
85 * @return Number of tokens that can be added before reaching maximum count.
86 */
87 UBaseType_t getRemainingCapacity() const {
88 auto current = getCount();
89 return (current >= maxCount) ? 0 : (maxCount - current);
90 }
91
92private:
93 const UBaseType_t maxCount; /*!< Maximum count value for the semaphore. */
94 const UBaseType_t initialCount; /*!< Initial count value for the semaphore. */
95};
