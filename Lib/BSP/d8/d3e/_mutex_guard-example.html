MutexGuard RAII Example

This example demonstrates the use of MutexGuard for automatic locking and unlocking in a critical section.

Example usage:

#include "Mutex.hpp"
RTOS::Mutex sharedMutex(RTOS::Mutex::MutexType::NORMAL);
volatile int sharedData = 0;
void updateSharedData(int value) {
if (!sharedMutex.create()) {
// Handle creation failure
return;
}
MutexGuard guard(sharedMutex, 50); // Lock with 50ms timeout
if (guard.isLocked()) {
sharedData = value;
// Mutex is automatically unlocked when guard goes out of scope
} else {
// Handle lock failure, e.g., log error or retry
}
}
