188 void transmitBlock(const void* buffer, uint32_t x, uint32_t y, size_t width, size_t height) override {
189 assert(buffer /* buffer must not be nullptr */);
190 assert(width /* width must be nonzero */);
191 assert(height /* height must be nonzero */);
192 assert(width * height <= 0xffff /* pixel count must fit in 16 bits */);
193 const auto* pixelData = static_cast<const uint16_t*>(buffer);
194 auto spi = spiPort.getHandle();
195 // Set address window (8-bit transfers)
196 if (spi->Init.DataSize != SPI_DATASIZE_8BIT) {
197 spi->Init.DataSize = SPI_DATASIZE_8BIT;
198 HAL_SPI_Init(spi);
199 }
200 setAddressWindow(x, y, width, height);
201 sendCommand(Registers::RAMWR);
202 // Switch to 16-bit for pixel data
203 if (spi->Init.DataSize != SPI_DATASIZE_16BIT) {
204 spi->Init.DataSize = SPI_DATASIZE_16BIT;
205 HAL_SPI_Init(spi);
206 }
207 setTransferActive(1);
208 spiPort.sendBlock(pixelData, width * height);
209 }
