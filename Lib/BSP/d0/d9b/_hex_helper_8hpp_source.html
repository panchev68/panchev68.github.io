13/** HexHelper utility class.
14 * Provides static methods for converting between hexadecimal, ASCII, and BCD formats.
15 */
16class HexHelper {
17public:
18 /** Converts a 4-bit value to its ASCII hexadecimal representation.
19 * @param value The 4-bit value to convert (only lower 4 bits are used).
20 * @return The ASCII character representing the hexadecimal digit.
21 */
22 static constexpr char toAscii(uint8_t value) {
23 value &= 0x0F;
24 return value < 10 ? static_cast<char>('0' + value) : static_cast<char>('A' + value - 10);
25 }
26
27 /** Converts a 16-bit value to its BCD representation.
28 * @param value The 16-bit value to convert.
29 * @return The BCD representation as a 32-bit value.
30 */
31 static constexpr uint32_t toBcd(uint16_t value) {
32 uint32_t result = 0;
33 uint8_t index = 0;
34 while (value) {
35 result |= ((value % 10) << ((index % 2) ? 4 : 0)) << (index / 2 * 8);
36 value /= 10;
37 ++index;
38 }
39 return result;
40 }
41
42 /** Converts an 8-bit value to its BCD representation.
43 * @param value The 8-bit value to convert (must be less than 100).
44 * @return The BCD representation as an 8-bit value.
45 */
46 static constexpr uint8_t toBcd8(uint8_t value) {
47 return static_cast<uint8_t>(((value / 10) << 4) | (value % 10));
48 }
49
50 /** Converts an ASCII hexadecimal character to its 4-bit value.
51 * @param ascii The ASCII character to convert ('0'-'9', 'A'-'F', 'a'-'f').
52 * @return The 4-bit value.
53 * @pre ascii must be a valid hexadecimal digit.
54 */
55 static constexpr uint8_t fromAscii(char ascii) {
56 if (ascii >= '0' && ascii <= '9') return static_cast<uint8_t>(ascii - '0');
57 if (ascii >= 'A' && ascii <= 'F') return static_cast<uint8_t>(ascii - 'A' + 10);
58 if (ascii >= 'a' && ascii <= 'f') return static_cast<uint8_t>(ascii - 'a' + 10);
59 assert(false);
60 return 0;
61 }
62
63private:
64 HexHelper() = delete;
65 ~HexHelper() = delete;
66};
