25 /** Event-driven SPI port abstraction for STM32 HAL.
26 * Async SPI comms with event sync, callback registration, and device management.
27 */
28 class SpiPort {
29 public:
30 using State = HAL_SPI_StateTypeDef; /**< HAL SPI state type. */
31 using Handle = Lib::HAL::Handle<SPI_HandleTypeDef>; /**< HAL SPI handle type. */
32
33 /** SPI callback ID type for STM32F4 (custom, since HAL does not provide).
34 * Enumerates callback types for SPI events.
35 */
36 enum class CallbackId : uint8_t {
37 RX_COMPLETE, /**< RX complete callback. */
38 TX_COMPLETE, /**< TX complete callback. */
39 TX_RX_COMPLETE, /**< TX/RX complete callback. */
40 ABORT, /**< Abort callback. */
41 ERROR /**< Error callback. */
42 };
43
44 /** SPI event bits for RTOS event group signaling.
45 * Defines event bits for SPI operation completion and error signaling.
46 */
47 enum EventBit : uint8_t {
48 TX_COMPLETED = 1, /**< TX completed event bit. */
49 RX_COMPLETED = 2, /**< RX completed event bit. */
50 TX_RX_COMPLETED = 4, /**< TX/RX completed event bit. */
51 ABORT_COMPLETED = 8, /**< Abort completed event bit. */
52 ERROR = 16 /**< Error event bit. */
53 };
54
55 /** SPI error codes mapped from HAL error values.
56 * Enumerates error codes for SPI operations.
57 */
58 enum class ErrorCode : uint8_t {
59 NONE = HAL_SPI_ERROR_NONE, /**< No error. */
60 MODF = HAL_SPI_ERROR_MODF, /**< Mode fault. */
61 CRCE = HAL_SPI_ERROR_CRC, /**< CRC error. */
62 OVR = HAL_SPI_ERROR_OVR, /**< Overrun error. */
63 FRE = HAL_SPI_ERROR_FRE, /**< Frame error. */
64 DMA = HAL_SPI_ERROR_DMA, /**< DMA error. */
65 FLAG = HAL_SPI_ERROR_FLAG, /**< Flag error. */
66 ABORT = HAL_SPI_ERROR_ABORT /**< Abort error. */
67 };
68
69 /** Constructor of SpiPort class.
70 * @param handle SPI peripheral handle pointer.
71 */
72 explicit SpiPort(SPI_HandleTypeDef* handle) : handle(handle) {}
73
74 /** Initialize SPI port and register with device manager. */
75 void init() {
76 /* Create event group for SPI signaling. */
77 if (!eventGroup.create())
78 assert(false);
79 /* Initialize SPI peripheral. */
80 if (HAL_SPI_Init(handle.getHandle()) != HAL_OK)
81 assert(false);
82 /* Register SPI device with device manager. */
83 if (!deviceManager.addDevice(this))
84 assert(false);
85 }
86
87 /** Deinitialize SPI port and remove from device manager. */
88 void deinit() {
89 /* Remove SPI device from device manager. */
90 if (!deviceManager.removeDevice(this))
91 assert(false);
92 /* Destroy event group if created. */
93 if (eventGroup.isCreated())
94 eventGroup.destroy();
95 /* Deinitialize SPI peripheral. */
96 if (HAL_SPI_DeInit(handle.getHandle()) != HAL_OK)
97 assert(false);
98 }
99
100 /** Send data asynchronously and wait for completion.
101 * @param buffer Data buffer.
102 * @param bufferSize Bytes to send.
103 * @param timeout RTOS ticks.
104 * @return `true` if successful, `false` otherwise.
105 */
106 bool send(const void* buffer, size_t bufferSize, uint32_t timeout = portMAX_DELAY) {
107 assert(buffer && bufferSize > 0);
108 /* Start async transmit and wait for TX completed event. */
109 return sendData(buffer, bufferSize) && waitForEvent<TX_COMPLETED>(timeout);
110 }
111
112 /** Receive data asynchronously and wait for completion.
113 * @param buffer Receive buffer.
114 * @param bufferSize Bytes to receive.
115 * @param timeout RTOS ticks.
116 * @return `true` if successful, `false` otherwise.
117 */
118 bool receive(void* buffer, size_t bufferSize, uint32_t timeout = portMAX_DELAY) {
119 assert(buffer && bufferSize > 0);
120 /* Start async receive and wait for RX completed event. */
121 return receiveData(buffer, bufferSize) && waitForEvent<RX_COMPLETED>(timeout);
122 }
123
124 /** Send and receive data asynchronously, wait for completion.
125 * @param txBuffer Transmit buffer.
126 * @param rxBuffer Receive buffer.
127 * @param bufferSize Bytes to transfer.
128 * @param timeout RTOS ticks.
129 * @return `true` if successful, `false` otherwise.
130 */
131 bool sendReceive(const void* txBuffer, void* rxBuffer, size_t bufferSize, uint32_t timeout = portMAX_DELAY) {
132 assert(txBuffer && rxBuffer && bufferSize > 0);
133 /* Start async transmit/receive and wait for TX/RX completed event. */
134 return sendReceiveData(txBuffer, rxBuffer, bufferSize) && waitForEvent<TX_RX_COMPLETED>(timeout);
135 }
136
137 /** Abort ongoing SPI operation and wait for completion.
138 * @param timeout RTOS ticks.
139 * @return `true` if successful, `false` otherwise.
140 */
141 bool abort(uint32_t timeout = portMAX_DELAY) { return abortOperation() && waitForEvent<ABORT_COMPLETED>(timeout); }
142
143 /** Get underlying SPI handle.
144 * @return SPI_HandleTypeDef pointer.
145 */
146 SPI_HandleTypeDef* getHandle() const { return handle.getHandle(); }

148 protected:
149 /** Start async receive.
150 * @param buffer Receive buffer.
151 * @param bufferSize Bytes to receive.
152 * @return `true` if started, `false` otherwise.
153 */
154 virtual bool receiveData(void* buffer, size_t bufferSize) {
155 const auto status = HAL_SPI_Receive_IT(handle.getHandle(), static_cast<uint8_t*>(buffer), bufferSize);
156 assert(status == HAL_OK);
157 /* Return true if receive started successfully. */
158 return status == HAL_OK;
159 }
160
161 /** Start async transmit.
162 * @param buffer Transmit buffer.
163 * @param bufferSize Bytes to send.
164 * @return `true` if started, `false` otherwise.
165 */
166 virtual bool sendData(const void* buffer, size_t bufferSize) {
167 const auto status = HAL_SPI_Transmit_IT(handle.getHandle(), static_cast<const uint8_t*>(buffer), bufferSize);
168 assert(status == HAL_OK);
169 /* Return true if transmit started successfully. */
170 return status == HAL_OK;
171 }
172
173 /** Start async transmit/receive.
174 * @param txBuffer Transmit buffer.
175 * @param rxBuffer Receive buffer.
176 * @param bufferSize Bytes to transfer.
177 * @return `true` if started, `false` otherwise.
178 */
179 virtual bool sendReceiveData(const void* txBuffer, void* rxBuffer, size_t bufferSize) {
180 const auto status = HAL_SPI_TransmitReceive_IT(
181 handle.getHandle(), static_cast<const uint8_t*>(txBuffer), static_cast<uint8_t*>(rxBuffer), bufferSize
182 );
183 assert(status == HAL_OK);
184 /* Return true if transmit/receive started successfully. */
185 return status == HAL_OK;
186 }
187
188 /** Start async abort.
189 * @return `true` if started, `false` otherwise.
190 */
191 virtual bool abortOperation() {
192 const auto status = HAL_SPI_Abort_IT(handle.getHandle());
193 assert(status == HAL_OK);
194 /* Return true if abort started successfully. */
195 return status == HAL_OK;
196 }

198 /** Handle RX complete event. */
199 virtual void handleRxCompleteEvent() {}
200
201 /** Handle TX complete event. */
202 virtual void handleTxCompleteEvent() {}
203
204 /** Handle TX/RX complete event. */
205 virtual void handleTxRxCompleteEvent() {}
206
207 /** Handle abort complete event. */
208 virtual void handleAbortCompleteEvent() {}

210 /** Handle error event.
211 * @param error SPI error code.

216 * @return HAL SPI state.

220 /** Wait for a specific event bit.
221 * @tparam T_BIT Event bit to wait for.
222 * @param timeout Timeout in RTOS ticks.
223 * @return `true` if event occurred, `false` otherwise.
224 */
225 template <EventBit T_BIT> bool waitForEvent(uint32_t timeout) {
226 if (timeout == 0)
227 return true;
228 const auto bits = eventGroup.waitForBits(T_BIT, true, false, timeout).to_ulong();
229 assert(bits & T_BIT);
230 /* Return true if the specified event bit was set. */
231 return bits & T_BIT;
232 }
233
234 /** Handle SPI event callback.
235 * @param callbackId Callback ID.
236 */
237 void eventHandler(CallbackId callbackId) {
238 switch (callbackId) {
239 case CallbackId::RX_COMPLETE:
240 /* Set RX completed event bit from ISR. */
241 if (eventGroup.setBitsFromISR(RX_COMPLETED) == pdFAIL)
242 assert(false);
243 handleRxCompleteEvent();
244 break;
245 case CallbackId::TX_COMPLETE:
246 /* Set TX completed event bit from ISR. */
247 if (eventGroup.setBitsFromISR(TX_COMPLETED) == pdFAIL)
248 assert(false);
249 handleTxCompleteEvent();
250 break;
251 case CallbackId::TX_RX_COMPLETE:
252 /* Set TX/RX completed event bit from ISR. */
253 if (eventGroup.setBitsFromISR(TX_RX_COMPLETED) == pdFAIL)
254 assert(false);
255 handleTxRxCompleteEvent();
256 break;
257 case CallbackId::ABORT:
258 /* Set abort completed event bit from ISR. */
259 if (eventGroup.setBitsFromISR(ABORT_COMPLETED) == pdFAIL)
260 assert(false);
261 handleAbortCompleteEvent();
262 break;
263 case CallbackId::ERROR:
264 /* Set error event bit from ISR and handle error. */
265 if (eventGroup.setBitsFromISR(ERROR) == pdFAIL)
266 assert(false);
267 handleErrorEvent(static_cast<ErrorCode>(HAL_SPI_GetError(handle.getHandle())));
268 break;
269 default:
270 assert(false);
271 break;
272 }
273 }
274
275 private:
276 using DeviceManager = DeviceManagerBase<SpiPort, SPI_HandleTypeDef>;
277 Handle handle; /**< SPI handle pointer. */
278 RTOS::EventGroup eventGroup; /**< RTOS event group for signaling. */
279 static inline DeviceManager deviceManager; /**< Device manager instance. */
280 };
