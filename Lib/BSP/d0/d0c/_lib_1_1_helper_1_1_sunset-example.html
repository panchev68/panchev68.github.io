Lib::Helper::Sunset

Example of using the Sunset class for sunrise, sunset, and moon phase calculation.

This example demonstrates how to use the Sunset class to calculate sunrise, sunset, and moon phase for a specific geographic location and date. The output shows the calculated times and moon phase index.

Example usage:
#include "Lib/Helper/Sunset.hpp"
#include <iostream>
int main() {
Lib::Helper::Sunset sunset(42.7, 23.3, 2.0); // Sofia, Bulgaria, UTC+2
sunset.setCurrentDate(2024, 6, 21); // Summer solstice
double sunrise = sunset.calculateSunrise();
double sunsetTime = sunset.calculateSunset();
int moonPhase = sunset.calculateCurrentMoonPhase();
Lib::Helper::DateTime::Time sunriseTime = Lib::Helper::Sunset::convertToTime(sunrise);
Lib::Helper::DateTime::Time sunsetTimeObj = Lib::Helper::Sunset::convertToTime(sunsetTime);
std::cout << "Sunrise: " << sunriseTime.hour << ":" << sunriseTime.minute << std::endl;
std::cout << "Sunset: " << sunsetTimeObj.hour << ":" << sunsetTimeObj.minute << std::endl;
std::cout << "Moon phase index: " << moonPhase << std::endl;
return 0;
}
