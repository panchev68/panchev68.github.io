16/** Abstract base class for display drivers.
17 * Defines the required interface and base functionality for display driver implementations.
18 */
19class DriverBase {
20public:
21 static constexpr int MAX_TRANSFER_HEIGHT = 240; /*!< Maximum allowed height for a transfer block. */
22 static inline DriverBase * instance = nullptr; /*!< Pointer to the current driver instance. */
23
24 /** Constructor of the DriverBase class. */
25 DriverBase() { instance = this; }
26
27 /** Destructor of the DriverBase class. */
28 virtual ~DriverBase() = default;
29
30 /** Initializes the display driver.
31 * @return `true` if initialization succeeded, `false` otherwise.
32 */
33 virtual bool setup() = 0;
34
35 /** Transmits a rectangular block of data to the display.
36 * @param buffer Pointer to the data buffer.
37 * @param x X-coordinate of the block.
38 * @param y Y-coordinate of the block.
39 * @param width Width of the block.
40 * @param height Height of the block.
41 */
42 virtual void transmitBlock(const void * buffer, uint32_t x, uint32_t y, size_t width, size_t height) = 0;
43
44 /** Handles the transfer complete event. */
45 virtual void handleTransferCompleteEvent() { DisplayDriver_TransferCompleteCallback(); }
46
47 /** Returns whether a transfer is currently active.
48 * @return `1` if a transfer is active, `0` otherwise.
49 */
50 virtual int getTransferActive() const { return isTransferActive ? 1 : 0; }
51
52 /** Checks if a block can be transferred based on its bottom coordinate.
53 * @param bottom Bottom coordinate of the block.
54 * @return `1` if the block can be transferred, `0` otherwise.
55 */
56 virtual int canTransferBlock(int16_t bottom) const { return bottom <= MAX_TRANSFER_HEIGHT; }
57
58protected:
59 /** Sets the transfer active state.
60 * @param value `1` to set transfer active, `0` to set inactive.
61 */
62 void setTransferActive(int value) { isTransferActive = value != 0; }
63
64private:
65 bool isTransferActive = false; /*!< Indicates if a transfer is in progress. */
66};
