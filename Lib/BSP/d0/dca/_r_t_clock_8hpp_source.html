27/** STM32 RTC peripheral device abstraction.
28 * Provides thread-safe access to the STM32 RTC peripheral, supporting date/time read/write and backup register
29 * operations. Integrates with RTOS mutex and backup SRAM domain for safe concurrent access.
30 */
31class RTClock : Handle<RTC_HandleTypeDef> {
32public:
33 using DateTime = Lib::Helper::DateTime;
34 static constexpr uint32_t NUMBER_OF_REGISTERS = 20; /*!< Number of RTC backup registers. */
35
36 /** Constructor of RTClock class.
37 * @param handle Pointer to RTC_HandleTypeDef.
38 */
39 explicit RTClock(RTC_HandleTypeDef * handle) : Handle(handle) { assert(handle); }
40
41 /** Destructor of RTClock class. */
42 virtual ~RTClock() = default;
43
44 /** Initialize RTC device and enable backup SRAM. */
45 void setup() {
46 assert(!mutex.isCreated());
47 if (!mutex.create()) assert(0);
48 BKPSRAM::Domain::getInstance().enable();
49 }
50
51 /** Release device resources. */
52 void release() { mutex.destroy(); }
53
54 /** Read current date and time from RTC.
55 * @return DateTime structure with current date and time.
56 */
57 DateTime read() {
58 RTC_TimeTypeDef time{};
59 RTC_DateTypeDef date{};
60 MutexGuard guard(mutex);
61 if (HAL_RTC_GetTime(getHandle(), &time, FORMAT_BIN) != HAL_OK) assert(0);
62 if (HAL_RTC_GetDate(getHandle(), &date, FORMAT_BIN) != HAL_OK) assert(0);
63 return {{date.Date, date.Month, date.Year + DateTime::Date::FIRST_YEAR}, {time.Hours, time.Minutes, time.Seconds}};
64 }
65
66 /** Write date and time to RTC.
67 * @param dateTime DateTime structure to write.
68 */
69 void write(const DateTime & dateTime) {
70 RTC_TimeTypeDef time{};
71 RTC_DateTypeDef date{};
72 time.Hours = dateTime.time.hours;
73 time.Minutes = dateTime.time.minutes;
74 time.Seconds = dateTime.time.seconds;
75 date.Date = dateTime.date.day;
76 date.Month = dateTime.date.month;
77 date.Year = dateTime.date.year - DateTime::Date::FIRST_YEAR;
78 MutexGuard guard(mutex);
79 if (HAL_RTC_SetTime(getHandle(), &time, FORMAT_BIN) != HAL_OK) assert(0);
80 if (HAL_RTC_SetDate(getHandle(), &date, FORMAT_BIN) != HAL_OK) assert(0);
81 }
82
83 /** Read RTC backup register.
84 * @param index Register index.
85 * @param[out] value Output value.
86 * @return `true` if read was successful, `false` otherwise.
87 */
88 bool readRegister(uint32_t index, uint32_t & value) {
89 if (index >= NUMBER_OF_REGISTERS) return false;
90 MutexGuard guard(mutex);
91 value = HAL_RTCEx_BKUPRead(getHandle(), index);
92 return true;
93 }
94
95 /** Write RTC backup register.
96 * @param index Register index.
97 * @param value Value to write.
98 * @return `true` if write was successful, `false` otherwise.
99 */
100 bool writeRegister(uint32_t index, uint32_t value) {
101 if (index >= NUMBER_OF_REGISTERS) return false;
102 MutexGuard guard(mutex);
103 HAL_RTCEx_BKUPWrite(getHandle(), index, value);
104 return true;
105 }
106
107private:
108 RTOS::Mutex mutex; /*!< RTOS mutex for thread safety. */
109};
