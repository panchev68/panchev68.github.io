20 /** Digi message checksum calculator.
21 *
22 * Calculates the checksum for Digi API frames using the specified buffer or raw data pointer.
23 */
24 class Checksum {
25 public:
26 /** Constructor of Checksum class.
27 * @param apiFrame Digi API frame buffer.
28 */
29 explicit Checksum(const ApiFrameBuffer & apiFrame) {
30 const auto * header = reinterpret_cast<const Frame::Header *>(apiFrame.getData());
31 size_t payloadLen = (static_cast<size_t>(header->lengthMSB) << 8) | header->lengthLSB;
32 size_t msgSize = payloadLen + sizeof(Frame::Header);
33 if (msgSize > apiFrame.getSize()) { assert(false); }
34 uint8_t crc = 0xFF;
35 for (size_t i = sizeof(Frame::Header); i < msgSize; ++i) crc -= apiFrame.getData()[i];
36 result = crc;
37 }
38
39 /** Constructor of Checksum class.
40 * @param data Pointer to the raw frame data.
41 */
42 explicit Checksum(const uint8_t * data) {
43 if (!data) { assert(false); }
44 const auto * header = reinterpret_cast<const Frame::Header *>(data);
45 size_t payloadLen = (static_cast<size_t>(header->lengthMSB) << 8) | header->lengthLSB;
46 uint8_t crc = 0xFF;
47 for (size_t i = sizeof(Frame::Header); i < payloadLen + sizeof(Frame::Header); ++i) crc -= data[i];
48 result = crc;
49 }
50
51 /** Conversion operator to uint8_t.
52 * @return The calculated checksum value.
53 */
54 operator uint8_t() const { return result; }
55
56 private:

58 };
