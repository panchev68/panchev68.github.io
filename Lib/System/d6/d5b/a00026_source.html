188 * @brief Sets up the system parameters by reading from storage and validating the data.
189 *
190 * This function opens the storage, reads the data, and checks if the data is valid.
191 * If the data is not valid, it restores the default values and writes them back to storage.

195 virtual bool setup () {
196 _storage->open ();
197
198 // read the data from storage
199 if (read () == false) {
200 assert (0 && "Reading error");
201 return false;
202 }
203 // check if the data is valid and restore by default if not
204 if (_data->isValid () == false) {
205 restoreByDefault ();
206 if (write () == false) {
207 assert (0 && "Writing error");
208 return false;
209 }
210 }
