#pragma once
#include "Project/BSP/M24C64W.hpp"
#include "Project/System/ParametersData.hpp"
#include <Lib/System/ParametersBase.hpp>
namespace BSP
{
extern M24C64W eeprom;
}
namespace System
{
using ParametersBase = Lib::System::ParametersBase<ParametersData>;
class Parameters : public ParametersBase
{
public:
Parameters ();
~Parameters ();
virtual bool setup () override;
virtual ParametersData & getData ();
virtual bool write ();
bool restoreByDefault () override;
private:
class Storage;
Storage * _storage;
private:
};
#pragma region Implementation
// ------------------------- Implementation ---------------------------
class Parameters::Storage : public ParametersBase::StorageBase
{
constexpr static uint32_t FLASH_ADDRESS = 0;
public:
bool open () override
{
return BSP::eeprom.open ();
}
bool close () override
{
return BSP::eeprom.close ();
}
bool read (void * dst, size_t len) override
{
return BSP::eeprom.read (FLASH_ADDRESS, dst, len);
}
bool write (const void * src, size_t len) override
{
return BSP::eeprom.write (FLASH_ADDRESS, src, len);
}
};
// Parameters constructor
inline Parameters::Parameters () : ParametersBase ((new Storage))
{
_storage = reinterpret_cast<Storage *> (ParametersBase::getStorage ());
}
// Parameters destructor
inline Parameters::~Parameters ()
{
delete _storage;
}
// Initial initialization of System parameters
inline bool Parameters::setup ()
{
return ParametersBase::setup ();
}
// Gets the reference of parameters data area
inline ParametersData & Parameters::getData ()
{
return ParametersBase::getData ();
}
// Writes the system parameters data area
inline bool Parameters::write ()
{
return ParametersBase::write ();
}
// Restores default parameters
inline bool Parameters::restoreByDefault ()
{
auto & data = getData ();
data.parameter1 = 1234;
return true;
}
#pragma endregion
} // namespace System
