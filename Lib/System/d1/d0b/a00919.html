FirmwareVersion.cpp
#include "FirmwareVersion.hpp"
#include <iostream>
int main () {
// Create a FirmwareVersion object
Lib::System::FirmwareVersion firmwareVersion ("1.0.0", 1);
// Retrieve and print the build date
std::cout << "Build Date: " << firmwareVersion.getBuildDate () << std::endl;
// Retrieve and print the version value
std::cout << "Version Value: " << firmwareVersion.getValue () << std::endl;
// Retrieve and print the sub-version number
std::cout << "Sub-Version: " << firmwareVersion.getSubVersion () << std::endl;
// Convert to string and print
std::cout << "Version String: " << firmwareVersion.toString () << std::endl;
// Convert to uint32_t and print
uint32_t versionAsInt = firmwareVersion;
std::cout << "Version as uint32_t: " << versionAsInt << std::endl;
// Convert to std::string and print
std::string versionAsString = firmwareVersion;
std::cout << "Version as std::string: " << versionAsString << std::endl;
return 0;
}
