32 /// \brief Reads a system produce date

47 /// \brief Writes a system produce date in OTP ROM
48 /// \param value - reference to a DateTime instance
49 /// \return true in a successful record
50 /// \note This function is executed once.
51 virtual bool write (const DateTime & value)
52 {
53 bool result;
54 if (isEmpty () == false) return false;
55 if (value.isValid () == false) return false;
56 auto & otp = Lib::HAL::OTP::getInstance ();
57 if (otp.unlock () == false) assert (0);
58 auto date = DateTime::pack (value);
59 result = otp.write (_flashAddress, date);
60 if (result == false) result = otp.write (_flashAddress, date); //+ 03.04.2024 try again
61 assert (result);
62 if (otp.lock () == false) assert (0);
63 return result;
64 }
