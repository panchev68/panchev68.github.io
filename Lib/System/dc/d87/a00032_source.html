25 /**
26 * @class Configuration
27 * @brief Manages configuration data stored in OTP (One-Time Programmable) memory.
28 *
29 * This class provides methods to read and write configuration data to a specific
30 * address in OTP memory. It ensures that the data written is masked with a valid
31 * mask and provides methods to check if the configuration is empty or valid.
32 */
33 class Configuration {
34 constexpr static inline uint32_t VALID_MASK = 0x1000'0000;
35
36 public:
37
38 /**
39 * @typedef Bitset
40 * @brief A type alias for std::bitset with a size of 31 bits.
41 *
42 * This type alias defines Bitset as a std::bitset with 31 bits. It can be used
43 * to represent and manipulate a fixed-size sequence of bits.
44 */
45 using Bitset = std::bitset<31>;
46
47 /**
48 * @brief Constructs a Configuration object with a specified offset.
49 *
50 * This constructor initializes the Configuration object by setting the
51 * flash address to the beginning address of the OTP (One-Time Programmable)
52 * memory plus the given offset. It also asserts that the offset is within
53 * the valid range of the OTP memory.
54 *
55 * @param offset The offset to be added to the beginning address of the OTP memory.
56 *
57 * @note The offset must be less than the difference between the end address
58 * and the beginning address of the OTP memory.
59 */
60 Configuration (uint32_t offset) : _flashAddress (Lib::HAL::OTP::beginAddr () + offset) {
61 assert (offset < (Lib::HAL::OTP::endAddr () - Lib::HAL::OTP::beginAddr ()));
62 }
63
64 /**
65 * @brief Virtual destructor for the Configuration class.
66 *
67 * This destructor is defined as default, which means it will use the compiler-generated
68 * implementation. Declaring a virtual destructor ensures that the destructors of derived
69 * classes are called correctly when an object is deleted through a pointer to the base class.
70 */
71 virtual ~Configuration () = default;
72
73 /**
74 * @brief Reads a value from the specified flash address and returns it as a Bitset.
75 *
76 * This function reads a 32-bit unsigned integer from the flash memory at the
77 * address specified by _flashAddress using the OTP (One-Time Programmable)
78 * memory interface. The read value is then converted into a Bitset and returned.
79 *
80 * @return Bitset The value read from the flash memory, represented as a Bitset.
81 */
82 const Bitset read () const {
83 return Bitset (Lib::HAL::OTP::getInstance ().read<uint32_t> (_flashAddress));
84 }
85
86 /**
87 * @brief Writes a Bitset value to a specified flash address.
88 *
89 * This function writes the provided Bitset value to a flash memory address
90 * using the OTP (One-Time Programmable) memory interface. It ensures that
91 * the OTP memory is unlocked before writing and locked after writing.
92 *
93 * @param value The Bitset value to be written to the flash address.
94 * @return true if the write operation was successful, false otherwise.
95 */
96 bool write (const Bitset & value) {
97 auto & otp = Lib::HAL::OTP::getInstance ();
98
99 auto write = [&otp] (uint32_t addr, uint32_t value) -> bool {
100 return otp.write (addr, value & !VALID_MASK);
101 };
102
103 if (otp.unlock () == false) assert (0);
104 auto result = write (_flashAddress, value.to_ulong ());
105 if (!result) write (_flashAddress, value.to_ulong ());
106 if (otp.lock () == false) {
107 assert (0);
108 return false;
109 }
110 return result;
111 }
112
113 /**
114 * @brief Checks if the configuration is empty.
115 *
116 * This function reads the value at the specified flash address and
117 * compares it to 0xffffffff. If the value is 0xffffffff, it indicates
118 * that the configuration is empty.
119 *
120 * @return true if the configuration is empty, false otherwise.
121 */
122 virtual bool isEmpty () const { return (readValue (_flashAddress) == 0xffff'ffff); }
123
124 /**
125 * @brief Checks if the value at the specified flash address is valid.
126 *
127 * This function reads the value from the flash memory at the address specified
128 * by `_flashAddress` and checks if it matches the `VALID_MASK`. If the value
129 * matches the mask, the function returns `true`, indicating that the value is
130 * valid. Otherwise, it returns `false`.
131 *
132 * @return true if the value at the flash address is valid, false otherwise.
133 */
134 virtual bool isValid () { return ((readValue (_flashAddress) & VALID_MASK) == 0); }
135
136 private:
137
138 /**
139 * @brief Reads a 32-bit value from the specified flash address.
140 *
141 * This function reads a 32-bit value from the flash memory at the given address.
142 * It uses the Lib::HAL::OTP::getInstance().read<uint32_t> method to perform the read operation.
143 *
144 * @param flashAddr The address in flash memory from which to read the 32-bit value.
145 * @return uint32_t The 32-bit value read from the specified flash address.
146 */
147 uint32_t readValue (uint32_t flashAddr) const {
148 return Lib::HAL::OTP::getInstance ().read<uint32_t> (flashAddr);
149 }
150
151 /**
152 * @brief The flash memory address.
153 *
154 * This constant holds the address of the flash memory. It is used to
155 * reference the starting point of the flash memory in the system.
156 */
157 const uint32_t _flashAddress;
158 };
