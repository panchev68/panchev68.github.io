89 *
90 * This function writes a DateTime value to a specific flash memory address.
91 * It first checks if the memory is empty and if the DateTime value is valid.
92 * If both conditions are met, it unlocks the OTP (One-Time Programmable) memory,
93 * packs the DateTime value, and writes it to the flash memory address.
94 * If the first write attempt fails, it retries once. The function asserts
95 * if the write operation fails after the retry or if the OTP lock operation fails.
96 *
97 * @param value The DateTime value to be written to the flash memory.
98 * @return true if the write operation is successful, false otherwise.
99 */
100 virtual bool write (const Date & value) {
101 DateTime dt;
102 dt.date = value;
103 bool result;
104 if (isEmpty () == false) return false;
105 if (value.isValid () == false) return false;
106 auto & otp = Lib::HAL::OTP::getInstance ();
107 if (otp.unlock () == false) assert (0);
108 auto pack = DateTime::pack (dt);
109 result = otp.write (_flashAddress, pack);
110 if (result == false) result = otp.write (_flashAddress, pack);
111 assert (result);
112 if (otp.lock () == false) assert (0);
113 return result;
114 }
