/**
* @file Parameters.cpp
* @brief Implementation of the Parameters class for managing system parameters.
*
* This file contains the implementation of the Parameters class, which is a singleton
* that manages the parameters for the system. The Parameters class inherits from
* Lib::System::ParametersBase and provides methods to setup, retrieve, and write
* parameter data. It also contains a nested Storage class that handles the storage
* operations.
*
* The ParametersData structure is defined as a packed structure to ensure that there
* is no padding between its members, which is important for memory-constrained
* environments or when interfacing with hardware or network protocols that require
* a specific data layout.
*
* The file also includes example functions to demonstrate how to setup the parameters
* and change a specific parameter.
*
* @date 08.01.2025
*/
#pragma once
#include "ParametersData.hpp"
#include <Lib/System/ParametersBase.hpp>
namespace System {
/**
* @class Parameters
* @brief Manages the parameters for the system.
*
* The Parameters class is a singleton that manages the parameters for the system.
* It inherits from Lib::System::ParametersBase and provides methods to setup,
* retrieve, and write parameter data. It also contains a nested Storage class
* that handles the storage operations.
*/
class Parameters : public Lib::System::ParametersBase<ParametersData> {
using ParametersBase = Lib::System::ParametersBase<ParametersData>;
public:
/**
* @brief Retrieves the singleton instance of the Parameters class.
*
* This function returns a reference to the singleton instance of the Parameters class.
* The instance is created the first time this function is called and will be destroyed
* when the program exits.
*
* @return Parameters& Reference to the singleton instance of the Parameters class.
*/
static Parameters & getInstance () {
static Parameters instance;
return instance;
}
/**
* @brief Sets up the parameters.
*
* This function overrides the setup function from the base class.
* It initializes the parameters by calling the setup function
* from the Lib::System::ParametersBase class with ParametersData.
*
* @return true if the setup is successful, false otherwise.
*/
virtual bool setup () override {
return Lib::System::ParametersBase<ParametersData>::setup ();
}
/**
* @brief Retrieves the parameter data.
*
* This function overrides the base class implementation to return
* a pointer to the parameter data.
*
* @return ParametersData* Pointer to the parameter data.
*/
ParametersData * getData () const override {
return ParametersBase::getData ();
}
/**
* @brief Writes the parameters to the storage.
*
* This function overrides the write method from the base class ParametersBase.
* It calls the base class's write method to perform the actual write operation.
*
* @return true if the write operation is successful, false otherwise.
*/
bool write () override { return ParametersBase::write (); }
private:
/**
* @class Storage
* @brief A class that provides storage functionalities by inheriting from StorageBase.
*
* The Storage class implements methods to open, close, read from, and write to a storage medium.
*/
class Storage : public StorageBase {
public:
/**
* @brief Default constructor for the Storage class.
*
* This constructor initializes the Storage class with default values.
*/
Storage () = default;
/**
* @brief Virtual destructor for the Storage class.
*
* This destructor is declared as default, meaning it will use the compiler-generated
* default implementation. It ensures that derived class destructors are called
* correctly when an object is deleted through a pointer to the base class.
*/
virtual ~Storage () = default;
/**
* @brief Opens the resource.
*
* This function overrides the base class implementation to open the resource.
*
* @return true if the resource was successfully opened, false otherwise.
*/
virtual bool open () override;
/**
* @brief Closes the resource.
*
* This function overrides the base class implementation to close the resource.
*
* @return true if the resource was successfully closed, false otherwise.
*/
virtual bool close () override;
/**
* @brief Reads data into the provided destination buffer.
*
* This function reads a specified number of bytes into the buffer pointed to by `dst`.
*
* @param dst Pointer to the destination buffer where the read data will be stored.
* @param len The number of bytes to read into the destination buffer.
* @return true if the read operation was successful, false otherwise.
*/
virtual bool read (void * dst, size_t len) override;
/**
* @brief Writes data to a destination.
*
* This function writes a specified number of bytes from the source buffer to the destination.
*
* @param src Pointer to the source buffer containing the data to be written.
* @param len The number of bytes to write from the source buffer.
* @return true if the write operation was successful, false otherwise.
*/
virtual bool write (const void * src, size_t len) override;
};
/**
* @brief Constructs a Parameters object.
*
* This constructor initializes a Parameters object by calling the base class
* constructor with a unique pointer to a Storage object.
*/
Parameters () : ParametersBase (std::make_unique<Storage> ()) {}
/**
* @brief Restores default parameters.
*
* This function overrides the base class implementation to restore
* the default values for parameters. It sets `parameter1` to 1 and
* `parameter2` to 2.
*
* @return true if the parameters were successfully restored, false otherwise.
*/
bool restoreByDefault () override {
auto data = getData ();
data->parameter1 = 1;
data->parameter2 = 2;
}
};
