* protocols that require a specific data layout.
*
* @note The structure is defined within the System namespace.
*/
#pragma once
namespace System {
/**
* @brief A packed structure to hold parameter data.
*
* This structure is packed to ensure that there is no padding between its members,
* which is important for memory-constrained environments or when interfacing with
* hardware or network protocols that require a specific data layout.
*/
__packed struct ParametersData {
uint16_t parameter1;
uint32_t parameter2;
};
} // namespace System

/**
* @file System.cpp
* @brief Implementation of the System class and its nested Parameters class.
*
* This file contains the implementation of the System class, which is a singleton
* that provides system-level functionalities. It also includes the nested Parameters
* class, which handles system parameters and their storage in EEPROM.
*
* @note This file is part of the IAR Embedded Workbench project.
*
* @details
* The System class inherits from Lib::System::System and provides a singleton
* instance through the getInstance() method. It contains a nested Parameters class
* that manages system parameters using a storage mechanism implemented with EEPROM.
*
* The Parameters class inherits from Lib::System::ParametersBase and overrides
* methods for setup, data retrieval, writing, and restoring default parameters.
* The storage mechanism is implemented in the nested Storage class, which interacts
* with the BSP::M24C64W EEPROM.
*
* @date 08.01.2025
*/
#pragma once
#include "ParametersData.hpp"
#include "Project/BSP/M24C64W.hpp"
#include "VersionDef.hpp"
#include <Lib/System/ParametersBase.hpp>
#include <Lib/System/System.hpp>
namespace System {
/// @typedef SystemBase
/// @brief Alias for the Lib::System::System class.
///
/// This typedef provides a shorthand notation for referring to the
/// Lib::System::System class, making the code more concise and readable.
using SystemBase = Lib::System::System;
/**
* @class System
* @brief A class representing the system, inheriting from SystemBase.
*
* The System class provides a singleton instance of the system and manages its parameters.
*/
class System : public SystemBase {
public:
/**
* @typedef ParametersBase
* @brief Alias for the Lib::System::ParametersBase template class specialized with ParametersData.
*
* This typedef provides a convenient alias for the Lib::System::ParametersBase template class,
* which is specialized with the ParametersData type. It is used to simplify the code and improve readability.
*/
using ParametersBase = Lib::System::ParametersBase<ParametersData>;
/**
* @class Parameters
* @brief A class to handle system parameters.
*
* This class is designed to manage and store various system parameters.
* It provides an interface to access and modify these parameters as needed.
*/
class Parameters;
/**
* @brief Get the singleton instance of the System class.
*
* This function returns a reference to the single instance of the System class,
* ensuring that only one instance of the class exists throughout the program.
*
* @return System& Reference to the singleton instance of the System class.
*/
static System & getInstance () {
static System instance;
return instance;
}
/// @brief Reference to a Parameters object.
/// This member holds a reference to a Parameters object, which
/// contains configuration or settings used by the system.
Parameters & parameters;
/**
* @brief Sets up the system.
*
* This function initializes the system and prepares it for operation.
*
* @return true if the setup was successful, false otherwise.
*/
bool setup () { parameters.setup (); }
private:
/**
* @brief Constructor for the System class.
*
* This constructor initializes the System object.
*/
System () : Lib::System::System (FIRMWARE_VERSION, FIRMWARE_SUBVERSION, HARDWARE_VERSION, HARDWARE_SUBVERSION),
parameters (*new Parameters) {}
/**
* @brief Destructor for the System class.
*
* This destructor is responsible for cleaning up any resources
* that the System class may have allocated during its lifetime.
*/
~System () = default;
};
/**
* @class Parameters
* @brief A class to manage system parameters.
*
* The Parameters class is a singleton that manages system parameters.
* It inherits from Lib::System::ParametersBase and provides methods to
* set up, retrieve, and write parameter data. It also contains a nested
* Storage class that handles the storage operations.
*/
class Parameters : public ParametersBase {
public:
/**
* @brief Constructor for the Parameters class.
*
* This constructor initializes the Parameters object by calling the
* constructor of the base class ParametersBase with a unique pointer
* to a Storage object.
*/
Parameters () : ParametersBase (std::make_unique<Storage> ()) {
}
/**
* @brief Default destructor for the Parameters class.
*
* This destructor is explicitly defaulted, meaning it will use the compiler-generated
* default implementation. It ensures that any resources managed by an instance of
* Parameters are properly released when the instance is destroyed.
*/
~Parameters () = default;
/**
* @brief Sets up the system parameters.
*
* This function overrides the setup function from the base class.
* It calls the setup function of the ParametersBase class to
* initialize the system parameters.
*
* @return true if the setup is successful, false otherwise.
*/
virtual bool setup () override { return ParametersBase::setup (); }
/**
* @brief Retrieves the data associated with the parameters.
*
* This function overrides the base class implementation to return a reference
* to the ParametersData object. It calls the base class's getData() method.
*
* @return ParametersData& Reference to the ParametersData object.
*/
virtual ParametersData & getData () { return ParametersBase::getData (); }
/**
* @brief Writes the current parameters to the underlying storage.
*
* This function overrides the write method from the ParametersBase class.
* It performs the necessary operations to save the current state of the parameters.
*
* @return true if the write operation was successful, false otherwise.
*/
virtual bool write () { return ParametersBase::write (); }
/**
* @brief Restores the system to its default state.
*
* This function overrides the base class implementation to restore
* the system to its default state. It modifies the `parameter1` of
* the data object to a default value of 1234.
*
* @return true Always returns true indicating the restore operation
* was successful.
*/
bool restoreByDefault () override {
auto data = getData ();
data->parameter1 = 1234;
return true;
}
private:
/**
* @class Storage
* @brief A class that provides storage operations using EEPROM.
*
* This class inherits from ParametersBase::StorageBase and implements
* methods to open, close, read from, and write to EEPROM storage.
*
* @note The EEPROM used is M24C64W from the BSP library.
*/
class Storage : public ParametersBase::StorageBase {
/**
* @brief Starting address for parameter storage.
*
* This constant defines the starting address in memory where parameters
* are stored. It is set to 0, indicating the beginning of the memory space.
*/
constexpr static uint32_t START_PARM_ADDRESS = 0;
public:
/**
* @brief Constructor for the Storage class.
*
* This constructor initializes the Storage object by obtaining an instance
* of the M24C64W EEPROM from the BSP (Board Support Package).
*/
Storage () : _eeprom (BSP::M24C64W::getInstance ()) {}
/**
* @brief Default destructor for the Storage class.
*
* This destructor is explicitly defaulted, meaning it will be automatically
* generated by the compiler. It ensures that any resources managed by the
* Storage class are properly released when an instance of the class is destroyed.
*/
~Storage () = default;
/**
* @brief Opens the EEPROM.
*
* This function overrides the base class implementation to open the EEPROM.
*
* @return true if the EEPROM was successfully opened, false otherwise.
*/
bool open () override { return _eeprom.open (); }
/**
* @brief Closes the EEPROM interface.
*
* This function overrides the base class implementation to close the EEPROM interface.
*
* @return true if the EEPROM interface was successfully closed, false otherwise.
*/
bool close () override { return _eeprom.close (); }
/**
* @brief Reads data from the EEPROM.
*
* This function reads a specified number of bytes from the EEPROM starting
* from a predefined flash address and stores the data in the provided
* destination buffer.
*
* @param dst Pointer to the destination buffer where the read data will be stored.
* @param len Number of bytes to read from the EEPROM.
* @return true if the read operation was successful, false otherwise.
*/
bool read (void * dst, size_t len) override { return _eeprom.read (START_PARM_ADDRESS, dst, len); }
/**
* @brief Writes data to the EEPROM.
*
* This function writes a specified number of bytes from the source buffer to the EEPROM
* at a predefined flash address.
*
* @param src Pointer to the source buffer containing the data to be written.
* @param len The number of bytes to write from the source buffer.
* @return True if the write operation was successful, false otherwise.
*/
bool write (const void * src, size_t len) override { return _eeprom.write (START_PARM_ADDRESS, src, len); }
private:
/**
* @brief Reference to an instance of the M24C64W EEPROM.
*/
BSP::M24C64W & _eeprom;
};
};
} // namespace System
