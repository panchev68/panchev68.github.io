92 /**
93 * @brief Converts the version information to a string representation.
94 *
95 * This function formats the version information stored in the object into
96 * a string in the format "major.minor.subversion". The major and minor
97 * versions are extracted from the _version member variable, while the
98 * subversion is obtained from the getSubVersion() method.

100 * @return A string representing the version in the format "major.minor.subversion".

103 std::array<char, 10> result;
104 auto len = snprintf (result.data (), result.size (), "%u.%02u.%02u", (_version >> 8) & 0xff, (_version) & 0xff, getSubVersion ());
105 return (std::string (result.begin (), result.begin () + len));

133 /**
134 * @brief Converts a version string in the format "X.Y" to a 32-bit unsigned integer.
135 *
136 * This function takes a version string with two components separated by a dot (e.g., "1.2")
137 * and converts it to a 32-bit unsigned integer. The first component is placed in the higher
138 * byte and the second component is placed in the lower byte of the resulting integer.
139 *
140 * @param string The version string to convert. It should be in the format "X.Y".
141 * @return A 32-bit unsigned integer representing the version.
142 */
143 static uint32_t stringToU32 (const std::string & string) {
144 std::array<uint8_t, 2> array ({0, 0});
145 auto begin = 0;
146 for (auto i = 0; i < array.size (); i++) {
147 auto found = string.find ('.', begin);
148 if (found == std::string::npos) found = string.size ();
149 array [i] = std::stol (string.substr (begin, found));
150 begin = found + 1;
151 if (found == string.size ()) break;
152 }
153 return static_cast<uint32_t> (array.at (0)) << 8 | static_cast<uint32_t> (array.at (1));
154 }

198 * @brief Retrieves the build date and time of the application.
199 *
200 * This function uses the predefined macros __TIME__ and __DATE__ to obtain the
201 * build time and date, respectively. It constructs a DateTime object with the
202 * extracted values and returns the date as a string.
203 *
204 * @return A string representing the build date in the format "YYYY-MM-DD".
205 */
206 virtual const std::string getBuildDate () const {
207 auto buildDateTime = [] (const char * ts = __TIME__, const char * ds = __DATE__) {
208 auto getDigit = [] (const char * ptr, uint32_t offset) -> uint8_t {
209 assert (ptr);
210 return ptr [offset] - '0';
211 };
212 Lib::Helper::DateTime dt;
213 dt.time.hours = (uint8_t) (10 * getDigit (ts, 0) + getDigit (ts, 1));
214 dt.time.minutes = (uint8_t) (10 * getDigit (ts, 3) + getDigit (ts, 4));
215 dt.time.seconds = ((uint8_t) (10 * getDigit (ts, 6) + getDigit (ts, 7)));
216 dt.date.year = ((uint16_t) (1000 * getDigit (ds, 7) + 100 * getDigit (ds, 8) + 10 * getDigit (ds, 9) + getDigit (ds, 10)));
217 dt.date.month = ((uint8_t) (ds [2] == 'b' ? 2 : (ds [2] == 'y' ? 5 : (ds [2] == 'l' ? 7 : (ds [2] == 'g' ? 8 : (ds [2] == 'p' ? 9 : (ds [2] == 't' ? 10 : (ds [2] == 'v' ? 11 : (ds [2] == 'c' ? 12 : (ds [2] == 'n' ? (ds [1] == 'a' ? 1 : 6) : (ds [1] == 'a' ? 3 : 4)))))))))));
218 dt.date.day = ((uint8_t) (10 * (ds [4] == ' ' ? 0 : getDigit (ds, 4)) + getDigit (ds, 5)));
219 return dt;
220 };
221 return buildDateTime ().date.toString ();
