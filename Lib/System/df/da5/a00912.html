Example of using the ParametersBase class
/**
* @file ParametersBase.hpp
* @brief This file contains the definition and implementation of the Parameters class and its nested Storage class.
*
* The Parameters class is a singleton that manages the parameters for the system. It inherits from
* Lib::System::ParametersBase and provides methods to setup, retrieve, and write parameter data.
* The nested Storage class handles the storage operations using an external EEPROM memory chip of type M24C64W.
*
* @note The EEPROM operations are performed at a fixed flash address defined by FLASH_ADDRESS.
*
* @author Bobby
* @date 08.01.2025
*/
#pragma once
#include "Project/BSP/M24C64W.hpp"
#include <Lib/System/ParametersBase.hpp>
namespace System {
__packed struct ParametersData {
uint32_t parameter1;
};
} // namespace System
namespace BSP {
/**
* @brief External EEPROM instance of type M24C64W.
*
* This variable represents an external EEPROM memory chip of type M24C64W.
* It is declared as an external variable, meaning it is defined in another
* translation unit and can be used across multiple files.
*/
extern M24C64W eeprom;
} // namespace BSP
namespace System {
using ParametersBase = Lib::System::ParametersBase<ParametersData>;
/**
* @class Parameters
* @brief Manages the parameters for the system.
*
* The Parameters class is a singleton that manages the parameters for the system.
* It inherits from Lib::System::ParametersBase and provides methods to setup,
* retrieve, and write parameter data. It also contains a nested Storage class
* that handles the storage operations.
*/
class Parameters : public ParametersBase {
/**
* @class Parameters::Storage
* @brief A class that provides storage functionalities for parameters using EEPROM.
*
* This class inherits from ParametersBase::StorageBase and implements methods to
* open, close, read from, and write to EEPROM storage.
*
* @note The EEPROM operations are performed at a fixed flash address defined by FLASH_ADDRESS.
*/
class Parameters::Storage : public ParametersBase::StorageBase {
constexpr static uint32_t FLASH_ADDRESS = 0;
public:
/**
* @brief Opens the EEPROM interface.
*
* This function overrides the base class implementation to open the EEPROM
* interface using the BSP (Board Support Package) layer.
*
* @return true if the EEPROM interface was successfully opened, false otherwise.
*/
bool open () override { return BSP::eeprom.open (); }
/**
* @brief Closes the EEPROM interface.
*
* This function overrides the base class implementation to close the EEPROM
* interface using the BSP (Board Support Package) functionality.
*
* @return true if the EEPROM interface was successfully closed, false otherwise.
*/
bool close () override { return BSP::eeprom.close (); }
/**
* @brief Reads data from EEPROM into the provided destination buffer.
*
* This function overrides the base class read function. It reads a specified
* length of data from the EEPROM starting at a predefined flash address and
* stores it in the provided destination buffer.
*
* @param dst Pointer to the destination buffer where the read data will be stored.
* @param len The number of bytes to read from the EEPROM.
* @return true if the read operation was successful, false otherwise.
*/
bool read (void * dst, size_t len) override { return BSP::eeprom.read (FLASH_ADDRESS, dst, len); }
/**
* @brief Writes data to the EEPROM.
*
* This function writes a block of data to the EEPROM at a specified address.
*
* @param src Pointer to the source data to be written.
* @param len Length of the data to be written in bytes.
* @return true if the write operation was successful, false otherwise.
*/
bool write (const void * src, size_t len) override { return BSP::eeprom.write (FLASH_ADDRESS, src, len); }
};
public:
/**
* @brief Constructs a Parameters object.
*
* This constructor initializes a Parameters object by calling the
* constructor of the base class ParametersBase with a unique pointer
* to a Storage object.
*/
Parameters () : ParametersBase (std::make_unique<Storage> ()) {}
/**
* @brief Destructor for the Parameters class.
*/
~Parameters () = default;
/**
* @brief Sets up the necessary parameters or configurations.
*
* This function overrides the base class setup function to provide
* specific setup functionality for the derived class.
*
* @return true if the setup is successful, false otherwise.
*/
virtual bool setup () override { return ParametersBase::setup (); }
/**
* @brief Retrieves the data.
*
* This function overrides the base class getData function to provide
* access to the data of type ParametersData.
*
* @return A reference to the data of type ParametersData.
*/
virtual ParametersData * getData () const override { return ParametersBase::getData (); }
/**
* @brief Writes the data to storage.
*
* This function overrides the base class write function to write the
* data to storage using the EEPROM interface.
*
* @return true if the write operation was successful, false otherwise.
*/
virtual bool write () override { return ParametersBase::write (); }
/**
* @brief Restores the parameters to their default values.
*
* This function overrides the base class restoreByDefault function to
* restore the parameters to their default values. It sets the parameter1
* value to 1234.
*
* @return true if the parameters were successfully restored to their default values, false otherwise.
*/
bool restoreByDefault () override {
auto data = getData (); // get the pointer to the data
data.parameter1 = 1234; // set the parameter1 value to 1234
return true;
}
private:
};
} // namespace System
