#pragma once
#include "ParametersData.hpp"
#include <Lib/System/ParametersBase.hpp>
namespace System {
class Parameters : public Lib::System::ParametersBase<ParametersData> {
using ParametersBase = Lib::System::ParametersBase<ParametersData>;
public:
static Parameters & getInstance () {
static Parameters instance;
return instance;
}
virtual bool setup () override {
return Lib::System::ParametersBase<ParametersData>::setup ();
}
ParametersData * getData () const override {
return ParametersBase::getData ();
}
bool write () override { return ParametersBase::write (); }
private:
class Storage : public StorageBase {
public:
Storage () = default;
virtual ~Storage () = default;
virtual bool open () override;
virtual bool close () override;
virtual bool read (void * dst, size_t len) override;
virtual bool write (const void * src, size_t len) override;
};
Parameters () : ParametersBase (std::make_unique<Storage> ()) {}
bool restoreByDefault () override {
auto data = getData ();
data->parameter1 = 1;
data->parameter2 = 2;
}
};
System::Parameters parameter;
void setupExample () {
parameter.setup ();
auto data = parameter.getData ();
}
void changeParameter1Example () {
auto data = parameter.getData ();
data->parameter1 = 2;
parameter.write ();
}
} // namespace System
