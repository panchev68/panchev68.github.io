81 static uint32_t stringToU32 (const std::string & string)
82 {
83 std::array<uint8_t, 2> array ({0, 0});
84 auto begin = 0;
85 for (auto i = 0; i < array.size (); i++)
86 {
87 auto found = string.find ('.', begin);
88 if (found == std::string::npos) found = string.size ();
89 array [i] = std::stol (string.substr (begin, found));
90 begin = found + 1;
91 if (found == string.size ()) break;
92 }
93 return static_cast<uint32_t> (array.at (0)) << 8 | static_cast<uint32_t> (array.at (1));
