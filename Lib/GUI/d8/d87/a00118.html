Example of Colors structure usage

This example demonstrates the usage of the Colors structure to set the foreground and background colors of a GUI element.

#include "Colors.hpp"
#include "SomeGUIFramework.hpp"
int main()
{
Lib::GUI::Colors colors;
SomeGUIFramework::Button button;
button.setForegroundColor(colors.foreground);
button.setBackgroundColor(colors.background);
button.display();
return 0;
}
