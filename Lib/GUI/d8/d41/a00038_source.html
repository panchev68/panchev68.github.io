19 /**
20 * @class ProgressBar
21 * @brief A class to represent and manage a graphical progress bar.
22 *
23 * The ProgressBar class provides functionality to display and manage a graphical progress bar
24 * with customizable value ranges and colors. It inherits from the Canvas class and includes
25 * methods to set the progress bar value, value margins, and colors.
26 *
27 */
28 class ProgressBar : public Canvas
29 {
30 public:
31
32 /**
33 * @brief ValueMargins struct
34 *
35 * The ValueMargins struct defines the minimum and maximum values for the progress bar.
36 * It includes two uint32_t values representing the minimum and maximum values of the progress bar.
37 *
38 */
39 struct ValueMargins
40 {
41 uint32_t min; ///< The minimum value of the progress bar
42 uint32_t max; ///< The maximum value of the progress bar
43 };
44
45 /**
46 * @brief Constructs a ProgressBar object with the given Canvas.
47 *
48 * @param canvas The Canvas object to be used for the ProgressBar.
49 */
50 ProgressBar (const Canvas & canvas) : Canvas (canvas),
51 _margins ({0, 100})
52 {}

55 * @brief Destroys the ProgressBar object.
56 */
57 ~ProgressBar () = default;

65 /**
66 * @brief Invalidates the ProgressBar object.
67 */
68 void invalidate ()
69 {
70 auto calcWidtht = [this] {
71 auto & margins = getValueMargins ();
72 return ((getValue () + 1) - margins.min) * width / (margins.max - margins.min);
73 };
74
75 Canvas::fill (_colors.background, x, y, calcWidtht (), height);
76
77 Position a1 (x, y);
78 Position a2 (x, y + height - 1);
79 Position a3 (x + width - 1, y);
80 Position a4 (x + width - 1, y + height - 1);
81 Line (_colors.foreground, a1, a3);
82 Line (_colors.foreground, a2, a4);
83 Line (_colors.foreground, a1, a2);
84 Line (_colors.foreground, a3, a4);
85 }
86
87 /**
88 * @brief Sets the value of the progress bar.
89 *
90 * Sets the value of the progress bar to the specified value. The value is clamped to the
91 * minimum and maximum values defined by the ValueMargins struct.
92 *
93 * @param value The new value of the progress bar.
94 */
95 void setValue (uint32_t value)
96 {
97 if (value < _margins.min) value = _margins.min;
98 else if (value > _margins.max) value = _margins.max;
99
100 if (value != _value)
101 {
102 _value = value;
103 invalidate ();
104 }
105 }
106
107 /**
108 * @brief Gets the current value of the progress bar.
109 *
110 * Returns the current value of the progress bar.
111 *
112 * @return The current value of the progress bar.
113 */
114 uint32_t getValue () const
115 {
116 return _value;
117 }
118
119 /**
120 * @brief Gets the value margins of the progress bar.
121 *
122 * Returns the value margins of the progress bar.
123 *
124 * @return The value margins of the progress bar.
125 */
126 const ValueMargins & getValueMargins ()
127 {
128 return _margins;
129 }
130
131 /**
132 * @brief Sets the value margins of the progress bar.
133 *
134 * Sets the value margins of the progress bar to the specified ValueMargins struct.
135 *
136 * @param value The new value margins of the progress bar.
137 */
138 void setValueMargins (const ValueMargins & value)
139 {
140 _margins = value;
141 }
142
143 /**
144 * @brief Sets the colors for the progress bar.
145 *
146 * This function assigns the provided Colors object to the internal
147 * _colors member, updating the colors used by the progress bar.
148 *
149 * @param colors A reference to a Colors object containing the new colors.
150 */
151 void setColors (const Colors & colors)
152 {
153 _colors = colors;
154 }
155
156 private:
157
158 uint32_t _value; ///< The current value of the progress bar
159 ValueMargins _margins; ///< The value margins of the progress bar
160 Colors _colors; ///< The colors used by the progress bar
161 Text _text; ///< The text object used to display the progress bar value
162 };
