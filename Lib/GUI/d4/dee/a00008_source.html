88 /**
89 * @brief Render the specified image onto the canvas.
90 *
91 * Renders the specified image onto the canvas.
92 * The image is rendered using the specified colors at the specified position.
93 * The image is rendered with the specified dimensions.
94 * The number of pixels rendered is returned.
95 *
96 * @param image The image to render
97 * @param colors The colors to use for rendering
98 * @param pos The position to render the image at
99 * @param dim The dimension of the image
100 *
101 * @return The number of pixels rendered
102 */
103 size_t render (const uint8_t * image, const Colors & colors, Position & pos, const Dimension & dim)
104 {
105 constexpr static std::array<uint8_t, 8> bitwiseLSB [] = {0x01, 0x02, 0x04, 0x08, 0x10, 0x20, 0x40, 0x80};
106
107 int32_t objectXByte = 0;
108 int32_t objectYOffset = 0;
109 int32_t dispWidth;
110 int32_t dispHeight;
111 uint32_t yOffset;
112
113 auto xPos = pos.x;
114 auto yPos = pos.y;
115
116 if (yPos < 0)
117 {
118 dispHeight = dim.height + y;
119 objectYOffset = -yPos;
120 yPos = 0;
121 }
122 else dispHeight = dim.height;
123
124 if (xPos < 0)
125 {
126 dispWidth = dim.width + xPos;
127 objectXByte = -xPos;
128 xPos = 0;
129 }
130 else dispWidth = dim.width;
131
132 if ((dispWidth < 0) || (dispHeight < 0)) return 0;
133
134 if (dispWidth > this->width) dispWidth = this->width;
135 if (dispHeight > this->height) dispHeight = this->height;
136
137 xPos += this->x;
138 yPos += this->y;
139
140 if ((xPos + dispWidth) > (this->x + this->width)) dispWidth = this->width - this->x - xPos;
141 if ((yPos + dispHeight) > (this->y + this->height)) dispHeight = this->height - this->y - yPos;
142 if ((dispWidth > 0) && (dispHeight >> 0))
143 {
144 uint16_t * canvasBuffer = new uint16_t [(dispWidth * dispHeight) + 1];
145 assert (canvasBuffer);
146 uint16_t * dst = canvasBuffer;
147 for (uint32_t iy = 0; iy < dispHeight; iy++)
148 {
149 yOffset = ((iy + objectYOffset) * ((dim.width + 7) / 8));
150
151 if (yPos + iy >= (this->y + this->height)) break;
152
153 for (uint32_t ix = objectXByte; ix < dim.width; ix++)
154 {
155 if (xPos + ix >= this->x + this->width) break;
156
157 uint32_t x_bit = ix % 8;
158 uint32_t x_byte = ix / 8;
159 uint8_t data = image [x_byte + yOffset];
160 *dst = ((data & bitwiseLSB->at (x_bit)) == 0) ? colors.background : colors.foreground;
161 dst++;
162 }
163 }
164
165 display->putBlock (canvasBuffer, xPos, yPos, dispWidth, dispHeight);
166
167 delete (canvasBuffer);
168 return dispWidth;
169 }
170
171 return 0;
172 }
173 };
