Example of Char class usage

This is an example of how to use the Char class.

#include "Char.hpp"
#include <iostream>
int main()
{
// Example bitmap for the character face
const uint8_t face[] = {0x3C, 0x42, 0xA9, 0x85, 0xA9, 0x91, 0x42, 0x3C};
// Create a Char object with the face, width, and height
Lib::GUI::Char character(face, 8, 8);
// Output the face of the character
const uint8_t * charFace = character.getFace();
for (size_t i = 0; i < 8; ++i)
{
std::cout << std::hex << static_cast<int>(charFace[i]) << " ";
}
std::cout << std::endl;
return 0;
}
