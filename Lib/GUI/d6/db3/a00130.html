Example of ProgressBar class usage

This example demonstrates the usage of the ProgressBar class to create and manage a graphical progress bar.

// Example usage of the ProgressBar class
#include "ProgressBar.hpp"
#include "Canvas.hpp"
#include "Colors.hpp"
int main() {
Canvas canvas;
Lib::GUI::ProgressBar progressBar(canvas);
// Set value margins
Lib::GUI::ProgressBar::ValueMargins margins = {0, 100};
progressBar.setValueMargins(margins);
// Set colors
Colors colors = {0xFFFFFF, 0x000000}; // Example colors
progressBar.setColors(colors);
// Set progress bar value
progressBar.setValue(50);
// Initialize and invalidate to render the progress bar
progressBar.initialize();
progressBar.invalidate();
return 0;
}
