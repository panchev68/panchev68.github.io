32 /**

79 /**
80 * @brief Align operator
81 *
82 * Aligns the object within the specified dimension based on the alignment mode.
83 *
84 * @param source The source dimension
85 * @param object The object dimension
86 * @param position The object position
87 *
88 * @return The aligned position
89 *
90 */
91 Position operator() (const Dimension & source, const Dimension & object, const Position & position)
92 {
93 Position result = position;
94
95 switch (_mode & 0x0f)
96 {
97 default:
98 break;
99 case Mode::LEFT:
100 result.x = 0;
101 break;
102 case Mode::RIGHT:
103 result.x = source.width - object.width;
104 break;
105 case Mode::HCENTER:
106 result.x = (source.width - object.width) / 2;
107 break;
108 }
109
110 switch (_mode & 0xf0)
111 {
112 default:
113 break;
114 case Mode::TOP:
115 result.y = 0;
116 break;
117 case Mode::BOTTOM:
118 result.y = source.height - object.height;
119 break;
120 case Mode::VCENTER:
121 result.y = (source.height - object.height) / 2;
122 break;
123 }
124
125 return result;
126 }
