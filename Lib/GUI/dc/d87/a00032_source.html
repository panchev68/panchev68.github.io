74 /**
75 * @brief Draw a line on the display.
76 *
77 * Draws a line on the display using the specified color and coordinates.
78 *
79 * @param color The color of the line
80 * @param x0 The x-coordinate of the starting point of the line
81 * @param y0 The y-coordinate of the starting point of the line
82 * @param x1 The x-coordinate of the ending point of the line
83 * @param y1 The y-coordinate of the ending point of the line
84 */
85 void draw (Color color, int16_t x0, int16_t y0, int16_t x1, int16_t y1)
86 {
87 auto swap = [] (int16_t & a, int16_t & b) {
88 int16_t temp = a;
89 a = b;
90 b = temp;
91 };
92 if (x0 == x1)
93 {
94 if (y0 > y1) swap (y0, y1);
95 display->fill (color, x0, y0, 1, y1 - y0);
96 }
97 else if (y0 == y1)
98 {
99 if (x0 > x1) swap (x0, x1);
100 display->fill (color, x0, y0, x1 - x0, 1);
101 }
102 else
103 {
104 bool steep = std::abs (y1 - y0) > std::abs (x1 - x0);
105 if (steep) swap (x0, y0), swap (x1, y1);
106 if (x0 > x1) swap (x0, x1), swap (y0, y1);
107 int16_t dx = x1 - x0;
108 int16_t dy = std::abs (y1 - y0);
109 int16_t err = dx / 2;
110 int16_t ystep = (y0 < y1) ? 1 : -1;
111
112 for (; x0 <= x1; x0++)
113 {
114 if (steep) display->putPixel (color, y0, x0);
115 else display->putPixel (color, x0, y0);
116 err -= dy;
117 if (err < 0) y0 += ystep, err += dx;
118 }
119 }
120 }
121 };
