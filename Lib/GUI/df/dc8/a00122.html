Example of DisplayDriverBase class usage

This is an example of how to use the DisplayDriverBase class.

#include "DisplayDriverBase.hpp"
#include <iostream>
class MyDisplayDriver : public Lib::GUI::DisplayDriverBase
{
public:
void putPixel(uint16_t color, int16_t x, int16_t y) override
{
// Implementation for drawing a pixel
std::cout << "Drawing pixel at (" << x << ", " << y << ") with color " << color << std::endl;
}
void putBlock(const uint16_t* data, uint16_t x, uint16_t y, size_t w, size_t h) override
{
// Implementation for drawing a block of pixels
std::cout << "Drawing block at (" << x << ", " << y << ") with width " << w << " and height " << h << std::endl;
}
void fill(uint16_t color, uint16_t x, uint16_t y, size_t w, size_t h) override
{
// Implementation for filling an area with color
std::cout << "Filling area at (" << x << ", " << y << ") with width " << w << " and height " << h << " with color " << color << std::endl;
}
void cls() override
{
// Implementation for clearing the screen
std::cout << "Clearing the screen" << std::endl;
}
size_t getHeight() const override
{
// Implementation for getting the height of the display
return 480;
}
size_t getWidth() const override
{
// Implementation for getting the width of the display
return 640;
}
};
int main()
{
MyDisplayDriver driver;
driver.putPixel(0xFFFF, 10, 10);
driver.putBlock(nullptr, 20, 20, 100, 50);
driver.fill(0x0000, 0, 0, 640, 480);
driver.cls();
std::cout << "Display width: " << driver.getWidth() << std::endl;
std::cout << "Display height: " << driver.getHeight() << std::endl;
return 0;
}
