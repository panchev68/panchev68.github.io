This is an example of how to use the Position class.

#include "Position.hpp"
int main() {
// Create a Position object using the default constructor
Lib::GUI::Position pos1;
// Create a Position object using the parameterized constructor

// Access the coordinates

// Output the coordinates

std::cout << "Position 2: (" << x << ", " << y << ")" << std::endl;
return 0;
}
