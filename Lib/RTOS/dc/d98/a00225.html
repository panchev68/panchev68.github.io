Example usage of the RTOS::MessageBuffer class.

Example usage of the RTOS::MessageBuffer class.Demonstrates how to create, send, and receive messages using the RTOS::MessageBuffer class in a FreeRTOS-based embedded system.

#include "MessageBuffer.hpp"
#include <cstdio>
void messageBufferExample()
{
RTOS::MessageBuffer msgBuffer;
constexpr size_t BUFFER_SIZE = 128;
if (!msgBuffer.create(BUFFER_SIZE)) {
printf("Failed to create message buffer.\n");
return;
}
const char* message = "Hello, RTOS!";
size_t sent = msgBuffer.send(message, strlen(message) + 1, 10);
printf("Sent %zu bytes.\n", sent);
char recvBuffer[32];
size_t received = msgBuffer.receive(recvBuffer, sizeof(recvBuffer), 10);
if (received > 0) {
printf("Received: %s\n", recvBuffer);
}
msgBuffer.destroy();
}
