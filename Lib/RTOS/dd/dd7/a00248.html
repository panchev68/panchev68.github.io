RTOS::TickService and RTOS::TickService::ClientBase.

RTOS::TickService and RTOS::TickService::ClientBase.Shows how to implement a client for periodic tick events.

#include "TickService.hpp"
#include <cstdio>
class MyTickClient : public RTOS::TickService::ClientBase {
public:
void handleTickEvent() override {
printf("Tick event received!\n");
}
};
int main() {
MyTickClient client;
RTOS::TickService::getInstance().start();
while (true) {
// Application main loop
}
return 0;
}
