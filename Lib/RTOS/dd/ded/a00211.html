Example usage of the Semaphore class for synchronizing tasks in FreeRTOS.

#include "RTOS.hpp"
#include <cstdint>
RTOS::Semaphore semaphore;
void senderTask(void *) {
semaphore.create();
// Signal the semaphore to unblock waiting tasks
semaphore.give();
}
void receiverTask(void *) {
semaphore.create();
// Wait for the semaphore to be given
if (semaphore.take(100)) {
// Semaphore was taken, proceed with critical section
}
}
