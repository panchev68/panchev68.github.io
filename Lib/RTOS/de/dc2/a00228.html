usage of RTOS::MemoryManager.

usage of RTOS::MemoryManager.Demonstrates allocation, deallocation, and memory monitoring in embedded real-time systems.

// Basic allocation
void* buffer = RTOS::MemoryManager::allocate(1024);
if (buffer) {
memset(buffer, 0, 1024);
RTOS::MemoryManager::deallocate(buffer);
}
// Allocation with status reporting
RTOS::MemoryManager::Status status;
void* safeBuffer = RTOS::MemoryManager::allocate(2048, status);
switch (status) {
case RTOS::MemoryManager::Status::SUCCESS: break;
case RTOS::MemoryManager::Status::OUT_OF_MEMORY: break;
case RTOS::MemoryManager::Status::SIZE_TOO_LARGE: break;
case RTOS::MemoryManager::Status::INVALID_SIZE: break;
}
// Safe deallocation with status
if (safeBuffer) {
RTOS::MemoryManager::Status deallocStatus;
RTOS::MemoryManager::deallocate(safeBuffer, deallocStatus);
if (deallocStatus == RTOS::MemoryManager::Status::SUCCESS) {
// Deallocation successful
}
}
// Memory monitoring
size_t freeHeap = RTOS::MemoryManager::getFreeHeapSize();
size_t minFreeHeap = RTOS::MemoryManager::getMinimumEverFreeHeapSize();
uint8_t usagePercent = RTOS::MemoryManager::getHeapUsagePercent();
printf("Free heap: %zu bytes\n", freeHeap);
printf("Minimum ever free: %zu bytes\n", minFreeHeap);
printf("Heap utilization: %u%%\n", usagePercent);
// Allocation size validation
size_t requestedSize = 4096;
if (RTOS::MemoryManager::isValidSize(requestedSize)) {
void* validatedBuffer = RTOS::MemoryManager::allocate(requestedSize);
if (validatedBuffer) {
RTOS::MemoryManager::deallocate(validatedBuffer);
}
}
#if MEMORY_MANAGER_DIAG
// Diagnostic tracking (when enabled)
const auto& diag = RTOS::MemoryManager::getDiagnostics();
printf("Allocations: %lu\n", diag.allocated);
printf("Deallocations: %lu\n", diag.deallocated);
printf("Active allocations: %lu\n", diag.active);
printf("Failed allocations: %lu\n", diag.failedAllocations);
printf("Peak usage: %zu bytes\n", diag.peakUsage);
printf("Current usage: %zu bytes\n", diag.currentUsage);
RTOS::MemoryManager::resetDiagnostics();
#endif
// RAII-style management with custom deleter
auto customDeleter = [](void* ptr) {
if (ptr) RTOS::MemoryManager::deallocate(ptr);
};
std::unique_ptr<uint8_t[], decltype(customDeleter)> managedBuffer(
static_cast<uint8_t*>(RTOS::MemoryManager::allocate(512)),
customDeleter
);
if (managedBuffer) {
std::fill(managedBuffer.get(), managedBuffer.get() + 512, 0xFF);
}
// Memory pool simulation
class MemoryPool {
static constexpr size_t BLOCK_SIZE = 64;
static constexpr size_t POOL_SIZE = 10;
void* blocks[POOL_SIZE];
bool allocated[POOL_SIZE];
public:
MemoryPool() {
for (size_t i = 0; i < POOL_SIZE; ++i) {
blocks[i] = RTOS::MemoryManager::allocate(BLOCK_SIZE);
allocated[i] = false;
}
}
~MemoryPool() {
for (size_t i = 0; i < POOL_SIZE; ++i) {
if (blocks[i]) RTOS::MemoryManager::deallocate(blocks[i]);
}
}
void* acquire() {
for (size_t i = 0; i < POOL_SIZE; ++i) {
if (blocks[i] && !allocated[i]) {
allocated[i] = true;
return blocks[i];
}
}
return nullptr;
}
void release(void* ptr) {
for (size_t i = 0; i < POOL_SIZE; ++i) {
if (blocks[i] == ptr) {
allocated[i] = false;
break;
}
}
}
};
MemoryPool pool;
void* poolBlock = pool.acquire();
if (poolBlock) {
pool.release(poolBlock);
}
