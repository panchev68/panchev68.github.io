Example usage of the Queue class template for passing integer messages between tasks.

#include "RTOS.hpp"
#include <cstdint>
void senderTask(void *) {
RTOS::Queue<int32_t, 10> queue;
queue.create();
int32_t value = 42;
queue.send(value, 10);
}
void receiverTask(void *) {
RTOS::Queue<int32_t, 10> queue;
queue.create();
int32_t received = 0;
if (queue.receive(received, 10)) {
// Process received value
}
}
