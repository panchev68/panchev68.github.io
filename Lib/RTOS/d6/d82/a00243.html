RTOS::Kernel

.Demonstrates enhanced kernel management, RAII guards, and system monitoring.

#include "Karnel.hpp"
#include <cstdio>
void kernelExample() {
// Check kernel state
if (RTOS::Kernel::isSchedulerRunning()) {
printf("Scheduler is running\n");
printf("Number of tasks: %lu\n", RTOS::Kernel::getNumberOfTasks());
printf("System uptime: %lu seconds\n", RTOS::Kernel::getUptimeSeconds());
}
// Memory monitoring
auto memInfo = RTOS::Kernel::getMemoryInfo();
printf("Free heap: %zu bytes (%u%% used)\n",
memInfo.freeHeap, RTOS::Kernel::getHeapUsagePercent());
printf("Minimum free heap: %zu bytes\n", memInfo.minimumFreeHeap);
// CPU usage monitoring
uint8_t cpuUsage = RTOS::Kernel::getCpuUsagePercent();
printf("CPU usage: %u%%\n", cpuUsage);
// Critical section with RAII
{
RTOS::Kernel::CriticalSection cs;
// Critical code here - interrupts disabled
printf("In critical section\n");
// Interrupts automatically restored when cs goes out of scope
}
// Function-based critical section
RTOS::Kernel::withCriticalSection([]() {
// Critical code here
printf("In lambda critical section\n");
});
// Scheduler suspension with RAII
{
RTOS::Kernel::SchedulerSuspension suspension;
// Scheduler suspended - no task switches
printf("Scheduler suspended\n");
// Scheduler automatically resumed when suspension goes out of scope
}
// Function-based scheduler suspension
RTOS::Kernel::withSchedulerSuspended([]() {
// Code with scheduler suspended
printf("In suspended scheduler section\n");
});
// Time utilities
printf("Tick rate: %lu Hz\n", RTOS::Kernel::getTickRateHz());
printf("Current tick: %lu\n", RTOS::Kernel::getTickCount());
printf("Current time: %lu ms\n", RTOS::Kernel::getTickCountMs());
// Task management
printf("Max priority: %lu\n", RTOS::Kernel::getMaxPriority());
if (RTOS::Kernel::isValidPriority(5)) {
printf("Priority 5 is valid\n");
}
// Delays
RTOS::Kernel::delayMs(100); // Delay 100ms
RTOS::Kernel::delaySeconds(1); // Delay 1 second
}
void periodicTask() {
auto lastWakeTime = RTOS::Kernel::getTickCount();
const auto period = RTOS::TickCounter::msToTicks(1000); // 1 second
while (true) {
// Do periodic work
printf("Periodic task running at %lu ms\n", RTOS::Kernel::getTickCountMs());
// Wait for next period
RTOS::Kernel::delayUntil(&lastWakeTime, period);
}
}

void timerISR() {
// Use ISR-specific critical section
RTOS::Kernel::CriticalSectionISR cs;
// ISR-specific operations
auto currentTick = RTOS::Kernel::getTickCountFromISR();
// Signal higher priority task wake-up if needed
BaseType_t woken = pdFALSE;
// ... other ISR operations that might wake tasks ...
RTOS::Kernel::yieldFromISR(woken);
}
void systemMonitorTask() {
while (true) {
// System health monitoring
printf("=== System Status ===\n");
printf("Kernel version: %s\n", RTOS::Kernel::getVersion());
printf("Uptime: %lu seconds\n", RTOS::Kernel::getUptimeSeconds());
printf("Tasks: %lu\n", RTOS::Kernel::getNumberOfTasks());
printf("CPU usage: %u%%\n", RTOS::Kernel::getCpuUsagePercent());
printf("Heap usage: %u%%\n", RTOS::Kernel::getHeapUsagePercent());
printf("Free heap: %zu bytes\n", RTOS::Kernel::getFreeHeapSize());
// Sleep for 5 seconds
RTOS::Kernel::delaySeconds(5);
}
}
