415
416 uint8_t bytes[2] = {0};

419 // ---------------------------------------
420 com.InstructionMode = QSPI_INSTRUCTION_1_LINE;
421 com.AddressMode = QSPI_ADDRESS_1_LINE;
422 com.AddressSize = QSPI_ADDRESS_24_BITS;
423 com.Address = 0;
424
425 com.AlternateByteMode = QSPI_ALTERNATE_BYTES_NONE;
426 com.AlternateBytes = QSPI_ALTERNATE_BYTES_NONE;
427 com.AlternateBytesSize = QSPI_ALTERNATE_BYTES_NONE;
428
429 com.DummyCycles = 0;
430 com.DataMode = QSPI_DATA_2_LINES; // 2 lines 8.2.24 Read Manufacturer / Device ID Dual I/O (92h)
431 com.NbData = sizeof (bytes);
432
433 com.DdrMode = QSPI_DDR_MODE_DISABLE;
434 com.DdrHoldHalfCycle = QSPI_DDR_HHC_ANALOG_DELAY;
435 com.SIOOMode = QSPI_SIOO_INST_EVERY_CMD;
436
437 auto res = qspi.command ();

439 if (res == false) return false;
440 id = bytes[0] << 8 | bytes[1];
441 return true;
