126 _HighFrame.play (engine_speed); // play the high engine speed frame

128 // --- The two frames are loaded
129 if ((engine_speed >= _LowFrame.getFrameEngineSpeed ()) && (engine_speed < _HighFrame.getFrameEngineSpeed ()))
130 {
131 // --- current engine speeds are within the range of charged frames
132 ratio = ((engine_speed - _LowFrame.getFrameEngineSpeed ()) << 16) / (_HighFrame.getFrameEngineSpeed () - _LowFrame.getFrameEngineSpeed ());
133 }
134 else
135 {
136 ratio = (engine_speed < _LowFrame.getFrameEngineSpeed ()) ? 0 : 0x10000;
137 }
138
139 // --- ccalculates the volume gain values for the respective frames
140 _HighFrame.VolumeRatio = (ratio > 0) ? ((ratio <= 0xffff) ? ratio[0] : 0xffff) : 0;
141 _LowFrame.VolumeRatio = _HighFrame.VolumeRatio ^ 0xffff;
142
143 // --- mixed
144 result = (_LowFrame.getPlayResult () * _LowFrame.VolumeRatio) + (_HighFrame.getPlayResult () * _HighFrame.VolumeRatio) >> 16;
145 }
146 else if (_LowFrame.isLoaded ())
147 {
148 // --- the low engine speed frame is loaded but the high speed frame is missing
149 result = _LowFrame.play (engine_speed); // play the low engine speed frame
150 }
151 else if (_HighFrame.isLoaded ())
152 {
153 // --- the high speed frame is loaded but the low engine speed frame is missing
154 result = _HighFrame.play (engine_speed); // play the high engine speed frame
