219 virtual inline auto open () -> bool
220 {
221 I2S.play (); // play begin
222 vTaskDelay (100);
223
224 bool res;
225 I2C.open ();
226 // --------------------------------------------------
227 // initalize TAS6424
228
229 // --------------------------------------------------
230 // 1. set all outputs to Hi-Z state on TAS6424 ...
231 // [ Channel State Control Register ]
232 // CH1 STATE CONTROL = 01: Hi-Z
233 // CH2 STATE CONTROL = 01: Hi-Z
234 // CH3 STATE CONTROL = 01: Hi-Z
235 // CH4 STATE CONTROL = 01: Hi-Z
236 res = _writeRegister (XRegister::CH_STATE_CTRL, 0x55);
237 assert (res);
238 // --------------------------------------------------
239 // 2. Master Reset TAS6424.
240 res = _writeRegister (XRegister::MODE_CTRL, 0x80);
241 assert (res);
242
243 vTaskDelay (10);
244
245 res = _readRegisters ();
246 assert (res);
247
248 // --------------------------------------------------
249 // 3. Set All Channels to BTL Mode
250 // [ Mode control ] (MODE_CTRL)
251 // RESET = 0 // 0: Normal operation
252 // RESERVED = 0 // RESERVED
253 // PBTL CH34 = 1 // Channels 3 and 4 are in parallel BTL mode
254 // PBTL CH10 = 1 // Channels 3 and 4 are in parallel BTL mode
255 // CH1 LO MODE = 0 // Channel 1 is in normal/speaker mode
256 // CH2 LO MODE = 0 // Channel 1 is in normal/speaker mode
257 // CH3 LO MODE = 0 // Channel 1 is in normal/speaker mode
258 // CH4 LO MODE = 0 // Channel 1 is in normal/speaker mode
259 res = _writeRegister (XRegister::MODE_CTRL, 0b00110000);
260 assert (res);
261
262 // --------------------------------------------------
263 // 4. Set Gain to Level 2 (15V) on TAS6424
264 // [ Misc Control 1 ] (MISC_CTRL1)
265 // HPF BYPASS = 1 // High pass filter eneabled
266 // OTW CONTROL = 01 // Global overtemperature warning set to 130C
267 // OC CONTROL = 1 // Overcurrent is level 2
268 // VOLUME RATE = 00 // Volume update rate is 1 step / FSYNC
269 // GAIN = 01 // Gain Level 2 = 15-V peak output voltage
270 res = _writeRegister (XRegister::MISC_CTRL1, 0b10110001); // 0xB1
271 assert (res);
272
273 // --------------------------------------------------
274 // 5. Set PWM for 44x FS on TAS6424
275 // [ Misc Control 2 ]
276 // PWM FREQUENCY = 110: 44 x fS (1.94 MHz / 2.11 MHz)
277 // OUTPUT PHASE = 10: 45 degrees output-phase switching offset
278 res = _writeRegister (XRegister::MISC_CTRL2, 0b11000011);
279 assert (res);
280
281 vTaskDelay (1);
282
283 mute (true);
284
285 return res;
286 }
